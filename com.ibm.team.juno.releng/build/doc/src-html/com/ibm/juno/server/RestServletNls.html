009package com.ibm.juno.server;
010
011import java.util.*;
012
013import com.ibm.juno.core.utils.*;
014import com.ibm.juno.server.annotation.*;
015
016/**
017 * Utility class for extracting messages from resource bundles identified through the {@link RestResource#messages()} annotation.
018 * <p>
019 * When mutiple servlets are using the same resource bundle, the key names can be prefixed with the servlet class name
020 *    to prevent name conflicts.
021 * 
022 * @author James Bognar (jbognar@us.ibm.com)
023 */
024public final class RestServletNls {
025
026   private List<ClassBundleDef> defs = new LinkedList<ClassBundleDef>();
027   private Map<Locale,SafeResourceBundle> bundles = new HashMap<Locale,SafeResourceBundle>();
028
029   /**
030    * Create an NLS bundle for the specified servlet.
031    *
032    * @param servletClass The servlet class.
033    * @param bundlePath The bundle path.
034    */
035   protected RestServletNls(Class<?> servletClass, String bundlePath) {
036      addDef(servletClass, bundlePath);
037   }
038
039   /**
040    * Add another bundle path to this NLS bundle.
041    * Order of property lookup is first-to-last.
042    *
043    * @param servletClass The servlet class.
044    * @param bundlePath The bundle path.
045    * @return This object (for method chaining).
046    */
047   protected RestServletNls addDef(Class<?> servletClass, String bundlePath) {
048      defs.add(new ClassBundleDef(servletClass, bundlePath.replace('/', '.')));
049      return this;
050   }
051
052   /**
053    * A single entry of {@link RestResource#messages()}.
054    */
055   static class ClassBundleDef {
056   private Class<?> servletClass;
057   private String bundlePath;
058      private ClassBundleDef(Class<?> servletClass, String bundlePath) {
059      this.servletClass = servletClass;
060         this.bundlePath = bundlePath;
061      }
062   }
063
064   String getMessage(Locale locale, String key, Object...args) {
065      SafeResourceBundle rb = getBundle(locale);
066      return rb.getString(key, args);
067   }
068
069   String findMessage(Locale locale, String...keys) {
070      SafeResourceBundle rb = getBundle(locale);
071      return rb.findFirstString(keys);
072   }
073
074   /**
075    * Returns the resource bundle for the specified locale.
076    *
077    * @param locale The client locale.
078    * @return The resource bundle for the specified locale.  Never <jk>null</jk>.
079    */
080   protected SafeResourceBundle getBundle(Locale locale) {
081      if (! bundles.containsKey(locale)) {
082         List<SafeResourceBundle> l = new ArrayList<SafeResourceBundle>(defs.size());
083         for (ClassBundleDef cbd : defs) {
084         ResourceBundle rb = null;
085            ClassLoader cl = cbd.servletClass.getClassLoader();
086         try {
087               rb = ResourceBundle.getBundle(cbd.bundlePath, locale, cl);
088         } catch (MissingResourceException e) {
089            try {
090                  rb = ResourceBundle.getBundle(cbd.servletClass.getPackage().getName() + '.' + cbd.bundlePath, locale, cl);
091            } catch (MissingResourceException e2) {
092            }
093         }
094            l.add(new SafeResourceBundle(rb, cbd.servletClass));
095         }
096         SafeResourceBundle b = (l.size() == 1 ? l.get(0) : new SafeResourceMultiBundle(l));
097         bundles.put(locale, b);
098
099      }
100      return bundles.get(locale);
101   }
102}
