009package com.ibm.juno.server;
010
011import com.ibm.juno.core.parser.*;
012import com.ibm.juno.core.serializer.*;
013import com.ibm.juno.server.annotation.*;
014
015/**
016 * Configurable properties for the {@link RestServlet} class.
017 * <p>
018 * Properties can be set on the {@link RestServlet} class using the {@link RestResource#properties} or {@link RestMethod#properties} annotations.
019 * <p>
020 * These properties can also be passed in as servlet init parameters.
021 * <p>
022 * These properties are only valid at the class level, not the method level.  Setting them on {@link RestMethod#properties()} has no effect.
023 *
024 * @author James Bognar (jbognar@us.ibm.com)
025 */
026public final class RestServletProperties {
027
028   /**
029    * Allow header URL parameters ({@link Boolean}, default=<jk>true</jk>).
030    * <p>
031    * When enabled, headers such as <js>"Accept"</js> and <js>"Content-Type"</js> to be passed in as URL query parameters.
032    * For example:  <js>"?Accept=text/json&Content-Type=text/json"</js>
033    * <p>
034    * Parameter names are case-insensitive.
035    * <p>
036    * Useful for debugging REST interface using only a browser.
037    */
038   public static final String REST_allowHeaderParams = "RestServlet.allowHeaderParams";
039
040   /**
041    * Allow <js>"method"</js> URL parameter for specific HTTP methods (String, default=<js>""</js>, example=<js>"HEAD,OPTIONS"</js>).
042    * <p>
043    * When specified, the HTTP method can be overridden by passing in a <js>"method"</js> URL parameter on a regular GET request.
044    * For example:  <js>"?method=OPTIONS"</js>
045    * <p>
046    * Parameter name is case-insensitive.  Use "*" to represent all methods.  For backwards compatibility, "true" also means "*".
047    * <p>
048    * Note that per the <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec9.html">HTTP specification</a>, special care should
049    *    be taken when allowing non-safe (POST, PUT, DELETE) methods to be invoked through GET requests.
050    */
051   public static final String REST_allowMethodParam = "RestServlet.allowMethodParam";
052
053   /**
054    * Allow <js>"content"</js> URL parameter ({@link Boolean}, default=<jk>true</jk>).
055    * <p>
056    * When enabled, the HTTP body content on PUT and POST requests can be passed in as text using the <js>"content"</js> URL parameter.
057    * For example:  <js>"?content={name:'John%20Smith',age:45}"</js>
058    * <p>
059    * Parameter name is case-insensitive.
060    * <p>
061    * Useful for debugging PUT and POST methods using only a browser.
062    */
063   public static final String REST_allowContentParam = "RestServlet.allowContentParam";
064
065   /**
066    * Render stack traces in HTTP response bodies when errors occur ({@link Boolean}, default=<jk>false</jk>).
067    * <p>
068    * When enabled, Java stack traces will be rendered in the output response.
069    * Useful for debugging, although allowing stack traces to be rendered may cause security concerns.
070    */
071   public static final String REST_renderResponseStackTraces = "RestServlet.renderResponseStackTraces";
072
073   /**
074    * Use stack trace hashes ({@link Boolean}, default=<jk>true</jk>).
075    * <p>
076    * When enabled, the number of times an exception has occurred will be determined based on stack trace hashsums,
077    * made available through the {@link RestException#getOccurrence()} method.
078    */
079   public static final String REST_useStackTraceHashes = "RestServlet.useStackTraceHashes";
080
081   /**
082    * The default character encoding for the request and response if not specified on the request ({@link String}>, default=<js>"utf-8"</js>).
083    */
084   public static final String REST_defaultCharset = "RestServlet.defaultCharset";
085
086
087   //--------------------------------------------------------------------------------
088   // Automatically added properties.
089   //--------------------------------------------------------------------------------
090
091   /**
092    * The request servlet path.
093    * <p>
094    * Automatically added to properties return by {@link RestServlet#createRequestProperties(com.ibm.juno.core.ObjectMap, RestRequest)}
095    *    and are therefore available through {@link SerializerContext#getProperties()} and {@link ParserContext#getProperties()}.
096    * <p>
097    * Equivalent to the value returned by {@link RestRequest#getServletPath()}
098    */
099   public static final String REST_servletPath = "RestServlet.servletPath";
100
101   /**
102    * The request servlet URI.
103    * <p>
104    * Equivalent to the value returned by {@link RestRequest#getServletURI()}
105    */
106   public static final String REST_servletURI = "RestServlet.servletURI";
107
108   /**
109    * The request servlet URI.
110    * <p>
111    * Equivalent to the value returned by {@link RestRequest#getRelativeServletURI()}
112    */
113   public static final String REST_relativeServletURI = "RestServlet.relativeServletURI";
114
115   /**
116    * The request URI path info.
117    * <p>
118    * Automatically added to properties return by {@link RestServlet#createRequestProperties(com.ibm.juno.core.ObjectMap, RestRequest)}
119    *    and are therefore available through {@link SerializerContext#getProperties()} and {@link ParserContext#getProperties()}.
120    * <p>
121    * Equivalent to the value returned by {@link RestRequest#getPathInfo()}
122    */
123   public static final String REST_pathInfo = "RestServlet.pathInfo";
124
125   /**
126    * The request URI.
127    * <p>
128    * Automatically added to properties return by {@link RestServlet#createRequestProperties(com.ibm.juno.core.ObjectMap, RestRequest)}
129    *    and are therefore available through {@link SerializerContext#getProperties()} and {@link ParserContext#getProperties()}.
130    * <p>
131    * Equivalent to the value returned by {@link RestRequest#getRequestURI()}
132    */
133   public static final String REST_requestURI = "RestServlet.requestURI";
134
135   /**
136    * The request method.
137    * <p>
138    * Automatically added to properties return by {@link RestServlet#createRequestProperties(com.ibm.juno.core.ObjectMap, RestRequest)}
139    *    and are therefore available through {@link SerializerContext#getProperties()} and {@link ParserContext#getProperties()}.
140    * <p>
141    * Equivalent to the value returned by {@link RestRequest#getMethod()}
142    */
143   public static final String REST_method = "RestServlet.method";
144}
