009package com.ibm.juno.server.response;
010
011import static javax.servlet.http.HttpServletResponse.*;
012
013import java.io.*;
014
015import com.ibm.juno.core.*;
016import com.ibm.juno.core.annotation.*;
017import com.ibm.juno.core.serializer.*;
018import com.ibm.juno.server.*;
019
020/**
021 * Response handler for POJOs not handled by other handlers.
022 * <p>
023 * This uses the serializers defined on the response to serialize the POJO.
024 * <p>
025 * The {@link Serializer} used is based on the <code>Accept</code> header on the request.
026 * <p>
027 * The <code>Content-Type</code> header is set to the mime-type defined on the selected
028 *    serializer based on the {@link Produces#contentType() @Produces.contentType} annotation.
029 * <p>
030 * This handler is registered by default on {@link RestServlet RestServlets} via the
031 *    default implementation of the {@link RestServlet#createResponseHandlers} method.
032 *
033 * @author James Bognar (jbognar@us.ibm.com)
034 */
035public class DefaultHandler implements ResponseHandler {
036
037   @Override /* ResponseHandler */
038   public boolean handle(RestRequest req, RestResponse res, Object output) throws IOException, RestException {
039      SerializerGroup g = res.getSerializerGroup();
040      String accept = req.getHeader("Accept", "");
041      String matchingAccept = g.findMatch(accept);
042      if (matchingAccept != null) {
043         Serializer<?> r = g.getSerializer(matchingAccept);
044         String contentType = r.getResponseContentType();
045         if (contentType == null)
046            contentType = res.getContentType();
047         if (contentType == null)
048            contentType = matchingAccept;
049         res.setContentType(contentType);
050         ObjectMap headers = r.getResponseHeaders(res.getProperties());
051         if (headers != null)
052            for (String key : headers.keySet())
053               res.setHeader(key, headers.getString(key));
054
055         try {
056            ObjectMap p = res.getProperties();
057            if (req.isPlainText()) {
058               p.put(SerializerProperties.SERIALIZER_useIndentation, true);
059               res.setContentType("text/plain");
060            }
061            p.append("mediaType", matchingAccept).append("characterEncoding", res.getCharacterEncoding());
062            SerializerContext ctx = r.createContext(p, req.getJavaMethod());
063            if (! r.isWriterSerializer()) {
064               OutputStreamSerializer s2 = (OutputStreamSerializer)r;
065               OutputStream os = res.getNegotiatedOutputStream();
066               s2.serialize(output, os, ctx);
067               os.close();
068            } else {
069               WriterSerializer s2 = (WriterSerializer)r;
070               Writer w = res.getNegotiatedWriter();
071               s2.serialize(output, w, ctx);
072               w.close();
073            }
074         } catch (SerializeException e) {
075            throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
076         }
077      } else {
078         throw new RestException(SC_NOT_ACCEPTABLE,
079            "Unsupported media-type in request header ''Accept'': ''{0}''\n\tSupported media-types: {1}",
080            req.getHeader("Accept", ""), g.getSupportedMediaTypes()
081         );
082      }
083      return true;
084   }
085}
