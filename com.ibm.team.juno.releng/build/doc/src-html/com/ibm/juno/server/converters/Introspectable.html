009package com.ibm.juno.server.converters;
010
011import static javax.servlet.http.HttpServletResponse.*;
012
013import com.ibm.juno.core.*;
014import com.ibm.juno.core.json.*;
015import com.ibm.juno.core.utils.*;
016import com.ibm.juno.server.*;
017
018/**
019 * Converter for enablement of {@link PojoIntrospector} support on response objects returned by a <code>@RestMethod</code> method.
020 * <p>
021 *    When enabled, public methods can be called on objects returned through the {@link RestResponse#setOutput(Object)} method.
022 * <p>
023 *    Note that opening up public methods for calling through a REST interface can be dangerous, and should
024 *    be done with caution.
025 * <p>
026 *    Java methods are invoked by passing in the following URL parameters:
027 * <ul>
028 *    <li><code>&amp;invokeMethod</code> - The Java method name, optionally with arguments if necessary to differentiate between methods.
029 *    <li><code>&amp;invokeArgs</code> - The arguments as a JSON array.
030 * </ul>
031 * <p>
032 *    See {@link PojoIntrospector} for additional information on introspecting POJO methods.
033 */
034public final class Introspectable implements RestConverter {
035
036   @Override /* RestConverter */
037   @SuppressWarnings({"unchecked", "rawtypes"})
038   public Object convert(RestRequest req, Object o, ClassMeta cm) throws RestException {
039      String method = req.getParameter("invokeMethod");
040      String args = req.getParameter("invokeArgs");
041      if (method == null)
042         return o;
043      try {
044         if (cm.getPojoFilter() != null)
045            o = cm.getPojoFilter().filter(o);
046         return new PojoIntrospector(o, JsonParser.DEFAULT).invokeMethod(method, args);
047      } catch (Exception e) {
048         e.printStackTrace();
049         return new RestException(SC_INTERNAL_SERVER_ERROR,
050            "Error occurred trying to invoke method: {0}",
051            e.getLocalizedMessage()
052         ).initCause(e);
053      }
054   }
055}
