022/**
023 * JAX-RS provider for the same serialize/parse support provided by the {@link RestServletDefault} class.
024 *
025 * @author James Bognar (jbognar@us.ibm.com)
026 */
027@Provider
028@Produces({
029   "application/json", "text/json",                 // JsonSerializer
030   "application/json+simple", "text/json+simple",   // JsonSerializer.Simple
031   "application/json+schema",                       // JsonSchemaSerializer
032   "text/xml",                                      // XmlDocSerializer
033   "text/xml+simple",                               // XmlDocSerializer.Simple
034   "text/xml+schema",                               // XmlSchemaDocSerializer
035   "text/html",                                     // HtmlDocSerializer
036   "application/x-www-form-urlencoded",             // UrlEncodingSerializer
037   "text/xml+soap",                                 // SoapXmlSerializer
038   "application/x-java-serialized-object"           // JavaSerializedObjectSerializer
039})
040@Consumes({
041   "application/json", "text/json",                 // JsonParser
042   "text/xml",                                      // XmlParser
043   "text/html",                                     // HtmlParser
044   "application/x-www-form-urlencoded",             // UrlEncodingParser
045   "application/x-java-serialized-object"           // JavaSerializedObjectParser
046})
047@JunoProvider(
048   serializers={
049      JsonSerializer.class,
050      JsonSerializer.Simple.class,
051      JsonSchemaSerializer.class,
052      XmlDocSerializer.class,
053      XmlDocSerializer.Simple.class,
054      XmlSchemaDocSerializer.class,
055      HtmlDocSerializer.class,
056      UrlEncodingSerializer.class,
057      SoapXmlSerializer.class,
058      JavaSerializedObjectSerializer.class
059   },
060   parsers={
061      JsonParser.class,
062      XmlParser.class,
063      HtmlParser.class,
064      UrlEncodingParser.class,
065      JavaSerializedObjectParser.class
066   }
067)
068public final class DefaultProvider extends BaseProvider {}
