009package com.ibm.juno.server.response;
010
011import java.io.*;
012import java.util.zip.*;
013
014import com.ibm.juno.core.utils.*;
015import com.ibm.juno.core.utils.ZipFileList.ZipFileEntry;
016import com.ibm.juno.server.*;
017import com.ibm.juno.server.annotation.*;
018
019/**
020 * Response handler for ZipFileList objects.
021 * <p>
022 * Can be associated with a REST resource using the {@link RestResource#responseHandlers} annotation.
023 * <p>
024 * Sets the following headers:
025 * <ul>
026 *    <li><code>Content-Type</code> - <code>application/zip</code>
027 *    <li><code>Content-Disposition=attachment;filename=X</code> - Sets X to the file name passed in through
028 *       the constructor {@link ZipFileList#ZipFileList(String)}.
029 * </ul>
030 */
031public class ZipFileListResponseHandler implements ResponseHandler {
032
033   @Override /* ResponseHandler */
034   public boolean handle(RestRequest req, RestResponse res, Object output) throws IOException, RestException {
035      if (output.getClass() == ZipFileList.class) {
036         ZipFileList m = (ZipFileList)output;
037         res.setContentType("application/zip"); //$NON-NLS-1$
038         res.setHeader("Content-Disposition", "attachment;filename=" + m.fileName); //$NON-NLS-1$ //$NON-NLS-2$
039         OutputStream os = res.getOutputStream();
040         try {
041            ZipOutputStream zos = new ZipOutputStream(os);
042            try {
043               for (ZipFileEntry e : m)
044                  e.write(zos);
045            } catch (Exception e) {
046               e.printStackTrace();
047            } finally {
048               zos.flush();
049               zos.close();
050            }
051         } finally {
052            os.flush();
053         }
054         return true;
055      }
056      return false;
057   }
058}
