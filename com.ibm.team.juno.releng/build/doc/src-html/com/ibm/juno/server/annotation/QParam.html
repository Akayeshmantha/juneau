009package com.ibm.juno.server.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015
016import com.ibm.juno.server.*;
017
018/**
019 * Identical to {@link Param @Param}, but only retrieves the parameter from the
020 *    URL string, not URL-encoded form posts.
021 * <p>
022 * Unlike {@link Param @Param}, using this annotation does not result in the servlet reading the contents
023 *    of URL-encoded form posts.
024 * Therefore, this annotation can be used in conjunction with the {@link Content @Content} annotation
025 *    or {@link RestRequest#getInput(Class)} method for <code>application/x-www-form-urlencoded POST</code> calls.
026 *
027 * <h6 class='topic'>Example</h6>
028 * <p class='bcode'>
029 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>)
030 *    <jk>public void</jk> doGet(RestRequest req, RestResponse res,
031 *             <ja>@QParam</ja>(<js>"p1"</js>) <jk>int</jk> p1, <ja>@QParam</ja>(<js>"p2"</js>) String p2, <ja>@QParam</ja>(<js>"p3"</js>) UUID p3) {
032 *       ...
033 *    }
034 * </p>
035 * <p>
036 *    This is functionally equivalent to the following code...
037 * </p>
038 * <p class='bcode'>
039 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>)
040 *    <jk>public void</jk> doGet(RestRequest req, RestResponse res) {
041 *       <jk>int</jk> p1 = req.getQueryParameter(<jk>int</jk>.<jk>class</jk>, <js>"p1"</js>, 0);
042 *       String p2 = req.getQueryParameter(String.<jk>class</jk>, <js>"p2"</js>);
043 *       UUID p3 = req.getQueryParameter(UUID.<jk>class</jk>, <js>"p3"</js>);
044 *       ...
045 *    }
046 * </p>
047 *
048 * @author James Bognar (jbognar@us.ibm.com)
049 */
050@Documented
051@Target(PARAMETER)
052@Retention(RUNTIME)
053@Inherited
054public @interface QParam {
055
056   /**
057    * URL parameter name.
058    */
059   String value();
060
061   /**
062    * Specify <jk>true</jk> if using multi-part parameters to represent collections and arrays.
063    * <p>
064    *    Normally, we expect single parameters to be specified in UON notation for representing
065    *    collections of values (e.g. <js>"&key=(1,2,3)"</js>.
066    *    This annotation allows the use of multi-part parameters to represent collections
067    *    (e.g. <js>"&key=1&key=2&key=3"</js>.
068    * <p>
069    *    This setting should only be applied to Java parameters of type array or Collection.
070    */
071   boolean multipart() default false;
072}
