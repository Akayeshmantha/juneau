009package com.ibm.juno.server.labels;

011import java.util.*;
012
013import com.ibm.juno.core.annotation.*;
014import com.ibm.juno.server.*;

016/**
017 * Simple bean for describing REST methods.
018 * <p>
019 *    Primarily used for constructing tables with name/path/description/... columns on REST OPTIONS requests.
020 *
021 * @author James Bognar (jbognar@us.ibm.com)
022 */
023@Bean(properties={"httpMethod","path","javaMethod","description","input","responses","consumes","produces","matchers","guards","converters"})
024public final class MethodDescription {
025   private String javaMethod, httpMethod, path, description;
026   private String[] guards, converters, matchers;
027   private Set<Var> requestVars = new TreeSet<Var>();
028   private Map<Integer,Response> responses = new TreeMap<Integer,Response>();
029   private Collection<String> consumes, produces;
030
031   /**
032    * A possible response status.
033    */
034   public static class Response {
035
036      /** HTTP status code */
037      public int status;
038
039      /** Response description */
040      public String description;
041
042      /** Response headers set */
043      public Set<Var> output = new TreeSet<Var>();
044
045      /** Bean constructor */
046      public Response() {}
047
048      /**
049       * Constructor.
050       *
051       * @param status HTTP status code.
052       */
053      public Response(int status) {
054         this.status = status;
055         this.description = RestUtils.getHttpResponseText(status);
056      }
057
058      /**
059       * Add a response variable to this response.
060       *
061       * @param category The response variable category (typically only <js>"header"</js>).
062       * @param name The response variable name.
063       * @return The new variable object whose fields can be updated.
064       */
065      public Var addResponseVar(String category, String name) {
066         for (Var v : output)
067            if (v.matches(category, name))
068               return v;
069         Var v = new Var(category, name);
070         output.add(v);
071         return v;
072      }
073
074      /**
075       * Sets the description for this response.
076       *
077       * @param description The new description.
078       * @return This object (for method chaining).
079       */
080      public Response setDescription(String description) {
081         this.description = description;
082         return this;
083      }
084   }
085
086   /** Constructor. */
087   public MethodDescription() {}
088
089   /**
090    * Returns the javaMethod field on this label.
091    *
092    * @return The name.
093    */
094   public String getJavaMethod() {
095      return javaMethod;
096   }
097
098   /**
099    * Sets the javaMethod field on this label to a new value.
100    *
101    * @param javaMethod The new name.
102    * @return This object (for method chaining).
103    */
104   public MethodDescription setJavaMethod(String javaMethod) {
105      this.javaMethod = javaMethod;
106      return this;
107   }
108
109   /**
110    * Returns the httpMethod field on this label.
111    *
112    * @return The name.
113    */
114   public String getHttpMethod() {
115      return httpMethod;
116   }
117
118   /**
119    * Sets the httpMethod field on this label to a new value.
120    *
121    * @param httpMethod The new name.
122    * @return This object (for method chaining).
123    */
124   public MethodDescription setHttpMethod(String httpMethod) {
125      this.httpMethod = httpMethod;
126      return this;
127   }
128
129   /**
130    * Returns the path field on this label.
131    *
132    * @return The path.
133    */
134   public String getPath() {
135      return path;
136   }
137
138   /**
139    * Sets the path field on this label to a new value.
140    *
141    * @param path The new path.
142    * @return This object (for method chaining).
143    */
144   public MethodDescription setPath(String path) {
145      this.path = path;
146      return this;
147   }
148
149   /**
150    * Returns the description field on this label.
151    *
152    * @return The description.
153    */
154   public String getDescription() {
155      return description;
156   }
157
158   /**
159    * Sets the description field on this label to a new value.
160    *
161    * @param description The new description.
162    * @return This object (for method chaining).
163    */
164   public MethodDescription setDescription(String description) {
165      this.description = description;
166      return this;
167   }
168
169   /**
170    * Returns the vars field on this label.
171    *
172    * @return The vars.
173    */
174   public Collection<Var> getInput() {
175      return requestVars;
176   }
177
178   /**
179    * Returns the possible response codes returned by this method.
180    *
181    * @return The possible response codes returned by this method.
182    */
183   public Collection<Response> getResponses() {
184      return responses.values();
185   }
186
187   /**
188    * Sets the list of <code>Accept</code> header values that this method accepts if it's different
189    *    from the servlet.
190    *
191    * @param consumes The list of <code>Accept</code> header values.
192    * @return This object (for method chaining).
193    */
194   public MethodDescription setConsumes(Collection<String> consumes) {
195      this.consumes = consumes;
196      return this;
197   }
198
199   /**
200    * Returns the list of <code>Accept</code> header values that this method accepts if it's different
201    *    from the servlet.
202    *
203    * @return The list of <code>Accept</code> header values.
204    */
205   public Collection<String> getConsumes() {
206      return consumes;
207   }
208
209   /**
210    * Sets the list of <code>Content-Type</code> header values that this method accepts if it's different
211    *    from the servlet.
212    *
213    * @param produces The list of <code>Content-Type</code> header values.
214    * @return This object (for method chaining).
215    */
216   public MethodDescription setProduces(Collection<String> produces) {
217      this.produces = produces;
218      return this;
219   }
220
221   /**
222    * Returns the list of <code>Content-Type</code> header values that this method accepts if it's different
223    *    from the servlet.
224    *
225    * @return The list of <code>Content-Type</code> header values.
226    */
227   public Collection<String> getProduces() {
228      return produces;
229   }
230   /**
231    * Sets the guards field on this label to a new value.
232    *
233    * @param guards The guards associated with this method.
234    * @return This object (for method chaining).
235    */
236   public MethodDescription setGuards(Class<?>...guards) {
237      this.guards = new String[guards.length];
238      for (int i = 0; i < guards.length; i++)
239         this.guards[i] = guards[i].getSimpleName();
240      return this;
241   }
242
243   /**
244    * Returns the guards field on this label.
245    *
246    * @return The guards.
247    */
248   public String[] getGuards() {
249      return guards;
250   }
251
252   /**
253    * Sets the matchers field on this label to a new value.
254    *
255    * @param matchers The matchers associated with this method.
256    * @return This object (for method chaining).
257    */
258   public MethodDescription setMatchers(Class<?>...matchers) {
259      this.matchers = new String[matchers.length];
260      for (int i = 0; i < matchers.length; i++)
261         this.matchers[i] = matchers[i].getSimpleName();
262      return this;
263   }
264
265   /**
266    * Returns the matchers field on this label.
267    *
268    * @return The matchers.
269    */
270   public String[] getMatchers() {
271      return matchers;
272   }
273
274   /**
275    * Sets the converters field on this label to a new value.
276    *
277    * @param converters The converters associated with this method.
278    * @return This object (for method chaining).
279    */
280   public MethodDescription setConverters(Class<?>...converters) {
281      this.converters = new String[converters.length];
282      for (int i = 0; i < converters.length; i++)
283         this.converters[i] = converters[i].getSimpleName();
284      return this;
285   }
286
287   /**
288    * Returns the converters field on this label.
289    *
290    * @return The converters.
291    */
292   public String[] getConverters() {
293      return converters;
294   }
295
296   /**
297    * Add a request variable to this method description.
298    *
299    * @param category The variable category (e.g. <js>"attr"</js>, <js>"attr"</js>, <js>"header"</js>, <js>"content"</js>).
300    * @param name The variable name.
301    *    Can be <jk>null</jk> in the case of <js>"content"</js> category.
302    * @return The new variable whose fields can be modified.
303    */
304   public Var addRequestVar(String category, String name) {
305      for (Var v : requestVars)
306         if (v.matches(category, name))
307            return v;
308      Var v = new Var(category, name);
309      requestVars.add(v);
310      return v;
311   }
312
313   /**
314    * Add a possible HTTP response code from this method.
315    *
316    * @param httpStatus The HTTP status code.
317    * @return The new response object whose fields can be modified.
318    */
319   public Response addResponse(int httpStatus) {
320      if (! responses.containsKey(httpStatus))
321         responses.put(httpStatus, new Response(httpStatus));
322      return responses.get(httpStatus);
323   }
