009package com.ibm.juno.server.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015
016import com.ibm.juno.server.*;
017
018/**
019 * Annotation that can be applied to a parameter of a {@link RestMethod} annotated method
020 *    to identify it as a URL query parameter converted to a POJO.
021 *
022 * <h6 class='topic'>Example</h6>
023 * <p class='bcode'>
024 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>)
025 *    <jk>public void</jk> doGet(RestRequest req, RestResponse res,
026 *             <ja>@Param</ja>(<js>"p1"</js>) <jk>int</jk> p1, <ja>@Param</ja>(<js>"p2"</js>) String p2, <ja>@Param</ja>(<js>"p3"</js>) UUID p3) {
027 *       ...
028 *    }
029 * </p>
030 * <p>
031 *    This is functionally equivalent to the following code...
032 * </p>
033 * <p class='bcode'>
034 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>)
035 *    <jk>public void</jk> doGet(RestRequest req, RestResponse res) {
036 *       <jk>int</jk> p1 = req.getParam(<jk>int</jk>.<jk>class</jk>, <js>"p1"</js>, 0);
037 *       String p2 = req.getParam(String.<jk>class</jk>, <js>"p2"</js>);
038 *       UUID p3 = req.getParam(UUID.<jk>class</jk>, <js>"p3"</js>);
039 *       ...
040 *    }
041 * </p>
042 *
043 * <h6 class='topic'>Important note concerning FORM posts</h6>
044 * <p>
045 * This annotation should not be combined with the {@link Content @Content} annotation or {@link RestRequest#getInput(Class)} method
046 *    for <code>application/x-www-form-urlencoded POST</code> posts, since it will trigger the underlying servlet
047 *    API to parse the body content as key-value pairs resulting in empty content.
048 * <p>
049 * The {@link QParam @QParam} annotation can be used to retrieve a URL parameter
050 *    in the URL string without triggering the servlet to drain the body content.
051 *
052 * @author James Bognar (jbognar@us.ibm.com)
053 */
054@Documented
055@Target(PARAMETER)
056@Retention(RUNTIME)
057@Inherited
058public @interface Param {
059
060   /**
061    * URL parameter name.
062    */
063   String value();
064
065   /**
066    * Specify <jk>true</jk> if using multi-part parameters to represent collections and arrays.
067    * <p>
068    *    Normally, we expect single parameters to be specified in UON notation for representing
069    *    collections of values (e.g. <js>"&key=(1,2,3)"</js>.
070    *    This annotation allows the use of multi-part parameters to represent collections
071    *    (e.g. <js>"&key=1&key=2&key=3"</js>.
072    * <p>
073    *    This setting should only be applied to Java parameters of type array or Collection.
074    */
075   boolean multipart() default false;
076}
