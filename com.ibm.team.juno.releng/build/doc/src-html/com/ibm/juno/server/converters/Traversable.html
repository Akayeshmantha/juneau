018/**
019 * Converter for enablement of {@link PojoRest} support on response objects returned by a <code>@RestMethod</code> method.
020 * <p>
021 *    When enabled, objects in a POJO tree returned by the REST method can be addressed through additional URL path information.
022 *
023 * <p class='bcode'>
024 *    <jc>// Resource method on resource "http://localhost:8080/sample/addressBook"</jc>
025 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, converters=Traversable.<jk>class</jk>)
026 *    <jk>public void</jk> doGet(RestRequest req, RestResponse res) {
027 *       <jk>return new</jk> AddressBook();
028 *    }
029 *
030 *    <jc>// Sample usage</jc>
031 *    <jk>public static void</jk> main(String[] args) {
032 *       <jc>// Get the zip code of the 2nd address of the first person in the address book.</jc>
033 *       RestCall r = <jk>new</jk> RestClient().doGet(<js>"http://localhost:8080/sample/addressBook/0/addresses/1/zip"</js>);
034 *       <jk>int</jk> zip = r.getResponse(Integer.<jk>class</jk>);
035 *    }
036 * </p>
037 * <p>
038 *    See {@link PojoRest} for additional information on addressing elements in a POJO tree using URL notation.
039 */
040public final class Traversable implements RestConverter {
041
042   @Override /* RestConverter */
043   @SuppressWarnings({"rawtypes", "unchecked"})
044   public Object convert(RestRequest req, Object o, ClassMeta cm) throws RestException {
045      if (o == null)
046         return null;
047
048      if (req.getPathRemainder() != null) {
049         try {
050            if (cm.getPojoFilter() != null)
051               o = cm.getPojoFilter().filter(o);
052            PojoRest p = new PojoRest(o, req.getReaderParser());
053            o = p.get(req.getPathRemainder());
054         } catch (SerializeException e) {
055            throw new RestException(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, e);
056         } catch (PojoRestException e) {
057            throw new RestException(e.getStatus(), e);
058         }
059      }
060
061      return o;
062   }
063}
