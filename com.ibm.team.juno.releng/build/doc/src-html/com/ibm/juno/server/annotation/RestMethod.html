009package com.ibm.juno.server.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015
016import com.ibm.juno.core.encoders.*;
017import com.ibm.juno.core.parser.*;
018import com.ibm.juno.core.serializer.*;
019import com.ibm.juno.server.*;
020
021/**
022 * Identifies a REST Java method on a {@link RestServlet} implementation class.
023 * <p>
024 *    Refer to {@link com.ibm.juno.server} doc for information on using this class.
025 *
026 * @author James Bognar (jbognar@us.ibm.com)
027 */
028@Documented
029@Target(METHOD)
030@Retention(RUNTIME)
031@Inherited
032public @interface RestMethod {
033
034   /**
035    * REST method name.
036    * <p>
037    *    Typically <js>"GET"</js>, <js>"PUT"</js>, <js>"POST"</js>, <js>"DELETE"</js>, or <js>"OPTIONS"</js>.
038    * <p>
039    *    Can also be a non-HTTP-standard name that is passed in through a <code>&amp;method=methodName</code> URL parameter.
040    * <p>
041    *    Method names are case-insensitive (always folded to upper-case).
042    * <p>
043    *    If a method name is not specified, then the method name is determined based on the Java method name.<br>
044    *    For example, if the method is <code>doPost(...)</code>, then the method name is automatically detected as <js>"POST"</js>.
045
046    */
047   String name() default "";
048
049   /**
050    * Optional path pattern for the specified method.
051    * <p>
052    *    Appending <js>"/*"</js> to the end of the path pattern will make it match any remainder too.<br>
053    *    Not appending <js>"/*"</js> to the end of the pattern will cause a 404 (Not found) error to occur
054    *    if the exact pattern is not found.
055    */
056   String path() default "/*";
057
058   /**
059    * URL path pattern priority.
060    * <p>
061    *    To force path patterns to be checked before other path patterns, use a higher priority number.
062    * <p>
063    *    By default, it's <code>0</code>, which means it will use an internal heuristic to
064    *    determine a best match.
065    */
066   int priority() default 0;
067
068   /**
069    * Method guards.
070    * <p>
071    *    Associates one or more {@link RestGuard RestGuards} with a method call.
072    *    These guards get called immediately before execution of the REST method.
073    * <p>
074    *    Typically, guards will be used for permissions checking on the user making the request,
075    *    but it can also be used for other purposes like pre-call validation of a request.
076    */
077   Class<? extends RestGuard>[] guards() default {};
078
079   /**
080    * Method response converters.
081    * <p>
082    *    Associates one or more {@link RestConverter RestConverters} with a method call.
083    *    These converters get called immediately after execution of the REST method in the same
084    *       order specified in the annotation.
085    * <p>
086    *    Can be used for performing post-processing on the response object before serialization.
087    * <p>
088    *    Default converters are available in the {@link com.ibm.juno.server.converters} package.
089    */
090   Class<? extends RestConverter>[] converters() default {};
091
092   /**
093    * Method matchers.
094    * <p>
095    *    Associates one more more {@link RestMatcher RestMatchers} with this method.
096    * <p>
097    *    Matchers are used to allow multiple Java methods to handle requests assigned to the same
098    *       URL path pattern, but differing based on some request attribute, such as a specific header value.
099    * <p>
100    *    See {@link RestMatcher} for details.
101    */
102   Class<? extends RestMatcher>[] matchers() default {};
103
104   /**
105    * Overrides the list of serializers assigned at the method level.
106    * <p>
107    * Use this annotation when the list of serializers assigned to a method differs from the list of serializers assigned at the servlet level.
108    * <p>
109    * To append to the list of serializers assigned at the servlet level, use <code>serializersInherit=<jsf>SERIALIZERS</jsf></code>.
110    *
111    * <p class='bcode'>
112    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServlet {
113    *
114    *       <ja>@RestMethod</ja>(
115    *          name=<js>"GET"</js>,
116    *          path=<js>"/foo"</js>,
117    *          serializers=MySpecialSerializer.<jk>class</jk>,
118    *          serializersInherit=<jsf>SERIALIZERS</jsf>
119    *       )
120    *       <jk>public</jk> Object doGetWithSpecialAcceptType() {
121    *          <jc>// Handle request for special Accept type</jc>
122    *       }
123    *    }
124    * </p>
125    */
126   Class<? extends Serializer<?>>[] serializers() default {};
127
128   /**
129    * Used in conjunction with {@link #serializers()} to identify what class-level settings are inherited by the method serializer group.
130    * <p>
131    * Possible values:
132    * </p>
133    * <ul>
134    *    <li>{@link Inherit#SERIALIZERS} - Inherit class-level serializers.
135    *    <li>{@link Inherit#PROPERTIES} - Inherit class-level properties.
136    *    <li>{@link Inherit#FILTERS} - Inherit class-level filters.
137    * </ul>
138    * <p>
139    *    For example, to inherit all serializers, properties, and filters from the servlet class:
140    * </p>
141    * <p class='bcode'>
142    *    <ja>@RestMethod</ja>(
143    *       path=<js>"/foo"</js>,
144    *       serializers=MySpecialSerializer.<jk>class</jk>,
145    *       serializersInherit={<jsf>SERIALIZERS</jsf>,<jsf>PROPERTIES</jsf>,<jsf>FILTERS</jsf>}
146    *    )
147    * </p>
148    */
149   Inherit[] serializersInherit() default {};
150
151   /**
152    * Overrides the list of parsers assigned at the method level.
153    * <p>
154    * Use this annotation when the list of parsers assigned to a method differs from the list of parsers assigned at the servlet level.
155    * <p>
156    * To append to the list of serializers assigned at the servlet level, use <code>serializersInherit=<jsf>SERIALIZERS</jsf></code>.
157    *
158    * <p class='bcode'>
159    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServlet {
160    *
161    *       <ja>@RestMethod</ja>(
162    *          name=<js>"PUT"</js>,
163    *          path=<js>"/foo"</js>,
164    *          parsers=MySpecialParser.<jk>class</jk>,
165    *          parsersInherit=<jsf>PARSERS</jsf>
166    *       )
167    *       <jk>public</jk> Object doGetWithSpecialAcceptType() {
168    *          <jc>// Handle request for special Accept type</jc>
169    *       }
170    *    }
171    * </p>
172    */
173   Class<? extends Parser<?>>[] parsers() default {};
174
175   /**
176    * Used in conjunction with {@link #parsers()} to identify what class-level settings are inherited by the method parser group.
177    * <p>
178    * Possible values:
179    * <ul>
180    *    <li>{@link Inherit#PARSERS} - Inherit class-level parsers.
181    *    <li>{@link Inherit#PROPERTIES} - Inherit class-level properties.
182    *    <li>{@link Inherit#FILTERS} - Inherit class-level filters.
183    * </ul>
184    * <p>
185    *    For example, to inherit all parsers, properties, and filters from the servlet class:
186    * </p>
187    * <p class='bcode'>
188    *    <ja>@RestMethod</ja>(
189    *       path=<js>"/foo"</js>,
190    *       parsers=MySpecialParser.<jk>class</jk>,
191    *       parsersInherit={<jsf>PARSERS</jsf>,<jsf>PROPERTIES</jsf>,<jsf>FILTERS</jsf>}
192    *    )
193    * </p>
194    */
195   Inherit[] parsersInherit() default {};
196
197   /**
198    * Appends to the list of {@link Encoder encoders} specified on the servlet.
199    * <p>
200    * Use this annotation when the list of encoders assigned to a method differs from the list of encoders assigned at the servlet level.
201    * <p>
202    * These can be used to enable various kinds of compression (e.g. <js>"gzip"</js>) on requests and responses.
203    *
204    * <p class='bcode'>
205    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServlet {
206    *
207    *       <ja>@RestMethod</ja>(
208    *          name=<js>"PUT"</js>,
209    *          path=<js>"/foo"</js>,
210    *          encoders={GzipEncoder.<jk>class</jk>}
211    *       )
212    *       <jk>public</jk> Object doGetWithSpecialEncoding() {
213    *          <jc>// Handle request with special encoding</jc>
214    *       }
215    *    }
216    * </p>
217    *
218    * If you want to OVERRIDE the set of encoders specified by the servlet, combine this annotation with <code><ja>@RestMethod</ja>(inheritEncoders=<jk>false</jk>)</code>.
219    */
220   Class<? extends Encoder>[] encoders() default {};
221
222   /**
223    * Specifies whether the method should inherit encoders from the servlet.
224    */
225   boolean inheritEncoders() default true;
226
227   /**
228    * Same as {@link RestResource#properties()}, except defines property values by default when this method is called.
229    * <p>
230    * This is equivalent to simply calling <code>res.addProperties()</code> in the Java method, but is provided for convenience.
231    */
232   Property[] properties() default {};
233
234   /**
235    * Appends the specified filters to all serializers and parsers used by this method.
236    */
237   Class<?>[] filters() default {};
238
239   /**
240    * Possible HTTP response codes from this method.
241    * <p>
242    *    This annotation is provided for documentation purposes.
243    *    It is used in conjunction with the following NLS resource bundle keys to populate options pages:
244    * <ul>
245    *    <li><js>"(className.?)[javaMethodName].res.[rc]"</js> - Response description (defaults to HTTP status message).
246    *       e.g. <js>"MyClass.myMethod.res.404 = Sorry...I couldn't find that."</js> or <js>"myMethod.res.404 = Sorry...I couldn't find that."</js>
247    *    <li><js>"(className.?)[javaMethodName].res.[rc].[varCategory].[varName]"</js> - Response variable .
248    *       e.g. <js>"MyClass.myMethod.res.302.[header].[Location] = Set to the location where the thing can be found."</js> or  <js>"myMethod.res.302.[header].[Location] = Set to the location where the thing can be found."</js>
249    * </ul>
250    * <p>
251    * Note that you can still define the keys in the resource bundle and not use this annotation.
252    * However, the annotation is provided if you just want the OPTIONS page to show the possible status codes with
253    * default HTTP status messages.
254    */
255   int[] rc() default {};
256
257   /**
258    * Specifies default values for request headers.
259    * <p>
260    * Strings are of the format <js>"Header-Name: header-value"</js>.
261    * <p>
262    * Affects values returned by {@link RestRequest#getHeader(String)} when the header is not present on the request.
263    * <p>
264    * The most useful reason for this annotation is to provide a default <code>Accept</code> header when one is not specified
265    *    so that a particular default {@link Serializer} is picked.
266    * <p>
267    * Only one header value can be specified per entry (i.e. it's not a delimited list of header entries).
268    * <p>
269    * Header values specified at the method level override header values specified at the servlet level.
270    *
271    * <dl>
272    *    <dt>Example:</dt>
273    *    <dd>
274    * <p class='bcode'>
275    *    <jc>// Assume "text/json" Accept value when Accept not specified</jc>
276    *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/*"</js>, defaultRequestHeaders={<js>"Accept: text/json"</js>})
277    *    <jk>public</jk> String doGet() {
278    *       ...
279    *    }
280    * </p>
281    *    </dd>
282    * </dl>
283    */
284   String[] defaultRequestHeaders() default {};
285
286   /**
287    * Optional description.
288    * <p>
289    *    This description is used in the following locations:
290    *    <ul>
291    *       <li>The value returned by {@link RestRequest#getMethodDescription()}.
292    *       <li>The <js>"$R{methodDescription}"</js> variable.
293    *       <li>The description of the method in the OPTIONS page.
294    *    </ul>
295    * <p>
296    *    The default value pulls the description from the <code>(className.?)[javaMethodName]</code> entry in the servlet resource bundle.
297    *    (e.g. <js>"MyClass.myMethod = foo"</js> or <js>"myMethod = foo"</js>).
298    * <p>
299    *    This field value can contain variables (e.g. "$L{my.localized.variable}").
300    *    See {@link RestServlet#createRequestVarResolver(RestRequest)}.
301    */
302   String description() default "";
303
304   /**
305    * Optional input description.
306    * <p>
307    *    This annotation is provided for documentation purposes and is used to populate the method <js>"input"</js> column
308    *       on the OPTIONS page.
309    * <p>
310    * Example:
311    * <p class='bcode'>
312    *    <ja>@RestMethod</ja>(
313    *       name=<js>"POST"</js>, path=<js>"/{a}"</js>,
314    *       description=<js>"This is my method."</js>,
315    *       input={
316    *          <ja>@Var</ja>(category=<js>"attr"</js>, name=<js>"a"</js>, description=<js>"The 'a' attribute"</js>),
317    *          <ja>@Var</ja>(category=<js>"param"</js>, name=<js>"b"</js>, description=<js>"The 'b' parameter"</js>),
318    *          <ja>@Var</ja>(category=<js>"content"</js>, description=<js>"The HTTP content"</js>),
319    *          <ja>@Var</ja>(category=<js>"header"</js>, name=<js>"D"</js>, description=<js>"The 'D' header"</js>),
320    *          <ja>@Var</ja>(category=<js>"foo"</js>, name=<js>"bar"</js>, description=<js>"An arbitrary category"</js>)
321    *       }
322    *    )
323    * </p>
324    * This is functionally equivalent to specifying the following keys in the resource bundle for the class, except in this case
325    * the strings are internationalized.
326    * <p class='bcode'>
327    *    <jk>MyClass.myMethod</jk> = <js>This is my method.</js>
328    *    <jk>MyClass.myMethod.req.attr.a</jk> = <js>The 'a' attribute</js>
329    *    <jk>MyClass.myMethod.req.param.b</jk> = <js>The 'b' parameter</js>
330    *    <jk>MyClass.myMethod.req.content</jk> = <js>The HTTP content</js>
331    *    <jk>MyClass.myMethod.req.header.d</jk> = <js>The 'D' header</js>
332    *    <jk>MyClass.myMethod.req.foo.bar</jk> = <js>An arbitrary category</js>
333    * <p>
334    * As a general rule, use annotations when you don't care about internationalization (i.e. you only want to support English),
335    * and use resource bundles if you need to support localization.
336    * <p>
337    *    These annotations can contain variables (e.g. "$L{my.localized.variable}").
338    *    See {@link RestServlet#createRequestVarResolver(RestRequest)}.
339    */
340   Var[] input() default {};
341
342
343   /**
344    * Optional output description.
345    * <p>
346    *    This annotation is provided for documentation purposes and is used to populate the method <js>"responses"</js> column
347    *       on the OPTIONS page.
348    * <p>
349    * Example:
350    * <p class='bcode'>
351    *    <ja>@RestMethod</ja>(
352    *       name=<js>"GET"</js>, path=<js>"/"</js>,
353    *       responses={
354    *          <ja>@Response</ja>(200),
355    *          <ja>@Response</ja>(
356    *             value=302,
357    *             description=<js>"Thing wasn't found here"</js>,
358    *             output={
359    *                <ja>@Var</ja>(category=<js>"header"</js>, name=<js>"Location"</js>, description=<js>"The place to find the thing"</js>)
360    *             }
361    *          )
362    *       }
363    *    )
364    * </p>
365    * This is functionally equivalent to specifying the following keys in the resource bundle for the class, except in this case
366    * the strings are internationalized.
367    * <p class='bcode'>
368    *    <jk>MyClass.myMethod.res.200</jk> = <js>OK</js>
369    *    <jk>MyClass.myMethod.res.302</jk> = <js>Thing wasn't found here</js>
370    *    <jk>MyClass.myMethod.res.302.header.Location</jk> = <js>The place to find the thing</js>
371    * <p>
372    * As a general rule, use annotations when you don't care about internationalization (i.e. you only want to support English),
373    * and use resource bundles if you need to support localization.
374    * <p>
375    *    These annotations can contain variables (e.g. "$L{my.localized.variable}").
376    *    See {@link RestServlet#createRequestVarResolver(RestRequest)}.
377    */
378   Response[] responses() default {};
