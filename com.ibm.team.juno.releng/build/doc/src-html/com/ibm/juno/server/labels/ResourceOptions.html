009package com.ibm.juno.server.labels;
010
011import static javax.servlet.http.HttpServletResponse.*;
012
013import java.util.*;
014
015import com.ibm.juno.core.annotation.*;
016import com.ibm.juno.server.*;
017import com.ibm.juno.server.annotation.*;

019/**
020 * Default POJO bean used for generating OPTIONS page results.
021 * <p>
022 * @author James Bognar (jbognar@us.ibm.com)
023 */
024@Bean(properties={"label","description","className","methods","children","consumes","produces","guards","filters","converters"})
025public class ResourceOptions {
026   private String label, description;
027   private String className;
028   private Collection<MethodDescription> methods;
029   private ChildResourceDescriptions children;
030   private Collection<String> consumes, produces;
031   private Collection<String> guards;
032   private Collection<String> filters;
033   private Collection<String> converters;
034
035   /**
036    * Constructor.
037    * @param servlet The servlet that this bean describes.
038    * @param req The HTTP servlet request.
039    */
040   public ResourceOptions(RestServlet servlet, RestRequest req) {
041      try {
042         setClassName(servlet.getClass().getName());
043         setLabel(servlet.getLabel(req));
044         setDescription(servlet.getDescription(req));
045         setMethods(servlet.getMethodDescriptions(req));
046         setConsumes(servlet.getSupportedAcceptTypes());
047         setProduces(servlet.getSupportedContentTypes());
048         setChildren(new ChildResourceDescriptions(servlet, req));
049         setGuards(servlet.getGuards());
050         setFilters(servlet.getFilters());
051         setConverters(servlet.getConverters());
052      } catch (RestServletException e) {
053         throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
054      }
055   }
056
057   /**
058    * Bean constructor.
059    */
060   public ResourceOptions() {}
061

065    */
066   public String getLabel() {
067      return label;
068   }
069
070   /**
071    * Sets the label of the REST resource.
072    * @param label The new bean property value.
073    * @return This object (for method chaining).
074    */
075   public ResourceOptions setLabel(String label) {
076      this.label = label;
077      return this;
078   }
079
080   /**

082    * @return The current bean property value.
083    */

085      return description;
086   }
087
088   /**
089    * Sets the description of the REST resource.
090    * @param description The new bean property value.
091    * @return This object (for method chaining).
092    */
093   public ResourceOptions setDescription(String description) {
094      this.description = description;
095      return this;
096   }
097
098   /**
099    * Returns the class name of the REST resource.
100    * @return The current bean property value.
101    */
102   public String getClassName() {
103      return className;
104   }
105
106   /**
107    * Sets the class name of the REST resource.
108    * @param className The new bean property value.
109    * @return This object (for method chaining).
110    */
111   public ResourceOptions setClassName(String className) {
112      this.className = className;
113      return this;
114   }
115
116   /**
117    * Returns the methods provided on this REST resource.
118    * @return The current bean property value.
119    */
120   public Collection<MethodDescription> getMethods() {
121      return methods;
122   }
123
124   /**
125    * Sets the methods provided on this REST resource.
126    * @param methods The new bean property value.
127    * @return This object (for method chaining).
128    */
129   public ResourceOptions setMethods(Collection<MethodDescription> methods) {
130      this.methods = methods;
131      return this;
132   }
133
134   /**
135    * Returns the list of allowable <code>Accept</code> header values on requests.
136    * @return The current bean property value.
137    */
138   public Collection<String> getConsumes() {
139      return consumes;
140   }
141
142   /**
143    * Sets the list of allowable <code>Accept</code> header values on requests.
144    * @param consumes The new bean property value.
145    * @return This object (for method chaining).
146    */
147   public ResourceOptions setConsumes(Collection<String> consumes) {
148      this.consumes = consumes;
149      return this;
150   }
151
152   /**
153    * Returns the list of allowable <code>Content-Type</code> header values on requests.
154    * @return The current bean property value.
155    */
156   public Collection<String> getProduces() {
157      return produces;
158   }
159
160   /**
161    * Sets the list of allowable <code>Content-Type</code> header values on requests.
162    * @param produces The new bean property value.
163    * @return This object (for method chaining).
164    */
165   public ResourceOptions setProduces(Collection<String> produces) {
166      this.produces = produces;
167      return this;
168   }
169
170   /**
171    * Returns the description of child resources with this resource (typically through {@link RestResource#children()} annotation).
172    * @return The description of child resources of this resource.
173    */
174   public ChildResourceDescriptions getChildren() {
175      return children;
176   }
177
178   /**
179    * Sets the child resource descriptions associated with this resource.
180    * @param children The child resource descriptions.

184      this.children = children;
185      return this;
186   }
187
188
189   /**
190    * Returns the list of class-wide guards associated with this resource (typically through {@link RestResource#guards()} annotation).
191    * @return The simple class names of the guards.
192    */
193   public Collection<String> getGuards() {
194      return guards;
195   }
196
197   /**
198    * Sets the simple class names of the guards associated with this resource.
199    * @param guards The simple class names of the guards.
200    * @return This object (for method chaining).
201    */
202   public ResourceOptions setGuards(Collection<String> guards) {
203      this.guards = guards;
204      return this;
205   }
206
207   /**
208    * Shortcut for calling {@link #setGuards(Collection)} from {@link RestGuard} instances.
209    * @param guards Instances of guards associated with this resource.
210    * @return This object (for method chaining).
211    */
212   public ResourceOptions setGuards(RestGuard[] guards) {
213      Collection<String> l = new ArrayList<String>(guards.length);
214      for (RestGuard g : guards)
215         l.add(g.getClass().getSimpleName());
216      return setGuards(l);
217   }
218

224      return filters;
225   }
226
227   /**
228    * Sets the simple class names of the filters associated with this resource.
229    * @param filters The simple class names of the filters.
230    * @return This object (for method chaining).
231    */
232   public ResourceOptions setFilters(Collection<String> filters) {
233      this.filters = filters;
234      return this;
235   }
236
237   /**
238    * Shortcut for calling {@link #setFilters(Collection)} from {@link Class} instances.
239    * @param filters Filter classes associated with this resource.
240    * @return This object (for method chaining).
241    */
242   public ResourceOptions setFilters(Class<?>[] filters) {
243      Collection<String> l = new ArrayList<String>(filters.length);
244      for (Class<?> c : filters)
245         l.add(c.getSimpleName());
246      return setFilters(l);
247   }

252    */
253   public Collection<String> getConverters() {
254      return converters;
255   }
256
257   /**
258    * Sets the simple class names of the converters associated with this resource.
259    * @param converters The simple class names of the converters.
260    * @return This object (for method chaining).
261    */
262   public ResourceOptions setConverters(Collection<String> converters) {
263      this.converters = converters;
264      return this;
265   }
266
267   /**
268    * Shortcut for calling {@link #setConverters(Collection)} from {@link RestConverter} instances.
269    * @param converters Converter classes associated with this resource.
270    * @return This object (for method chaining).
271    */
272   public ResourceOptions setConverters(RestConverter[] converters) {
273      Collection<String> l = new ArrayList<String>(converters.length);
274      for (RestConverter c : converters)
275         l.add(c.getClass().getSimpleName());
276      return setConverters(l);
277   }
