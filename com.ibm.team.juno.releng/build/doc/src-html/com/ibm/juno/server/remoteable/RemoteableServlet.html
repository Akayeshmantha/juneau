009package com.ibm.juno.server.remoteable;
010
011import static javax.servlet.http.HttpServletResponse.*;
012
013import java.util.*;
014import java.util.concurrent.*;
015
016import com.ibm.juno.core.*;
017import com.ibm.juno.core.dto.*;
018import com.ibm.juno.core.parser.*;
019import com.ibm.juno.server.*;
020import com.ibm.juno.server.annotation.*;
021
022/**
023 * Abstract class for defining Remoteable services.
024 * <p>
025 * Remoteable services are POJOs whose methods can be invoked remotely through proxy interfaces.
026 * <p>
027 * To implement a remoteable service, developers must simply subclass from this class and implement the {@link #getServiceMap()} method that
028 *    maps java interfaces to POJO instances.
029 *
030 * See {@link com.ibm.juno.server.remoteable} for details.
031 *
032 * @author James Bognar (jbognar@us.ibm.com)
033 */
034@SuppressWarnings("serial")
035public abstract class RemoteableServlet extends RestServletDefault {
036
037   private Map<String,Class<?>> classNameMap = new ConcurrentHashMap<String,Class<?>>();
038
039   //--------------------------------------------------------------------------------
040   // Abstract methods
041   //--------------------------------------------------------------------------------
042
043   /**
044    * Returns the list of interfaces to their implementation objects.
045    * <p>
046    * This class is called often and not cached, so any caching should occur in the subclass if necessary.
047    *
048    * @return The service map.
049    * @throws Exception
050    */
051   protected abstract Map<Class<?>,Object> getServiceMap() throws Exception;
052
053   //--------------------------------------------------------------------------------
054   // REST methods
055   //--------------------------------------------------------------------------------
056
057   /**
058    * [GET /] - Get the list of all remote interfaces.
059    *
060    * @param req The HTTP servlet request.
061    * @return The list of links to the remote interfaces.
062    * @throws Exception
063    */
064   @RestMethod(name="GET", path="/")
065   public List<Link> getInterfaces(RestRequest req) throws Exception {
066      List<Link> l = new LinkedList<Link>();
067      boolean useAll = ! useOnlyAnnotated();
068      for (Class<?> c : getServiceMap().keySet()) {
069         if (useAll || getBeanContext().getClassMeta(c).isRemoteable())
070            l.add(new Link(c.getName(), "{0}/{1}", req.getRequestURI(), c.getName())); //$NON-NLS-1$
071      }
072      return l;
073   }
074
075   /**
076    * [GET /{javaInterface] - Get the list of all remoteable methods on the specified interface name.
077    *
078    * @param javaInterface The Java interface name.
079    * @return The methods defined on the interface.
080    * @throws Exception
081    */
082   @RestMethod(name="GET", path="/{javaInterface}")
083   public Collection<String> listMethods(@Attr String javaInterface) throws Exception {
084      return getMethods(javaInterface).keySet();
085   }
086
087   /**
088    * [POST /{javaInterface}/{javaMethod}] - Invoke the specified service method.
089    *
090    * @param req The HTTP request.
091    * @param javaInterface The Java interface name.
092    * @param javaMethod The Java method name or signature.
093    * @return The results from invoking the specified Java method.
094    * @throws Exception
095    */
096   @RestMethod(name="POST", path="/{javaInterface}/{javaMethod}")
097   public Object invoke(RestRequest req, @Attr String javaInterface, @Attr String javaMethod) throws Exception {
098
099      // Find the parser.
100      ReaderParser p = req.getReaderParser();
101      if (p == null)
102         throw new RestException(SC_UNSUPPORTED_MEDIA_TYPE, "Could not find parser for media type ''{0}''", req.getMediaType()); //$NON-NLS-1$
103      Class<?> c = getInterfaceClass(javaInterface);
104
105      // Find the service.
106      Object service = getServiceMap().get(c);
107      if (service == null)
108         throw new RestException(SC_NOT_FOUND, "Service not found"); //$NON-NLS-1$
109
110      // Find the method.
111      java.lang.reflect.Method m = getMethods(javaInterface).get(javaMethod);
112      if (m == null)
113         throw new RestException(SC_NOT_FOUND, "Method not found"); //$NON-NLS-1$
114
115      // Parse the args and invoke the method.
116      ClassMeta<?>[] argTypes = p.getBeanContext().getClassMetas(m.getParameterTypes());
117      Object[] params = p.parseArgs(req.getReader(), -1, argTypes);
118      return m.invoke(service, params);
119   }
120
121   //--------------------------------------------------------------------------------
122   // Other methods
123   //--------------------------------------------------------------------------------
124
125   private boolean useOnlyAnnotated() {
126      return getProperties().getBoolean(RemoteableServiceProperties.REMOTEABLE_includeOnlyRemotableMethods, false);
127   }
128
129   private Map<String,java.lang.reflect.Method> getMethods(String javaInterface) throws Exception {
130      Class<?> c = getInterfaceClass(javaInterface);
131      ClassMeta<?> cm = getBeanContext().getClassMeta(c);
132      return (useOnlyAnnotated() ? cm.getRemoteableMethods() : cm.getPublicMethods());
133   }
134
135   /**
136    * Return the <code>Class</code> given it's name if it exists in the services map.
137    */
138   private Class<?> getInterfaceClass(String javaInterface) throws Exception {
139      Class<?> c = classNameMap.get(javaInterface);
140      if (c == null) {
141         for (Class<?> c2 : getServiceMap().keySet())
142            if (c2.getName().equals(javaInterface)) {
143               classNameMap.put(javaInterface, c2);
144               return c2;
145            }
146         throw new RestException(SC_NOT_FOUND, "Interface class not found");
147      }
148      return c;
149   }
150}
