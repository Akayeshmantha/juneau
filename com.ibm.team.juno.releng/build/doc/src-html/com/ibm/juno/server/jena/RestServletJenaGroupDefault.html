009package com.ibm.juno.server.jena;
010
011import com.ibm.juno.server.*;
012import com.ibm.juno.server.annotation.*;
013import com.ibm.juno.server.labels.*;
014
015/**
016 * Specialized subclass of {@link RestServletJenaDefault} for showing "group" pages.
017 * <p>
018 *    Group pages consist of simple lists of child resource URLs and their labels.
019 *    They're meant to be used as jumping-off points for child resources.
020 * <p>
021 *    Child resources are specified using the {@link RestResource#children()} annotation.
022 *
023 * @author James Bognar (jbognar@us.ibm.com)
024 */
025@RestResource()
026public abstract class RestServletJenaGroupDefault extends RestServletJenaDefault {
027   private static final long serialVersionUID = 1L;
028
029   /**
030    * [GET /] - Get child resources.
031    *
032    * @param req The HTTP request
033    * @return The bean containing links to the child resources.
034    */
035   @RestMethod(name="GET", path="/", description="Child resources")
036   public ChildResourceDescriptions getChildren(RestRequest req) {
037      return new ChildResourceDescriptions(this, req);
038   }
039}
