024/**
025 * JAX-RS provider for the same serialize/parse support provided by the {@link RestServletJenaDefault} class.
026 *
027 * @author James Bognar (jbognar@us.ibm.com)
028 */
029@Provider
030@Produces({
031   "application/json", "text/json",                 // JsonSerializer
032   "application/json+simple","text/json+simple",    // JsonSerializer.Simple
033   "application/json+schema","text/json+schema",    // JsonSchemaSerializer
034   "text/xml",                                      // XmlDocSerializer
035   "text/xml+simple",                               // XmlDocSerializer.Simple
036   "text/xml+schema",                               // XmlSchemaDocSerializer
037   "text/html",                                     // HtmlDocSerializer
038   "application/x-www-form-urlencoded",             // UrlEncodingSerializer
039   "text/xml+soap",                                 // SoapXmlSerializer
040   "text/xml+rdf",                                  // RdfSerializer.Xml
041   "text/xml+rdf+abbrev",                           // RdfSerializer.XmlAbbrev
042   "text/n-triple",                                 // RdfSerializer.NTriple
043   "text/turtle",                                   // RdfSerializer.Turtle
044   "text/n3",                                       // RdfSerializer.N3
045   "application/x-java-serialized-object"           // JavaSerializedObjectSerializer
046})
047@Consumes({
048   "application/json", "text/json",                 // JsonParser
049   "text/xml",                                      // XmlParser
050   "text/html",                                     // HtmlParser
051   "application/x-www-form-urlencoded",             // UrlEncodingParser
052   "text/xml+rdf",                                  // RdfParser.Xml
053   "text/n-triple",                                 // RdfParser.NTriple
054   "text/turtle",                                   // RdfParser.Turtle
055   "text/n3",                                       // RdfParser.N3
056   "application/x-java-serialized-object"           // JavaSerializedObjectParser
057})
058@JunoProvider(
059   serializers={
060      JsonSerializer.class,
061      JsonSerializer.Simple.class,
062      JsonSchemaSerializer.class,
063      XmlDocSerializer.class,
064      XmlDocSerializer.Simple.class,
065      XmlSchemaDocSerializer.class,
066      HtmlDocSerializer.class,
067      UrlEncodingSerializer.class,
068      SoapXmlSerializer.class,
069      RdfSerializer.Xml.class,
070      RdfSerializer.XmlAbbrev.class,
071      RdfSerializer.NTriple.class,
072      RdfSerializer.Turtle.class,
073      RdfSerializer.N3.class,
074      JavaSerializedObjectSerializer.class
075   },
076   parsers={
077      JsonParser.class,
078      XmlParser.class,
079      HtmlParser.class,
080      UrlEncodingParser.class,
081      RdfParser.Xml.class,
082      RdfParser.NTriple.class,
083      RdfParser.Turtle.class,
084      RdfParser.N3.class,
085      JavaSerializedObjectParser.class,
086   }
087)
088public final class DefaultJenaProvider extends BaseProvider {}
