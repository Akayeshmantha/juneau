010
011import static javax.servlet.http.HttpServletResponse.*;
012
013import java.util.*;
014
015import com.ibm.juno.core.*;
016import com.ibm.juno.core.parser.*;
017import com.ibm.juno.core.serializer.*;
018import com.ibm.juno.core.utils.*;
019import com.ibm.juno.server.*;

021/**
022 * Converter for enablement of {@link PojoQuery} support on response objects returned by a <code>@RestMethod</code> method.
023 * <p>
024 *    When enabled, objects in a POJO tree can be filtered using the functionality described in the {@link PojoQuery}
025 *    class.
026 * <p>
027 *    The following HTTP request parameters are available for tabular data (e.g. {@code Collections} of {@code Maps}, arrays of beans, etc...):
028 * <ul>
029 *    <li><b>&amp;q=<i>JSON-object</i></b> - Query parameter.  Only return rows that match the specified search string. <br>
030 *          The JSON object keys are column names, and the values are search parameter strings.<br>
031 *          Example:  <code>&amp;s=(name=Bill*,birthDate=&gt;2000)</code><br>
032 *          <br>
033 *    <li><b>&amp;v=<i>JSON-array or comma-delimited list</i></b> - View parameter.  Only return the specified columns.<br>
034 *          Example:  <code>&amp;v=(name,birthDate)</code><br>
035 *          <br>
036 *    <li><b>&amp;s=<i>JSON-object</i></b> - Sort parameter.  Sort the results by the specified columns.<br>
037 *          The JSON object keys are the column names, and the values are either {@code 'A'} for ascending or {@code 'D'} for descending.
038 *          Example:  <code>&amp;s=(name=A,birthDate=D)</code><br>
039 *          <br>
040 *    <li><b>&amp;i=<i>true/false</i></b> - Case-insensitive parameter.  Specify <jk>true</jk> for case-insensitive matching on the {@code &amp;q} parameter. <br>
041 *          <br>
042 *    <li><b>&amp;p=<i>number</i></b> - Position parameter.  Only return rows starting at the specified index position (zero-indexed).  Default is {@code 0}.<br>
043 *          <br>
044 *    <li><b>&amp;q=<i>number</i></b> - Limit parameter.  Only return the specified number of rows. Default is {@code 0} (meaning return all rows).
045 * </ul>
046 *
047 * <p>
048 *    The <b>&amp;v</b> parameter can also be used on {@code Maps} and beans.
049 *
050 * <p>
051 *    See {@link PojoQuery} for additional information on filtering POJO models.
052 */
053public final class Queryable implements RestConverter {
054
055   @Override /* RestConverter */
056   @SuppressWarnings({ "unchecked", "rawtypes" })
057   public Object convert(RestRequest req, Object o, ClassMeta cm) {
058      if (o == null)
059         return null;
060
061      try {
062
063         // If no actual filtering parameters have been passed in, and there is no map augmenter specified,
064         // then just pass the original object back.
065         if (req.hasAnyQueryParameters("q","v","s","g","i","p","l")) {
066            BeanContext bc = req.getBeanContext();
067
068            if (cm.getPojoFilter() != null)
069               o = cm.getPojoFilter().filter(o);
070
071            PojoQuery f = new PojoQuery(o, bc);
072
073            if (o instanceof Collection || o.getClass().isArray()) {
074               ObjectMap query = req.getQueryParameter("q", ObjectMap.class);
075               ClassMeta<List<String>> cm1 = bc.getCollectionClassMeta(List.class, String.class);
076               List<String> view = req.getQueryParameter("v", cm1);
077               ClassMeta<List<Object>> cm2 = bc.getCollectionClassMeta(List.class, String.class);
078               List sort = req.getQueryParameter("s", cm2);
079               boolean ignoreCase = req.getQueryParameter("i", Boolean.class, false);
080               int pos = req.getQueryParameter("p", Integer.class, 0);
081               int limit = req.getQueryParameter("l", Integer.class, 0);
082               o = f.filterCollection(query, view, sort, pos, limit, ignoreCase);
083
084            } else {
085               ClassMeta<List<String>> cm2 = bc.getCollectionClassMeta(List.class, String.class);
086               List<String> view = req.getQueryParameter("v", cm2);
087               o = f.filterMap(view);
088            }
089         }
090         return o;
091      } catch (SerializeException e) {
092         throw new RestException(SC_BAD_REQUEST, e);
093      } catch (ParseException e) {
094         throw new RestException(SC_BAD_REQUEST, e);
095      }
096   }
097}
