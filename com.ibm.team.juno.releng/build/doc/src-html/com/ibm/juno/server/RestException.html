009package com.ibm.juno.server;
010
011import java.text.*;
012
013/**
014 * Exception thrown to trigger an error HTTP status.
015 * <p>
016 *    REST methods on subclasses of {@link RestServlet} can throw
017 *    this exception to trigger an HTTP status other than the automatically-generated
018 *    <code>404</code>, <code>405</code>, and <code>500</code> statuses.
019 *
020 * @author James Bognar (jbognar@us.ibm.com)
021 */
022public class RestException extends RuntimeException {
023
024   private static final long serialVersionUID = 1L;
025
026   private int status;
027   private int occurrence;
028
029   /**
030    * Constructor.
031    *
032    * @param status The HTTP status code.
033    * @param msg The status message.
034    * @param args Optional string format arguments.
035    */
036   public RestException(int status, String msg, Object...args) {
037      super(args.length == 0 ? msg : MessageFormat.format(msg, args));
038      this.status = status;
039   }
040
041   /**
042    * Constructor.
043    *
044    * @param status The HTTP status code.
045    * @param cause The root exception.
046    */
047   public RestException(int status, Throwable cause) {
048      this(status, cause.getLocalizedMessage());
049      initCause(cause);
050   }
051
052
053   /**
054    * Sets the inner cause for this exception.
055    *
056    * @param cause The inner cause.
057    * @return This object (for method chaining).
058    */
059   @Override /* Throwable */
060   public synchronized RestException initCause(Throwable cause) {
061      super.initCause(cause);
062      return this;
063   }
064
065
066   /**
067    * Returns all error messages from all errors in this stack.
068    * <p>
069    * Typically useful if you want to render all the error messages in the stack, but don't
070    * want to render all the stack traces too.
071    *
072    * @param scrubForXssVulnerabilities If <jk>true</jk>, replaces <js>'&lt;'</js>, <js>'&gt;'</js>, and <js>'&amp;'</js> characters with spaces.
073    * @return All error messages from all errors in this stack.
074    */
075   public String getFullStackMessage(boolean scrubForXssVulnerabilities) {
076      String msg = getMessage();
077      StringBuilder sb = new StringBuilder();
078      if (msg != null) {
079         if (scrubForXssVulnerabilities)
080            msg = msg.replace('<', ' ').replace('>', ' ').replace('&', ' ');
081         sb.append(msg);
082      }
083      Throwable e = getCause();
084      while (e != null) {
085         msg = e.getMessage();
086         if (msg != null && scrubForXssVulnerabilities)
087            msg = msg.replace('<', ' ').replace('>', ' ').replace('&', ' ');
088         String cls = e.getClass().getSimpleName();
089         if (msg == null)
090            sb.append(MessageFormat.format("\nCaused by ({0})", cls));
091         else
092            sb.append(MessageFormat.format("\nCaused by ({0}): {1}", cls, msg));
093         e = e.getCause();
094      }
095      return sb.toString();
096   }
097
098   @Override /* Object */
099   public int hashCode() {
100      int i = 0;
101      Throwable t = this;
102      while (t != null) {
103         for (StackTraceElement e : t.getStackTrace())
104         i ^= e.hashCode();
105         t = t.getCause();
106      }
107      return i;
108   }
109
110   void setOccurrence(int occurrence) {
111      this.occurrence = occurrence;
112   }
113
114   /**
115    * Returns the number of times this exception occurred on this servlet.
116    * <p>
117    * This only gets set if {@link RestServletProperties#REST_useStackTraceHashes} is enabled on the servlet.
118    *
119    * @return The occurrence number if {@link RestServletProperties#REST_useStackTraceHashes} is enabled, or <code>0</code> otherwise.
120    */
121   public int getOccurrence() {
122      return occurrence;
123   }
124
125   /**
126    * Returns the HTTP status code.
127    *
128    * @return The HTTP status code.
129    */
130   public int getStatus() {
131      return status;
132   }
133}
