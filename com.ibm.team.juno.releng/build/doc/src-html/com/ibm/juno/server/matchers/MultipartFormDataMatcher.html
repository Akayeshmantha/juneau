009package com.ibm.juno.server.matchers;
010
011import com.ibm.juno.server.*;
012
013/**
014 * Predefined matcher for matching requests with content type <js>"multipart/form-data"</js>.
015 *
016 * @author James Bognar (jbognar@us.ibm.com)
017 */
018public class MultipartFormDataMatcher extends RestMatcher {
019   @Override /* RestMatcher */
020   public boolean matches(RestRequest req) {
021      String contentType = req.getContentType();
022      return contentType != null && contentType.startsWith("multipart/form-data"); //$NON-NLS-1$
023   }
024}
