009package com.ibm.juno.server.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.io.*;
015import java.lang.annotation.*;
016
017/**
018 * Annotation that can be applied to a parameter of a {@link RestMethod} annotated method
019 *    to identify it as the HTTP request body converted to a POJO.
020 *
021 * <h6 class='topic'>Example</h6>
022 * <p class='bcode'>
023 *    <ja>@RestMethod</ja>(name=<js>"POST"</js>)
024 *    <jk>public void</jk> doPostPerson(RestRequest req, RestResponse res, <ja>@Content</ja> Person person) {
025 *       ...
026 *    }
027 * </p>
028 * <p>
029 *    This is functionally equivalent to the following code...
030 * </p>
031 * <p class='bcode'>
032 *    <ja>@RestMethod</ja>(name=<js>"POST"</js>)
033 *    <jk>public void</jk> doPostPerson(RestRequest req, RestResponse res) {
034 *       Person person = req.getInput(Person.<jk>class</jk>);
035 *       ...
036 *    }
037 * </p>
038 * <p>
039 *    {@link Reader Readers} and {@link InputStream InputStreams} can also be specified as content parameters.
040 *    When specified, any registered parsers are bypassed.
041 * </p>
042 * <p class='bcode'>
043 *    <ja>@RestMethod</ja>(name=<js>"POST"</js>)
044 *    <jk>public void</jk> doPostPerson(<ja>@Header</ja> String mediaType, <ja>@Content</ja> InputStream input) {
045 *       ...
046 *    }
047 * </p>
048 * @author James Bognar (jbognar@us.ibm.com)
049 */
050@Documented
051@Target(PARAMETER)
052@Retention(RUNTIME)
053@Inherited
054public @interface Content {}
