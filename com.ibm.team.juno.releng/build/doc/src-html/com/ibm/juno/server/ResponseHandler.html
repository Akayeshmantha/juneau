009package com.ibm.juno.server;
010
011import java.io.*;
012
013import javax.servlet.http.*;
014
015import com.ibm.juno.server.annotation.*;
016import com.ibm.juno.server.response.*;
017
018/**
019 * Defines the interface for handlers that convert POJOs to appropriate HTTP responses.
020 * <p>
021 * The {@link RestServlet} API uses the concept of registered response handlers for
022 *    converting objects returned by REST methods or set through {@link RestResponse#setOutput(Object)}
023 *    into appropriate HTTP responses.
024 * <p>
025 * Response handlers can be associated with {@link RestServlet RestServlets} through the following ways:
026 * <ul>
027 *    <li>Through the {@link RestResource#responseHandlers @RestResource.responseHandlers} annotation.
028 *    <li>By overriding {@link RestServlet#createResponseHandlers()} and augmenting or creating your
029 *       own list of handlers.
030 * </ul>
031 * <p>
032 * By default, {@link RestServlet RestServlets} are registered with the following response handlers:
033 * <ul>
034 *    <li>{@link DefaultHandler} - Serializes POJOs using the Juno serializer API.
035 *    <li>{@link ReaderHandler} - Pipes the output of {@link Reader Readers} to the response writer ({@link RestResponse#getWriter()}).
036 *    <li>{@link InputStreamHandler} - Pipes the output of {@link InputStream InputStreams} to the response output stream ({@link RestResponse#getOutputStream()}).
037 *    <li>{@link RedirectHandler} - Handles {@link Redirect} objects.
038 * </ul>
039 * <p>
040 * Response handlers can be used to process POJOs that cannot normally be handled through Juno serializers, or
041 *    because it's simply easier to define response handlers for special cases.
042 * <p>
043 * The following example shows how to create a response handler to handle special <code>Foo</code> objects outside the normal
044 *    Juno architecture.
045 * <p class='bcode'>
046 *    <ja>@RestResource</ja>(
047 *       path=<js>"/example"</js>,
048 *       responseHandlers=FooHandler.<jk>class</jk>
049 *    )
050 *    <jk>public class</jk> Example <jk>extends</jk> RestServlet {
051 *
052 *       <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/"</js>)
053 *       <jk>public</jk> Foo test1() {
054 *          <jk>return new</jk> Foo(<js>"123"</js>);
055 *       }
056 *
057 *       <jk>public static class</jk> FooHandler <jk>implements</jk> ResponseHandler {
058 *          <ja>@Override</ja>
059 *          <jk>public boolean</jk> handle(RestRequest req, RestResponse res, Object output) <jk>throws</jk> IOException, RestException {
060 *             <jk>if</jk> (output <jk>instanceof</jk> Foo) {
061 *                Foo foo = (Foo)output;
062 *                <jc>// Set some headers and body content.</jc>
063 *                res.setHeader(<js>"Foo-ID"</js>, foo.getId());
064 *                res.getWriter().write(<js>"foo.id="</js> + foo.getId());
065 *                <jk>return true</jk>;  <jc>// We handled it.</jc>
066 *             }
067 *             <jk>return false</jk>;  <jc>// We didn't handle it.</jc>
068 *          }
069 *       }
070 *    }
071 * </p>
072 */
073public interface ResponseHandler {
074
075   /**
076    * Process this response if possible.
077    * This method should return <jk>false</jk> if it wasn't able to process the response.
078    *
079    * @param req The HTTP servlet request.
080    * @param res The HTTP servlet response;
081    * @param output The POJO returned by the REST method that now needs to be sent to the response.
082    * @return true If this handler handled the response.
083    * @throws IOException - If low-level exception occurred on output stream.  Results in a {@link HttpServletResponse#SC_INTERNAL_SERVER_ERROR} error.
084    * @throws RestException - If some other exception occurred.  Can be used to provide an appropriate HTTP response code and message.
085    */
086   boolean handle(RestRequest req, RestResponse res, Object output) throws IOException, RestException;
087}
