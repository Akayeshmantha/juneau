009package com.ibm.juno.microservice;
010import static com.ibm.juno.core.html.HtmlDocSerializerProperties.*;
011
012import com.ibm.juno.core.utils.*;
013import com.ibm.juno.server.*;
014import com.ibm.juno.server.annotation.*;
015
016/**
017 * Superclass for all REST resources.
018 * <p>
019 * In additional to the functionality of the {@link RestServletDefault} group, 
020 * augments the {@link #createVarResolver()} method with the following additional variable types:
021 * <ul class='spaced-list'>
022 *    <li><code class='snippet'>$ARG{...}</code> - Command line arguments pulled from {@link Microservice#getArgs()}.<br>
023 *       <h6 class='figure'>Example:</h6>
024 *       <p class='bcode'>
025 *          String firstArg = request.getVarResolver().resolve(<js>"$ARG{0}"</js>);  <jc>// First argument.</jc> 
026 *          String namedArg = request.getVarResolver().resolve(<js>"$ARG{myarg}"</js>);  <jc>// Named argument (e.g. "myarg=foo"). </jc>
027 *       </p>
028 *    <li><code class='snippet'>$MF{...}</code> - Manifest file entries pulled from {@link Microservice#getManifest()}.<br>
029 *       <h6 class='figure'>Example:</h6>
030 *       <p class='bcode'>
031 *          String mainClass = request.getVarResolver().resolve(<js>"$MF{Main-Class}"</js>);  <jc>// Main class. </jc>
032 *       </p>
033 * </ul>
034 * 
035 * @author James Bognar (jbognar@us.ibm.com)
036 */
037@SuppressWarnings("serial")
038@RestResource(
039   properties={
040      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'$R{servletURI}?method=OPTIONS'}")
041   },
042   config="$S{juno.configFile}",
043   stylesheet="$C{REST/stylesheet,styles/juno.css}"
044)
045public abstract class Resource extends RestServletDefault {
046
047   /**
048    * Adds $ARG and $MF variables to variable resolver defined on {@link RestServlet#createVarResolver()}.
049    */
050   @Override
051   protected StringVarResolver createVarResolver() {
052      StringVarResolver r = super.createVarResolver();
053
054      // Command-line arguments.
055      r.addVar("ARG", new StringVarWithDefault() {
056         @Override /* StringVar */
057         public String resolve(String varVal) {
058            return Microservice.getArgs().getArg(varVal);
059         }
060      });
061
062      // Manifest file entries.
063      r.addVar("MF", new StringMapVar(Microservice.getManifest()));
064
065      return r;
066   }
067}
