054 */

112   @SuppressWarnings({"unchecked", "rawtypes"})
113   @Override /* ReaderParser */
114   protected <T> T doParse(Reader in, int estimatedSize, ClassMeta<T> type, ParserContext ctx) throws ParseException, IOException {
115
116      in = IOUtils.getBufferedReader(in, estimatedSize);
117
118      RdfParserContext rctx = (RdfParserContext)ctx;
119
120      type = ctx.getBeanContext().normalizeClassMeta(type);
121
122      Model model = rctx.getModel();
123      RDFReader r = rctx.getRdfReader();
124      r.read(model, in, null);
125      BeanContext bc = ctx.getBeanContext();
126
127      List<Resource> roots = getRoots(model, rctx);
128
129      try {
130         // Special case where we're parsing a loose collection of resources.
131         if (rctx.isLooseCollection() && (type.isCollection() || type.isArray())) {
132            Collection c = null;
133            if (type.isArray())
134               c = new ArrayList();
135            else
136               c = (type.canCreateNewInstance(ctx.getOuter()) ? (Collection<?>)type.newInstance(ctx.getOuter()) : new ObjectList(bc));
137            for (Resource resource : roots)
138               c.add(parseAnything(type.getElementType(), resource, null, rctx, ctx.getOuter(), null));
139
140            if (type.isArray())
141               return (T)bc.toArray(type, c);
142            return (T)c;
143         }
144
145         if (roots.isEmpty())
146            return null;
147         if (roots.size() > 1)
148            throw new ParseException("Too many root nodes found in model:  {0}", roots.size());
149         Resource resource = roots.get(0);
150
151         return parseAnything(type, resource, null, rctx, ctx.getOuter(), null);
152
153      } catch (ParseException e) {
154         throw e;
155      } catch (Exception e) {
156         throw new ParseException(e);
157      }
158   }

160   /*
161    * Finds the roots in the model using either the "root" property to identify it,
162    *    or by resorting to scanning the model for all nodes with no incoming predicates.
163    */
164   private List<Resource> getRoots(Model m, RdfParserContext ctx) {
165      List<Resource> l = new LinkedList<Resource>();
166
167      // First try to find the root using the "http://www.ibm.com/juno/root" property.
168      Property root = m.createProperty(ctx.getJunoNsUri(), RDF_junoNs_ROOT);
169      for (ResIterator i  = m.listResourcesWithProperty(root); i.hasNext();)
170         l.add(i.next());
171
172      if (! l.isEmpty())
173         return l;
174
175      // Otherwise, we need to find all resources that aren't objects.
176      // We want to explicitly ignore statements where the subject
177      // and object are the same node.
178      Set<RDFNode> objects = new HashSet<RDFNode>();
179      for (StmtIterator i = m.listStatements(); i.hasNext();) {
180         Statement st = i.next();
181         RDFNode subject = st.getSubject();
182         RDFNode object = st.getObject();
183         if (object.isResource() && ! object.equals(subject))
184            objects.add(object);
185      }
186      for (ResIterator i = m.listSubjects(); i.hasNext();) {
187         Resource r = i.next();
188         if (! objects.contains(r))
189            l.add(r);
190      }
191      return l;
192   }

194   private <T> BeanMap<T> parseIntoBeanMap(Resource r2, BeanMap<T> m, RdfParserContext ctx) throws ParseException {
195      BeanMeta<T> bm = m.getMeta();
196      if (bm.hasBeanUriProperty() && r2.getURI() != null)
197         m.putBeanUri(r2.getURI());
198      Property subTypeIdProperty = null;
199      BeanPropertyMeta<T> stp = bm.getSubTypeIdProperty();
200      if (stp != null) {
201         subTypeIdProperty = ctx.getProperty(stp.getName());
202         Statement st = r2.getProperty(subTypeIdProperty);
203         if (st == null)
204            throw new ParseException("Could not find subtype ID property for bean of type ''{0}''", m.getClassMeta());
205         String subTypeId = st.getLiteral().getString();
206         stp.set(m, subTypeId);
207      }
208      for (StmtIterator i = r2.listProperties(); i.hasNext();) {
209         Statement st = i.next();
210         Property p = st.getPredicate();
211         if (p.equals(subTypeIdProperty))
212            continue;
213         String key = ctx.decodeString(p.getLocalName());
214         BeanPropertyMeta<T> pMeta = m.getPropertyMeta(key);
215         if (pMeta != null) {
216            RDFNode o = st.getObject();
217            ClassMeta<?> cm = pMeta.getClassMeta();
218            if ((cm.isArray() || cm.isCollection()) && isMultiValuedCollections(ctx, pMeta)) {
219               Object val = parseAnything(cm.getElementType(), o, pMeta, ctx, m.getBean(false), key);
220               pMeta.add(m, val);
221            } else {
222               Object val = parseAnything(cm, o, pMeta, ctx, m.getBean(false), key);
223               pMeta.set(m, val);
224            }
225         } else if (! (p.equals(ctx.getRootProperty()) || p.equals(ctx.getClassProperty()) || p.equals(subTypeIdProperty))) {
226            if (bm.isSubTyped()) {
227               RDFNode o = st.getObject();
228               Object val = parseAnything(object(), o, null, ctx, m.getBean(false), key);
229               m.put(key, val);
230            } else {
231               onUnknownProperty(ctx, key, m, -1, -1);
232            }
233         }
234      }
235      return m;
236   }

238   private boolean isMultiValuedCollections(RdfParserContext ctx, BeanPropertyMeta<?> pMeta) {
239      if (pMeta != null && pMeta.getRdfMeta().getCollectionFormat() != RdfCollectionFormat.DEFAULT)
240         return pMeta.getRdfMeta().getCollectionFormat() == RdfCollectionFormat.MULTI_VALUED;
241      return ctx.getCollectionFormat() == RdfCollectionFormat.MULTI_VALUED;
242   }

244   @SuppressWarnings({ "unchecked", "rawtypes" })
245   private <T> T parseAnything(ClassMeta<T> nt, RDFNode n, BeanPropertyMeta<?> p, RdfParserContext ctx, Object outer, Object name) throws ParseException {
246
247      BeanContext bc = ctx.getBeanContext();
248
249      if (nt == null)
250         nt = (ClassMeta<T>)object();
251      PojoFilter<T,Object> filter = (PojoFilter<T,Object>)nt.getPojoFilter();
252      ClassMeta<?> ft = nt.getFilteredClassMeta();
253
254      if (! ft.canCreateNewInstance(outer)) {
255         if (n.isResource()) {
256            Statement st = n.asResource().getProperty(ctx.getClassProperty());
257            if (st != null) {
258               String c = st.getLiteral().getString();
259               ft = nt = (ClassMeta<T>)bc.getClassMetaFromString(c);
260            }
261         }
262      }
263
264      try {
265
266         Object o = null;
267         if (n.isResource() && n.asResource().getURI() != null && n.asResource().getURI().equals(RDF_NIL)) {
268            // Do nothing.  Leave o == null.
269         } else if (ft.isObject()) {
270            if (n.isLiteral()) {
271               o = n.asLiteral().getValue();
272               if (o instanceof String) {
273                  o = ctx.decodeString(o);
274               }
275            }
276            else if (n.isResource()) {
277               Resource r = n.asResource();
278               if (ctx.wasAlreadyProcessed(r))
279                  o = r.getURI();
280               else if (r.getProperty(ctx.getValueProperty()) != null) {
281                  o = parseAnything(object(), n.asResource().getProperty(ctx.getValueProperty()).getObject(), null, ctx, outer, null);
282               } else if (isSeq(r, ctx)) {
283                  o = new ObjectList(bc);
284                  parseIntoCollection(r.as(Seq.class), (Collection)o, ft.getElementType(), ctx);
285               } else if (isBag(r, ctx)) {
286                  o = new ObjectList(bc);
287                  parseIntoCollection(r.as(Bag.class), (Collection)o, ft.getElementType(), ctx);
288               } else if (r.canAs(RDFList.class)) {
289                  o = new ObjectList(bc);
290                  parseIntoCollection(r.as(RDFList.class), (Collection)o, ft.getElementType(), ctx);
291               } else {
292                  // If it has a URI and no child properties, we interpret this as an
293                  // external resource, and convert it to just a URL.
294                  String uri = r.getURI();
295                  if (uri != null && ! r.listProperties().hasNext()) {
296                     o = r.getURI();
297                  } else {
298                     o = new ObjectMap(bc);
299                     parseIntoMap(r, (Map)o, null, null, ctx);
300                  }
301               }
302            } else {
303               throw new ParseException("Unrecognized node type ''{0}'' for object", n);
304            }
305         } else if (ft.isBoolean()) {
306            o = bc.convertToType(getValue(n, ctx, outer), boolean.class);
307         } else if (ft.isCharSequence()) {
308            o = ctx.decodeString(getValue(n, ctx, outer));
309         } else if (ft.isChar()) {
310            o = ctx.decodeString(getValue(n, ctx, outer)).charAt(0);
311         } else if (ft.isNumber()) {
312            o = parseNumber(getValue(n, ctx, outer).toString(), (Class<? extends Number>)ft.getInnerClass());
313         } else if (ft.isMap()) {
314            Resource r = n.asResource();
315            if (ctx.wasAlreadyProcessed(r))
316               return null;
317            Map m = (ft.canCreateNewInstance(outer) ? (Map)ft.newInstance(outer) : new ObjectMap(bc));
318            o = parseIntoMap(r, m, nt.getKeyType(), nt.getValueType(), ctx);
319         } else if (ft.isCollection() || ft.isArray()) {
320            if (ft.isArray())
321               o = new ArrayList();
322            else
323               o = (ft.canCreateNewInstance(outer) ? (Collection<?>)ft.newInstance(outer) : new ObjectList(bc));
324            Resource r = n.asResource();
325            if (ctx.wasAlreadyProcessed(r))
326               return null;
327            if (isSeq(r, ctx)) {
328               parseIntoCollection(r.as(Seq.class), (Collection)o, ft.getElementType(), ctx);
329            } else if (isBag(r, ctx)) {
330               parseIntoCollection(r.as(Bag.class), (Collection)o, ft.getElementType(), ctx);
331            } else if (r.canAs(RDFList.class)) {
332               parseIntoCollection(r.as(RDFList.class), (Collection)o, ft.getElementType(), ctx);
333            } else {
334               throw new ParseException("Unrecognized node type ''{0}'' for collection", n);
335            }
336            if (ft.isArray())
337               o = bc.toArray(ft, (Collection)o);
338         } else if (ft.canCreateNewInstanceFromObjectMap(outer)) {
339            Resource r = n.asResource();
340            if (ctx.wasAlreadyProcessed(r))
341               return null;
342            Map m = new ObjectMap(bc);
343            parseIntoMap(r, m, nt.getKeyType(), nt.getValueType(), ctx);
344            o = ft.newInstanceFromObjectMap(outer, (ObjectMap)m);
345         } else if (ft.canCreateNewBean(outer)) {
346            Resource r = n.asResource();
347            if (ctx.wasAlreadyProcessed(r))
348               return null;
349            BeanMap<?> bm = bc.newBeanMap(outer, ft.getInnerClass());
350            o = parseIntoBeanMap(r, bm, ctx).getBean();
351         } else if (ft.isUri() && n.isResource()) {
352            o = ft.newInstanceFromString(outer, ctx.decodeString(n.asResource().getURI()));
353         } else if (ft.canCreateNewInstanceFromString(outer)) {
354            o = ft.newInstanceFromString(outer, ctx.decodeString(getValue(n, ctx, outer)));
355         } else {
356            throw new ParseException("Class ''{0}'' could not be instantiated.  Reason: ''{1}''", ft.getInnerClass().getName(), ft.getNotABeanReason());
357         }
358
359         if (filter != null && o != null)
360            o = filter.unfilter(o, nt);
361
362         if (outer != null)
363            setParent(nt, o, outer);
364
365         if (name != null)
366            setName(nt, o, name);
367
368         return (T)o;
369
370      } catch (RuntimeException e) {
371         throw e;
372      } catch (Exception e) {
373         if (p == null)
374            throw new ParseException("Error occurred trying to parse into class ''{0}''", ft).initCause(e);
375         throw new ParseException("Error occurred trying to parse value for bean property ''{0}'' on class ''{1}''",
376            p.getName(), p.getBeanMeta().getClassMeta()
377         ).initCause(e);
378      }
379   }

381   private boolean isSeq(RDFNode n, RdfParserContext ctx) {
382      if (n.isResource()) {
383         Statement st = n.asResource().getProperty(ctx.getTypeProperty());
384         if (st != null)
385            return RDF_SEQ.equals(st.getResource().getURI());
386      }
387      return false;
388   }
389
390   private boolean isBag(RDFNode n, RdfParserContext ctx) {
391      if (n.isResource()) {
392         Statement st = n.asResource().getProperty(ctx.getTypeProperty());
393         if (st != null)
394            return RDF_BAG.equals(st.getResource().getURI());
395      }
396      return false;
397   }

399   private Object getValue(RDFNode n, RdfParserContext ctx, Object outer) throws ParseException {
400      if (n.isLiteral())
401         return n.asLiteral().getValue();
402      if (n.isResource()) {
403         Statement st = n.asResource().getProperty(ctx.getValueProperty());
404         if (st != null) {
405            n = st.getObject();
406            if (n.isLiteral())
407               return n.asLiteral().getValue();
408            return parseAnything(object(), st.getObject(), null, ctx, outer, null);
409         }
410      }
411      throw new ParseException("Unknown value type for node ''{0}''", n);
412   }

414   private <K,V> Map<K,V> parseIntoMap(Resource r, Map<K,V> m, ClassMeta<K> keyType, ClassMeta<V> valueType, RdfParserContext ctx) throws ParseException {
415      // Add URI as "uri" to generic maps.
416      if (r.getURI() != null) {
417         K uri = convertAttrToType(m, "uri", keyType);
418         V value = convertAttrToType(m, r.getURI(), valueType);
419         m.put(uri, value);
420      }
421      for (StmtIterator i = r.listProperties(); i.hasNext();) {
422         Statement st = i.next();
423         Property p = st.getPredicate();
424         String key = p.getLocalName();
425         if (! (key.equals("root") && p.getURI().equals(ctx.getJunoNsUri()))) {
426            key = ctx.decodeString(key);
427            RDFNode o = st.getObject();
428            K key2 = convertAttrToType(m, key, keyType);
429            V value = parseAnything(valueType, o, null, ctx, m, key);
430            m.put(key2, value);
431         }
432
433      }
434      // TODO Auto-generated method stub
435      return m;
436   }

438   private <E> Collection<E> parseIntoCollection(Container c, Collection<E> l, ClassMeta<E> et, RdfParserContext ctx) throws ParseException {
439      for (NodeIterator ni = c.iterator(); ni.hasNext();) {
440         E e = parseAnything(et, ni.next(), null, ctx, l, null);
441         l.add(e);
442      }
443      return l;
444   }
445
446   private <E> Collection<E> parseIntoCollection(RDFList list, Collection<E> l, ClassMeta<E> et, RdfParserContext ctx) throws ParseException {
447      for (ExtendedIterator<RDFNode> ni = list.iterator(); ni.hasNext();) {
448         E e = parseAnything(et, ni.next(), null, ctx, l, null);
449         l.add(e);
450      }
451      return l;
452   }
