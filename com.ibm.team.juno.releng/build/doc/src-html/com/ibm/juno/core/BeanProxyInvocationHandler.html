009package com.ibm.juno.core;
010
011import java.lang.reflect.*;
012import java.util.*;
013
014/**
015 * Provides an {@link InvocationHandler} for creating beans from bean interfaces.
016 * <p>
017 *    If the {@code useInterfaceProxies} setting is enabled in {@link BeanContext}, this
018 *    is the class that creates instances of beans from interfaces.
019 *
020 * @author Barry M. Caceres
021 * @param <T> The interface class
022 */
023public class BeanProxyInvocationHandler<T> implements InvocationHandler {
024
025   private final BeanMeta<T> meta;                 // The BeanMeta for this instance
026   private Map<String, Object> beanProps;    // The map of property names to bean property values.
027
028   /**
029    * Constructs with the specified {@link BeanMeta}.
030    *
031    * @param meta The bean meta data.
032    */
033   public BeanProxyInvocationHandler(BeanMeta<T> meta) {
034      this.meta = meta;
035      this.beanProps = new HashMap<String, Object>();
036   }
037
038   /**
039    * Implemented to handle the method called.
040    */
041   @Override /* InvocationHandler */
042   public Object invoke(Object proxy, Method method, Object[] args) {
043      Class<?>[] paramTypes = method.getParameterTypes();
044      if (method.getName().equals("equals") && (paramTypes.length == 1) && (paramTypes[0] == java.lang.Object.class)) {
045         Object arg = args[0];
046         if (arg == null)
047            return false;
048         if (proxy == arg)
049            return true;
050         if (proxy.getClass() == arg.getClass()) {
051            InvocationHandler ih = Proxy.getInvocationHandler(arg);
052            if (ih instanceof BeanProxyInvocationHandler) {
053               return this.beanProps.equals(((BeanProxyInvocationHandler<?>)ih).beanProps);
054            }
055         }
056         BeanMap<Object> bean = this.meta.ctx.forBean(arg);
057         return this.beanProps.equals(bean);
058      }
059
060      if (method.getName().equals("hashCode") && (paramTypes.length == 0))
061         return Integer.valueOf(this.beanProps.hashCode());
062
063      if (method.getName().equals("toString") && (paramTypes.length == 0))
064         return this.beanProps.toString();
065
066      String prop = this.meta.getterProps.get(method);
067      if (prop != null)
068         return this.beanProps.get(prop);
069
070      prop = this.meta.setterProps.get(method);
071      if (prop != null) {
072         this.beanProps.put(prop, args[0]);
073         return null;
074      }
075
076      throw new UnsupportedOperationException("Unsupported bean method.  method=[ " + method + " ]");
077   }
078}
