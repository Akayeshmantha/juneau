019/**
020 * Top-level ATOM feed object.
021 * <p>
022 *    Represents an <code>atomFeed</code> construct in the RFC4287 specification.
023 * <p>
024 * <h6 class='figure'>Schema</h6>
025 * <p class='bcode'>
026 *    atomFeed =
027 *       element atom:feed {
028 *          atomCommonAttributes,
029 *          (atomAuthor*
030 *           & atomCategory*
031 *           & atomContributor*
032 *           & atomGenerator?
033 *           & atomIcon?
034 *           & atomId
035 *           & atomLink*
036 *           & atomLogo?
037 *           & atomRights?
038 *           & atomSubtitle?
039 *           & atomTitle
040 *           & atomUpdated
041 *           & extensionElement*),
042 *          atomEntry*
043 *       }
044 * </p>
045 * <p>
046 *    Refer to {@link com.ibm.juno.core.dto.atom} for further information about ATOM support.
047 * </p>
048 *
049 * @author James Bognar (jbognar@us.ibm.com)
050 */

175    * @param entries The entries in the feed.

178   public Feed setEntries(List<Entry> entries) {
179      this.entries = entries;
180      return this;
181   }
182
183   /**
184    * Adds an entry to the list of entries in the feed.
185    *
186    * @param entries The entries to add to the list of entries in the feed.s
187    * @return This object (for method chaining).
188    */
189   public Feed addEntries(Entry...entries) {
190      if (this.entries == null)
191         this.entries = new LinkedList<Entry>();
192      this.entries.addAll(Arrays.asList(entries));
193      return this;
194   }
