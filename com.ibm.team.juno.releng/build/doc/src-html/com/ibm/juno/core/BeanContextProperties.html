009package com.ibm.juno.core;
010
011import java.beans.*;
012import java.io.*;
013import java.lang.reflect.*;
014
015import com.ibm.juno.core.annotation.*;
016
017/**
018 * Configurable properties on the {@link BeanContextFactory} class.
019 * <p>
020 *    Use the {@link BeanContextFactory#setProperty(String, Object)} method to set properties on
021 *    bean contexts.
022 *
023 * @author James Bognar (jbognar@us.ibm.com)
024 */
025public final class BeanContextProperties {
026
027   /**
028    * Require no-arg constructor ({@link Boolean}, default=<jk>false</jk>).
029    * <p>
030    * If <jk>true</jk>, a Java class must implement a default no-arg constructor to be considered a bean.
031    * <p>
032    * The {@link Bean @Bean} annotation can be used on a class to override this setting when <jk>true</jk>.
033    */
034   public static final String BEAN_beansRequireDefaultConstructor = "BeanContext.beansRequireDefaultConstructor";
035
036   /**
037    * Require {@link Serializable} interface ({@link Boolean}, default=<jk>false</jk>).
038    * <p>
039    * If <jk>true</jk>, a Java class must implement the {@link Serializable} interface to be considered a bean.
040    * <p>
041    * The {@link Bean @Bean} annotation can be used on a class to override this setting when <jk>true</jk>.
042    */
043   public static final String BEAN_beansRequireSerializable = "BeanContext.beansRequireSerializable";
044
045   /**
046    * Require setters for getters ({@link Boolean}, default=<jk>false</jk>).
047    * <p>
048    * If <jk>true</jk>, only getters that have equivalent setters will be considered as properties on a bean.
049    * Otherwise, they will be ignored.
050    */
051   public static final String BEAN_beansRequireSettersForGetters = "BeanContext.beansRequireSettersForGetters";
052
053   /**
054    * Require some properties ({@link Boolean}, default=<jk>true</jk>).
055    * <p>
056    * If <jk>true</jk>, then a Java class must contain at least 1 property to be considered a bean.
057    * <p>
058    * The {@link Bean @Bean} annotation can be used on a class to override this setting when <jk>true</jk>.
059    */
060   public static final String BEAN_beansRequireSomeProperties = "BeanContext.beansRequireSomeProperties";
061
062   /**
063    * Put returns old value ({@link Boolean}, default=<jk>false</jk>).
064    * <p>
065    * If <jk>true</jk>, then the {@link BeanMap#put(String,Object) BeanMap.put()} method will return old property values.
066    * <p>
067    * Disabled by default because it introduces a slight performance penalty.
068    */
069   public static final String BEAN_beanMapPutReturnsOldValue = "BeanContext.beanMapPutReturnsOldValue";
070
071   /**
072    * Look for bean constructors with the specified minimum visibility ({@link Visibility}, default={@link Visibility#PUBLIC}).
073    */
074   public static final String BEAN_beanConstructorVisibility = "BeanContext.beanConstructorVisibility";
075
076   /**
077    * Look for bean classes with the specified minimum visibility ({@link Visibility}, default={@link Visibility#PUBLIC}).
078    * <p>
079    * Classes are not considered beans unless they meet the minimum visibility requirements.
080    * For example, if the visibility is <code>PUBLIC</code> and the bean class is <jk>protected</jk>, then
081    *    the class will not be interpreted as a bean class.
082    */
083   public static final String BEAN_beanClassVisibility = "BeanContext.beanClassVisibility";
084
085   /**
086    * Look for bean fields with the specified minimum visibility ({@link Visibility}, default={@link Visibility#PUBLIC}).
087    * <p>
088    * Fields are not considered bean properties unless they meet the minimum visibility requirements.
089    * For example, if the visibility is <code>PUBLIC</code> and the bean field is <jk>protected</jk>, then
090    *    the field will not be interpreted as a bean property.
091    * <p>
092    * Use {@link Visibility#NONE} to prevent bean fields from being interpreted as bean properties altogether.
093    */
094   public static final String BEAN_beanFieldVisibility = "BeanContext.beanFieldVisibility";
095
096   /**
097    * Look for bean methods with the specified minimum visibility ({@link Visibility}, default={@link Visibility#PUBLIC}).
098    * <p>
099    * Methods are not considered bean getters/setters unless they meet the minimum visibility requirements.
100    * For example, if the visibility is <code>PUBLIC</code> and the bean method is <jk>protected</jk>, then
101    *    the method will not be interpreted as a bean getter or setter.
102    */
103   public static final String BEAN_methodVisibility = "BeanContext.methodVisibility";
104
105   /**
106    * Use Java {@link Introspector} for determining bean properties ({@link Boolean}, default=<jk>false</jk>).
107    * <p>
108    * Using the built-in Java bean introspector will not pick up fields or non-standard getters/setters.
109    * Most {@link Bean @Bean} annotations will be ignored.
110    */
111   public static final String BEAN_useJavaBeanIntrospector = "BeanContext.useJavaBeanIntrospector";
112
113   /**
114    * Use interface proxies ({@link Boolean}, default=<jk>true</jk>).
115    * <p>
116    * If <jk>true</jk>, then interfaces will be instantiated as proxy classes through the use of an {@link InvocationHandler}
117    * if there is no other way of instantiating them.
118    */
119   public static final String BEAN_useInterfaceProxies = "BeanContext.useInterfaceProxies";
120
121   /**
122    * Ignore unknown properties ({@link Boolean}, default=<jk>false</jk>).
123    * <p>
124    * If <jk>true</jk>, trying to set a value on a non-existent bean property will silently be ignored.
125    * Otherwise, a {@code RuntimeException} is thrown.
126    */
127   public static final String BEAN_ignoreUnknownBeanProperties = "BeanContext.ignoreUnknownBeanProperties";
128
129   /**
130    * Ignore unknown properties with null values ({@link Boolean}, default=<jk>true</jk>).
131    * <p>
132    * If <jk>true</jk>, trying to set a <jk>null</jk> value on a non-existent bean property will silently be ignored.
133    * Otherwise, a {@code RuntimeException} is thrown.
134    */
135   public static final String BEAN_ignoreUnknownNullBeanProperties = "BeanContext.ignoreUnknownNullBeanProperties";
136
137   /**
138    * Ignore properties without setters ({@link Boolean}, default=<jk>true</jk>).
139    * <p>
140    * If <jk>true</jk>, trying to set a value on a bean property without a setter will silently be ignored.
141    * Otherwise, a {@code RuntimeException} is thrown.
142    */
143   public static final String BEAN_ignorePropertiesWithoutSetters = "BeanContext.ignorePropertiesWithoutSetters";
144
145   /**
146    * Ignore invocation errors on getters ({@link Boolean}, default=<jk>false</jk>).
147    * <p>
148    * If <jk>true</jk>, errors thrown when calling bean getter methods will silently be ignored.
149    * Otherwise, a {@code BeanRuntimeException} is thrown.
150    */
151   public static final String BEAN_ignoreInvocationExceptionsOnGetters = "BeanContext.ignoreInvocationExceptionsOnGetters";
152
153   /**
154    * Ignore invocation errors on setters ({@link Boolean}, default=<jk>false</jk>).
155    * <p>
156    * If <jk>true</jk>, errors thrown when calling bean setter methods will silently be ignored.
157    * Otherwise, a {@code BeanRuntimeException} is thrown.
158    */
159   public static final String BEAN_ignoreInvocationExceptionsOnSetters = "BeanContext.ignoreInvocationExceptionsOnSetters";
160
161   /**
162    * Add to the list of packages whose classes should not be considered beans ({@link String}, comma-delimited list).
163    * <p>
164    * When specified, the current list of ignore packages are appended to.
165    * The default list of ignore packages are as follows:
166    * <ul>
167    *    <li><code>java.lang</code>
168    *    <li><code>java.lang.annotation</code>
169    *    <li><code>java.lang.ref</code>
170    *    <li><code>java.lang.reflect</code>
171    *    <li><code>java.io</code>
172    *    <li><code>java.net</code>
173    *    <li><code>java.nio.*</code>
174    *    <li><code>java.util.*</code>
175    * </ul>
176    * Any classes within these packages will be serialized to strings using {@link Object#toString()}.
177    * <p>
178    * Note that you can specify prefix patterns to include all subpackages.
179    */
180   public static final String BEAN_addNotBeanPackages = "BeanContext.addNotBeanPackages";
181
182   /**
183    * Remove from the list of packages whose classes should not be considered beans ({@link String}, comma-delimited list).
184    * <p>
185    * Essentially the opposite of {@link #BEAN_addNotBeanPackages}.
186    */
187   public static final String BEAN_removeNotBeanPackages = "BeanContext.removeNotBeanPackages";
188}
