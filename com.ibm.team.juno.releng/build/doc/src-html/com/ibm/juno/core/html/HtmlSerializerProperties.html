019/**
020 * Configurable properties on the {@link HtmlSerializer} class.
021 * <p>
022 *    Use the {@link HtmlSerializer#setProperty(String, Object)} method to set property values.
023 * <p>
024 *    In addition to these properties, the following properties are also applicable for {@link HtmlSerializer}.
025 * <ul>
026 *    <li>{@link XmlSerializerProperties}
027 *    <li>{@link SerializerProperties}
028 *    <li>{@link BeanContextProperties}
029 * </ul>
030 *
031 * @author James Bognar (jbognar@us.ibm.com)
032 */
033public final class HtmlSerializerProperties implements Cloneable {
034
035   /**
036    * Anchor text source ({@link String}, default={@link #TO_STRING}).
037    * <p>
038    * When creating anchor tags (e.g. <code><xt>&lt;a</xt> <xa>href</xa>=<xs>'...'</xs><xt>&gt;</xt>text<xt>&lt;/a&gt;</xt></code>)
039    *    in HTML, this setting defines what to set the inner text to.
040    * <p>
041    * Possible values:
042    * <ul>
043    *    <li>{@link #TO_STRING} / <js>"toString"</js> - Set to whatever is returned by {@link #toString()} on the object.
044    *    <li>{@link #URI} / <js>"uri"</js> - Set to the URI value.
045    *    <li>{@link #LAST_TOKEN} / <js>"lastToken"</js> - Set to the last token of the URI value.
046    *    <li>{@link #PROPERTY_NAME} / <js>"propertyName"</js> - Set to the bean property name.
047    * </ul>
048    */
049   public static final String HTML_uriAnchorText = "HtmlSerializer.uriAnchorText";
050
051   /** Constant for {@link HtmlSerializerProperties#HTML_uriAnchorText} property. */
052   public static final String PROPERTY_NAME = "propertyName";
053   /** Constant for {@link HtmlSerializerProperties#HTML_uriAnchorText} property. */
054   public static final String TO_STRING = "toString";
055   /** Constant for {@link HtmlSerializerProperties#HTML_uriAnchorText} property. */
056   public static final String URI = "uri";
057   /** Constant for {@link HtmlSerializerProperties#HTML_uriAnchorText} property. */
058   public static final String LAST_TOKEN = "lastToken";
059
060   String uriAnchorText = TO_STRING, title, description, cssUrl;
061   String[] cssImports;
062   ObjectMap links;
063   boolean nowrap;
064
065   /**
066    * Sets the specified property value.
067    *
068    * @param property The property name.
069    * @param value The property value.
070    * @return <jk>true</jk> if property name was valid and property was set.
071    */
072   public boolean setProperty(String property, Object value) {
073      if (property.equals(HTML_uriAnchorText))
074         uriAnchorText = (value == null ? null : value.toString());
075      else if (property.equals(HTMLDOC_title))
076         title = (value == null ? null : value.toString());
077      else if (property.equals(HTMLDOC_description))
078         description = (value == null ? null : value.toString());
079      else if (property.equals(HTMLDOC_nowrap))
080         nowrap = Boolean.valueOf(value.toString());
081      else if (property.equals(HTMLDOC_links))
082         try {
083            links = new ObjectMap(value.toString());
084         } catch (ParseException e) {
085            e.printStackTrace();
086         }
087      else if (property.equals(HTMLDOC_addLinks))
088         try {
089            if (links == null)
090               links = new ObjectMap(value.toString());
091            else
092               links.putAll(new ObjectMap(value.toString()));
093         } catch (ParseException e) {
094            e.printStackTrace();
095         }
096      else if (property.equals(HTMLDOC_cssUrl))
097         cssUrl = (value == null ? null : value.toString());
098      else if (property.equals(HTMLDOC_cssImports))
099         cssImports = StringUtils.split(value == null ? null : value.toString(), ',');
100      else
101         return false;
102      return true;
103   }
104
105   @Override /* Cloneable */
106   public HtmlSerializerProperties clone() {
107      try {
108         return (HtmlSerializerProperties)super.clone();
109      } catch (CloneNotSupportedException e) {
110         throw new RuntimeException(e); // Shouldn't happen
111      }
112   }
