022/**
023 * Context object that lives for the duration of a single parse of {@link RdfParser}.
024 * <p>
025 *    See {@link ParserContext} for details.
026 * </p>
027 * <p>
028 *    This class is NOT thread safe.  It is meant to be discarded after one-time use.
029 * </p>
030 *
031 * @author James Bognar (jbognar@us.ibm.com)
032 */
033public class RdfParserContext extends ParserContext {
034
035   private final String rdfLanguage;
036   private final Namespace junoNs, junoBpNs;
037   private final Property pRoot, pValue, pClass, pType;
038   private final Model model;
039   private final boolean trimWhitespace, looseCollection;
040   private final RDFReader rdfReader;
041   private final Set<Resource> urisVisited = new HashSet<Resource>();
042   private final RdfCollectionFormat collectionFormat;
043
044   /**
045    * Constructor.
046    *
047    * @param beanContext The bean context being used by the parser.
048    * @param pp Default general parser properties.
049    * @param rpp Default Jena parser properties.
050    * @param op Override properties.
051    * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
052    * @param outer The outer object for instantiating top-level non-static inner classes.
053    */
054   protected RdfParserContext(BeanContext beanContext, ParserProperties pp, RdfParserProperties rpp, ObjectMap op, Method javaMethod, Object outer) {
055      super(beanContext, pp, op, javaMethod, outer);
056      ObjectMap jenaSettings = new ObjectMap();
057      jenaSettings.putAll(rpp.jenaSettings);
058      if (op == null || op.isEmpty()) {
059         this.rdfLanguage = rpp.rdfLanguage;
060         this.junoNs = rpp.junoNs;
061         this.junoBpNs = rpp.junoBpNs;
062         this.trimWhitespace = rpp.trimWhitespace;
063         this.collectionFormat = rpp.collectionFormat;
064         this.looseCollection = rpp.looseCollection;
065      } else {
066         this.rdfLanguage = op.getString(RDF_language, rpp.rdfLanguage);
067         this.junoNs = (op.containsKey(RDF_junoNs) ? NamespaceFactory.parseNamespace(op.get(RDF_junoNs)) : rpp.junoNs);
068         this.junoBpNs = (op.containsKey(RDF_junoBpNs) ? NamespaceFactory.parseNamespace(op.get(RDF_junoBpNs)) : rpp.junoBpNs);
069         this.trimWhitespace = op.getBoolean(RdfParserProperties.RDF_trimWhitespace, rpp.trimWhitespace);
070         this.collectionFormat = RdfCollectionFormat.valueOf(op.getString(RDF_collectionFormat, "DEFAULT"));
071         this.looseCollection = op.getBoolean(RDF_looseCollection, rpp.looseCollection);
072      }
073      this.model = ModelFactory.createDefaultModel();
074      addModelPrefix(junoNs);
075      addModelPrefix(junoBpNs);
076      this.pRoot = model.createProperty(junoNs.getUri(), RDF_junoNs_ROOT);
077      this.pValue = model.createProperty(junoNs.getUri(), RDF_junoNs_VALUE);
078      this.pClass = model.createProperty(junoNs.getUri(), RDF_junoNs_CLASS);
079      this.pType = model.createProperty("http://www.w3.org/1999/02/22-rdf-syntax-ns#type");
080      rdfReader = model.getReader(rdfLanguage);
081
082      // Note: NTripleReader throws an exception if you try to set any properties on it.
083      if (! rdfLanguage.equals(LANG_NTRIPLE)) {
084         for (Map.Entry<String,Object> e : jenaSettings.entrySet())
085            rdfReader.setProperty(e.getKey(), e.getValue());
086      }
087   }
088
089   final boolean wasAlreadyProcessed(Resource r) {
090      return ! urisVisited.add(r);
091   }
092
093   final Property getRootProperty() {
094      return pRoot;
095   }
096
097   final Property getValueProperty() {
098      return pValue;
099   }
100
101   final Property getClassProperty() {
102      return pClass;
103   }
104
105   final Property getTypeProperty() {
106      return pType;
107   }
108
109   final Model getModel() {
110      return model;
111   }
112
113   final RDFReader getRdfReader() {
114      return rdfReader;
115   }
116
117   final RdfCollectionFormat getCollectionFormat() {
118      return collectionFormat;
119   }
120
121   final boolean isLooseCollection() {
122      return looseCollection;
123   }
124
125   final String getJunoNsUri() {
126      return junoNs.getUri();
127   }
128
129   /**
130    * Adds the specified namespace as a model prefix.
131    *
132    * @param ns The XML namespace.
133    */
134   final void addModelPrefix(Namespace ns) {
135      model.setNsPrefix(ns.getName(), ns.getUri());
136   }
137
138   /**
139    * Constructs a <code>Property</code> in the Juno Bean namespace in this mode.
140    *
141    * @param name The property name.
142    * @return The new property object.
143    */
144   final Property getProperty(String name) {
145      return model.createProperty(junoBpNs.getUri(), name);
146   }
147
148   /**
149    * Decodes the specified string.
150    * <p>
151    *    If {@link RdfParserProperties#RDF_trimWhitespace} is <jk>true</jk>, the resulting string is trimmed before decoding.
152    * </p>
153    * <p>
154    *    If {@link #isTrimStrings()} is <jk>true</jk>, the resulting string is trimmed after decoding.
155    * </p>
156    *
157    * @param s The string to trim.
158    * @return The trimmed string, or <jk>null</jk> if the string was <jk>null</jk>.
159    */
160   final String decodeString(Object o) {
161      if (o == null)
162         return null;
163      String s = o.toString();
164      if (s.isEmpty())
165         return s;
166      if (trimWhitespace)
167         s = s.trim();
168      s = XmlUtils.decode(s);
169      if (isTrimStrings())
170         s = s.trim();
171      return s;
172   }
173}
