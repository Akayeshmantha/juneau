009package com.ibm.juno.core.urlencoding;
010
011import static com.ibm.juno.core.urlencoding.UonSerializerProperties.*;
012import static com.ibm.juno.core.urlencoding.UrlEncodingProperties.*;
013
014import java.io.*;
015import java.lang.reflect.*;
016
017import com.ibm.juno.core.*;
018import com.ibm.juno.core.serializer.*;
019
020/**
021 * Context object that lives for the duration of a single serialization of {@link UonSerializer} and {@link UrlEncodingSerializer}.
022 * <p>
023 *    See {@link SerializerContext} for details.
024 * </p>
025 * <p>
026 *    This class is NOT thread safe.  It is meant to be discarded after one-time use.
027 * </p>
028 *
029 * @author James Bognar (jbognar@us.ibm.com)
030 */
031public final class UonSerializerContext extends SerializerContext {
032
033   private final boolean simpleMode, useWhitespace, encodeChars, expandedParams;
034
035   /**
036    * Constructor.
037    *
038    * @param beanContext The bean context being used by the serializer.
039    * @param sp Default general serializer properties.
040    * @param usp Default UON serializer properties.
041    * @param uep Default URL-Encoding properties.
042    * @param op Override properties.
043    * @param javaMethod Java method that invoked this serializer.
044    *    When using the REST API, this is the Java method invoked by the REST call.
045    *    Can be used to access annotations defined on the method or class.
046    */
047   protected UonSerializerContext(BeanContext beanContext, SerializerProperties sp, UonSerializerProperties usp, UrlEncodingProperties uep, ObjectMap op, Method javaMethod) {
048      super(beanContext, sp, op, javaMethod);
049      if (op == null || op.isEmpty()) {
050         simpleMode = usp.simpleMode;
051         useWhitespace = usp.useWhitespace;
052         encodeChars = usp.encodeChars;
053         expandedParams = uep.expandedParams;
054      } else {
055         simpleMode = op.getBoolean(UON_simpleMode, usp.simpleMode);
056         useWhitespace = op.getBoolean(UON_useWhitespace, usp.useWhitespace);
057         encodeChars = op.getBoolean(UON_encodeChars, usp.encodeChars);
058         expandedParams = op.getBoolean(URLENC_expandedParams, uep.expandedParams);
059
060      }
061   }
062
063   final UonSerializerWriter getWriter(Writer out) {
064      if (out instanceof UonSerializerWriter)
065         return (UonSerializerWriter)out;
066      return new UonSerializerWriter(out, useWhitespace, isSimpleMode(), isEncodeChars(), isTrimStrings(), getRelativeUriBase(), getAbsolutePathUriBase());
067   }
068
069   final boolean isUseWhitespace() {
070      return useWhitespace;
071   }
072
073   final boolean isSimpleMode() {
074      return simpleMode;
075   }
076
077   final boolean isEncodeChars() {
078      return encodeChars;
079   }
080
081   final boolean isExpandedParams() {
082      return expandedParams;
083   }
084
085   /**
086    * Returns true if the specified bean property should be expanded as multiple key-value pairs.
087    */
088   final boolean shouldUseExpandedParams(BeanPropertyMeta<?> pMeta) {
089      ClassMeta<?> cm = pMeta.getClassMeta();
090      if (cm.isArray() || cm.isCollection()) {
091         if (expandedParams)
092            return true;
093         if (pMeta.getBeanMeta().getClassMeta().getUrlEncodingMeta().isExpandedParams())
094            return true;
095      }
096      return false;
097   }
098
099   final boolean shouldUseExpandedParams(Object value) {
100      if (value == null || ! expandedParams)
101         return false;
102      ClassMeta<?> cm = getBeanContext().getClassMetaForObject(value).getFilteredClassMeta();
103      if (cm.isArray() || cm.isCollection()) {
104         if (expandedParams)
105            return true;
106      }
107      return false;
108   }
109
110}
