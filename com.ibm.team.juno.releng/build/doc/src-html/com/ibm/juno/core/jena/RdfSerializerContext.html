023/**
024 * Context object that lives for the duration of a single serialization of {@link RdfSerializer}.
025 * <p>
026 *    See {@link SerializerContext} for details.
027 * </p>
028 * <p>
029 *    This class is NOT thread safe.  It is meant to be discarded after one-time use.
030 * </p>
031 *
032 * @author James Bognar (jbognar@us.ibm.com)
033 */
034public final class RdfSerializerContext extends XmlSerializerContext {
035
036   private final String rdfLanguage;
037   private final Namespace junoNs, junoBpNs;
038   private final boolean addLiteralTypes, addRootProperty, useXmlNamespaces, looseCollection;
039   private final Property pRoot, pValue, pClass;
040   private final Model model;
041   private final RDFWriter writer;
042   private final RdfCollectionFormat collectionFormat;
043
044   /**
045    * Constructor.
046    * @param beanContext The bean context being used by the serializer.
047    * @param sp Default general serializer properties.
048    * @param xsp Default XML serializer properties.
049    * @param jsp Default Jena serializer properties.
050    * @param op Override properties.
051    * @param javaMethod Java method that invoked this serializer.
052    *    When using the REST API, this is the Java method invoked by the REST call.
053    *    Can be used to access annotations defined on the method or class.
054    */
055   protected RdfSerializerContext(BeanContext beanContext, SerializerProperties sp, XmlSerializerProperties xsp, RdfSerializerProperties jsp, ObjectMap op, Method javaMethod) {
056      super(beanContext, sp, xsp, op, javaMethod);
057      ObjectMap jenaSettings = new ObjectMap();
058      jenaSettings.put("rdfXml.tab", isUseIndentation() ? 2 : 0);
059      jenaSettings.put("rdfXml.attributeQuoteChar", Character.toString(getQuoteChar()));
060      jenaSettings.putAll(jsp.jenaSettings);
061      if (op == null || op.isEmpty()) {
062         this.rdfLanguage = jsp.rdfLanguage;
063         this.junoNs = jsp.junoNs;
064         this.junoBpNs = jsp.junoBpNs;
065         this.addLiteralTypes = jsp.addLiteralTypes;
066         this.addRootProperty = jsp.addRootProperty;
067         this.collectionFormat = jsp.collectionFormat;
068         this.looseCollection = jsp.looseCollection;
069         this.useXmlNamespaces = jsp.useXmlNamespaces;
070      } else {
071         this.rdfLanguage = op.getString(RDF_language, jsp.rdfLanguage);
072         this.junoNs = (op.containsKey(RDF_junoNs) ? NamespaceFactory.parseNamespace(op.get(RDF_junoNs)) : jsp.junoNs);
073         this.junoBpNs = (op.containsKey(RDF_junoBpNs) ? NamespaceFactory.parseNamespace(op.get(RDF_junoBpNs)) : jsp.junoBpNs);
074         this.addLiteralTypes = op.getBoolean(RDF_addLiteralTypes, jsp.addLiteralTypes);
075         this.addRootProperty = op.getBoolean(RDF_addRootProperty, jsp.addRootProperty);
076         for (Map.Entry<String,Object> e : op.entrySet()) {
077            String key = e.getKey();
078            if (key.startsWith("Rdf.jena."))
079               jenaSettings.put(key.substring(9), e.getValue());
080         }
081         this.collectionFormat = RdfCollectionFormat.valueOf(op.getString(RDF_collectionFormat, "DEFAULT"));
082         this.looseCollection = op.getBoolean(RDF_looseCollection, jsp.looseCollection);
083         this.useXmlNamespaces = op.getBoolean(RDF_useXmlNamespaces, jsp.useXmlNamespaces);
084      }
085      this.model = ModelFactory.createDefaultModel();
086      addModelPrefix(junoNs);
087      addModelPrefix(junoBpNs);
088      for (Namespace ns : this.getNamespaces())
089         addModelPrefix(ns);
090      this.pRoot = model.createProperty(junoNs.getUri(), RDF_junoNs_ROOT);
091      this.pValue = model.createProperty(junoNs.getUri(), RDF_junoNs_VALUE);
092      this.pClass = model.createProperty(junoNs.getUri(), RDF_junoNs_CLASS);
093      writer = model.getWriter(rdfLanguage);
094
095      // Only apply properties with this prefix!
096      String propPrefix = RdfProperties.LANG_PROP_MAP.get(rdfLanguage);
097      if (propPrefix == null)
098         throw new RuntimeException("Unknown RDF language encountered: '"+rdfLanguage+"'");
099
100      for (Map.Entry<String,Object> e : jenaSettings.entrySet())
101         if (e.getKey().startsWith(propPrefix))
102            writer.setProperty(e.getKey().substring(propPrefix.length()), e.getValue());
103   }
104
105   /**
106    * Adds the specified namespace as a model prefix.
107    * @param ns The XML namespace.
108    */
109   public void addModelPrefix(Namespace ns) {
110      model.setNsPrefix(ns.getName(), ns.getUri());
111   }
112
113   final RdfCollectionFormat getCollectionFormat() {
114      return collectionFormat;
115   }
116
117   final boolean isUseXmlNamespaces() {
118      return useXmlNamespaces;
119   }
120
121   final boolean isLooseCollection() {
122      return looseCollection;
123   }
124
125   final String getRdfLanguage() {
126      return rdfLanguage;
127   }
128
129   final Namespace getJunoNs() {
130      return junoNs;
131   }
132
133   final Namespace getJunoBpNs() {
134      return junoBpNs;
135   }
136
137   final boolean isAddLiteralTypes() {
138      return addLiteralTypes;
139   }
140
141   final boolean isAddRootProperty() {
142      return addRootProperty;
143   }
144
145   final Property getRootProperty() {
146      return pRoot;
147   }
148
149   final Property getValueProperty() {
150      return pValue;
151   }
152
153   final Property getClassProperty() {
154      return pClass;
155   }
156
157   final Model getModel() {
158      return model;
159   }
160
161   final RDFWriter getWriter() {
162      return writer;
163   }
164
165   final String encodeTextInvalidChars(Object o) {
166      if (o == null)
167         return null;
168      String s = o.toString();
169      if (isTrimStrings())
170         s = s.trim();
171      return XmlUtils.encodeTextInvalidChars(s);
172   }
173
174   final String encodeElementName(Object o) {
175      if (isTrimStrings() && o != null)
176         o = o.toString().trim();
177      return XmlUtils.encodeElementName(o);
178   }
