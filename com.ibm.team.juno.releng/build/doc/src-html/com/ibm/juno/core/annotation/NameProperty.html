018/**
019 * Identifies a setter as a method for setting the name of a POJO as it's known by
020 * its parent object.
021 * <p>
022 * For example, the {@link Section} class must know the name it's known by it's parent
023 * {@link ConfigFileImpl} class, so parsers will call this method with the sectio name
024 * using the {@link Section#setName(String)} method.
025 *
026 * @author James Bognar (jbognar@us.ibm.com)
027 */
028@Target({METHOD})
029@Retention(RUNTIME)
030@Inherited
031public @interface NameProperty {}
