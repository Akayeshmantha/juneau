011import static com.ibm.juno.core.xml.XmlParserProperties.*;
012
013import java.io.*;
014import java.lang.reflect.*;
015
016import javax.xml.stream.*;
017import javax.xml.stream.util.*;
018
019import com.ibm.juno.core.*;
020import com.ibm.juno.core.parser.*;
021import com.ibm.juno.core.utils.*;

023/**
024 * Context object that lives for the duration of a single parsing of {@link XmlParser}.
025 * <p>
026 *    This class is NOT thread safe.  It is meant to be discarded after one-time use.
027 * </p>
028 *
029 * @author James Bognar (jbognar@us.ibm.com)
030 */
031public final class XmlParserContext extends ParserContext {
032
033   private final String xsiNs;
034   private final boolean
035      trimWhitespace,
036      validating,
037      coalescing,
038      replaceEntityReferences,
039      preserveRootElement;
040   private final XMLReporter reporter;
041   private final XMLResolver resolver;
042   private final XMLEventAllocator eventAllocator;
043   private XMLStreamReader xmlStreamReader;
044
045   /**
046    * Create a new parser context with the specified options.
047    *
048    * @param beanContext The bean context being used.
049    * @param pp The default parser properties.
050    * @param xpp The default XML parser properties.
051    * @param op The override properties.
052    * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
053    * @param outer The outer object for instantiating top-level non-static inner classes.
054    */
055   public XmlParserContext(BeanContext beanContext, ParserProperties pp, XmlParserProperties xpp, ObjectMap op, Method javaMethod, Object outer) {
056      super(beanContext, pp, op, javaMethod, outer);
057      if (op == null || op.isEmpty()) {
058         xsiNs = xpp.getXsiNs();
059         trimWhitespace = xpp.isTrimWhitespace();
060         validating = xpp.isValidating();
061         coalescing = xpp.isCoalescing();
062         replaceEntityReferences = xpp.isReplaceEntityReferences();
063         reporter = xpp.getReporter();
064         resolver = xpp.getResolver();
065         eventAllocator = xpp.getEventAllocator();
066         preserveRootElement = xpp.isPreserveRootElement();
067      } else {
068         xsiNs = op.getString(XML_xsiNs, xpp.getXsiNs());
069         trimWhitespace = op.getBoolean(XML_trimWhitespace, xpp.isTrimWhitespace());
070         validating = op.getBoolean(XML_validating, xpp.isValidating());
071         coalescing = op.getBoolean(XML_coalescing, xpp.isCoalescing());
072         replaceEntityReferences = op.getBoolean(XML_replaceEntityReferences, xpp.isReplaceEntityReferences());
073         reporter = (XMLReporter)op.get(XML_reporter, xpp.getReporter());
074         resolver = (XMLResolver)op.get(XML_resolver, xpp.getResolver());
075         eventAllocator = (XMLEventAllocator)op.get(XML_eventAllocator, xpp.getEventAllocator());
076         preserveRootElement = op.getBoolean(XML_preserveRootElement, xpp.isPreserveRootElement());
077      }
078   }
079
080   final String getXsiNs() {
081      return xsiNs;
082   }
083
084   final boolean isPreserveRootElement() {
085      return preserveRootElement;
086   }
087
088   /**
089    * Wrap the specified reader in a STAX reader based on settings in this context.
090    *
091    * @param r The input reader.
092    * @param estimatedSize The estimated size of the contents of the reader.
093    * @return The new STAX reader.
094    * @throws ParseException If problem occurred trying to create reader.
095    */
096   final XMLStreamReader getReader(Reader r, int estimatedSize) throws ParseException {
097      try {
098         r = IOUtils.getBufferedReader(r, estimatedSize);
099         XMLInputFactory factory = XMLInputFactory.newInstance();
100         factory.setProperty(XMLInputFactory.IS_VALIDATING, validating);
101         factory.setProperty(XMLInputFactory.IS_COALESCING, coalescing);
102         factory.setProperty(XMLInputFactory.IS_REPLACING_ENTITY_REFERENCES, replaceEntityReferences);
103         if (factory.isPropertySupported(XMLInputFactory.REPORTER) && reporter != null)
104            factory.setProperty(XMLInputFactory.REPORTER, reporter);
105         if (factory.isPropertySupported(XMLInputFactory.RESOLVER) && resolver != null)
106            factory.setProperty(XMLInputFactory.RESOLVER, resolver);
107         if (factory.isPropertySupported(XMLInputFactory.ALLOCATOR) && eventAllocator != null)
108            factory.setProperty(XMLInputFactory.ALLOCATOR, eventAllocator);
109         xmlStreamReader = factory.createXMLStreamReader(r);
110         xmlStreamReader.nextTag();
111      } catch (Error e) {
112         close();
113         throw new ParseException(e.getLocalizedMessage());
114      } catch (XMLStreamException e) {
115         close();
116         throw new ParseException(e);
117      }
118
119      return xmlStreamReader;
120   }
121
122   /**
123    * Decodes and trims the specified string.
124    */
125   final String decodeString(String s) {
126      if (s == null || s.isEmpty())
127         return s;
128      if (trimWhitespace)
129         s = s.trim();
130      s = XmlUtils.decode(s);
131      if (isTrimStrings())
132         s = s.trim();
133      return s;
134   }
135
136   /**
137    * Decodes the specified literal (e.g. <js>"true"</js>, <js>"123"</js>).
138    * <p>
139    *    Unlike <code>decodeString(String)</code>, the input string is ALWAYS trimmed before decoding, and
140    *    NEVER trimmed after decoding.
141    * </p>
142    *
143    * @param s The string to trim.
144    * @return The trimmed string, or <jk>null</jk> if the string was <jk>null</jk>.
145    */
146   final String decodeLiteral(String s) {
147      if (s == null || s.isEmpty())
148         return s;
149      s = s.trim();
150      s = XmlUtils.decode(s);
151      return s;
152   }
153
154   /**
155    * Silently closes the XML stream.
156    */
157   @Override /* ParserContext */
158   public void close() throws ParseException {
159      super.close();
160      try {
161         if (xmlStreamReader != null)
162            xmlStreamReader.close();
163      } catch (XMLStreamException e) {
164         // Ignore.
165      }
166   }
167}
