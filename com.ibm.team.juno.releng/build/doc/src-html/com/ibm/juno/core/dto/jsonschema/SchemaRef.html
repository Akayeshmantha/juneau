009package com.ibm.juno.core.dto.jsonschema;
010
011import java.net.*;
012
013/**
014 * Convenience class for representing a schema reference such as <js>"{'$ref':'/url/to/ref'}"</js>.
015 * <p>
016 *    An instance of this object is equivalent to calling...
017 *
018 * <p class='bcode'>
019 *    Schema s = <jk>new</jk> Schema().setRef(uri);
020 * </p>
021 *
022 * @author James Bognar (jbognar@us.ibm.com)
023 */
024public class SchemaRef extends Schema {
025
026   /**
027    * Constructor.
028    *
029    * @param uri The URI of the target reference.  Can be <jk>null</jk>.
030    */
031   public SchemaRef(String uri) {
032      this.setRef(uri == null ? null : URI.create(uri));
033   }
034}
