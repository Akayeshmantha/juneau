009package com.ibm.juno.core.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.beans.*;
015import java.lang.annotation.*;
016
017import com.ibm.juno.core.*;
018import com.ibm.juno.core.filter.*;
019
020/**
021 * Used to tailor how beans get interpreted by the framework.
022 * <p>
023 *    Can be used to do the following:
024 * <ul>
025 *    <li>Explicitly specify the set and order of properties on a bean.
026 *    <li>Associate a {@link PropertyNamer} with a class.
027 *    <li>Specify subtypes of a bean differentiated by a sub type property.
028 * </ul>
029 * <p>
030 *    This annotation can be applied to classes and interfaces.
031 *
032 * @author James Bognar (jbognar@us.ibm.com)
033 */
034@Documented
035@Target(TYPE)
036@Retention(RUNTIME)
037@Inherited
038public @interface Bean {
039
040   /**
041    * The set and order of names of properties associated with a bean class.
042    * <p>
043    *    The order specified is the same order that the entries will be returned by the {@link BeanMap#entrySet()} and related methods.
044    * <p>
045    *    This annotation is an alternative to using the {@link BeanFilter} class with an implemented {@link BeanFilter#getProperties()} method.
046    *
047    * <dl>
048    *    <dt>Example:</dt>
049    *    <dd>
050    *       <p class='bcode'>
051    *    <jc>// Address class with only street/city/state properties (in that order).</jc>
052    *    <jc>// All other properties are ignored.</jc>
053    *    <ja>@Bean</ja>(properties={<js>"street"</js>,<js>"city"</js>,<js>"state"</js>})
054    *    <jk>public class</jk> Address {
055    *    ...
056    *       </p>
057    *    </dd>
058    * </dl>
059    */
060   String[] properties() default {};
061
062   /**
063    * Specifies a list of properties that should be excluded from {@link BeanMap#entrySet()}.
064    * <p>
065    *    This annotation is an alternative to using the {@link BeanFilter} class with an implemented {@link BeanFilter#getExcludeProperties()} method.
066    *
067    * <dl>
068    *    <dt>Example:</dt>
069    *    <dd>
070    *       <p class='bcode'>
071    *    <jc>// Address class with only street/city/state properties (in that order).</jc>
072    *    <jc>// All other properties are ignored.</jc>
073    *    <ja>@Bean</ja>(excludeProperties={<js>"city"</js>,<js>"state"</js>})
074    *    <jk>public class</jk> Address {
075    *       ...
076    *       </p>
077    *    </dd>
078    * </dl>
079    */
080   String[] excludeProperties() default {};
081
082   /**
083    * Associates a {@link PropertyNamer} with this bean to tailor the names of the bean properties.
084    * <p>
085    *    Property namers are used to transform bean property names from standard form to some other form.
086    *    For example, the {@link PropertyNamerDashedLC} will convert property names to dashed-lowercase, and
087    *       these will be used as attribute names in JSON, and element names in XML.
088    * <p>
089    *    This annotation is an alternative to using the {@link BeanFilter} class with an implemented {@link BeanFilter#getPropertyNamer()} method.
090    *
091    * <dl>
092    *    <dt>Example:</dt>
093    *    <dd>
094    *       <p class='bcode'>
095    *    <jc>// Define a class with dashed-lowercase property names.</jc>
096    *    <ja>@Bean</ja>(propertyNamer=PropertyNamerDashedLC.<jk>class</jk>)
097    *    <jk>public class</jk> MyClass {
098    *       ...
099    *    }
100    *       </p>
101    *    </dd>
102    * </dl>
103    */
104   Class<? extends PropertyNamer> propertyNamer() default PropertyNamerDefault.class;
105
106   /**
107    * Defines a virtual property on a superclass that identifies bean subtype classes.
108    * <p>
109    *    In the following example, the abstract class has two subclasses that are differentiated
110    *       by a property called <code>subType</code>
111    * <p class='bcode'>
112    *    <jc>// Abstract superclass</jc>
113    *    <ja>@Bean</ja>(
114    *       subTypeProperty=<js>"subType"</js>,
115    *       subTypes={
116    *          <ja>@BeanSubType</ja>(type=A1.<jk>class</jk>, id=<js>"A1"</js>),
117    *          <ja>@BeanSubType</ja>(type=A2.<jk>class</jk>, id=<js>"A2"</js>)
118    *       }
119    *    )
120    *    <jk>public class</jk> A {
121    *       <jk>public</jk> String <jf>f0</jf> = <js>"f0"</js>;
122    *    }
123    *
124    *    <jc>// Subclass 1</jc>
125    *    <jk>public class</jk> A1 <jk>extends</jk> A {
126    *       <jk>public</jk> String <jf>f1</jf>;
127    *    }
128    *
129    *    <jc>// Subclass 2</jc>
130    *    <jk>public class</jk> A2 <jk>extends</jk> A {
131    *       <jk>public</jk> String <jf>f2</jf>;
132    *    }
133    * </p>
134    * <p>
135    *    The following shows what happens when serializing a subclassed object to JSON:
136    * <p>
137    * <p class='bcode'>
138    *    JsonSerializer s = JsonSerializer.<jsf>DEFAULT_LAX</jsf>;
139    *    A1 a1 = <jk>new</jk> A1();
140    *    a1.<jf>f1</jf> = <js>"f1"</js>;
141    *    String r = s.serialize(a1);
142    *    <jsm>assertEquals</jsm>(<js>"{subType:'A1',f1:'f1',f0:'f0'}"</js>, r);
143    * </p>
144    * <p>
145    *    The following shows what happens when parsing back into the original object.
146    * <p>
147    * <p class='bcode'>
148    *    JsonParser p = JsonParser.<jsf>DEFAULT</jsf>;
149    *    A a = p.parse(r, A.<jk>class</jk>);
150    *    <jsm>assertTrue</jsm>(a <jk>instanceof</jk> A1);
151    * </p>
152    * <p>
153    *    This annotation is an alternative to using the {@link BeanFilter} class with an implemented {@link BeanFilter#getSubTypeProperty()} method.
154    */
155   String subTypeProperty() default "";
156
157   /**
158    * Used in conjunction with {@link #subTypeProperty()} to set up bean subtypes.
159    */
160   BeanSubType[] subTypes() default {};
161
162   /**
163    * Identifies a class to be used as the interface class for this and all subclasses.
164    * <p>
165    *    When specified, only the list of properties defined on the interface class will be used during serialization.
166    *    Additional properties on subclasses will be ignored.
167    * <p class='bcode'>
168    *    <jc>// Parent class</jc>
169    *    <ja>@Bean</ja>(interfaceClass=A.<jk>class</jk>)
170    *    <jk>public abstract class</jk> A {
171    *       <jk>public</jk> String <jf>f0</jf> = <js>"f0"</js>;
172    *    }
173    *
174    *    <jc>// Sub class</jc>
175    *    <jk>public class</jk> A1 <jk>extends</jk> A {
176    *       <jk>public</jk> String <jf>f1</jf> = <js>"f1"</js>;
177    *    }
178    *
179    *    JsonSerializer s = JsonSerializer.<jsf>DEFAULT_LAX</jsf>;
180    *    A1 a1 = <jk>new</jk> A1();
181    *    String r = s.serialize(a1);
182    *    <jsm>assertEquals</jsm>(<js>"{f0:'f0'}"</js>, r);  // Note f1 is not serialized.
183    * </p>
184    * <p>
185    *    Note that this annotation can be used on the parent class so that it filters to all child classes,
186    *       or can be set individually on the child classes.
187    * <p>
188    *    This annotation is an alternative to using the {@link BeanFilter} class with an implemented {@link BeanFilter#getInterfaceClass()} method.
189    */
190   Class<?> interfaceClass() default Object.class;
191
192   /**
193    * Identifies a stop class for the annotated class.
194    * <p>
195    * Identical in purpose to the stop class specified by {@link Introspector#getBeanInfo(Class, Class)}.
196    * Any properties in the stop class or in its baseclasses will be ignored during analysis.
197    * <p>
198    * For example, in the following class hierarchy, instances of <code>C3</code> will include property <code>p3</code>, but
199    *    not <code>p1</code> or <code>p2</code>.
200    * <p class='bcode'>
201    *    <jk>public class</jk> C1 {
202    *       <jk>public int</jk> getP1();
203    *    }
204    *
205    *    <jk>public class</jk> C2 <jk>extends</jk> C1 {
206    *       <jk>public int</jk> getP2();
207    *    }
208    *
209    *    <ja>@Bean</ja>(stopClass=C2.<jk>class</jk>)
210    *    <jk>public class</jk> C3 <jk>extends</jk> C2 {
211    *       <jk>public int</jk> getP3();
212    *    }
213    * </p>
214    */
215   Class<?> stopClass() default Object.class;
216}
