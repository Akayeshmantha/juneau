009package com.ibm.juno.core.parser;
010
011import static com.ibm.juno.core.parser.ParserProperties.*;
012
013import java.lang.reflect.*;
014import java.util.*;
015
016import com.ibm.juno.core.*;
017import com.ibm.juno.core.utils.*;
018import com.ibm.juno.core.utils.log.*;
019
020/**
021 * Context object that lives for the duration of a single parsing of {@link Parser} and its subclasses.
022 * <p>
023 *    This class is NOT thread safe.  It is meant to be discarded after one-time use.
024 * </p>
025 *
026 * @author James Bognar (jbognar@us.ibm.com)
027 */
028public class ParserContext {
029
030   private static JunoLogger logger = JunoLogger.getLogger(ParserContext.class);
031
032   private final boolean debug, trimStrings;
033   private boolean closed;
034   private final BeanContext beanContext;
035   private final List<String> warnings = new LinkedList<String>();
036
037   private final ObjectMap properties;
038   private final Method javaMethod;
039   private final Object outer;
040
041   /**
042    * Create a new parser context with the specified options.
043    *
044    * @param beanContext The bean context being used.
045    * @param pp The default parser properties.
046    * @param op The override properties.
047    * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
048    * @param outer The outer object for instantiating top-level non-static inner classes.
049    */
050   public ParserContext(BeanContext beanContext, ParserProperties pp, ObjectMap op, Method javaMethod, Object outer) {
051      if (op == null || op.isEmpty()) {
052         debug = pp.debug;
053         trimStrings = pp.trimStrings;
054      } else {
055         debug = op.getBoolean(PARSER_debug, pp.debug);
056         trimStrings = op.getBoolean(PARSER_trimStrings, pp.trimStrings);
057      }
058      this.beanContext = beanContext;
059      this.properties = op;
060      this.javaMethod = javaMethod;
061      this.outer = outer;
062   }
063
064   /**
065    * Returns the bean context associated with this context.
066    *
067    * @return The bean context associated with this context.
068    */
069   public final BeanContext getBeanContext() {
070      return beanContext;
071   }
072
073   /**
074    * Returns the Java method that invoked this parser.
075    * <p>
076    * When using the REST API, this is the Java method invoked by the REST call.
077    * Can be used to access annotations defined on the method or class.
078    *
079    * @return The Java method that invoked this parser.
080   */
081   public final Method getJavaMethod() {
082      return javaMethod;
083   }
084
085   /**
086    * Returns the outer object used for instantiating top-level non-static member classes.
087    * When using the REST API, this is the servlet object.
088    *
089    * @return The outer object.
090   */
091   public final Object getOuter() {
092      return outer;
093   }
094
095   /**
096    * Returns the {@link ParserProperties#PARSER_debug} setting value in this context.
097    *
098    * @return The {@link ParserProperties#PARSER_debug} setting value in this context.
099    */
100   public final boolean isDebug() {
101      return debug;
102   }
103
104   /**
105    * Returns the {@link ParserProperties#PARSER_trimStrings} setting value in this context.
106    *
107    * @return The {@link ParserProperties#PARSER_trimStrings} setting value in this context.
108    */
109   public final boolean isTrimStrings() {
110      return trimStrings;
111   }
112
113   /**
114    * Returns the runtime properties associated with this context.
115    *
116    * @return The runtime properties associated with this context.
117    */
118   public final ObjectMap getProperties() {
119      return properties;
120   }
121
122   /**
123    * Logs a warning message.
124    *
125    * @param msg The warning message.
126    * @param args Optional printf arguments to replace in the error message.
127    */
128   public void addWarning(String msg, Object... args) {
129      logger.warning(msg, args);
130      msg = args.length == 0 ? msg : String.format(msg, args);
131      warnings.add((warnings.size() + 1) + ": " + msg);
132   }
133
134   /**
135    * Trims the specified object if it's a <code>String</code> and {@link #isTrimStrings()} returns <jk>true</jk>.
136    *
137    * @param o The object to trim.
138    * @return The trimmmed string if it's a string.
139    */
140   @SuppressWarnings("unchecked")
141   public final <K> K trim(K o) {
142      if (trimStrings && o instanceof String)
143         return (K)o.toString().trim();
144      return o;
145
146   }
147
148   /**
149    * Trims the specified string if {@link ParserContext#isTrimStrings()} returns <jk>true</jk>.
150    *
151    * @param s The input string to trim.
152    * @return The trimmed string, or <jk>null</jk> if the input was <jk>null</jk>.
153    */
154   public final String trim(String s) {
155      if (trimStrings && s != null)
156         return s.trim();
157      return s;
158   }
159
160   /**
161    * Perform cleanup on this context object if necessary.
162    *
163    * @throws ParseException
164    */
165   public void close() throws ParseException {
166      if (closed)
167         throw new ParseException("Attempt to close ParserContext more than once.");
168      if (debug && warnings.size() > 0)
169         throw new ParseException("Warnings occurred during parsing: \n" + StringUtils.join(warnings, "\n"));
170   }
171
172   @Override /* Object */
173   protected void finalize() throws Throwable {
174      if (! closed)
175         throw new RuntimeException("ParserContext was not closed.");
176   }
177}
