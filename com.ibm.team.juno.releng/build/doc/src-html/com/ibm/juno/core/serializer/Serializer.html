021/**
022 * Parent class for all Juno serializers.
023 *
024 *
025 * <h6 class='topic'>Description</h6>
026 * <p>
027 *    Base serializer class that serves as the parent class for all serializers.
028 * <p>
029 *    Subclasses should extend directly from {@link OutputStreamSerializer} or {@link WriterSerializer}.
030 *
031 *
032 * <h6 class='topic'>@Produces annotation</h6>
033 * <p>
034 *    The media types that this serializer can produce is specified through the {@link Produces @Produces} annotation.
035 * <p>
036 *    However, the media types can also be specified programmatically by overriding the {@link #getMediaTypes()}
037 *       and {@link #getResponseContentType()} methods.
038 *
039 *
040 * <h6 class='topic'>Configurable properties</h6>
041 *    See {@link SerializerProperties} for a list of configurable properties that can be set on this class
042 *    using the {@link #setProperty(String, Object)} method.
043 *
044 * @param <W> The output stream or writer class type.
045 * @author James Bognar (jbognar@us.ibm.com)
046 */
047public abstract class Serializer<W> extends CoreApi {
048
049   /** General serializer properties currently set on this serializer. */
050   protected transient SerializerProperties sp = new SerializerProperties();
051   private String[] mediaTypes;
052   private String contentType;
053
054   // Hidden constructor to force subclass from OuputStreamSerializer or WriterSerializer.
055   Serializer() {}
056
057   /**
058    * Returns <jk>true</jk> if this parser subclasses from {@link WriterSerializer}.
059    *
060    * @return <jk>true</jk> if this parser subclasses from {@link WriterSerializer}.
061    */
062   public abstract boolean isWriterSerializer();
063
064   //--------------------------------------------------------------------------------
065   // Abstract methods
066   //--------------------------------------------------------------------------------
067
068   /**
069    * Serializes a POJO to the specified output stream or writer.
070    * <p>
071    * This method should NOT close the context object.
072    *
073    * @param o The object to serialize.
074    * @param out The writer or output stream to write to.
075    * @param ctx The serializer context object return by {@link #createContext(ObjectMap, Method)}.<br>
076    *    If <jk>null</jk>, context is created using {@link #createContext()}.
077    *
078    * @throws IOException If a problem occurred trying to write to the writer.
079    * @throws SerializeException If a problem occurred trying to convert the output.
080    */
081   protected abstract void doSerialize(Object o, W out, SerializerContext ctx) throws IOException, SerializeException;
082
083   //--------------------------------------------------------------------------------
084   // Other methods
085   //--------------------------------------------------------------------------------
086
087   /**
088    * Calls {@link #serialize(Object, Object, SerializerContext)} but intercepts {@link StackOverflowError} exceptions
089    *    and wraps them in a useful message.
090    * @param o The object to serialize.
091    * @param out The writer or output stream to write to.
092    * @param ctx The serializer context object return by {@link #createContext(ObjectMap, Method)}.<br>
093    *    If <jk>null</jk>, context is created using {@link #createContext()}.
094    *
095    * @throws IOException If a problem occurred trying to write to the writer.
096    * @throws SerializeException If a problem occurred trying to convert the output.
097    */
098   public final void serialize(Object o, W out, SerializerContext ctx) throws IOException, SerializeException {
099      try {
100         doSerialize(o, out, ctx);
101      } catch (StackOverflowError e) {
102         throw new SerializeException("Stack overflow occurred.  This can occur when trying to serialize models containing loops.  It's recommended you use the SerializerProperties.SERIALIZER_detectRecursions setting to help locate the loop.").initCause(e);
103      } finally {
104         ctx.close();
105      }
106   }
107
108   /**
109    * Serializes a POJO to the specified output stream or writer.
110    * <p>
111    * Equivalent to calling <code>serializer.serialize(o, out, <jk>null</jk>);</code>
112    *
113    * @param o The object to serialize.
114    * @param out The writer or output stream to write to.
115    *
116    * @throws IOException If a problem occurred trying to write to the writer.
117    * @throws SerializeException If a problem occurred trying to convert the output.
118    */
119   public final void serialize(Object o, W out) throws IOException, SerializeException {
120      SerializerContext ctx = createContext();
121      serialize(o, out, ctx);
122   }
123
124   /**
125    * Create the context object that will be passed in to the serialize method.
126    * <p>
127    *    It's up to implementers to decide what the context object looks like, although typically
128    *    it's going to be a subclass of {@link SerializerContext}.
129    *
130    * @param properties Optional additional properties.
131    * @param javaMethod Java method that invoked this serializer.
132    *    When using the REST API, this is the Java method invoked by the REST call.
133    *    Can be used to access annotations defined on the method or class.
134    * @return The new context.
135    */
136   public SerializerContext createContext(ObjectMap properties, Method javaMethod) {
137      return new SerializerContext(getBeanContext(), sp, properties, javaMethod);
138   }
139
140   /**
141    * Create a basic context object without overriding properties or specifying <code>javaMethod</code>.
142    * <p>
143    * Equivalent to calling <code>createContext(<jk>null</jk>, <jk>null</jk>)</code>.
144    *
145    * @return The new context.
146    */
147   protected SerializerContext createContext() {
148      return createContext(null, null);
149   }
150
151   /**
152    * Converts the contents of the specified object array to a list.
153    * <p>
154    *    Works on both object and primitive arrays.
155    * <p>
156    *    In the case of multi-dimensional arrays, the outgoing list will
157    *    contain elements of type n-1 dimension.  i.e. if {@code type} is <code><jk>int</jk>[][]</code>
158    *    then {@code list} will have entries of type <code><jk>int</jk>[]</code>.
159    *
160    * @param type The type of array.
161    * @param array The array being converted.
162    * @return The array as a list.
163    */
164   protected final List<Object> toList(Class<?> type, Object array) {
165      Class<?> componentType = type.getComponentType();
166      if (componentType.isPrimitive()) {
167         int l = Array.getLength(array);
168         List<Object> list = new ArrayList<Object>(l);
169         for (int i = 0; i < l; i++)
170            list.add(Array.get(array, i));
171         return list;
172      }
173      return Arrays.asList((Object[])array);
174   }
175
176   /**
177    * Returns the media types handled based on the value of the {@link Produces} annotation on the serializer class.
178    * <p>
179    * This method can be overridden by subclasses to determine the media types programatically.
180    *
181    * @return The list of media types.  Never <jk>null</jk>.
182    */
183   public String[] getMediaTypes() {
184      if (mediaTypes == null) {
185         Produces p = ReflectionUtils.getAnnotation(Produces.class, getClass());
186         if (p == null)
187            throw new RuntimeException(MessageFormat.format("Class ''{0}'' is missing the @Produces annotation", getClass().getName()));
188         mediaTypes = p.value();
189      }
190      return mediaTypes;
191   }
192
193   /**
194    * Optional method that specifies HTTP request headers for this serializer.
195    * <p>
196    *    For example, {@link SoapXmlSerializer} needs to set a <code>SOAPAction</code> header.
197    * <p>
198    *    This method is typically meaningless if the serializer is being used standalone (i.e. outside of a REST server or client).
199    *
200    * @param properties Optional run-time properties (the same that are passed to {@link WriterSerializer#doSerialize(Object, Writer, SerializerContext)}.
201    *    Can be <jk>null</jk>.
202    * @return The HTTP headers to set on HTTP requests.
203    *    Can be <jk>null</jk>.
204    */
205   public ObjectMap getResponseHeaders(ObjectMap properties) {
206      return new ObjectMap(getBeanContext());
207   }
208
209   /**
210    * Optional method that returns the response <code>Content-Type</code> for this serializer if it is different from the matched media type.
211    * <p>
212    *    This method is specified to override the content type for this serializer.
213    *    For example, the {@link com.ibm.juno.core.json.JsonSerializer.Simple} class returns that it handles media type <js>"text/json+simple"</js>, but returns
214    *    <js>"text/json"</js> as the actual content type.
215    *    This allows clients to request specific 'flavors' of content using specialized <code>Accept</code> header values.
216    * <p>
217    *    This method is typically meaningless if the serializer is being used standalone (i.e. outside of a REST server or client).
218    *
219    * @return The response content type.  If <jk>null</jk>, then the matched media type is used.
220    */
221   public String getResponseContentType() {
222      if (contentType == null) {
223         Produces p = getClass().getAnnotation(Produces.class);
224         if (p == null)
225            contentType = "";
226         else {
227            contentType = p.contentType();
228            if (contentType.isEmpty())
229               contentType = p.value()[0];
230         }
231      }
232      return (contentType.isEmpty() ? null : contentType);
233   }
