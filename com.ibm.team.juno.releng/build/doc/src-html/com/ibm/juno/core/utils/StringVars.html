009package com.ibm.juno.core.utils;
010
011/**
012 * Resuable common StringVar resolvers.
013 * <p>
014 *
015 * @author James Bognar (jbognar@us.ibm.com)
016 */
017public class StringVars {
018
019   /**
020    * Reusable string variable for system properties.
021    */
022   public static final StringVar SYSTEM_PROPERTIES_VAR = new StringMapVar(System.getProperties());
023
024   /**
025    * Reusable string variable for environment variables.
026    */
027   public static final StringVar ENV_VARIABLES_VAR = new StringVarWithDefault() {
028      @Override /* StringVar */
029      public String resolve(String varVal) {
030         // Note that lookup is case-insensitive on windows.
031         return System.getenv(varVal);
032      }
033   };
034}
