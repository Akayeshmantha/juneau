009package com.ibm.juno.core.dto.cognos;
010
011import java.util.*;
012
013import com.ibm.juno.core.*;
014import com.ibm.juno.core.annotation.*;
015import com.ibm.juno.core.xml.annotation.*;
016
017/**
018 * Represents a Cognos dataset.
019 * <p>
020 *    When serialized to XML, creates the following construct (example pulled from <code>AddressBookResource</code>):
021 * <p class='bcode'>
022 *    <xt>&lt;?xml</xt> <xa>version</xa>=<xs>'1.0'</xs> <xa>encoding</xa>=<xs>'UTF-8'</xs><xt>?&gt;</xt>
023 *    <xt>&lt;c:dataset <xa>xmlns:c</xa>=<xs>'http://developer.cognos.com/schemas/xmldata/1/'</xs>&gt;</xt>
024 *       <xt>&lt;c:metadata&gt;</xt>
025 *          <xt>&lt;c:item</xt> <xa>name</xa>=<xs>'name'</xs> <xa>type</xa>=<xs>'xs:String'</xs> <xa>length</xa>=<xs>'255'</xs><xt>/&gt;</xt>
026 *          <xt>&lt;c:item</xt> <xa>name</xa>=<xs>'age'</xs> <xa>type</xa>=<xs>'xs:int'</xs><xt>/&gt;</xt>
027 *          <xt>&lt;c:item</xt> <xa>name</xa>=<xs>'numAddresses'</xs> <xa>type</xa>=<xs>'xs:int'</xs><xt>/&gt;</xt>
028 *       <xt>&lt;/c:metadata&gt;</xt>
029 *       <xt>&lt;c:data&gt;</xt>
030 *          <xt>&lt;c:row&gt;</xt>
031 *             <xt>&lt;c:value&gt;</xt>Barack Obama<xt>&lt;/c:value&gt;</xt>
032 *             <xt>&lt;c:value&gt;</xt>52<xt>&lt;/c:value&gt;</xt>
033 *             <xt>&lt;c:value&gt;</xt>2<xt>&lt;/c:value&gt;</xt>
034 *          <xt>&lt;/c:row&gt;</xt>
035 *          <xt>&lt;c:row&gt;</xt>
036 *             <xt>&lt;c:value&gt;</xt>George Walker Bush<xt>&lt;/c:value&gt;</xt>
037 *             <xt>&lt;c:value&gt;</xt>67<xt>&lt;/c:value&gt;</xt>
038 *             <xt>&lt;c:value&gt;</xt>2<xt>&lt;/c:value&gt;</xt>
039 *          <xt>&lt;/c:row&gt;</xt>
040 *       <xt>&lt;/c:data&gt;</xt>
041 *    <xt>&lt;/c:dataset&gt;</xt>
042 * </p>
043 * <p>
044 *    Only 2-dimentional POJOs (arrays or collections of maps or beans) can be serialized to Cognos.
045 *
046 * <h6 class='topic'>Example</h6>
047 * <p>
048 *    The construct shown above is a serialized <code>AddressBook</code> object which is a subclass of <code>LinkedList&lt;Person&gt;</code>.
049 *    The code for generating the XML is as follows...
050 * </p>
051 * <p class='bcode'>
052 *    Column[] items = {
053 *       <jk>new</jk> Column(<js>"name"</js>, <js>"xs:String"</js>, 255),
054 *       <jk>new</jk> Column(<js>"age"</js>, <js>"xs:int"</js>),
055 *       <jk>new</jk> Column(<js>"numAddresses"</js>, <js>"xs:int"</js>)
056 *          .addFilter(
057 *             <jk>new</jk> PojoFilter&ltPerson,Integer&gt;() {
058 *                <ja>@Override</ja>
059 *                <jk>public</jk> Integer filter(Person p) {
060 *                   <jk>return</jk> p.<jf>addresses</jf>.size();
061 *                }
062 *             }
063 *          )
064 *    };
065 *
066 *    DataSet ds = <jk>new</jk> DataSet(items, <jsf>addressBook</jsf>, BeanContext.<jsf>DEFAULT</jsf>);
067 *
068 *    String xml = XmlSerializer.<jsf>DEFAULT_SQ</jsf>.serialize(ds);
069 * </p>
070 *
071 * @author James Bognar (jbognar@us.ibm.com)
072 */
073@Xml(name="dataset")
074@SuppressWarnings("unchecked")
075public class DataSet {
076
077   private Column[] metaData;
078   private List<Row> data;
079
080   /** Bean constructor. */
081   public DataSet() {}
082
083   /**
084    * Constructor.
085    *
086    * @param columns The meta-data that represents the columns in the dataset.
087    * @param o The POJO being serialized to Cognos.
088    *    Must be an array/collection of beans/maps.
089    * @param beanContext The bean context used to convert POJOs to strings.
090    * @throws Exception An error occurred trying to serialize the POJO.
091    */
092   public DataSet(Column[] columns, Object o, BeanContext beanContext) throws Exception {
093      metaData = columns;
094      data = new LinkedList<Row>();
095      if (o != null) {
096         if (o.getClass().isArray())
097            o = Arrays.asList((Object[])o);
098         if (o instanceof Collection) {
099            Collection<?> c = (Collection<?>)o;
100            for (Object o2 : c) {
101               Row r = new Row();
102               Map<?,?> m = null;
103               if (o2 instanceof Map)
104                  m = (Map<?,?>)o2;
105               else
106                  m = beanContext.forBean(o2);
107               for (Column col : columns) {
108                  Object v;
109                  if (col.filter != null)
110                     v = col.filter.filter(o2);
111                  else
112                     v = m.get(col.getName());
113                  r.add(v == null ? null : v.toString());
114               }
115               data.add(r);
116            }
117         }
118      }
119   }
120
121   /**
122    * Represents a row of data.
123    * <p>
124    *    When serialized to XML, creates the following construct (example pulled from <code>AddressBookResource</code>):
125    * <p class='bcode'>
126    *    <xt>&lt;row&gt;</xt>
127    *       <xt>&lt;value&gt;</xt>Barack Obama<xt>&lt;/value&gt;</xt>
128    *       <xt>&lt;value&gt;</xt>52<xt>&lt;/value&gt;</xt>
129    *       <xt>&lt;value&gt;</xt>2<xt>&lt;/value&gt;</xt>
130    *    <xt>&lt;/row&gt;</xt>
131    * </p>
132    *
133    * @author James Bognar (jbognar@us.ibm.com)
134    */
135   @Xml(name="row", childName="value")
136   public static class Row extends LinkedList<String> {
137      private static final long serialVersionUID = 1L;
138   }
139
140
141   //--------------------------------------------------------------------------------
142   // Bean properties
143   //--------------------------------------------------------------------------------
144
145   /**
146    * Bean property getter:  <property>metadata</property>.
147    *
148    * @return The value of the <property>metadata</property> property on this bean, or <jk>null</jk> if it is not set.
149    */
150   @BeanProperty(name="metadata")
151   public Column[] getMetaData() {
152      return metaData;
153   }
154
155   /**
156    * Bean property setter:  <property>metadata</property>.
157    *
158    * @param metaData The new value for the <property>metadata</property> property on this bean.
159    * @return This object (for method chaining).
160    */
161   @BeanProperty(name="metadata")
162   public DataSet setMetaData(Column[] metaData) {
163      this.metaData = metaData;
164      return this;
165   }
166
167   /**
168    * Bean property getter:  <property>data</property>.
169    *
170    * @return The value of the <property>data</property> property on this bean, or <jk>null</jk> if it is not set.
171    */
172   @BeanProperty(name="data")
173   public List<Row> getData() {
174      return data;
175   }
176
177   /**
178    * Bean property setter:  <property>data</property>.
179    *
180    * @param data The new value for the <property>data</property> property on this bean.
181    * @return This object (for method chaining).
182    */
183   @BeanProperty(name="data")
184   public DataSet setData(List<Row> data) {
185      this.data = data;
186      return this;
187   }
188}
