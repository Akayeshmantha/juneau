009package com.ibm.juno.core.jena.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015
016import com.ibm.juno.core.jena.*;
017
018/**
019 * Annotation for specifying options for RDF serializers.
020 * <p>
021 *    Can be applied to Java packages, types, fields, and methods.
022 * <p>
023 *    Can be used for the following:
024 * <ul>
025 *    <li>Override the default behavior of how collections and arrays are serialized.
026 * </ul>
027 *
028 * @author James Bognar (jbognar@us.ibm.com)
029 */
030@Documented
031@Target({PACKAGE,TYPE,FIELD,METHOD})
032@Retention(RUNTIME)
033@Inherited
034public @interface Rdf {
035
036   /**
037    * Sets the XML prefix of this property or class.
038    * <p>
039    *    Must either be matched to a {@link #namespace()} annotation on the same object, parent object, or a {@link RdfNs} with the same name
040    *    through the {@link RdfSchema#rdfNs()} annotation on the package.
041    * </p>
042    */
043   String prefix() default "";
044
045   /**
046    * Sets the namespace URI of this property or class.
047    * <p>
048    *    Must be matched with a {@link #prefix()} annotation on this object, a parent object, or a {@link RdfNs} with the same name
049    *    through the {@link RdfSchema#rdfNs()} annotation on the package.
050    */
051   String namespace() default "";
052
053   /**
054    * The format for how collections (e.g. lists and arrays) are serialized in RDF.
055    * @see RdfCollectionFormat
056    */
057   RdfCollectionFormat collectionFormat() default RdfCollectionFormat.DEFAULT;
058}
