009package com.ibm.juno.core.filter;
010
011import java.util.*;
012
013import com.ibm.juno.core.annotation.*;
014
015/**
016 * Bean filter constructed from a {@link Bean @Bean} annotation found on a class.
017 * <p>
018 * <b>*** Internal class - Not intended for external use ***</b>
019 *
020 * @author James Bognar (jbognar@us.ibm.com)
021 * @param <T> The class type that this filter applies to.
022 */
023public final class AnnotationBeanFilter<T> extends BeanFilter<T> {
024
025   /**
026    * Constructor.
027    *
028    * @param annotatedClass The class found to have a {@link Bean @Bean} annotation.
029    * @param annotations The {@link Bean @Bean} annotations found on the class and all parent classes in child-to-parent order.
030    */
031   public AnnotationBeanFilter(Class<T> annotatedClass, List<Bean> annotations) {
032      super(annotatedClass);
033
034      ListIterator<Bean> li = annotations.listIterator(annotations.size());
035      while (li.hasPrevious()) {
036         Bean b = li.previous();
037
038         if (b.properties().length > 0)
039            setProperties(b.properties());
040
041         if (b.excludeProperties().length > 0)
042            setExcludeProperties(b.excludeProperties());
043
044         setPropertyNamer(b.propertyNamer());
045
046         if (b.interfaceClass() != Object.class)
047            setInterfaceClass(b.interfaceClass());
048
049         if (b.stopClass() != Object.class)
050            setStopClass(b.stopClass());
051
052         if (! b.subTypeProperty().isEmpty()) {
053            setSubTypeProperty(b.subTypeProperty());
054
055            LinkedHashMap<Class<?>,String> subTypes = new LinkedHashMap<Class<?>,String>();
056            for (BeanSubType bst : b.subTypes())
057               subTypes.put(bst.type(), bst.id());
058
059            setSubTypes(subTypes);
060         }
061      }
062   }
063}
