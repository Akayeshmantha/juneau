017/**
018 * Represents a group of {@link Parser Parsers} that can be looked up by media type.
019 *
020 *
021 * <h6 class='topic'>Description</h6>
022 * <p>
023 *    Provides the following features:
024 * <ul>
025 *    <li>Finds parsers based on HTTP <code>Content-Type</code> header values.
026 *    <li>Sets common properties on all parsers in a single method call.
027 *    <li>Locks all parsers in a single method call.
028 *    <li>Clones existing groups and all parsers within the group in a single method call.
029 * </ul>
030 *
031 *
032 * <h6 class='topic'>Match ordering</h6>
033 * <p>
034 *    Parsers are matched against <code>Content-Type</code> strings in the order they exist in this group.
035 * <p>
036 *    Adding new entries will cause the entries to be prepended to the group.
037 *    This allows for previous parsers to be overridden through subsequent calls.
038 * <p>
039 *    For example, calling <code>g.append(P1.<jk>class</jk>,P2.<jk>class</jk>).append(P3.<jk>class</jk>,P4.<jk>class</jk>)</code>
040 *    will result in the order <code>P3, P4, P1, P2</code>.
041 *
042 *
043 * <h6 class='topic'>Examples</h6>
044 * <p class='bcode'>
045 *    <jc>// Construct a new parser group</jc>
046 *    ParserGroup g = <jk>new</jk> ParserGroup();
047 *
048 *    <jc>// Add some parsers to it</jc>
049 *    g.append(JsonParser.<jk>class</jk>, XmlParser.<jk>class</jk>);
050 *
051 *    <jc>// Change settings on parsers simultaneously</jc>
052 *    g.setProperty(BeanContextProperties.<jsf>BEAN_beansRequireSerializable</jsf>, <jk>true</jk>)
053 *       .addFilters(CalendarFilter.ISO8601DT.<jk>class</jk>)
054 *       .lock();
055 *
056 *    <jc>// Find the appropriate parser by Content-Type</jc>
057 *    ReaderParser p = (ReaderParser)g.getParser(<js>"text/json"</js>);
058 *
059 *    <jc>// Parse a bean from JSON</jc>
060 *    String json = <js>"{...}"</js>;
061 *    AddressBook addressBook = p.parse(json, AddressBook.<jk>class</jk>);
062 * </p>
063 *
064 * @author James Bognar (jbognar@us.ibm.com)
065 */
066public final class ParserGroup extends Lockable {
067
068   private transient Map<String,ParserEntry> entryMap = new HashMap<String,ParserEntry>();
069   private transient LinkedList<ParserEntry> tempEntries = new LinkedList<ParserEntry>();
070   private transient ParserEntry[] entries;
071
072
073   /**
074    * Registers the specified REST parsers with this parser group.
075    *
076    * @param p The parsers to append to this group.
077    * @return This object (for method chaining).
078    */
079   public ParserGroup append(Parser<?>...p) {
080      checkLock();
081      entries = null;
082      for (Parser<?> pp : reverse(p))  {
083         ParserEntry e = new ParserEntry(pp);
084         tempEntries.addFirst(e);
085         for (String mediaType : e.mediaTypes)
086            entryMap.put(mediaType, e);
087      }
088      return this;
089   }
090
091   /**
092    * Same as {@link #append(Parser[])}, except specify classes instead of class instances
093    *     of {@link Parser}.
094    * <p>
095    * Note that this can only be used on {@link Parser Parsers} with no-arg constructors.
096    *
097    * @param p The parsers to append to this group.
098    * @return This object (for method chaining).
099    * @throws Exception Thrown if {@link Parser} could not be constructed.
100    */
101   public ParserGroup append(Class<? extends Parser<?>>...p) throws Exception {
102      checkLock();
103      for (Class<? extends Parser<?>> c : reverse(p)) {
104         try {
105         append(c.newInstance());
106         } catch (NoClassDefFoundError e) {
107            // Ignore if dependent library not found (e.g. Jena).
108            System.err.println(e);
109         }
110      }
111      return this;
112   }
113
114   /**
115    * Same as {@link #append(Class[])}, except specify a single class to avoid unchecked compile warnings.
116    *
117    * @param p The parser to append to this group.
118    * @return This object (for method chaining).
119    * @throws Exception Thrown if {@link Parser} could not be constructed.
120    */
121   public ParserGroup append(Class<? extends Parser<?>> p) throws Exception {
122      checkLock();
123      try {
124      append(p.newInstance());
125      } catch (NoClassDefFoundError e) {
126         // Ignore if dependent library not found (e.g. Jena).
127         System.err.println(e);
128      }
129      return this;
130   }
131
132   /**
133    * Returns the parser registered to handle the specified media type.
134    * <p>
135    * The media-type string must not contain any parameters such as <js>";charset=X"</js>.
136    *
137    * @param mediaType The media-type string (e.g. <js>"text/json"</js>).
138    * @return The REST parser that handles the specified request content type, or <jk>null</jk> if
139    *       no parser is registered to handle it.
140    */
141   public Parser<?> getParser(String mediaType) {
142      ParserEntry e = entryMap.get(mediaType);
143      return (e == null ? null : e.parser);
144   }
145
146   /**
147    * Searches the group for a parser that can handle the specified media type.
148    *
149    * @param mediaType The accept string.
150    * @return The media type registered by one of the parsers that matches the <code>mediaType</code> string,
151    *    or <jk>null</jk> if no media types matched.
152    */
153   public String findMatch(String mediaType) {
154      MediaRange[] mr = MediaRange.parse(mediaType);
155      if (mr.length == 0)
156         mr = MediaRange.parse("*/*");
157
158      for (MediaRange a : mr)
159         for (ParserEntry e : getEntries())
160            for (MediaRange a2 : e.mediaRanges)
161               if (a.matches(a2))
162                  return a2.getMediaType();
163
164      return null;
165   }
166
167   /**
168    * Returns the media types that all parsers in this group can handle
169    * <p>
170    * Entries are ordered in the same order as the parsers in the group.
171    *
172    * @return The list of media types.
173    */
174   public List<String> getSupportedMediaTypes() {
175      List<String> l = new ArrayList<String>();
176      for (ParserEntry e : getEntries())
177         for (String mt : e.mediaTypes)
178            if (! l.contains(mt))
179               l.add(mt);
180      return l;
181   }
182
183   private ParserEntry[] getEntries() {
184      if (entries == null)
185         entries = tempEntries.toArray(new ParserEntry[tempEntries.size()]);
186      return entries;
187   }
188
189   static class ParserEntry {
190      Parser<?> parser;
191      MediaRange[] mediaRanges;
192      String[] mediaTypes;
193
194      ParserEntry(Parser<?> p) {
195         parser = p;
196
197         mediaTypes = new String[p.getMediaTypes().length];
198         int i = 0;
199         for (String mt : p.getMediaTypes())
200            mediaTypes[i++] = mt.toLowerCase(Locale.ENGLISH);
201
202         List<MediaRange> l = new LinkedList<MediaRange>();
203         for (i = 0; i < mediaTypes.length; i++)
204            l.addAll(Arrays.asList(MediaRange.parse(mediaTypes[i])));
205         mediaRanges = l.toArray(new MediaRange[l.size()]);
206      }
207   }

211   // Convenience methods for setting properties on all parsers.
212   //--------------------------------------------------------------------------------
213
214   /**
215    * Shortcut for calling {@link Parser#setProperty(String, Object)} on all parsers in this group.
216    *
217    * @param property The property name.
218    * @param value The property value.
219    * @throws LockedException If {@link #lock()} was called on this object.
220    * @return This object (for method chaining).
221    */
222   public ParserGroup setProperty(String property, Object value) throws LockedException {
223      checkLock();
224      for (ParserEntry e : getEntries())
225         e.parser.setProperty(property, value);
226      return this;
227   }
228
229   /**
230    * Shortcut for calling {@link Parser#setProperties(ObjectMap)} on all parsers in this group.
231    *
232    * @param properties The properties to set.  Ignored if <jk>null</jk>.
233    * @throws LockedException If {@link #lock()} was called on this object.
234    * @return This object (for method chaining).
235    */
236   public ParserGroup setProperties(ObjectMap properties) {
237      checkLock();
238      if (properties != null)
239         for (Map.Entry<String,Object> e : properties.entrySet())
240            setProperty(e.getKey(), e.getValue());
241      return this;
242   }

244   /**
245    * Shortcut for calling {@link Parser#addNotBeanClasses(Class[])} on all parsers in this group.
246    *
247    * @param classes The classes to specify as not-beans to the underlying bean context of all parsers in this group.
248    * @throws LockedException If {@link #lock()} was called on this object.
249    * @return This object (for method chaining).
250    */
251   public ParserGroup addNotBeanClasses(Class<?>...classes) throws LockedException {
252      checkLock();
253      for (ParserEntry e : getEntries())
254         e.parser.addNotBeanClasses(classes);
255      return this;
256   }

258   /**
259    * Shortcut for calling {@link Parser#addFilters(Class[])} on all parsers in this group.
260    *
261    * @param classes The classes to add bean filters for to the underlying bean context of all parsers in this group.
262    * @throws LockedException If {@link #lock()} was called on this object.
263    * @return This object (for method chaining).

265   public ParserGroup addFilters(Class<?>...classes) throws LockedException {

270   }

272   /**
273    * Shortcut for calling {@link Parser#addImplClass(Class, Class)} on all parsers in this group.
274    *
275    * @param <T> The interface or abstract class type.
276    * @param interfaceClass The interface or abstract class.
277    * @param implClass The implementation class.
278    * @throws LockedException If {@link #lock()} was called on this object.
279    * @return This object (for method chaining).
280    */
281   public <T> ParserGroup addImplClass(Class<T> interfaceClass, Class<? extends T> implClass) throws LockedException {
282      checkLock();
283      for (ParserEntry e : getEntries())
284         e.parser.addImplClass(interfaceClass, implClass);
285      return this;
286   }

292   /**
293    * Locks this group and all parsers in this group.
294    */
295   @Override /* Lockable */
296   public ParserGroup lock() {
297      super.lock();
298      for (ParserEntry e : getEntries())
299         e.parser.lock();
300      return this;
301   }
302
303   /**
304    * Clones this group and all parsers in this group.
305    */
306   @Override /* Lockable */
307   public ParserGroup clone() throws CloneNotSupportedException {
308      ParserGroup c = (ParserGroup)super.clone();
309      c.entryMap = new HashMap<String,ParserEntry>();
310      c.tempEntries = new LinkedList<ParserEntry>();
311      c.entries = null;
312      ParserEntry[] e = getEntries();
313      for (int i = e.length-1; i >= 0; i--)
314         c.append(e[i].parser.clone());
315      return c;
316   }
