009package com.ibm.juno.core.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015
016import com.ibm.juno.core.filter.*;
017import com.ibm.juno.core.serializer.*;
018
019/**
020 * Annotation that can be applied to a class to associate a filter with it.
021 * <p>
022 *    Typically used to associate {@link PojoFilter PojoFilters} with classes using annotations
023 *       instead of programatically through a method such as {@link Serializer#addFilters(Class...)}.
024 *
025 * <h6 class='topic'>Example</h6>
026 * <p>
027 *    In this case, a filter is being applied to a bean that will force it to be serialized as a <code>String</code>
028 * <p class='bcode'>
029 *    <jc>// Our bean class</jc>
030 *    <ja>@Filter</ja>(BFilter.<jk>class</jk>)
031 *    <jk>public class</jk> B {
032 *       <jk>public</jk> String <jf>f1</jf>;
033 *    }
034 *
035 *    <jc>// Our filter to force the bean to be serialized as a String</jc>
036 *    <jk>public class</jk> BFilter <jk>extends</jk> PojoFilter&lt;B,String&gt; {
037 *       <jk>public</jk> String filter(B o) <jk>throws</jk> SerializeException {
038 *          <jk>return</jk> o.f1;
039 *       }
040 *       <jk>public</jk> B unfilter(String f, ClassMeta&lt;?&gt; hint) <jk>throws</jk> ParseException {
041 *          B b1 = <jk>new</jk> B();
042 *          b1.<jf>f1</jf> = f;
043 *          <jk>return</jk> b1;
044 *       }
045 *    }
046 *
047 *    <jk>public void</jk> testFilter() <jk>throws</jk> Exception {
048 *       WriterSerializer s = JsonSerializer.<jsf>DEFAULT</jsf>;
049 *       B b = <jk>new</jk> B();
050 *       b.<jf>f1</jf> = <js>"bar"</js>;
051 *       String json = s.serialize(b);
052 *       <jsm>assertEquals</jsm>(<js>"'bar'"</js>, json);
053 *
054 *       ReaderParser p = JsonParser.<jsf>DEFAULT</jsf>;
055 *       b = p.parse(json, B.<jk>class</jk>);
056 *       <jsm>assertEquals</jsm>(<js>"bar"</js>, t.<jf>f1</jf>);
057 *    }
058 * </p>
059 * <p>
060 *    Note that using this annotation is functionally equivalent to adding filters to the serializers and parsers:
061 * <p class='bcode'>
062 *    WriterSerializer s = <jk>new</jk> JsonSerializer.addFilters(BFilter.<jk>class</jk>);
063 *    ReaderParser p = <jk>new</jk> JsonParser.addFilters(BFilter.<jk>class</jk>);
064 * </p>
065 * <p>
066 *    It is technically possible to associate a {@link BeanFilter} with a bean class using this annotation.
067 *    However in practice, it's almost always less code to simply use the {@link Bean @Bean} annotation.
068 * </p>
069 *
070 * @author James Bognar (jbognar@us.ibm.com)
071 */
072@Documented
073@Target(TYPE)
074@Retention(RUNTIME)
075@Inherited
076public @interface Filter {
077
078   /**
079    * The filter class.
080    */
081   Class<? extends com.ibm.juno.core.filter.Filter> value() default com.ibm.juno.core.filter.Filter.NULL.class;
082}
