009package com.ibm.juno.core.jso;
010
011import java.io.*;
012
013import com.ibm.juno.core.*;
014import com.ibm.juno.core.annotation.*;
015import com.ibm.juno.core.parser.*;
016
017/**
018 * Parses POJOs from HTTP responses as Java {@link ObjectInputStream ObjectInputStreams}.
019 *
020 *
021 * <h6 class='topic'>Media types</h6>
022 * <p>
023 *    Consumes <code>Content-Type</code> types: <code>application/x-java-serialized-object</code>
024 *
025 *
026 * @author James Bognar (jbognar@us.ibm.com)
027 */
028@Consumes("application/x-java-serialized-object")
029public final class JavaSerializedObjectParser extends InputStreamParser {
030
031   //--------------------------------------------------------------------------------
032   // Overridden methods
033   //--------------------------------------------------------------------------------
034
035   @SuppressWarnings("unchecked")
036   @Override /* InputStreamParser */
037   protected <T> T doParse(InputStream in, int estimatedSize, ClassMeta<T> type, ParserContext ctx) throws ParseException, IOException {
038      try {
039         ObjectInputStream ois = new ObjectInputStream(in);
040         return (T)ois.readObject();
041      } catch (ClassNotFoundException e) {
042         throw new ParseException(e);
043      }
044   }
045
046
047   @Override /* Lockable */
048   public JavaSerializedObjectParser clone() {
049      try {
050         return (JavaSerializedObjectParser)super.clone();
051      } catch (CloneNotSupportedException e) {
052         throw new RuntimeException(e); // Shouldn't happen
053      }
054   }
055}
