009package com.ibm.juno.core.urlencoding;
010
011import com.ibm.juno.core.*;
012import com.ibm.juno.core.serializer.*;
013
014/**
015 * Configurable properties on the {@link UonSerializer} and {@link UrlEncodingSerializer} classes.
016 * <p>
017 *    Use the {@link UonSerializer#setProperty(String, Object)} method to set property values.
018 * <p>
019 *    In addition to these properties, the following properties are also applicable for {@link UonSerializer}.
020 * <ul>
021 *    <li>{@link SerializerProperties}
022 *    <li>{@link BeanContextProperties}
023 * </ul>
024 *
025 * @author James Bognar (jbognar@us.ibm.com)
026 */
027public final class UonSerializerProperties implements Cloneable {
028
029   /**
030    * Use simplified output ({@link Boolean}, default=<jk>false</jk>).
031    * <p>
032    * If <jk>true</jk>, type flags will not be prepended to values in most cases.
033    * <p>
034    * Use this setting if the data types of the values (e.g. object/array/boolean/number/string)
035    *    is known on the receiving end.
036    * <p>
037    * It should be noted that the default behavior produces a data structure that can
038    *    be losslessly converted into JSON, and any JSON can be losslessly represented
039    *    in a URL-encoded value.  However, this strict equivalency does not exist
040    *    when simple mode is used.
041    * <p>
042    * <table class='styled'>
043    *    <tr>
044    *       <th>Input (in JSON)</th>
045    *       <th>Normal mode output</th>
046    *       <th>Simple mode output</th>
047    *    </tr>
048    *    <tr>
049    *       <td class='code'>{foo:'bar',baz:'bing'}</td>
050    *       <td class='code'>$o(foo=bar,baz=bing)</td>
051    *       <td class='code'>(foo=bar,baz=bing)</td>
052    *    </tr>
053    *    <tr>
054    *       <td class='code'>{foo:{bar:'baz'}}</td>
055    *       <td class='code'>$o(foo=$o(bar=baz))</td>
056    *       <td class='code'>(foo=(bar=baz))</td>
057    *    </tr>
058    *    <tr>
059    *       <td class='code'>['foo','bar']</td>
060    *       <td class='code'>$a(foo,bar)</td>
061    *       <td class='code'>(foo,bar)</td>
062    *    </tr>
063    *    <tr>
064    *       <td class='code'>['foo',['bar','baz']]</td>
065    *       <td class='code'>$a(foo,$a(bar,baz))</td>
066    *       <td class='code'>(foo,(bar,baz))</td>
067    *    </tr>
068    *    <tr>
069    *       <td class='code'>true</td>
070    *       <td class='code'>$b(true)</td>
071    *       <td class='code'>true</td>
072    *    </tr>
073    *    <tr>
074    *       <td class='code'>123</td>
075    *       <td class='code'>$n(123)</td>
076    *       <td class='code'>123</td>
077    *    </tr>
078    * </table>
079    */
080   public static final String UON_simpleMode = "UonSerializer.simpleMode";
081
082   /**
083    * Use whitespace in output ({@link Boolean}, default=<jk>false</jk>).
084    * <p>
085    * If <jk>true</jk>, whitespace is added to the output to improve readability.
086    */
087   public static final String UON_useWhitespace = "UonSerializer.useWhitespace";
088
089   /**
090    * Encode non-valid URI characters to <js>"%xx"</js> constructs. ({@link Boolean}, default=<jk>false</jk> for {@link UonSerializer}, <jk>true</jk> for {@link UrlEncodingSerializer}).
091    * <p>
092    * If <jk>true</jk>, non-valid URI characters will be converted to <js>"%xx"</js> sequences.
093    * Set to <jk>false</jk> if parameter value is being passed to some other code that will already
094    *    perform URL-encoding of non-valid URI characters.
095    */
096   public static final String UON_encodeChars = "UonSerializer.encodeChars";
097
098   boolean
099      simpleMode = false,
100      useWhitespace = false,
101      encodeChars = false;
102
103   /**
104    * Sets the specified property value.
105    * @param property The property name.
106    * @param value The property value.
107    * @return <jk>true</jk> if property name was valid and property was set.
108    */
109   public boolean setProperty(String property, Object value) {
110      BeanContext bc = BeanContext.DEFAULT;
111      if (property.equals(UON_simpleMode))
112         simpleMode = bc.convertToType(value, Boolean.class);
113      else if (property.equals(UON_useWhitespace))
114         useWhitespace = bc.convertToType(value, Boolean.class);
115      else if (property.equals(UON_encodeChars))
116         encodeChars = bc.convertToType(value, Boolean.class);
117      else
118         return false;
119      return true;
120   }
121
122
123   //--------------------------------------------------------------------------------
124   // Overridden methods
125   //--------------------------------------------------------------------------------
126
127   @Override /* Cloneable */
128   public UonSerializerProperties clone() {
129      try {
130         return (UonSerializerProperties)super.clone();
131      } catch (CloneNotSupportedException e) {
132         throw new RuntimeException(e); // Shouldn't happen
133      }
134   }
135}
