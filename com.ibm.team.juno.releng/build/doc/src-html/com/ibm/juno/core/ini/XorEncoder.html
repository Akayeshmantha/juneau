009package com.ibm.juno.core.ini;
010
011import com.ibm.juno.core.utils.*;
012
013/**
014 * Simply XOR+Base64 encoder for obscuring passwords and other sensitive data in INI config files.
015 * <p>
016 * This is not intended to be used as strong encryption.
017 *
018 * @author James Bognar (jbognar@us.ibm.com)
019 */
020public final class XorEncoder implements Encoder {
021
022   /** Reusable XOR-Encoder instance. */
023   public static final XorEncoder INSTANCE = new XorEncoder();
024
025   private static final String key = System.getProperty("com.ibm.juno.core.ini.XorEncoder.key", "nuy7og796Vh6G9O6bG230SHK0cc8QYkH");   // The super-duper-secret key
026
027   @Override /* Encoder */
028   public String encode(String fieldName, String in) {
029      byte[] b = in.getBytes(IOUtils.UTF8);
030      for (int i = 0; i < b.length; i++) {
031            int j = i % key.length();
032         b[i] = (byte)(b[i] ^ key.charAt(j));
033      }
034      return StringUtils.base64Encode(b);
035   }
036
037   @Override /* Encoder */
038   public String decode(String fieldName, String in) {
039      byte[] b = StringUtils.base64Decode(in);
040      for (int i = 0; i < b.length; i++) {
041         int j = i % key.length();
042         b[i] = (byte)(b[i] ^ key.charAt(j));
043   }
044      return new String(b, IOUtils.UTF8);
045   }
046}
