009package com.ibm.juno.core.utils;
010
011/**
012 * Interface for the resolution of string variables using the {@link StringVarResolver} API.
013 *
014 * @author jbognar
015 */
016public abstract class StringVar {
017
018   /**
019    * The method called from {@link StringVarResolver}.
020    * Can be overridden to intercept the request and do special handling.
021    * Default implementation simply calls resolve(String).
022    *
023    * @param arg The inside argument of the variable.
024    * @return The resolved value.
025    */
026   protected String doResolve(String arg) {
027      return resolve(arg);
028   }
029
030   /**
031    * The interface that needs to be implemented for string vars.
032    *
033    * @param arg The inside argument of the variable.
034    * @return The resolved value.
035    */
036   public abstract String resolve(String arg);
037}
