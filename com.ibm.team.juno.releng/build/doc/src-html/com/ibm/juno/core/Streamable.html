009package com.ibm.juno.core;
010
011import java.io.*;
012
013/**
014 * Interface that identifies that an object can be serialized directly to an output stream.
015 * <p>
016 *    Instances must identify the media type of the content by implementing the
017 *    {@link #getMediaType()} method.
018 * </p>
019 *
020 * @author James Bognar (jbognar@us.ibm.com)
021 */
022public interface Streamable {
023
024   /**
025    * Serialize this object to the specified output stream.
026    *
027    * @param os The output stream to stream to.
028    * @throws IOException
029    */
030   void streamTo(OutputStream os) throws IOException;
031
032   /**
033    * Returns the serialized media type for this resource (e.g. <js>"text/html"</js>).
034    *
035    * @return The media type, or <jk>null</jk> if the media type is not known.
036    */
037   String getMediaType();
038}
