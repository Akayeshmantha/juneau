009package com.ibm.juno.core.xml;
010
011import static com.ibm.juno.core.xml.NamespaceFactory.*;
012import static com.ibm.juno.core.xml.XmlSerializerProperties.*;
013
014import java.io.*;
015import java.lang.reflect.*;
016
017import com.ibm.juno.core.*;
018import com.ibm.juno.core.serializer.*;
019import com.ibm.juno.core.utils.*;
020
021/**
022 * Context object that lives for the duration of a single serialization of the {@link XmlSerializer}.
023 * <p>
024 *    See {@link SerializerContext} for details.
025 * </p>
026 * <p>
027 *    This class is NOT thread safe.  It is meant to be discarded after one-time use.
028 * </p>
029 *
030 * @author James Bognar (jbognar@us.ibm.com)
031 */
032@SuppressWarnings("hiding")
033public class XmlSerializerContext extends SerializerContext {
034
035   private final boolean
036      addJsonTypeAttrs,
037      addJsonStringTypeAttrs,
038      autoDetectNamespaces,
039      enableNamespaces,
040      addNamespaceUrlsToRoot;
041
042   private Namespace
043      defaultNamespace;
044   private final Namespace
045      xsiNamespace,
046      xsNamespace;
047
048   private Namespace[] namespaces = new Namespace[0];
049
050   /**
051    * Constructor.
052    * @param beanContext The bean context being used by the serializer.
053    * @param sp Default general serializer properties.
054    * @param xsp Default XML serializer properties.
055    * @param op Override properties.
056    * @param javaMethod Java method that invoked this serializer.
057    *    When using the REST API, this is the Java method invoked by the REST call.
058    *    Can be used to access annotations defined on the method or class.
059    */
060   public XmlSerializerContext(BeanContext beanContext, SerializerProperties sp, XmlSerializerProperties xsp, ObjectMap op, Method javaMethod) {
061      super(beanContext, sp, op, javaMethod);
062      if (op == null || op.isEmpty()) {
063         addJsonTypeAttrs = xsp.addJsonTypeAttrs;
064         addJsonStringTypeAttrs = xsp.addJsonStringTypeAttrs;
065         enableNamespaces = xsp.enableNamespaces;
066         autoDetectNamespaces = xsp.autoDetectNamespaces;
067         addNamespaceUrlsToRoot = xsp.addNamespaceUrlsToRoot;
068         addNamespaces(xsp.namespaces);
069         defaultNamespace = findDefaultNamespace(xsp.defaultNamespace);
070         xsiNamespace = xsp.xsiNamespace;
071         xsNamespace = xsp.xsNamespace;
072      } else {
073         addJsonTypeAttrs = op.getBoolean(XML_addJsonTypeAttrs, xsp.addJsonTypeAttrs);
074         addJsonStringTypeAttrs = op.getBoolean(XML_addJsonStringTypeAttrs, xsp.addJsonStringTypeAttrs);
075         enableNamespaces = op.getBoolean(XML_enableNamespaces, xsp.enableNamespaces);
076         autoDetectNamespaces = op.getBoolean(XML_autoDetectNamespaces, xsp.autoDetectNamespaces);
077         addNamespaceUrlsToRoot = op.getBoolean(XML_addNamespaceUrisToRoot, xsp.addNamespaceUrlsToRoot);
078         namespaces = (op.containsKey(XML_namespaces) ? parseNamespaces(op.get(XML_namespaces)) : xsp.namespaces);
079         defaultNamespace = findDefaultNamespace(op.containsKey(XML_defaultNamespaceUri) ? op.getString(XML_defaultNamespaceUri) : xsp.defaultNamespace);
080         xsiNamespace = (op.containsKey(XML_xsiNamespace) ? parseNamespace(op.get(XML_xsiNamespace)) : xsp.xsiNamespace);
081         xsNamespace = (op.containsKey(XML_xsNamespace) ? parseNamespace(op.get(XML_xsNamespace)) : xsp.xsNamespace);
082      }
083   }
084
085   private Namespace findDefaultNamespace(String s) {
086      if (s == null)
087         return null;
088      if (StringUtils.startsWith(s, '{'))
089         return parseNamespace(s);
090      if (! s.startsWith("http://"))
091         return get(s, "http://unknown");
092      return get(null, s);
093   }
094
095   private void addNamespaces(Namespace...namespaces) {
096      for (Namespace ns : namespaces)
097         addNamespace(ns);
098   }
099
100   void addNamespace(Namespace ns) {
101      if (ns == defaultNamespace)
102         return;
103
104      for (Namespace n : namespaces)
105         if (n == ns)
106            return;
107
108      if (defaultNamespace != null && (ns.uri.equals(defaultNamespace.uri) || ns.name.equals(defaultNamespace.name)))
109         defaultNamespace = ns;
110      else
111         namespaces = ArrayUtils.append(namespaces, ns);
112   }
113
114   /**
115    * Returns the list of namespaces being used in the current XML serialization.
116    *
117    * @return The list of namespaces being used in the current XML serialization.
118    */
119   public Namespace[] getNamespaces() {
120      return namespaces;
121   }
122
123   /**
124    * Returns the {@link XmlSerializerProperties#XML_addJsonTypeAttrs} setting value in this context.
125    *
126    * @return The {@link XmlSerializerProperties#XML_addJsonTypeAttrs} setting value in this context.
127    */
128   public final boolean isAddJsonTypeAttrs() {
129      return addJsonTypeAttrs;
130   }
131
132   /**
133    * Returns the {@link XmlSerializerProperties#XML_addJsonStringTypeAttrs} setting value in this context.
134    *
135    * @return The {@link XmlSerializerProperties#XML_addJsonStringTypeAttrs} setting value in this context.
136    */
137   public final boolean isAddJsonStringTypeAttrs() {
138      return addJsonStringTypeAttrs;
139   }
140
141   /**
142    * Returns the {@link XmlSerializerProperties#XML_autoDetectNamespaces} setting value in this context.
143    *
144    * @return The {@link XmlSerializerProperties#XML_autoDetectNamespaces} setting value in this context.
145    */
146   public final boolean isAutoDetectNamespaces() {
147      return enableNamespaces && autoDetectNamespaces;
148   }
149
150   /**
151    * Returns the {@link XmlSerializerProperties#XML_enableNamespaces} setting value in this context.
152    *
153    * @return The {@link XmlSerializerProperties#XML_enableNamespaces} setting value in this context.
154    */
155   public final boolean isEnableNamespaces() {
156      return enableNamespaces;
157   }
158
159   /**
160    * Returns the {@link XmlSerializerProperties#XML_addNamespaceUrisToRoot} setting value in this context.
161    *
162    * @return The {@link XmlSerializerProperties#XML_addNamespaceUrisToRoot} setting value in this context.
163    */
164   public final boolean isAddNamespaceUrlsToRoot() {
165      return addNamespaceUrlsToRoot;
166   }
167
168   /**
169    * Returns the {@link XmlSerializerProperties#XML_defaultNamespaceUri} setting value in this context.
170    *
171    * @return The {@link XmlSerializerProperties#XML_defaultNamespaceUri} setting value in this context.
172    */
173   public final Namespace getDefaultNamespace() {
174      return defaultNamespace;
175   }
176
177   /**
178    * Returns the {@link XmlSerializerProperties#XML_xsiNamespace} setting value in this context.
179    *
180    * @return The {@link XmlSerializerProperties#XML_xsiNamespace} setting value in this context.
181    */
182   public final Namespace getXsiNamespace() {
183      return xsiNamespace;
184   }
185
186   /**
187    * Returns the {@link XmlSerializerProperties#XML_xsNamespace} setting value in this context.
188    *
189    * @return The {@link XmlSerializerProperties#XML_xsNamespace} setting value in this context.
190    */
191   public final Namespace getXsNamespace() {
192      return xsNamespace;
193   }
194
195   /**
196    * Wraps the specified writer in a {@link XmlSerializerWriter} if it is not already an instance of that class.
197    *
198    * @param out The writer being wrapped.
199    * @return The wrapped writer.
200    */
201   public XmlSerializerWriter getWriter(Writer out) {
202      if (out instanceof XmlSerializerWriter)
203         return (XmlSerializerWriter)out;
204      return new XmlSerializerWriter(out, isUseIndentation(), isTrimStrings(), getQuoteChar(), getRelativeUriBase(), getAbsolutePathUriBase(), isEnableNamespaces(), getDefaultNamespace());
205   }
206}
