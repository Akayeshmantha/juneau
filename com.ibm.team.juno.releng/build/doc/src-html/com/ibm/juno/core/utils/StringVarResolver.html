017/**
018 * Utility class for resolving  variables of the form <code>$X{key}</code> in strings.
019 * <p>
020 * This class implements the following two methods for resolving variables in strings:
021 * <ul>
022 *    <li>{@link #resolve(String)} - Returns a new string with variables replaced.
023 *    <li>{@link #writeTo(String,Writer)} - Resolves variables in the string and sends the result to the writer.
024 * </ul>
025 * <p>
026 * Variables are of the form <code>$X{key}</code>, where <code>X</code> can consist of zero or more ASCII characters.<br>
027 * The variable key can contain anything, even nested variables that get recursively resolved.
028 * <p>
029 * Variable types are defined through the {@link #addVar(String, StringVar)} method.
030 * <p>
031 * The {@link StringVar} interface defines a simple method for replacing a variable key with a value.
032 * <p>
033 * <h6 class='topic'>Example:</h6>
034 * <p class='bcode'>
035 *    <jc>// Create a variable resolver that resolves system properties (e.g. "$S{java.home}")</jc>
036 *    StringVarResolver r = <jk>new</jk> StringVarResolver()
037 *       .addVar(<js>"S"</js>, <jk>new</jk> StringVar() {
038 *          <ja>@Override</ja>
039 *          <jk>public</jk> String resolve(String varVal) {
040 *             <jk>return</jk> System.<jsm>getProperty</jsm>(varVal);
041 *          }
042 *       });
043 *
044 *    System.<jsf>out</jsf>.println(r.resolve(<js>"java.home is set to $S{java.home}"</js>));
045 * </p>
046 * <p>
047 * Subclasses of {@link StringVar} are provided for special purposes:
048 * <ul>
049 *    <li>{@link StringVarMultipart} - Interface for the resolution of vars that consist of a comma-delimited list (e.g. <js>"$X{foo, bar, baz}"</js>)
050 *    <li>{@link StringVarWithDefault} - Interface for the resolution of vars with a default value if the <code>resolve(String)</code> method returns <jk>null</jk> (e.g. <js>"$S{myProperty,not found}"</js>).
051 * </ul>
052 * <p>
053 * The {@link #DEFAULT} instance is a reusable variable resolver that includes support for system properties and environment variables.
054 * <p>
055 * <code>StringVarResolvers</code> can be extended by using the {@link #StringVarResolver(StringVarResolver)} constructor.
056 * <p>
057 * <h6 class='topic'>Example:</h6>
058 * <p class='bcode'>
059 *    <jc>// Create a var resolver that extends the default resolver and appends our own "$URLEncode{...}" variable</jc>
060 *    StringVarResolver r = <jk>new</jk> StringVarResolver(StringVarResolver.<jsf>DEFAULT</jsf>)
061 *       .addVar(<js>"URLEncode"</js>, <jk>new</jk> StringVar() {
062 *          <ja>@Override</ja>
063 *          <jk>public</jk> String resolve(String varVal) {
064 *             <jk>return</jk> URLEncoder.<jsm>encode</jsm>(varVal, <js>"UTF-8"</js>);
065 *          }
066 *       });
067 *
068 *    <jc>// Retrieve a system property and URL-encode it if necessary.</jc>
069 *    String myProperty = r.resolve(<js>"$URLEncode{$S{my.property}}"</js>);
070 * <p>
071 * Variables can be nested arbitrarily deep.
072 * <p>
073 * <h6 class='topic'>Example:</h6>
074 * <p class='bcode'>
075 *    <jc>// Look up a property in the following order:
076 *    // 1) MYPROPERTY environment variable.
077 *    // 2) 'my.property' system property if environment variable not found.
078 *    // 3) 'not found' string if system property not found.</jc>
079 *    String myproperty = StringVarResolver.<jsf>DEFAULT</jsf>.resolve(<js>"$E{MYPROPERTY,$S{my.property,not found}}"</js>);
080 * </p>
081 * <p>
082 * Resolved variables can also contain variables.
083 * <p class='bcode'>
084 *    <jc>// If MYPROPERTY is "$S{my.property}", and the system property "my.property" is "foo",
085 *    // then the following resolves to "foo".</jc>
086 *    String myproperty = StringVarResolver.<jsf>DEFAULT</jsf>.resolve(<js>"$E{MYPROPERTY}"</js>);
087 * </p>
088 * <p>
089 * <h6 class='topic'>Other notes:</h6>
090 * <ul class='spaced-list'>
091 *    <li>The escape character <js>'\'</js> can be used when necessary to escape the following characters: <code>$ , { }</code>
092 *    <li><b>WARNING:</b>  It is possible to cause {@link StackOverflowError StackOverflowErrors} if your nested variables result in
093 *       a recursive loop (e.g. the environment variable <code>'MYPROPERTY'</code> has the value <code>'$E{MYPROPERTY}'</code>).
094 *       So don't do that!
095 *    <li>As a general rule, this class tries to be as efficient as possible by not creating new strings when not needed.<br>
096 *       For example, calling the resolve method on a string that doesn't contain variables (e.g. <code>resolver.resolve(<js>"foobar"</js>)</code>)
097 *       will simply be a no-op and return the same string.
098 *
099 *
100 * @author James Bognar (jbognar@us.ibm.com)
101 */
102public class StringVarResolver {
103
104   /**
105    * Default string variable resolver with support for system properties and environment variables:
106    * <p>
107    * <ul>
108    *    <li><code>$S{key}</code>,<code>$S{key,default}</code> - System properties.
109    *    <li><code>$E{key}</code>,<code>$E{key,default}</code> - Environment variables.
110    * </ul>
111    */
112   public static final StringVarResolver DEFAULT = new StringVarResolver()
113      .addVar("S", StringVars.SYSTEM_PROPERTIES_VAR)
114      .addVar("E", StringVars.ENV_VARIABLES_VAR);
115
116   // Map of Vars added through addVar() method.
117   private Map<String,StringVar> varMap = new TreeMap<String,StringVar>();
118
119   private StringVarResolver parent;
120
121   /**
122    * Construct an empty var resolver.
123    */
124   public StringVarResolver() {}
125
126   /**
127    * Construct an empty var resolver with the specified parent.
128    *
129    * @param parent The parent string variable resolver.  Can be <jk>null</jk>.
130    */
131   public StringVarResolver(StringVarResolver parent) {
132      this.parent = parent;
133   }
134
135   /**
136    * Adds a parent var resolver to this var resolver.
137    *
138    * @param parent - The new parent object.
139    * @param replaceExisting - If <jk>true</jk>, then the existing parent (if there is one)
140    *    will be replaced with the new one.  Otherwise, we'll recursively call this method
141    *    to set it as the parent on the current top-level parent.
142    * @return This object (for method chaining).
143    */
144   public StringVarResolver setParent(StringVarResolver parent, boolean replaceExisting) {
145      if (this.parent == null || replaceExisting)
146         this.parent = parent;
147      else
148         this.parent.setParent(parent, false);
149      return this;
150   }
151
152   /**
153    * Register a new variable with this resolver.
154    *
155    * @param varName The variable name (e.g. <js>"X"</js> resolves <js>"$X{...}"</js> variables).
156    * @param v The variable resolver.
157    * @return This object (for method chaining).
158    */
159   public StringVarResolver addVar(String varName, StringVar v) {
160      assertFieldNotNull(v, "v");
161
162      // Need to make sure only ASCII characters are used.
163      for (int i = 0; i < varName.length(); i++) {
164         char c = varName.charAt(i);
165         if (c < 'A' || c > 'z' || (c > 'Z' && c < 'a'))
166            illegalArg("Invalid var name.  Must consist of only uppercase and lowercase ASCII letters.");
167      }
168      varMap.put(varName, v);
169      return this;
170   }
171
172   /**
173    * Resolve all variables in the specified string.
174    *
175    * @param s The string to resolve variables in.
176    * @return The new string with all variables resolved, or the same string if no variables were found.
177    *    Null input results in a blank string.
178    */
179   public String resolve(String s) {
180
181      if (s == null)
182         return "";
183      if (s.indexOf('$') == -1 && s.indexOf('\\') == -1)
184         return s;
185
186      // Special case where value consists of a single variable with no embedded variables (e.g. "$X{...}").
187      // This is a common case, so we want an optimized solution that doesn't involve string builders.
188      if (isSimpleVar(s)) {
189         String var = s.substring(1, s.indexOf('{'));
190         String val = s.substring(s.indexOf('{')+1, s.length()-1);
191         StringVar v = getVar(var);
192         if (v != null) {
193            s = v.doResolve(val);
194            return resolve(s);
195         }
196         return s;
197      }
198
199      try {
200         return writeTo(s, new StringWriter()).toString();
201      } catch (IOException e) {
202         throw new RuntimeException(e); // Never happens.
203      }
204   }
205
206   /**
207    * Checks to see if string is of the simple form "$X{...}" with no embedded variables.
208    * This is a common case, and we can avoid using StringWriters.
209    */
210   private boolean isSimpleVar(String s) {
211      int S1 = 1;    // Not in variable, looking for $
212      int S2 = 2;    // Found $, Looking for {
213      int S3 = 3;    // Found {, Looking for }
214      int S4 = 4;    // Found }
215
216      int length = s.length();
217      int state = S1;
218      for (int i = 0; i < length; i++) {
219         char c = s.charAt(i);
220         if (state == S1) {
221            if (c == '$') {
222               state = S2;
223            } else {
224               return false;
225            }
226         } else if (state == S2) {
227            if (c == '{') {
228               state = S3;
229            } else if (c < 'A' || c > 'z' || (c > 'Z' && c < 'a')) {   // False trigger "$X "
230               return false;
231            }
232         } else if (state == S3) {
233            if (c == '}')
234               state = S4;
235            else if (c == '{' || c == '$')
236               return false;
237         } else if (state == S4) {
238            return false;
239         }
240      }
241      return state == S4;
242   }
243
244   /**
245    * Resolves variables in the specified string and sends the output to the specified writer.
246    * More efficient than first parsing to a string and then serializing to the writer since this
247    * method doesn't need to construct a large string.
248    *
249    * @param s The string to resolve variables in.
250    * @param out The writer to write to.
251    * @return The same writer.
252    * @throws IOException
253    */
254   public Writer writeTo(String s, Writer out) throws IOException {
255
256      int S1 = 1;    // Not in variable, looking for $
257      int S2 = 2;    // Found $, Looking for {
258      int S3 = 3;    // Found {, Looking for }
259
260      int state = S1;
261      boolean isInEscape = false;
262      boolean hasInternalVar = false;
263      boolean hasInnerEscapes = false;
264      String varType = null;
265      String varVal = null;
266      int x = 0, x2 = 0;
267      int depth = 0;
268      int length = s.length();
269      for (int i = 0; i < length; i++) {
270         char c = s.charAt(i);
271         if (state == S1) {
272            if (isInEscape) {
273               if (c == '\\' || c == '$') {
274                  out.append(c);
275               } else {
276                  out.append('\\').append(c);
277               }
278               isInEscape = false;
279            } else if (c == '\\') {
280               isInEscape = true;
281            } else if (c == '$') {
282               x = i;
283               x2 = i;
284               state = S2;
285            } else {
286               out.append(c);
287            }
288         } else if (state == S2) {
289            if (isInEscape) {
290               isInEscape = false;
291            } else if (c == '\\') {
292               hasInnerEscapes = true;
293               isInEscape = true;
294            } else if (c == '{') {
295               varType = s.substring(x+1, i);
296               x = i;
297               state = S3;
298            } else if (c < 'A' || c > 'z' || (c > 'Z' && c < 'a')) {  // False trigger "$X "
299               if (hasInnerEscapes)
300                  out.append(unEscapeChars(s.substring(x, i+1), new char[]{'\\','{'}));
301               else
302                  out.append(s, x, i+1);
303               x = i + 1;
304               state = S1;
305               hasInnerEscapes = false;
306            }
307         } else if (state == S3) {
308            if (isInEscape) {
309               isInEscape = false;
310            } else if (c == '\\') {
311               isInEscape = true;
312               hasInnerEscapes = true;
313            } else if (c == '{') {
314               depth++;
315               hasInternalVar = true;
316            } else if (c == '}') {
317               if (depth > 0) {
318                  depth--;
319               } else {
320                  varVal = s.substring(x+1, i);
321                  varVal = (hasInternalVar ? resolve(varVal) : varVal);
322                  StringVar r = getVar(varType);
323                  if (r == null) {
324                     if (hasInnerEscapes)
325                        out.append(unEscapeChars(s.substring(x2, i+1), new char[]{'\\','$','{','}'}));
326                     else
327                        out.append(s, x2, i+1);
328                     x = i+1;
329                  } else {
330                     String replacement = r.doResolve(varVal);
331                     if (replacement == null)
332                        replacement = "";
333                     // If the replacement also contains variables, replace them now.
334                     if (replacement.indexOf('$') != -1)
335                        replacement = resolve(replacement);
336                     out.append(replacement);
337                     x = i+1;
338                  }
339                  state = 1;
340                  hasInnerEscapes = false;
341               }
342            }
343         }
344      }
345      if (isInEscape)
346         out.append('\\');
347      else if (state == S2)
348         out.append('$').append(unEscapeChars(s.substring(x+1), new char[]{'{', '\\'}));
349      else if (state == S3)
350         out.append('$').append(varType).append('{').append(unEscapeChars(s.substring(x+1), new char[]{'\\','$','{','}'}));
351      return out;
352   }
353
354   private StringVar getVar(String varType) {
355      StringVar v = varMap.get(varType);
356      if (v == null && parent != null)
357         v = parent.getVar(varType);
358      return v;
359   }
