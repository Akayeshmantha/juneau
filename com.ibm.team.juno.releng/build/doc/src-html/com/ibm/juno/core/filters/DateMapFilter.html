009package com.ibm.juno.core.filters;
010
011import java.util.*;
012
013import com.ibm.juno.core.*;
014import com.ibm.juno.core.filter.*;
015import com.ibm.juno.core.parser.*;
016
017/**
018 * Transforms {@link Date Dates} to {@link Map Maps} of the format <tt>{value:long}</tt>.
019 *
020 * @author James Bognar (jbognar@us.ibm.com)
021 */
022@SuppressWarnings("rawtypes")
023public class DateMapFilter extends PojoFilter<Date,Map> {
024
025   /**
026    * Converts the specified {@link Date} to a {@link Map}.
027    */
028   @Override /* PojoFilter */
029   public Map filter(Date o) {
030      ObjectMap m = new ObjectMap();
031      m.put("time", o.getTime());
032      return m;
033   }
034
035   /**
036    * Converts the specified {@link Map} to a {@link Date}.
037    */
038   @Override /* PojoFilter */
039   public Date unfilter(Map o, ClassMeta<?> hint) throws ParseException {
040      Class<?> c = (hint == null ? java.util.Date.class : hint.getInnerClass());
041      long l = Long.parseLong(((Map<?,?>)o).get("time").toString());
042      if (c == java.util.Date.class)
043         return new java.util.Date(l);
044      if (c == java.sql.Date.class)
045         return new java.sql.Date(l);
046      if (c == java.sql.Time.class)
047         return new java.sql.Time(l);
048      if (c == java.sql.Timestamp.class)
049         return new java.sql.Timestamp(l);
050      throw new ParseException("DateMapFilter is unable to narrow object of type ''{0}''", c);
051   }
052}
