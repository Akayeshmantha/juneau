009package com.ibm.juno.core.encoders;
010
011import static com.ibm.juno.core.utils.ArrayUtils.*;
012
013import java.util.*;
014
015import com.ibm.juno.core.*;
016
017/**
018 * Represents the group of {@link Encoder encoders} keyed by codings.
019 *
020 *
021 * <h6 class='topic'>Description</h6>
022 * <p>
023 *    Maintains a set of encoders and the codings that they can handle.
024 * <p>
025 *    The {@link #findMatch(String)} and {@link #getEncoder(String)} methods are then
026 *       used to find appropriate encoders for specific <code>Accept-Encoding</code>
027 *       and <code>Content-Encoding</code> header values.
028 *
029 *
030 * <h6 class='topic'>Match ordering</h6>
031 * <p>
032 *    Encoders are matched against <code>Accept-Encoding</code> strings in the order they exist in this group.
033 * <p>
034 *    Adding new entries will cause the entries to be prepended to the group.
035 *    This allows for previous encoders to be overridden through subsequent calls.
036 * <p>
037 *    For example, calling <code>g.append(E1.<jk>class</jk>,E2.<jk>class</jk>).append(E3.<jk>class</jk>,E4.<jk>class</jk>)</code>
038 *    will result in the order <code>E3, E4, E1, E2</code>.
039 *
040 *
041 * <h6 class='topic'>Example</h6>
042 * <p class='bcode'>
043 *    <jc>// Create an encoder group with support for gzip compression.</jc>
044 *    EncoderGroup g = <jk>new</jk> EncoderGroup().append(GzipEncoder.<jk>class</jk>);
045 *
046 *    <jc>// Should return "gzip"</jc>
047 *    String matchedCoding = g.findMatch(<js>"compress;q=1.0, gzip;q=0.8, identity;q=0.5, *;q=0"</js>);
048 *
049 *    <jc>// Get the encoder</jc>
050 *    IEncoder encoder = g.getEncoder(matchedCoding);
051 * </p>
052 *
053 *
054 * @author James Bognar (jbognar@us.ibm.com)
055 */
056public final class EncoderGroup {
057
058   private Map<String,EncoderEntry> entryMap = new TreeMap<String,EncoderEntry>(String.CASE_INSENSITIVE_ORDER);
059   private LinkedList<EncoderEntry> tempEntries = new LinkedList<EncoderEntry>();
060   private EncoderEntry[] entries;
061
062   /**
063    * Returns the coding string for the matching encoder that can handle the specified <code>Accept-Encoding</code>
064    *    or <code>Content-Encoding</code> header value.
065    * <p>
066    *    Returns <jk>null</jk> if no encoders can handle it.
067    * <p>
068    *    This method is fully compliant with the RFC2616/14.3 and 14.11 specifications.
069    *
070    * @param acceptEncoding The <code>Accept-Encoding</code> or <code>Content-Encoding</code> value.
071    * @return The coding value (e.g. <js>"gzip"</js>).
072    */
073   public String findMatch(String acceptEncoding) {
074      if (getEntries().length == 0)
075         return null;
076
077      MediaRange[] ae = MediaRange.parse(acceptEncoding);
078
079      if (ae.length == 0)
080         ae = MediaRange.parse("*");
081
082      for (MediaRange a : ae)
083         for (EncoderEntry e : getEntries())
084            for (MediaRange a2 : e.encodingRanges)
085               if (a.matches(a2))
086                  return a2.getType();
087
088      return null;
089   }
090
091   /**
092    * Adds the specified encoders to this group.
093    *
094    * @param e The encoders to instantiate and add to this group.
095    * @return This object (for method chaining).
096    * @throws Exception If an instantiation error occurred.
097    */
098   public EncoderGroup append(Class<? extends Encoder>...e) throws Exception {
099      for (Class<? extends Encoder> r : reverse(e))
100         append(r.newInstance());
101      return this;
102   }
103
104   /**
105    * Adds the specified encoders to this group.
106    *
107    * @param e The encoder to instantiate and add to this group.
108    * @return This object (for method chaining).
109    * @throws Exception If an instantiation error occurred.
110    */
111   public EncoderGroup append(Class<? extends Encoder> e) throws Exception {
112      append(e.newInstance());
113      return this;
114   }
115
116   /**
117    * Adds the specified encoders to this group.
118    *
119    * @param e The encoders to add to this group.
120    * @return This object (for method chaining).
121    */
122   public EncoderGroup append(Encoder...e) {
123      entries = null;
124      for (Encoder r : reverse(e)) {
125         EncoderEntry ee = new EncoderEntry(r);
126         tempEntries.addFirst(ee);
127         for (String s : ee.encodings)
128            this.entryMap.put(s, ee);
129      }
130      return this;
131   }
132
133   /**
134    * Adds the encoders in the specified group to this group.
135    *
136    * @param g The group containing the encoders to add to this group.
137    * @return This object (for method chaining).
138    */
139   public EncoderGroup append(EncoderGroup g) {
140      for (EncoderEntry e : reverse(g.getEntries()))
141         append(e.encoder);
142      return this;
143   }
144
145   /**
146    * Returns the encoder registered with the specified coding (e.g. <js>"gzip"</js>).
147    *
148    * @param coding The coding string.
149    * @return The encoder, or <jk>null</jk> if encoder isn't registered with that coding.
150    */
151   public Encoder getEncoder(String coding) {
152      EncoderEntry e = entryMap.get(coding);
153      return (e == null ? null : e.encoder);
154   }
155
156   /**
157    * Returns the set of codings supported by all encoders in this group.
158    *
159    * @return The set of codings supported by all encoders in this group.  Never <jk>null</jk>.
160    */
161   public List<String> getSupportedEncodings() {
162      List<String> l = new ArrayList<String>();
163      for (EncoderEntry e : getEntries())
164         for (String enc : e.encodings)
165            if (! l.contains(enc))
166               l.add(enc);
167      return l;
168   }
169
170   private EncoderEntry[] getEntries() {
171      if (entries == null)
172         entries = tempEntries.toArray(new EncoderEntry[tempEntries.size()]);
173      return entries;
174   }
175
176   static class EncoderEntry {
177      Encoder encoder;
178      MediaRange[] encodingRanges;
179      String[] encodings;
180
181      EncoderEntry(Encoder e) {
182         encoder = e;
183
184         encodings = new String[e.getCodings().length];
185         int i = 0;
186         for (String enc : e.getCodings())
187            encodings[i++] = enc;
188
189         List<MediaRange> l = new LinkedList<MediaRange>();
190         for (i = 0; i < encodings.length; i++)
191            l.addAll(Arrays.asList(MediaRange.parse(encodings[i])));
192         encodingRanges = l.toArray(new MediaRange[l.size()]);
193      }
194   }
195}
