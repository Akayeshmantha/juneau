012/**
013 * Properties associated with the {@link HtmlDocSerializer} class.
014 * <p>
015 *    These are typically specified via <ja>@RestResource.properties()</ja> and <ja>@RestMethod.properties()</ja> annotations,
016 *       although they can also be set programmatically via the <code>RestREsponse.setProperty()</code> method.
017 *
018 * <h6 class='topic'>Example</h6>
019 * <p class='bcode'>
020 *    <ja>@RestResource</ja>(
021 *       messages=<js>"nls/AddressBookResource"</js>,
022 *       properties={
023 *          <ja>@Property</ja>(name=HtmlDocSerializerProperties.<jsf>HTMLDOC_title</jsf>, value=<js>"$L{title}"</js>),
024 *          <ja>@Property</ja>(name=HtmlDocSerializerProperties.<jsf>HTMLDOC_description</jsf>, value=<js>"$L{description}"</js>),
025 *          <ja>@Property</ja>(name=HtmlDocSerializerProperties.<jsf>HTMLDOC_links</jsf>, value=<js>"{options:'?method=OPTIONS',doc:'doc'}"</js>)
026 *       }
027 *    )
028 *    <jk>public class</jk> AddressBookResource <jk>extends</jk> RestServletJenaDefault {
029 * </p>
030 * <p>
031 *    The <code>$L{...}</code> variable represent localized strings pulled from the resource bundle identified by the <code>messages</code> annotation.
032 *    These variables are replaced at runtime based on the HTTP request locale.
033 *    Several built-in runtime variable types are defined, and the API can be extended to include user-defined variables.
034 * </p>
035 *
036 * @author James Bognar (jbognar@us.ibm.com)
037 */
038public final class HtmlDocSerializerProperties {
039
040   /**
041    * Adds a title at the top of a page.
042    *
043    * <dl>
044    *    <dt>Example:</dt>
045    *    <dd>
046    * <p>
047    *    The <code>AddressBookResource</code> sample class uses this property...
048    * </p>
049    * <p class='bcode'>
050    *    <ja>@RestResource</ja>(
051    *       messages=<js>"nls/AddressBookResource"</js>,
052    *       properties={
053    *          <ja>@Property</ja>(name=HtmlDocSerializerProperties.<jsf>HTMLDOC_title</jsf>, value=<js>"$L{title}"</js>)
054    *       }
055    *    )
056    *    <jk>public class</jk> AddressBookResource <jk>extends</jk> RestServletJenaDefault {
057    * </p>
058    * <p>
059    *    ...with this property in <code>AddressBookResource.properties</code>...
060    * </p>
061    * <p class='bcode'>
062    *    title = <js>AddressBook sample resource</js>
063    * </p>
064    * <p>
065    *    ...to produce this title on the HTML page...
066    * </p>
067    *       <img class='bordered' src='doc-files/HTML_TITLE.png'>
068    *    </dd>
069    * </dl>
070    */
071   public static final String HTMLDOC_title = "HtmlSerializer.title";
072
073   /**
074    * Adds a description right below the title of a page.
075    *
076    * <dl>
077    *    <dt>Example:</dt>
078    *    <dd>
079    * <p>
080    *    The <code>AddressBookResource</code> sample class uses this property...
081    * </p>
082    * <p class='bcode'>
083    *    <ja>@RestResource</ja>(
084    *       messages=<js>"nls/AddressBookResource"</js>,
085    *       properties={
086    *          <ja>@Property</ja>(name=HtmlDocSerializerProperties.<jsf>HTMLDOC_description</jsf>, value=<js>"description"</js>, type=<jsf>NLS</jsf>)
087    *       }
088    *    )
089    *    <jk>public class</jk> AddressBookResource <jk>extends</jk> RestServletJenaDefault {
090    * </p>
091    * <p>
092    *    ...with this property in <code>AddressBookResource.properties</code>...
093    * </p>
094    * <p class='bcode'>
095    *    description = <js>Simple address book POJO sample resource</js>
096    * </p>
097    * <p>
098    *    ...to produce this description on the HTML page...
099    * </p>
100    *       <img class='bordered' src='doc-files/HTML_DESCRIPTION.png'>
101    *    </dd>
102    * </dl>
103    */
104   public static final String HTMLDOC_description = "HtmlSerializer.description";
105
106   /**
107    * Adds a list of hyperlinks immediately under the title and description but above the content of the page.
108    * <p>
109    *    This can be used to provide convenient hyperlinks when viewing the REST interface from a browser.
110    * <p>
111    *    The value is a JSON object string where the keys are anchor text and the values are URLs.
112    * <p>
113    *    Relative URLs are considered relative to the servlet path.
114    *    For example, if the servlet path is <js>"http://localhost/myContext/myServlet"</js>, and the
115    *       URL is <js>"foo"</js>, the link becomes <js>"http://localhost/myContext/myServlet/foo"</js>.
116    *    Absolute (<js>"/myOtherContext/foo"</js>) and fully-qualified (<js>"http://localhost2/foo"</js>) URLs
117    *       can also be used.
118    *
119    * <dl>
120    *    <dt>Example:</dt>
121    *    <dd>
122    * <p>
123    *    The <code>AddressBookResource</code> sample class uses this property...
124    * </p>
125    * <p class='bcode'>
126    *    <ja>@RestResource</ja>(
127    *       messages=<js>"nls/AddressBookResource"</js>,
128    *       properties={
129    *          <ja>@Property</ja>(name=HtmlDocSerializerProperties.<jsf>HTMLDOC_links</jsf>, value=<js>"{options:'?method=OPTIONS',doc:'doc'}"</js>)
130    *       }
131    *    )
132    *    <jk>public class</jk> AddressBookResource <jk>extends</jk> RestServletJenaDefault {
133    * </p>
134    * <p>
135    *    ...to produce this list of links on the HTML page...
136    * </p>
137    *       <img class='bordered' src='doc-files/HTML_LINKS.png'>
138    *    </dd>
139    * </dl>
140    */
141   public static final String HTMLDOC_links = "HtmlDocSerializer.links";
142
143   /**
144    * Similar to {@link #HTMLDOC_links} except appends on to the existing list of links.
145    */
146   public static final String HTMLDOC_addLinks = "HtmlDocSerializer.addLinks";
147
148   /**
149    * Adds a link to the specified stylesheet URL.
150    * <p>
151    *    If not specified, defaults to the built-in stylesheet located at <js>"/servletPath/style.css"</js>.
152    *    Note that this stylesheet is controlled by the <code><ja>@RestResource</ja>.style()</code> annotation.
153    */
154   public static final String HTMLDOC_cssUrl = "HtmlDocSerializer.cssUrl";
155
156   /**
157    * Imports the specified CSS page URLs into the page.
158    */
159   public static final String HTMLDOC_cssImports = "HtmlDocSerializer.cssImports";
160
161   /**
162    * Adds <js>"* {white-space:nowrap}"</js> to the style header to prevent word wrapping.
163    */
164   public static final String HTMLDOC_nowrap = "HtmlDocSerializer.nowrap";
165}
