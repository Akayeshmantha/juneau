009package com.ibm.juno.core.utils;
010
011import java.io.*;
012import java.net.*;
013import java.util.*;
014import java.util.jar.*;
015
016import com.ibm.juno.core.*;
017
018/**
019 * Utility class for working with Jar manifest files.
020 * <p>
021 * Copies the contents of a {@link Manifest} into an {@link ObjectMap} so that the various
022 *    convenience methods on that class can be used to retrieve values.
023 *
024 * @author James Bognar (jbognar@us.ibm.com)
025 */
026public class ManifestFile extends ObjectMap {
027
028   private static final long serialVersionUID = 1L;
029   private transient volatile StringVar stringVar;
030
031   /**
032    * Create an instance of this class from a manifest file on the file system.
033    *
034    * @param f The manifest file.
035    * @throws IOException If a problem occurred while trying to read the manifest file.
036    */
037   public ManifestFile(File f) throws IOException {
038      Manifest mf = new Manifest();
039      FileInputStream fis = new FileInputStream(f);
040      try {
041         mf.read(fis);
042         load(mf);
043      } catch (IOException e) {
044         throw new IOException("Problem detected in MANIFEST.MF.  Contents below:\n" + IOUtils.read(f), e);
045      } finally {
046         IOUtils.closeQuietly(fis);
047      }
048   }
049
050   /**
051    * Create an instance of this class from a {@link Manifest} object.
052    *
053    * @param f The manifest to read from.
054    */
055   public ManifestFile(Manifest f) {
056      load(f);
057   }
058
059   /**
060    * Finds and loads the manifest file of the jar file that the specified class is contained within.
061    *
062    * @param c The class to get the manifest file of.
063    * @throws IOException If a problem occurred while trying to read the manifest file.
064    */
065   public ManifestFile(Class<?> c) throws IOException {
066      String className = c.getSimpleName() + ".class";
067      String classPath = c.getResource(className).toString();
068      if (! classPath.startsWith("jar")) {
069         return;
070      }
071      String manifestPath = classPath.substring(0, classPath.lastIndexOf("!") + 1) +  "/META-INF/MANIFEST.MF";
072      try {
073         Manifest mf = new Manifest(new URL(manifestPath).openStream());
074         load(mf);
075      } catch (MalformedURLException e) {
076         throw new IOException(e);
077      } catch (IOException e) {
078         e.printStackTrace();
079      }
080   }
081
082   /**
083    * Create an instance of a {@link StringVarWithDefault} that returns values in this manifest file.
084    *
085    * @return A new {@link StringVarWithDefault}.
086    */
087   public StringVar getStringVar() {
088      if (stringVar == null)
089         stringVar = new StringMapVar(this);
090      return stringVar;
091   }
092
093   private void load(Manifest mf) {
094      for (Map.Entry<Object,Object> e : mf.getMainAttributes().entrySet())
095         put(e.getKey().toString(), e.getValue().toString());
096   }
097}
