009package com.ibm.juno.core.utils;
010
011import java.util.*;
012
013/**
014 * Combination of a {@link LinkedList} and <code>IdentitySet</code>.
015 * <ul>
016 *    <li>Duplicate objects (by identity) will be skipped during insertion.
017 *    <li>Order of insertion maintained.
018 * </ul>
019 * <p>
020 *    Note:  This class is NOT thread safe, and is intended for use on small lists.
021 *
022 * @author James Bognar (jbognar@us.ibm.com)
023 * @param <T> Entry type.
024 */
025public class IdentityList<T> extends LinkedList<T> {
026
027   private static final long serialVersionUID = 1L;
028
029   @Override /* List */
030   public boolean add(T t) {
031      for (T t2 : this)
032         if (t2 == t)
033            return false;
034      super.add(t);
035      return true;
036   }
037
038   @Override /* List */
039   public boolean contains(Object t) {
040      for (T t2 : this)
041         if (t2 == t)
042            return true;
043      return false;
044   }
045}
