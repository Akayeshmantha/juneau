009package com.ibm.juno.core.dto.jsonschema;
010
011import java.io.*;
012import java.net.*;
013import java.util.concurrent.*;
014
015import com.ibm.juno.core.json.*;
016
017/**
018 * A container for retrieving JSON {@link Schema} objects by URI.
019 * <p>
020 *    Subclasses must implement one of the following methods to load schemas from external sources:
021 * <ul>
022 *    <li>{@link #getReader(URI)} - If schemas should be loaded from readers and automatically parsed.
023 *    <li>{@link #load(URI)} - If you want control over construction of {@link Schema} objects.
024 * </ul>
025 *
026 * @author James Bognar (jbognar@us.ibm.com)
027 */
028public abstract class SchemaMap extends ConcurrentHashMap<URI,Schema> {
029
030   private static final long serialVersionUID = 1L;
031
032   @Override /* Map */
033   public Schema get(Object uri) {
034      if (uri == null)
035         return null;
036      return get(URI.create(uri.toString()));
037   }
038
039   /**
040    * Return the {@link Schema} object at the specified URI.
041    * If this schema object has not been loaded yet, calls {@link #load(URI)}.
042    *
043    * @param uri The URI of the schema to retrieve.
044    * @return The Schema, or <jk>null</jk> if schema was not located and could not be loaded.
045    */
046   public Schema get(URI uri) {
047      Schema s = super.get(uri);
048      if (s != null)
049         return s;
050      synchronized(this) {
051         s = load(uri);
052         if (s != null) {
053            // Note:  Can't use add(Schema...) since the ID property may not be set.
054            s.setSchemaMap(this);
055            put(uri, s);
056         }
057         return s;
058      }
059   }
060
061   /**
062    * Convenience method for prepopulating this map with the specified schemas.
063    * <p>
064    * The schemas passed in through this method MUST have their ID properties set.
065    *
066    * @param schemas The set of schemas to add to this map.
067    * @return This object (for method chaining).
068    * @throws RuntimeException If one or more schema objects did not have their ID property set.
069    */
070   public SchemaMap add(Schema...schemas) {
071      for (Schema schema : schemas) {
072         if (schema.getId() == null)
073            throw new RuntimeException("Schema with no ID passed to SchemaMap.add(Schema...)");
074         put(schema.getId(), schema);
075         schema.setSchemaMap(this);
076      }
077      return this;
078   }
079
080   /**
081    * Subclasses must implement either this method or {@link #getReader(URI)} to load the schema with the specified URI.
082    * It's up to the implementer to decide where these come from.
083    * <p>
084    * The default implementation calls {@link #getReader(URI)} and parses the schema document.
085    * If {@link #getReader(URI)} returns <jk>null</jk>, this method returns <jk>null</jk> indicating this is an unreachable document.
086    *
087    * @param uri The URI to load the schema from.
088    * @return The parsed schema.
089    */
090   public Schema load(URI uri) {
091      Reader r = getReader(uri);
092      if (r == null)
093         return null;
094      try {
095         return JsonParser.DEFAULT.parse(r, -1, Schema.class);
096      } catch (Exception e) {
097         throw new RuntimeException(e);
098      } finally {
099         try {
100            r.close();
101         } catch (IOException e) {
102            // Ignore
103         }
104      }
105   }
106
107   /**
108    * Subclasses must implement either this method or {@link #load(URI)} to load the schema with the specified URI.
109    * It's up to the implementer to decide where these come from.
110    * <p>
111    * The default implementation returns <jk>null</jk>.
112    *
113    * @param uri The URI to connect to and retrieve the contents.
114    * @return The reader from reading the specified URI.
115    */
116   public Reader getReader(URI uri) {
117      return null;
118   }
119}
