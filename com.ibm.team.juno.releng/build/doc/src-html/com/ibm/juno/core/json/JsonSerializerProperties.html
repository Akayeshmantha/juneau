009package com.ibm.juno.core.json;
010
011import com.ibm.juno.core.*;
012import com.ibm.juno.core.serializer.*;
013
014/**
015 * Configurable properties on the {@link JsonSerializer} class.
016 * <p>
017 *    Use the {@link JsonSerializer#setProperty(String, Object)} method to set property values.
018 * <p>
019 *    In addition to these properties, the following properties are also applicable for {@link JsonSerializer}.
020 * <ul>
021 *    <li>{@link SerializerProperties}
022 *    <li>{@link BeanContextProperties}
023 * </ul>
024 *
025 * @author James Bognar (jbognar@us.ibm.com)
026 */
027public final class JsonSerializerProperties implements Cloneable {
028
029   /**
030    * Simple JSON mode ({@link Boolean}, default=<jk>false</jk>).
031    * <p>
032    * If <jk>true</jk>, JSON attribute names will only be quoted when necessary.
033    * Otherwise, they are always quoted.
034    */
035   public static final String JSON_simpleMode = "JsonSerializer.simpleMode";
036
037   /**
038    * Use whitespace in output ({@link Boolean}, default=<jk>false</jk>).
039    * <p>
040    * If <jk>true</jk>, whitespace is added to the output to improve readability.
041    */
042   public static final String JSON_useWhitespace = "JsonSerializer.useWhitespace";
043
044   /**
045    * Prefix solidus <js>'/'</js> characters with escapes ({@link Boolean}, default=<jk>false</jk>).
046    * <p>
047    * If <jk>true</jk>, solidus (e.g. slash) characters should be escaped.
048    * The JSON specification allows for either format.
049    * However, if you're embedding JSON in an HTML script tag, this setting prevents
050    *    confusion when trying to serialize <xt>&lt;\/script&gt;</xt>.
051    */
052   public static final String JSON_escapeSolidus = "JsonSerializer.escapeSolidus";
053
054   boolean
055      simpleMode = false,
056      useWhitespace = false,
057      escapeSolidus = false;
058
059   /**
060    * Sets the specified property value.
061    * @param property The property name.
062    * @param value The property value.
063    * @return <jk>true</jk> if property name was valid and property was set.
064    */
065   public boolean setProperty(String property, Object value) {
066      BeanContext bc = BeanContext.DEFAULT;
067      if (property.equals(JSON_simpleMode))
068         simpleMode = bc.convertToType(value, Boolean.class);
069      else if (property.equals(JSON_useWhitespace))
070         useWhitespace = bc.convertToType(value, Boolean.class);
071      else if (property.equals(JSON_escapeSolidus))
072         escapeSolidus = bc.convertToType(value, Boolean.class);
073      else
074         return false;
075      return true;
076   }
077
078
079   //--------------------------------------------------------------------------------
080   // Overridden methods
081   //--------------------------------------------------------------------------------
082
083   @Override /* Cloneable */
084   public JsonSerializerProperties clone() {
085      try {
086         return (JsonSerializerProperties)super.clone();
087      } catch (CloneNotSupportedException e) {
088         throw new RuntimeException(e); // Shouldn't happen
089      }
090   }
091}
