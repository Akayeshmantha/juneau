026/**
027 * Serializes POJO models to XML.
028 *
029 *
030 * <h6 class='topic'>Media types</h6>
031 * <p>
032 *    Handles <code>Accept</code> types: <code>text/xml</code>
033 * <p>
034 *    Produces <code>Content-Type</code> types: <code>text/xml</code>
035 *
036 *
037 * <h6 class='topic'>Description</h6>
038 * <p>
039 *    See the {@link JsonSerializer} class for details on how Java models map to JSON.
040 * <p>
041 *    For example, the following JSON...
042 * <p class='bcode'>
043 *    {
044 *       name:<js>'John Smith'</js>,
045 *       address: {
046 *          streetAddress: <js>'21 2nd Street'</js>,
047 *          city: <js>'New York'</js>,
048 *          state: <js>'NY'</js>,
049 *          postalCode: <js>10021</js>
050 *       },
051 *       phoneNumbers: [
052 *          <js>'212 555-1111'</js>,
053 *          <js>'212 555-2222'</js>
054 *       ],
055 *       additionalInfo: <jk>null</jk>,
056 *       remote: <jk>false</jk>,
057 *       height: <js>62.4</js>,
058 *       <js>'fico score'</js>:  <js>' &gt; 640'</js>
059 *    }
060 * <p>
061 *    ...maps to the following XML...
062 * <p class='bcode'>
063 *    <xt>&lt;object&gt;</xt>
064 *       <xt>&lt;name</xt> <xa>type</xa>=<xs>'string'</xs><xt>&gt;</xt>John Smith<xt>&lt;/name&gt;</xt>
065 *       <xt>&lt;address</xt> <xa>type</xa>=<xs>'object'</xs><xt>&gt;</xt>
066 *          <xt>&lt;streetAddress</xt> <xa>type</xa>=<xs>'string'</xs><xt>&gt;</xt>21 2nd Street<xt>&lt;/streetAddress&gt;</xt>
067 *          <xt>&lt;city</xt> <xa>type</xa>=<xs>'string'</xs><xt>&gt;</xt>New York<xt>&lt;/city&gt;</xt>
068 *          <xt>&lt;state</xt> <xa>type</xa>=<xs>'string'</xs><xt>&gt;</xt>NY<xt>&lt;/state&gt;</xt>
069 *          <xt>&lt;postalCode</xt> <xa>type</xa>=<xs>'number'</xs><xt>&gt;</xt>10021<xt>&lt;/postalCode&gt;</xt>
070 *       <xt>&lt;/address&gt;</xt>
071 *       <xt>&lt;phoneNumbers</xt> <xa>type</xa>=<xs>'array'</xs><xt>&gt;</xt>
072 *          <xt>&lt;string&gt;</xt>212 555-1111<xt>&lt;/string&gt;</xt>
073 *          <xt>&lt;string&gt;</xt>212 555-2222<xt>&lt;/string&gt;</xt>
074 *       <xt>&lt;/phoneNumbers&gt;</xt>
075 *       <xt>&lt;additionalInfo</xt> <xa>type</xa>=<xs>'null'</xs><xt>&gt;&lt;/additionalInfo&gt;</xt>
076 *       <xt>&lt;remote</xt> <xa>type</xa>=<xs>'boolean'</xs><xt>&gt;</xt>false<xt>&lt;/remote&gt;</xt>
077 *       <xt>&lt;height</xt> <xa>type</xa>=<xs>'number'</xs><xt>&gt;</xt>62.4<xt>&lt;/height&gt;</xt>
078 *       <xt>&lt;fico_x0020_score</xt> <xa>type</xa>=<xs>'string'</xs><xt>&gt;</xt> &amp;gt; 640<xt>&lt;/fico_x0020_score&gt;</xt>
079 *    <xt>&lt;/object&gt;</xt>
080 * <p>
081 *    This serializer provides several serialization options.  Typically, one of the predefined <jsf>DEFAULT</jsf> serializers will be sufficient.
082 *    However, custom serializers can be constructed to fine-tune behavior.
083 * <p>
084 *    If an attribute name contains any non-valid XML element characters, they will be escaped using standard {@code _x####_} notation.
085 *
086 *
087 * <h6 class='topic'>Configurable properties</h6>
088 * <p>
089 *    This class has the following properties associated with it:
090 * <ul>
091 *    <li>{@link XmlSerializerProperties}
092 *    <li>{@link SerializerProperties}
093 *    <li>{@link BeanContextProperties}
094 * </ul>
095 *
096 *
097 * <h6 class='topic'>Behavior-specific subclasses</h6>
098 * <p>
099 *    The following direct subclasses are provided for convenience:
100 * <ul>
101 *    <li>{@link Sq} - Default serializer, single quotes.
102 *    <li>{@link SqReadable} - Default serializer, single quotes, whitespace added.
103 *    <li>{@link XmlJson} - Default serializer with JSON attribute tags.
104 *    <li>{@link XmlJsonSq} - Default serializer with JSON attribute tags, single quotes.
105 * </ul>
106 *
107 *
108 * @author James Bognar (jbognar@us.ibm.com)
109 */
110@SuppressWarnings({ "rawtypes", "unchecked" })

201   /**
202    * Recursively searches for the XML namespaces on the specified POJO and adds them to the serializer context object.
203    *
204    * @param o The POJO to check.
205    * @param ctx The context that exists for the duration of a single serialization.
206    * @throws SerializeException
207    */
208   protected void findNsfMappings(Object o, XmlSerializerContext ctx) throws SerializeException {
209      BeanContext bc = ctx.getBeanContext();
210      ClassMeta<?> aType = null;                // The actual type
211      aType = ctx.push(null, o, null);
212
213      if (aType != null) {
214         Namespace ns = aType.getXmlMeta().getNamespace();
215         if (ns != null) {
216            if (ns.uri != null)
217               ctx.addNamespace(ns);
218            else
219               ns = null;
220         }
221      }
222
223      // Handle recursion
224      if (aType != null && ! aType.isPrimitive()) {
225
226         BeanMap bm = null;
227         if (aType.isBeanMap()) {
228            bm = (BeanMap)o;
229         } else if (aType.isBean()) {
230            bm = bc.forBean(o);
231         } else if (aType.isDelegate()) {
232            ClassMeta innerType = ((Delegate)o).getClassMeta();
233            Namespace ns = innerType.getXmlMeta().getNamespace();
234            if (ns != null) {
235               if (ns.uri != null)
236                  ctx.addNamespace(ns);
237               else
238                  ns = null;
239            }
240
241            if (innerType.isBean()) {
242               for (BeanPropertyMeta bpm : (Collection<BeanPropertyMeta>)innerType.getBeanMeta().getPropertyMetas()) {
243                  ns = bpm.getXmlMeta().getNamespace();
244                  if (ns != null && ns.uri != null)
245                     ctx.addNamespace(ns);
246               }
247
248            } else if (innerType.isMap()) {
249               for (Object o2 : ((Map)o).values())
250                  findNsfMappings(o2, ctx);
251            } else if (innerType.isCollection()) {
252               for (Object o2 : ((Collection)o))
253                  findNsfMappings(o2, ctx);
254            }
255
256         } else if (aType.isMap()) {
257            for (Object o2 : ((Map)o).values())
258               findNsfMappings(o2, ctx);
259         } else if (aType.isCollection()) {
260            for (Object o2 : ((Collection)o))
261               findNsfMappings(o2, ctx);
262         } else if (aType.isArray() && ! aType.getElementType().isPrimitive()) {
263            for (Object o2 : ((Object[])o))
264               findNsfMappings(o2, ctx);
265         }
266         if (bm != null) {
267            for (BeanMapEntry p : (Set<BeanMapEntry>)bm.entrySet()) {
268
269               Namespace ns = p.getMeta().getXmlMeta().getNamespace();
270               if (ns != null && ns.uri != null)
271                  ctx.addNamespace(ns);
272
273               try {
274                  findNsfMappings(p.getValue(), ctx);
275               } catch (Throwable x) {
276                  // Ignore
277               }
278            }
279         }
280      }
281
282      ctx.pop();
283   }

285   /**
286    * Workhorse method.
287    *
288    * @param out The writer to send the output to.
289    * @param o The object to serialize.
290    * @param eType The expected type if this is a bean property value being serialized.
291    * @param ctx The serializer context.
292    * @param elementName The root element name.
293    * @param elementNamespace The namespace of the element.
294    * @param addNamespaceUris Flag indicating that namespace URIs need to be added.
295    * @param format The format to serialize the output to.
296    * @param pMeta The bean property metadata if this is a bean property being serialized.
297    * @return The same writer passed in so that calls to the writer can be chained.
298    * @throws IOException If a problem occurred trying to write to the writer.
299    * @throws SerializeException If a problem occurred trying to convert the output.
300    */
301   protected XmlSerializerWriter serializeAnything(XmlSerializerWriter out, Object o, ClassMeta eType,
302         XmlSerializerContext ctx, String elementName, Namespace elementNamespace, boolean addNamespaceUris,
303         XmlFormat format, BeanPropertyMeta<?> pMeta) throws IOException, SerializeException {
304
305      BeanContext bc = ctx.getBeanContext();
306      String ts = null;              // The type string (e.g. <type> or <x x='type'>
307      int indent = ctx.indent;       // Current indentation
308      ClassMeta<?> aType = null;     // The actual type
309      ClassMeta<?> wType = null;     // The wrapped type
310      ClassMeta<?> gType = object(); // The generic type
311
312      aType = ctx.push(elementName, o, eType);
313
314      if (eType == null)
315         eType = object();
316
317      // Handle recursion
318      if (aType == null) {
319         o = null;
320         aType = object();
321      }
322
323      if (o != null) {
324
325         if (aType.isDelegate()) {
326            wType = aType;
327            aType = ((Delegate)o).getClassMeta();
328         }
329
330         gType = aType.getFilteredClassMeta();
331
332         // Filter if necessary
333         PojoFilter filter = aType.getPojoFilter();
334         if (filter != null) {
335            o = filter.filter(o);
336
337            // If the filter's getFilteredClass() method returns Object, we need to figure out
338            // the actual type now.
339            if (gType.isObject())
340               gType = bc.getClassMetaForObject(o);
341         }
342      } else {
343         gType = eType.getFilteredClassMeta();
344      }
345
346      String classAttr = null;
347      if (ctx.isAddClassAttrs()) {
348         if (o != null && ! eType.equals(aType))
349            classAttr = aType.toString();
350         else if (o == null)
351            classAttr = eType.toString();
352      }
353
354      // char '\0' is interpreted as null.
355      if (o != null && gType.isChar() && ((Character)o).charValue() == 0)
356         o = null;
357
358      boolean isCollapsed = false;     // If 'true', this is a collection and we're not rendering the outer element.
359
360      // Get the JSON type string.
361      if (gType.isCharSequence() || gType.isChar())
362         ts = "string";
363      else if (gType.isNumber())
364         ts = "number";
365      else if (gType.isBoolean())
366         ts = "boolean";
367      else if (gType.isMap() || gType.isBean() || gType.hasToObjectMapMethod()) {
368         isCollapsed = gType.getXmlMeta().getFormat() == XmlFormat.COLLAPSED;
369         ts = "object";
370      }
371      else if (gType.isCollection() || gType.isArray()) {
372         isCollapsed = (format == COLLAPSED && ! addNamespaceUris);
373         ts = "array";
374      }
375      else
376         ts = "string";
377
378
379      // Is there a name associated with this bean?
380      if (elementName == null)
381         elementName = gType.getXmlMeta().getElementName();
382      if (elementName == null)
383         elementName = aType.getXmlMeta().getElementName();
384
385      // If the value is null then it's either going to be <null/> or <XmlSerializer nil='true'/>
386      // depending on whether the element has a name.
387      boolean isNullTag = (elementName == null && o == null);
388
389      if (isNullTag)
390         ts = "null";
391
392      if (ctx.isEnableNamespaces()) {
393         if (elementNamespace == null)
394            elementNamespace = gType.getXmlMeta().getNamespace();
395         if (elementNamespace == null)
396            elementNamespace = aType.getXmlMeta().getNamespace();
397         if (elementNamespace != null && elementNamespace.uri == null)
398            elementNamespace = null;
399         if (elementNamespace == null)
400            elementNamespace = ctx.getDefaultNamespace();
401      } else {
402         elementNamespace = null;
403      }
404
405      // Do we need a carriage return after the start tag?
406      boolean cr = o != null && (gType.isMap() || gType.isCollection() || gType.isArray() || gType.isBean() || gType.hasToObjectMapMethod());
407
408      String en = (elementName == null ? ts : elementName);
409      boolean encodeEn = elementName != null;
410      String ns = (elementNamespace == null ? null : elementNamespace.name);
411      String xsi = null, dns = null, elementNs = null;
412      if (ctx.isEnableNamespaces()) {
413         xsi = ctx.getXsiNamespace().name;
414         dns = elementName == null && ctx.getDefaultNamespace() != null ? ctx.getDefaultNamespace().name : null;
415         elementNs = elementName == null ? dns : ns;
416         if (elementName == null)
417            elementNamespace = null;
418      }
419
420      // Render the start tag.
421      if (! isCollapsed) {
422         out.oTag(indent, elementNs, en, encodeEn);
423         if (addNamespaceUris) {
424            out.attr((String)null, "xmlns", ctx.getDefaultNamespace().getUri());
425
426            for (Namespace n : ctx.getNamespaces())
427               out.attr("xmlns", n.getName(), n.getUri());
428
429            Namespace xsiNs = ctx.getXsiNamespace();
430            if (xsiNs != null)
431               out.attr("xmlns", xsiNs.name, xsiNs.uri);
432         }
433         if (elementName != null && ctx.isAddJsonTypeAttrs() && (ctx.isAddJsonStringTypeAttrs() || ! ts.equals("string")))
434            out.attr(dns, "type", ts);
435         if (classAttr != null)
436            out.attr(dns, "_class", classAttr);
437         if (o == null) {
438            if (! isNullTag)
439               out.attr(xsi, "nil", "true");
440            if ((gType.isBoolean() || gType.isNumber()) && ! gType.isNullable())
441               o = gType.getPrimitiveDefault();
442         }
443
444         if (o != null && !(gType.isMap() || gType.isBean() || gType.hasToObjectMapMethod()))
445            out.append('>');
446
447         if (cr && !(gType.isMap() || gType.isBean() || gType.hasToObjectMapMethod()))
448            out.nl();
449      }
450
451      boolean hasChildren = true;
452
453      // Render the tag contents.
454      if (o != null) {
455         if (gType.isUri() || (pMeta != null && pMeta.isUri()))
456            out.appendUri(o);
457         else if (gType.isCharSequence() || gType.isChar())
458            out.encodeText(ctx.trim(o));
459         else if (gType.isNumber() || gType.isBoolean())
460            out.append(o);
461         else if (gType.isMap() || (wType != null && wType.isMap())) {
462            if (o instanceof BeanMap)
463               hasChildren = serializeBeanMap(out, (BeanMap)o, elementNamespace, ctx, isCollapsed);
464            else
465               hasChildren = serializeMap(out, (Map)o, gType, ctx);
466         }
467         else if (gType.hasToObjectMapMethod())
468            hasChildren = serializeMap(out, gType.toObjectMap(o), gType, ctx);
469         else if (gType.isBean())
470            hasChildren = serializeBeanMap(out, bc.forBean(o), elementNamespace, ctx, isCollapsed);
471         else if (gType.isCollection() || (wType != null && wType.isCollection())) {
472            if (isCollapsed)
473               ctx.indent--;
474            serializeCollection(out, (Collection)o, gType, ctx, pMeta);
475            if (isCollapsed)
476               ctx.indent++;
477         }
478         else if (gType.isArray()) {
479            if (isCollapsed)
480               ctx.indent--;
481            serializeCollection(out, toList(gType.getInnerClass(), o), gType, ctx, pMeta);
482            if (isCollapsed)
483               ctx.indent++;
484         }
485         else
486            out.encodeText(o);
487      }
488
489      ctx.pop();
490
491      // Render the end tag.
492      if (! isCollapsed) {
493         if (o == null || ! hasChildren)
494            out.append('/').append('>').nl();
495         else
496            out.i(cr ? indent : 0).eTag(elementNs, en, encodeEn).nl();
497      }
498
499      return out;
500   }

502   private boolean serializeMap(XmlSerializerWriter out, Map m, ClassMeta<?> type, XmlSerializerContext ctx) throws IOException, SerializeException {
503
504      m = ctx.sort(m);
505
506      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
507
508      boolean hasChildren = false;
509      for (Iterator i = m.entrySet().iterator(); i.hasNext();) {
510         Map.Entry e = (Map.Entry)i.next();
511
512         Object k = e.getKey();
513         if (k == null) {
514            k = "\u0000";
515         } else {
516            k = ctx.generalize(k, keyType);
517            if (ctx.isTrimStrings() && k instanceof String)
518               k = k.toString().trim();
519         }
520
521         Object value = e.getValue();
522
523         if (! hasChildren) {
524            hasChildren = true;
525            out.append('>').nl();
526         }
527         serializeAnything(out, value, valueType, ctx, k.toString(), null, false, NORMAL, null);
528      }
529      return hasChildren;
530   }

532   private boolean serializeBeanMap(XmlSerializerWriter out, BeanMap m, Namespace elementNs, XmlSerializerContext ctx, boolean isCollapsed) throws IOException, SerializeException {
533      boolean hasChildren = false;
534      BeanMeta bm = m.getMeta();
535
536      Map<String,BeanPropertyMeta> xmlAttrs = bm.getXmlMeta().getXmlAttrProperties();
537      Object content = null;
538      for (BeanPropertyMeta p : xmlAttrs.values()) {
539
540         String key = p.getName();
541         Object value = null;
542         try {
543            value = p.get(m);
544         } catch (StackOverflowError e) {
545            throw e;
546         } catch (Throwable x) {
547            ctx.addBeanGetterWarning(p, x);
548         }
549
550         if (ctx.canIgnoreValue(p.getClassMeta(), key, value))
551            continue;
552
553         Namespace ns = (ctx.isEnableNamespaces() && p.getXmlMeta().getNamespace() != elementNs ? p.getXmlMeta().getNamespace() : null);
554
555         if (p.isBeanUri() || p.isUri())
556            out.attrUri(ns, key, value);
557         else
558            out.attr(ns, key, value);
559      }
560
561      boolean hasContent = false;
562
563      for (BeanMapEntry p : (Set<BeanMapEntry>)m.entrySet()) {
564         BeanPropertyMeta pMeta = p.getMeta();
565         XmlFormat xf = pMeta.getXmlMeta().getXmlFormat();
566
567         if (xf == CONTENT) {
568            content = p.getValue();
569            hasContent = true;
570         } else if (xf == ATTR) {
571            // Do nothing
572         } else {
573            String key = p.getKey();
574            Object value = null;
575            try {
576               value = p.getValue();
577            } catch (StackOverflowError e) {
578               throw e;
579            } catch (Throwable x) {
580               ctx.addWarning("Could not call getValue() on property ''{0}'', {1}", key, x.getLocalizedMessage());
581            }
582
583            if (ctx.canIgnoreValue(pMeta.getClassMeta(), key, value))
584               continue;
585
586            if (! hasChildren) {
587               hasChildren = true;
588               out.appendIf(! isCollapsed, '>').nl();
589            }
590            serializeAnything(out, value, pMeta.getClassMeta(), ctx, key, pMeta.getXmlMeta().getNamespace(), false, pMeta.getXmlMeta().getXmlFormat(), pMeta);
591         }
592      }
593      if ((! hasContent) || ctx.canIgnoreValue(string(), null, content))
594         return hasChildren;
595      out.append('>').cr(ctx.indent);
596
597      // Serialize XML content.
598      XmlContentHandler h = bm.getXmlMeta().getXmlContentHandler();
599      if (h != null)
600         try {
601            h.serialize(out, m.getBean());
602         } catch (Exception e) {
603            throw new SerializeException(e);
604         }
605      else
606         out.encodeText(content);
607      out.nl();
608      return true;
609   }

611   private XmlSerializerWriter serializeCollection(XmlSerializerWriter out, Collection c, ClassMeta<?> type, XmlSerializerContext ctx, BeanPropertyMeta<?> ppMeta) throws IOException, SerializeException {
612
613      c = ctx.sort(c);
614
615      ClassMeta<?> elementType = type.getElementType();
616
617      String eName = null;
618      Namespace eNs = null;
619
620      if (ppMeta != null) {
621         eName = ppMeta.getXmlMeta().getChildName();
622         eNs = ppMeta.getXmlMeta().getNamespace();
623      }
624
625      if (eName == null) {
626         eName = type.getXmlMeta().getChildName();
627         eNs = type.getXmlMeta().getNamespace();
628      }
629
630      if (eName == null && ! elementType.isObject()) {
631         eName = elementType.getXmlMeta().getElementName();
632         eNs = elementType.getXmlMeta().getNamespace();
633      }
634
635      for (Iterator i = c.iterator(); i.hasNext();) {
636         Object value = i.next();
637         serializeAnything(out, value, elementType, ctx, eName, eNs, false, NORMAL, null);
638      }
639      return out;
640   }
