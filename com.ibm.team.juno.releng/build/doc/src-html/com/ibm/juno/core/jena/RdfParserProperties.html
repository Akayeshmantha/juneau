009package com.ibm.juno.core.jena;
010
011import com.ibm.juno.core.*;
012import com.ibm.juno.core.parser.*;
013
014/**
015 * Configurable properties on the {@link RdfParser} class.
016 * <p>
017 *    Use the {@link RdfParser#setProperty(String, Object)} method to set property values.
018 * <p>
019 *    In addition to these properties, the following properties are also applicable for {@link RdfParser}.
020 * <ul>
021 *    <li>{@link RdfProperties}
022 *    <li>{@link ParserProperties}
023 *    <li>{@link BeanContextProperties}
024 * </ul>
025 *
026 * @author James Bognar (jbognar@us.ibm.com)
027 */
028public final class RdfParserProperties extends RdfProperties implements Cloneable {
029
030
031   /**
032    * Trim whitespace from text elements ({@link Boolean}, default=<jk>false</jk>).
033    * <p>
034    * If <jk>true</jk>, whitespace in text elements will be automatically trimmed.
035    */
036   public static final String RDF_trimWhitespace = "RdfParser.trimWhitespace";
037
038   boolean trimWhitespace = false;
039
040   /**
041    * Sets the specified property value.
042    * @param property The property name.
043    * @param value The property value.
044    * @return <jk>true</jk> if property name was valid and property was set.
045    */
046   @Override /* RdfProperties */
047   public boolean setProperty(String property, Object value) {
048      if (property.equals(RDF_trimWhitespace))
049         trimWhitespace = BeanContext.DEFAULT.convertToType(value, boolean.class);
050      else
051         return super.setProperty(property, value);
052      return true;
053   }
054
055
056   //--------------------------------------------------------------------------------
057   // Overridden methods
058   //--------------------------------------------------------------------------------
059
060   @Override /* Cloneable */
061   public RdfParserProperties clone() {
062      try {
063         return (RdfParserProperties)super.clone();
064      } catch (CloneNotSupportedException e) {
065         throw new RuntimeException(e); // Shouldn't happen.
066      }
067   }
068}
