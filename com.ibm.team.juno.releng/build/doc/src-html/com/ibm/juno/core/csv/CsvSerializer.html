009package com.ibm.juno.core.csv;
010
011import java.io.*;
012import java.util.*;
013
014import com.ibm.juno.core.*;
015import com.ibm.juno.core.annotation.*;
016import com.ibm.juno.core.serializer.*;
017
018/**
019 * TODO - Work in progress.  CSV serializer.
020 */
021@Produces("text/csv")
022@SuppressWarnings({"unchecked","rawtypes"})
023public final class CsvSerializer extends WriterSerializer {
024
025   //--------------------------------------------------------------------------------
026   // Overridden methods
027   //--------------------------------------------------------------------------------
028
029   @Override /* Serializer */
030   protected void doSerialize(Object o, Writer out, SerializerContext ctx) throws IOException, SerializeException {
031      BeanContext bc = ctx.getBeanContext();
032      ClassMeta cm = bc.getClassMetaForObject(o);
033      Collection l = null;
034      if (cm.isArray()) {
035         l = Arrays.asList((Object[])o);
036      } else
037         l = (Collection)o;
038      if (l.size() > 0) {
039         ClassMeta entryType = bc.getClassMetaForObject(l.iterator().next());
040         if (entryType.isBean()) {
041            BeanMeta<?> bm = entryType.getBeanMeta();
042            int i = 0;
043            for (BeanPropertyMeta pm : bm.getPropertyMetas()) {
044               if (i++ > 0)
045                  out.append(',');
046               append(out, pm.getName());
047            }
048            out.append('\n');
049            for (Object o2 : l) {
050               i = 0;
051               BeanMap bean = bc.forBean(o2);
052               for (BeanPropertyMeta pm : bm.getPropertyMetas()) {
053                  if (i++ > 0)
054                     out.append(',');
055                  append(out, pm.get(bean));
056               }
057               out.append('\n');
058            }
059         }
060
061      }
062   }
063
064   private void append(Writer w, Object o) throws IOException {
065      if (o == null)
066         w.append("null");
067      else {
068         String s = o.toString();
069         boolean mustQuote = false;
070         for (int i = 0; i < s.length() && ! mustQuote; i++) {
071            char c = s.charAt(i);
072            if (Character.isWhitespace(c) || c == ',')
073               mustQuote = true;
074         }
075         if (mustQuote)
076            w.append('"').append(s).append('"');
077         else
078            w.append(s);
079      }
080   }
081
082   @Override /* Serializer */
083   public CsvSerializer clone() {
084      try {
085         return (CsvSerializer)super.clone();
086      } catch (CloneNotSupportedException e) {
087         throw new RuntimeException(e); // Shouldn't happen.
088      }
089   }
090}
