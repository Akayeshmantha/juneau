009package com.ibm.juno.core.parser;
010
011import java.io.*;
012
013import com.ibm.juno.core.*;
014import com.ibm.juno.core.annotation.*;
015
016/**
017 * Subclass of {@link Parser} for byte-based parsers.
018 *
019 *
020 * <h6 class='topic'>Description</h6>
021 * <p>
022 *    This class is typically the parent class of all byte-based parsers.
023 *    It has 1 abstract method to implement...
024 * <ul>
025 *    <li><code>parse(InputStream, ClassMeta, ParserContext)</code>
026 * </ul>
027 *
028 *
029 * <h6 class='topic'>@Consumes annotation</h6>
030 * <p>
031 *    The media types that this parser can handle is specified through the {@link Consumes @Consumes} annotation.
032 * <p>
033 *    However, the media types can also be specified programmatically by overriding the {@link #getMediaTypes()} method.
034 *
035 *
036 * @author James Bognar (jbognar@us.ibm.com)
037 */
038public abstract class InputStreamParser extends Parser<InputStream> {
039
040   @Override /* Parser */
041   public boolean isReaderParser() {
042      return false;
043   }
044
045   //--------------------------------------------------------------------------------
046   // Abstract methods
047   //--------------------------------------------------------------------------------
048
049   @Override /* Parser */
050   protected abstract <T> T doParse(InputStream in, int estimatedSize, ClassMeta<T> type, ParserContext ctx) throws ParseException, IOException;
051
052}
