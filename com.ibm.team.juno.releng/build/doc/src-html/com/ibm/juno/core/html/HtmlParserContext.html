009package com.ibm.juno.core.html;
010
011import java.io.*;
012import java.lang.reflect.*;
013
014import javax.xml.stream.*;
015
016import com.ibm.juno.core.*;
017import com.ibm.juno.core.parser.*;
018import com.ibm.juno.core.utils.*;
019
020/**
021 * Context object that lives for the duration of a single parsing of {@link HtmlParser}.
022 * <p>
023 *    This class is NOT thread safe.  It is meant to be discarded after one-time use.
024 * </p>
025 *
026 * @author James Bognar (jbognar@us.ibm.com)
027 */
028public final class HtmlParserContext extends ParserContext {
029
030   private XMLEventReader xmlEventReader;
031
032   /**
033    * Create a new parser context with the specified options.
034    *
035    * @param beanContext The bean context being used.
036    * @param pp The default generic parser properties.
037    * @param hpp The default HTML parser properties.
038    * @param properties The override properties.
039    * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
040    * @param outer The outer object for instantiating top-level non-static inner classes.
041    */
042   public HtmlParserContext(BeanContext beanContext, ParserProperties pp, HtmlParserProperties hpp, ObjectMap properties, Method javaMethod, Object outer) {
043      super(beanContext, pp, properties, javaMethod, outer);
044   }
045
046   /**
047    * Wraps the specified reader in an {@link XMLEventReader}.
048    * This event reader gets closed by the {@link #close()} method.
049    *
050    * @param in The reader to read from.
051    * @param estimatedSize The estimated size of the input.  If <code>-1</code>, uses a default size of <code>8196</code>.
052    * @return A new XML event reader using a new {@link XMLInputFactory}.
053    * @throws ParseException
054    */
055   final XMLEventReader getReader(Reader in, int estimatedSize) throws ParseException {
056      try {
057         in = IOUtils.getBufferedReader(in, estimatedSize);
058         XMLInputFactory factory = XMLInputFactory.newInstance();
059         factory.setProperty(XMLInputFactory.IS_REPLACING_ENTITY_REFERENCES, false);
060         this.xmlEventReader = factory.createXMLEventReader(in);
061      } catch (Error e) {
062         throw new ParseException(e.getLocalizedMessage());
063      } catch (XMLStreamException e) {
064         throw new ParseException(e);
065      }
066      return xmlEventReader;
067   }
068
069   @Override /* ParserContext */
070   public void close() throws ParseException {
071      if (xmlEventReader != null) {
072         try {
073            xmlEventReader.close();
074         } catch (XMLStreamException e) {
075            throw new ParseException(e);
076         }
077      }
078      super.close();
079   }
080}
