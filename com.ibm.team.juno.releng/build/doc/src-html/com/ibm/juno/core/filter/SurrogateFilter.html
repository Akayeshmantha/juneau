020/**
021 * Specialized {@link PojoFilter} for surrogate classes.
022 * <p>
023 * Surrogate classes are used in place of other classes during serialization.
024 * For example, you may want to use a surrogate class to change the names or order of bean
025 * properties on a bean.
026 * <p>
027 * The following is an example of a surrogate class change changes a property name:
028 * <p class='bcode'>
029 *    <jk>public class</jk> SurrogateClass {
030 *       <jk>public</jk> String surrogateField;  <jc>// New bean property</jc>
031 *
032 *       <jk>public</jk> SurrogateClass(NormalClass normalClass) {
033 *          <jk>this</jk>.surrogateField = normalClass.normalField;
034 *       }
035 *    }
036 * </p>
037 * <p>
038 * Optionally, a public static method can be used to unfilter a class during parsing:
039 * <p class='bcode'>
040 *    <jk>public class</jk> SurrogateClass {
041 *       ...
042 *       <jk>public static</jk> NormalClass <jsm>toNormalClass</jsm>(SurrogateClass surrogateClass) {
043 *          <jk>return new</jk> NormalClass(surrogateClass.filteredField);
044 *       }
045 *    }
046 * </p>
047 * <p>
048 * Surrogate classes must conform to the following:
049 * <ul>
050 *    <li>It must have a one or more public constructors that take in a single parameter whose type is the normal types.
051 *       (It is possible to define a class as a surrogate for multiple class types by using multiple constructors with
052 *       different parameter types).
053 *    <li>It optionally can have a public static method that takes in a single parameter whose type is the filtered type
054 *       and returns an instance of the normal type.  This is called the unfilter method.  The method can be called anything.
055 *    <li>If an unfilter method is present, the class must also contain a no-arg constructor (so that the filtered class
056 *       can be instantiated by the parser before being converted into the normal class by the unfilter method).
057 * </ul>
058 * <p>
059 * Surrogate classes are associated with serializers and parsers using the {@link CoreApi#addFilters(Class...)} method.
060 * <p class='bcode'>
061 *    <ja>@Test</ja>
062 *    <jk>public void</jk> test() <jk>throws</jk> Exception {
063 *       JsonSerializer s = <jk>new</jk> JsonSerializer.Simple().addFilters(Surrogate.<jk>class</jk>);
064 *       JsonParser p = <jk>new</jk> JsonParser().addFilters(Surrogate.<jk>class</jk>);
065 *       String r;
066 *       Normal n = Normal.<jsm>create</jsm>();
067 *
068 *       r = s.serialize(n);
069 *       assertEquals(<js>"{f2:'f1'}"</js>, r);
070 *
071 *       n = p.parse(r, Normal.<jk>class</jk>);
072 *       assertEquals(<js>"f1"</js>, n.f1);
073 *    }
074 *
075 *    <jc>// The normal class</jc>
076 *    <jk>public class</jk> Normal {
077 *       <jk>public</jk> String f1;
078 *
079 *       <jk>public static</jk> Normal <jsm>create</jsm>() {
080 *          Normal n = <jk>new</jk> Normal();
081 *          n.f1 = <js>"f1"</js>;
082 *          <jk>return</jk> n;
083 *       }
084 *    }
085 *
086 *    <jc>// The surrogate class</jc>
087 *    <jk>public static class</jk> Surrogate {
088 *       <jk>public</jk> String f2;
089 *
090 *       <jc>// Surrogate constructor</jc>
091 *       <jk>public</jk> Surrogate(Normal n) {
092 *          f2 = n.f1;
093 *       }
094 *
095 *       <jc>// Constructor used during parsing (only needed if unfilter method specified)</jc>
096 *       <jk>public</jk> Surrogate() {}
097 *
098 *       <jc>// Unfilter method (optional)</jc>
099 *       <jk>public static</jk> Normal <jsm>toNormal</jsm>(Surrogate f) {
100 *          Normal n = <jk>new</jk> Normal();
101 *          n.f1 = f.f2;
102 *          <jk>return</jk> n;
103 *       }
104 *    }
105 * </p>
106 * <p>
107 * It should be noted that a surrogate class is functionally equivalent to the following {@link PojoFilter} implementation:
108 * <p class='bcode'>
109 *    <jk>public static class</jk> SurrogateFilter <jk>extends</jk> PojoFilter&lt;Normal,Surrogate&gt; {
110 *       <jk>public</jk> Surrogate filter(Normal n) <jk>throws</jk> SerializeException {
111 *          <jk>return new</jk> Surrogate(n);
112 *       }
113 *       <jk>public</jk> Normal unfilter(Surrogate s, ClassMeta<?> hint) <jk>throws</jk> ParseException {
114 *          <jk>return</jk> Surrogate.<jsm>toNormal</jsm>(s);
115 *       }
116 *    }
117 * </p>
118 *
119 * @author James Bognar (jbognar@us.ibm.com)
120 * @param <T> The class type that this filter applies to.
121 * @param <F> The filtered class type.
122 */
123public class SurrogateFilter<T,F> extends PojoFilter<T,F> {
124
125   private Constructor<F> constructor;   // public F(T t);
126   private Method unfilterMethod;        // public static T valueOf(F f);
127
128   /**
129    * Constructor.
130    *
131    * @param forClass The normal class.
132    * @param constructor The constructor on the surrogate class that takes the normal class as a parameter.
133    * @param unfilterMethod The static method that converts surrogate objects into normal objects.
134    */
135   protected SurrogateFilter(Class<T> forClass, Constructor<F> constructor, Method unfilterMethod) {
136      super(forClass, constructor.getDeclaringClass());
137      this.constructor = constructor;
138      this.unfilterMethod = unfilterMethod;
139   }
140
141   /**
142    * Given the specified surrogate class, return the list of POJO filters.
143    * A filter is returned for each public 1-arg constructor found.
144    * Returns an empty list if no public 1-arg constructors are found.
145    *
146    * @param c The surrogate class.
147    * @return The list of POJO filters that apply to this class.
148    */
149   @SuppressWarnings({"unchecked", "rawtypes"})
150   public static List<SurrogateFilter<?,?>> findFilters(Class<?> c) {
151      List<SurrogateFilter<?,?>> l = new LinkedList<SurrogateFilter<?,?>>();
152      for (Constructor<?> cc : c.getConstructors()) {
153         if (cc.getAnnotation(BeanIgnore.class) == null) {
154            Class<?>[] pt = cc.getParameterTypes();
155
156            // Only constructors with one parameter.
157            // Ignore instance class constructors.
158            if (pt.length == 1 && pt[0] != c.getDeclaringClass()) {
159               int mod = cc.getModifiers();
160               if (Modifier.isPublic(mod)) {  // Only public constructors.
161
162                  // Find the unfilter method if there is one.
163                  Method unfilterMethod = null;
164                  for (Method m : c.getMethods()) {
165                     if (pt[0].equals(m.getReturnType())) {
166                        Class<?>[] mpt = m.getParameterTypes();
167                        if (mpt.length == 1 && mpt[0].equals(c)) { // Only methods with one parameter and where the return type matches this class.
168                           int mod2 = m.getModifiers();
169                           if (Modifier.isPublic(mod2) && Modifier.isStatic(mod2))  // Only public static methods.
170                              unfilterMethod = m;
171                        }
172                     }
173                  }
174
175                  l.add(new SurrogateFilter(pt[0], cc, unfilterMethod));
176               }
177            }
178         }
179      }
180      return l;
181   }
182
183   @Override /* PojoFilter */
184   public F filter(T o) throws SerializeException {
185      try {
186         return constructor.newInstance(o);
187      } catch (Exception e) {
188         throw new SerializeException(e);
189      }
190   }
191
192   @Override /* PojoFilter */
193   @SuppressWarnings("unchecked")
194   public T unfilter(F f, ClassMeta<?> hint) throws ParseException {
195      if (unfilterMethod == null)
196         throw new ParseException("static valueOf({0}) method not implement on surrogate class ''{1}''", f.getClass().getName(), getNormalClass().getName());
197      try {
198         return (T)unfilterMethod.invoke(null, f);
199      } catch (Exception e) {
200         throw new ParseException(e);
201      }
202   }
203}
