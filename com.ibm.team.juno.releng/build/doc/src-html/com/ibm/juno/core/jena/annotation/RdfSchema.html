009package com.ibm.juno.core.jena.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015
016/**
017 * Identifies the default RDF namespaces at the package level.
018 *
019 * @author James Bognar (jbognar@us.ibm.com)
020 */
021@Documented
022@Target(PACKAGE)
023@Retention(RUNTIME)
024@Inherited
025public @interface RdfSchema {
026
027   /**
028    * Sets the default RDF prefix for all classes in this and child packages.
029    * <p>
030    *    Must either be matched with a {@link #namespace()} annotation, or an {@link #rdfNs()} mapping with the
031    *    same {@link RdfNs#prefix} value.
032    * </p>
033    */
034   public String prefix() default "";
035
036   /**
037    * Sets the default RDF namespace URL for all classes in this and child packages.
038    * <p>
039    *    Must either be matched with a {@link #prefix()} annotation, or an {@link #rdfNs()} mapping with the
040    *    same {@link RdfNs#namespaceURI} value.
041    * </p>
042    */
043   public String namespace() default "";
044
045   /**
046    * Lists all namespace mappings to be used on all classes within this package.
047    * <p>
048    *    The purpose of this annotation is to allow namespace mappings to be defined in a single location
049    *    and referred to by name through just the {@link Rdf#prefix()} annotation.
050    * <p>
051    *    Inherited by child packages.
052    *
053    * <dl>
054    *    <dt>Example:</dt>
055    *    <dd>
056    * <p>
057    *    Contents of <code>package-info.java</code>...
058    * </p>
059    * <p class='bcode'>
060    *    <jc>// XML namespaces used within this package.</jc>
061    *    <ja>@RdfSchema</ja>(prefix=<js>"ab"</js>,
062    *       namespaces={
063    *          <ja>@RdfNs</ja>(prefix=<js>"ab"</js>, namespaceURI=<js>"http://www.ibm.com/addressBook/"</js>),
064    *          <ja>@RdfNs</ja>(prefix=<js>"per"</js>, namespaceURI=<js>"http://www.ibm.com/person/"</js>),
065    *          <ja>@RdfNs</ja>(prefix=<js>"addr"</js>, namespaceURI=<js>"http://www.ibm.com/address/"</js>),
066    *          <ja>@RdfNs</ja>(prefix=<js>"mail"</js>, namespaceURI="<js>http://www.ibm.com/mail/"</js>)
067    *       }
068    *    )
069    *    <jk>package</jk> com.ibm.sample.addressbook;
070    *    <jk>import</jk> com.ibm.juno.core.rdf.annotation.*;
071    * </p>
072    * <p>
073    *    Class in package using defined namespaces...
074    * </p>
075    * <p class='bcode'>
076    *    <jk>package</jk> com.ibm.sample.addressbook;
077    *
078    *    <jc>// Bean class, override "ab" namespace on package.</jc>
079    *    <ja>@Rdf</ja>(prefix=<js>"addr"</js>)
080    *    <jk>public class</jk> Address {
081    *
082    *       <jc>// Bean property, use "addr" namespace on class.</jc>
083    *       <jk>public int</jk> <jf>id</jf>;
084    *
085    *       <jc>// Bean property, override with "mail" namespace.</jc>
086    *       <ja>@Rdf</ja>(prefix=<js>"mail"</js>)
087    *       <jk>public</jk> String <jf>street</jf>, <jf>city</jf>, <jf>state</jf>;
088    *    }
089    * </p>
090    *    </dd>
091    * </dl>
092    */
093   public RdfNs[] rdfNs() default {};
094}
