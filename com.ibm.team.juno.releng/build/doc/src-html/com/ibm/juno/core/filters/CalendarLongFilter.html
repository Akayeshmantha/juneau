009package com.ibm.juno.core.filters;
010
011import java.util.*;
012
013import com.ibm.juno.core.*;
014import com.ibm.juno.core.filter.*;
015import com.ibm.juno.core.parser.*;
016
017/**
018 * Transforms {@link Calendar Calendars} to {@link Long Longs} using {@code Calender.getTime().getTime()}.
019 * <p>
020 *    TODO:  This class does not handle timezones correctly when parsing {@code GregorianCalendar} objects.
021 *
022 * @author James Bognar (jbognar@us.ibm.com)
023 */
024public class CalendarLongFilter extends PojoFilter<Calendar,Long> {
025
026   /**
027    * Converts the specified {@link Calendar} to a {@link Long}.
028    */
029   @Override /* PojoFilter */
030   public Long filter(Calendar o) {
031      return o.getTime().getTime();
032   }
033
034   /**
035    * Converts the specified {@link Long} to a {@link Calendar}.
036    */
037   @Override /* PojoFilter */
038   @SuppressWarnings("unchecked")
039   public Calendar unfilter(Long o, ClassMeta<?> hint) throws ParseException {
040      ClassMeta<? extends Calendar> tt;
041      try {
042         if (hint == null || ! hint.canCreateNewInstance())
043            hint = getBeanContext().getClassMeta(GregorianCalendar.class);
044         tt = (ClassMeta<? extends Calendar>)hint;
045         Calendar c = tt.newInstance();
046         c.setTimeInMillis(o);
047         return c;
048      } catch (Exception e) {
049         throw new ParseException(e);
050      }
051   }
052}
