009package com.ibm.juno.core.urlencoding.annotation;
010
011import static java.lang.annotation.ElementType.*;
012import static java.lang.annotation.RetentionPolicy.*;
013
014import java.lang.annotation.*;
015
016import com.ibm.juno.core.urlencoding.*;
017
018/**
019 * Annotation that can be applied to classes, fields, and methods to tweak how
020 * they are handled by {@link UrlEncodingSerializer} and {@link UrlEncodingParser}.
021 *
022 * @author James Bognar (jbognar@us.ibm.com)
023 */
024@Documented
025@Target({TYPE})
026@Retention(RUNTIME)
027@Inherited
028public @interface UrlEncoding {
029
030   /**
031    * When true, bean properties of type array or Collection will be expanded into multiple key=value pairings.
032    * <p>
033    * This annotation is identical in behavior to using the {@link UrlEncodingProperties#URLENC_expandedParams} 
034    * property, but applies to only instances of this bean.  
035    */
036   boolean expandedParams() default false;
037}
