009package com.ibm.juno.core.jena;
010
011import com.ibm.juno.core.*;
012import com.ibm.juno.core.serializer.*;
013
014/**
015 * Configurable properties on the {@link RdfSerializer} class.
016 * <p>
017 *    Use the {@link RdfSerializer#setProperty(String, Object)} method to set property values.
018 * <p>
019 *    In addition to these properties, the following properties are also applicable for {@link RdfSerializer}.
020 * <ul>
021 *    <li>{@link RdfProperties}
022 *    <li>{@link SerializerProperties}
023 *    <li>{@link BeanContextProperties}
024 * </ul>
025 *
026 * @author James Bognar (jbognar@us.ibm.com)
027 */
028public final class RdfSerializerProperties extends RdfProperties implements Cloneable {
029
030   /**
031    * Add XSI data types to non-<code>String</code> literals ({@link Boolean}, default=<jk>false</jk>).
032    */
033   public static final String RDF_addLiteralTypes = "RdfSerializer.addLiteralTypes";
034
035   /**
036    * Add RDF root identifier property to root node ({@link Boolean}, default=<jk>false</jk>).
037    * <p>
038    *    When enabled an RDF property <code>http://www.ibm.com/juno/root</code> is added with a value of <js>"true"</js>
039    *       to identify the root node in the graph.
040    *    This helps locate the root node during parsing.
041    * <p>
042    *    If disabled, the parser has to search through the model to find any resources without
043    *       incoming predicates to identify root notes, which can introduce a considerable performance
044    *       degradation.
045    */
046   public static final String RDF_addRootProperty = "RdfSerializer.addRootProperty";
047
048   boolean addLiteralTypes = false, addRootProperty = false;
049
050   /**
051    * Sets the specified property value.
052    * @param property The property name.
053    * @param value The property value.
054    * @return <jk>true</jk> if property name was valid and property was set.
055    */
056   @Override /* RdfProperties */
057   public boolean setProperty(String property, Object value) {
058      if (property.equals(RDF_addLiteralTypes))
059         addLiteralTypes = BeanContext.DEFAULT.convertToType(value, boolean.class);
060      else if (property.equals(RDF_addRootProperty))
061         addRootProperty = BeanContext.DEFAULT.convertToType(value, boolean.class);
062      else
063         return super.setProperty(property, value);
064      return true;
065   }
066
067
068   //--------------------------------------------------------------------------------
069   // Overridden methods
070   //--------------------------------------------------------------------------------
071
072   @Override /* Cloneable */
073   public RdfSerializerProperties clone() {
074      try {
075         return (RdfSerializerProperties)super.clone();
076      } catch (CloneNotSupportedException e) {
077         throw new RuntimeException(e); // Shouldn't happen.
078      }
079   }
080}
