022/**
023 * Serializes POJO metadata to HTTP responses as JSON.
024 *
025 *
026 * <h6 class='topic'>Media types</h6>
027 * <p>
028 *    Handles <code>Accept</code> types: <code>application/json+schema, text/json+schema</code>
029 * <p>
030 *    Produces <code>Content-Type</code> types: <code>application/json</code>
031 *
032 *
033 * <h6 class='topic'>Description</h6>
034 * <p>
035 *    Produces the JSON-schema for the JSON produced by the {@link JsonSerializer} class with the same properties.
036 *
037 *
038 * @author James Bognar (jbognar@us.ibm.com)
039 */
040@Produces(value={"application/json+schema","text/json+schema"},contentType="application/json")
041public final class JsonSchemaSerializer extends JsonSerializer {
042
043   /**
044    * Constructor.
045    */
046   public JsonSchemaSerializer() {
047      setProperty(SERIALIZER_detectRecursions, true);
048      setProperty(SERIALIZER_ignoreRecursions, true);
049   }
050
051   //--------------------------------------------------------------------------------
052   // Overridden methods
053   //--------------------------------------------------------------------------------
054
055   @Override /* JsonSerializer */
056   protected void doSerialize(Object o, Writer out, SerializerContext ctx) throws IOException, SerializeException {
057      JsonSerializerContext jctx = (JsonSerializerContext)ctx;
058      ObjectMap schema = getSchema(ctx.getBeanContext().getClassMetaForObject(o), jctx, "root", null);
059      serializeAnything(jctx.getWriter(out), schema, null, jctx, "root", null);
060   }
061
062   /*
063    * Creates a schema representation of the specified class type.
064    *
065    * @param eType The class type to get the schema of.
066    * @param ctx Serialize context used to prevent infinite loops.
067    * @param attrName The name of the current attribute.
068    * @return A schema representation of the specified class.
069    * @throws SerializeException If a problem occurred trying to convert the output.
070    */
071   @SuppressWarnings({ "unchecked", "rawtypes" })
072   private ObjectMap getSchema(ClassMeta<?> eType, JsonSerializerContext ctx, String attrName, String[] pNames) throws SerializeException {
073      try {
074
075         ObjectMap out = new ObjectMap();
076
077         if (eType == null)
078            eType = object();
079
080         ClassMeta<?> aType;        // The actual type (will be null if recursion occurs)
081         ClassMeta<?> gType;        // The generic type
082
083         aType = ctx.push(attrName, eType, null);
084
085         gType = eType.getFilteredClassMeta();
086         String type = null;
087
088         if (gType.isEnum() || gType.isCharSequence() || gType.isChar())
089            type = "string";
090         else if (gType.isNumber())
091            type = "number";
092         else if (gType.isBoolean())
093            type = "boolean";
094         else if (gType.isBean() || gType.isMap())
095            type = "object";
096         else if (gType.isCollection() || gType.isArray())
097            type = "array";
098         else
099            type = "any";
100
101         out.put("type", type);
102         out.put("description", eType.toString());
103         PojoFilter f = eType.getPojoFilter();
104         if (f != null)
105            out.put("filter", f);
106
107         if (aType != null) {
108            if (gType.isEnum())
109               out.put("enum", getEnumStrings((Class<Enum<?>>)gType.getInnerClass()));
110            else if (gType.isCollection() || gType.isArray()) {
111               ClassMeta componentType = gType.getElementType();
112               if (gType.isCollection() && isParentClass(Set.class, gType.getInnerClass()))
113                  out.put("uniqueItems", true);
114               out.put("items", getSchema(componentType, ctx, "items", pNames));
115            } else if (gType.isBean()) {
116               ObjectMap properties = new ObjectMap();
117               BeanMeta bm = ctx.getBeanContext().getBeanMeta(gType.getInnerClass());
118               if (pNames != null)
119                  bm = new BeanMetaFiltered(bm, pNames);
120               for (Iterator<BeanPropertyMeta<?>> i = bm.getPropertyMetas().iterator(); i.hasNext();) {
121                  BeanPropertyMeta p = i.next();
122                  properties.put(p.getName(), getSchema(p.getClassMeta(), ctx, p.getName(), p.getProperties()));
123               }
124               out.put("properties", properties);
125            }
126         }
127         ctx.pop();
128         return out;
129      } catch (StackOverflowError e) {
130         throw e;
131      } catch (Throwable e) {
132         throw new SerializeException("Exception occured trying to process object of type ''{0}''", eType).initCause(e);
133      }
134   }
135
136   @SuppressWarnings({ "unchecked", "rawtypes" })
137   private List<String> getEnumStrings(Class<? extends Enum> c) {
138      List<String> l = new LinkedList<String>();
139      try {
140         for (Object e : EnumSet.allOf(c))
141            l.add(e.toString());
142      } catch (Exception e) {
143         e.printStackTrace();
144      }
145      return l;
146   }
147
148
149   @Override /* Lockable */
150   public JsonSchemaSerializer lock() {
151      super.lock();
152      return this;
153   }
154}
