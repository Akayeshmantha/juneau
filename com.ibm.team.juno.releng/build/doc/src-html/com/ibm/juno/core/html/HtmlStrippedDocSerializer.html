009package com.ibm.juno.core.html;
010
011import java.io.*;
012import java.lang.reflect.*;
013import java.util.*;
014
015import com.ibm.juno.core.annotation.*;
016import com.ibm.juno.core.serializer.*;
017
018/**
019 * Serializes POJOs to HTTP responses as stripped HTML.
020 *
021 *
022 * <h6 class='topic'>Media types</h6>
023 * <p>
024 *    Handles <code>Accept</code> types: <code>text/html+stripped</code>
025 * <p>
026 *    Produces <code>Content-Type</code> types: <code>text/html</code>
027 *
028 *
029 * <h6 class='topic'>Description</h6>
030 * <p>
031 *    Produces the same output as {@link HtmlDocSerializer}, but without the header and body tags and page title and description.
032 *    Used primarily for JUnit testing the {@link HtmlDocSerializer} class.
033 *
034 *
035 * @author James Bognar (jbognar@us.ibm.com)
036 */
037@Produces(value="text/html+stripped",contentType="text/html")
038public class HtmlStrippedDocSerializer extends HtmlSerializer {
039
040   //---------------------------------------------------------------------------
041   // Overridden methods
042   //---------------------------------------------------------------------------
043
044   @Override /* Serializer */
045   protected void doSerialize(Object o, Writer out, SerializerContext ctx) throws IOException, SerializeException {
046      HtmlSerializerContext hctx = (HtmlSerializerContext)ctx;
047      HtmlSerializerWriter w = hctx.getWriter(out);
048      if (o == null
049         || (o instanceof Collection && ((Collection<?>)o).size() == 0)
050         || (o.getClass().isArray() && Array.getLength(o) == 0))
051         w.sTag(1, "p").append("No Results").eTag("p").nl();
052      else
053         super.doSerialize(o, w, hctx);
054   }
055}
