030   public static ObjectList getReadableClassNames(Object[] o) {
031      ObjectList l = new ObjectList();
032      for (int i = 0; i < o.length; i++)
033         l.add(o[i] == null ? "null" : getReadableClassName(o[i].getClass()));
034      return l;
035   }

037   /**
038    * Shortcut for calling <code><jsm>getReadableClassName</jsm>(c.getName())</code>
039    *
040    * @param c The class.
041    * @return A readable class type name, or <jk>null</jk> if parameter is <jk>null</jk>.

043   public static String getReadableClassName(Class<?> c) {
044      if (c == null)
045         return null;
046      return getReadableClassName(c.getName());
047   }

049   /**
050    * Converts the specified class name to a readable form when class name is a special construct like <js>"[[Z"</js>.
051    * <p>
052    * Examples:
053    * <p class='bcode'>
054    *    <jsm>getReadableClassName</jsm>(<js>"java.lang.Object"</js>);  <jc>// Returns "java.lang.Object"</jc>
055    *    <jsm>getReadableClassName</jsm>(<js>"boolean"</js>);  <jc>// Returns "boolean"</jc>
056    *    <jsm>getReadableClassName</jsm>(<js>"[Z"</js>);  <jc>// Returns "boolean[]"</jc>
057    *    <jsm>getReadableClassName</jsm>(<js>"[[Z"</js>);  <jc>// Returns "boolean[][]"</jc>
058    *    <jsm>getReadableClassName</jsm>(<js>"[Ljava.lang.Object;"</js>);  <jc>// Returns "java.lang.Object[]"</jc>
059    *    <jsm>getReadableClassName</jsm>(<jk>null</jk>);  <jc>// Returns null</jc>
060    * </p>
061    *
062    * @param className The class name.
063    * @return A readable class type name, or <jk>null</jk> if parameter is <jk>null</jk>.
064    */
065   public static String getReadableClassName(String className) {
066      if (className == null)
067         return null;
068      if (! StringUtils.startsWith(className, '['))
069         return className;
070      int depth = 0;
071      for (int i = 0; i < className.length(); i++) {
072         if (className.charAt(i) == '[')
073            depth++;
074         else
075            break;
076      }
077      char type = className.charAt(depth);
078      String c;
079      switch (type) {
080         case 'Z': c = "boolean"; break;
081         case 'B': c = "byte"; break;
082         case 'C': c = "char"; break;
083         case 'D': c = "double"; break;
084         case 'F': c = "float"; break;
085         case 'I': c = "int"; break;
086         case 'J': c = "long"; break;
087         case 'S': c = "short"; break;
088         default: c = className.substring(depth+1, className.length()-1);
089      }
090      StringBuilder sb = new StringBuilder(c.length() + 2*depth).append(c);
091      for (int i = 0; i < depth; i++)
092         sb.append("[]");
093      return sb.toString();
094   }

096   /**
097    * Returns <jk>true</jk> if <code>parent</code> is a parent class of <code>child</code>.
098    *
099    * @param parent The parent class.
100    * @param child The child class.
101    * @param strict If <jk>true</jk> returns <jk>false</jk> if the classes are the same.
102    * @return <jk>true</jk> if <code>parent</code> is a parent class of <code>child</code>.
103    */
104   public static boolean isParentClass(Class<?> parent, Class<?> child, boolean strict) {
105      return parent.isAssignableFrom(child) && ((!strict) || ! parent.equals(child));
106   }
107
108   /**
109    * Returns <jk>true</jk> if <code>parent</code> is a parent class or the same as <code>child</code>.
110    *
111    * @param parent The parent class.
112    * @param child The child class.
113    * @return <jk>true</jk> if <code>parent</code> is a parent class or the same as <code>child</code>.
114    */
115   public static boolean isParentClass(Class<?> parent, Class<?> child) {
116      return isParentClass(parent, child, false);
117   }

134   /**
135    * Returns the signature of the specified method.
136    * For no-arg methods, the signature will be a simple string such as <js>"toString"</js>.
137    * For methods with one or more args, the arguments will be fully-qualified class names (e.g. <js>"append(java.util.StringBuilder,boolean)"</js>)
138    *
139    * @param m The methods to get the signature on.
140    * @return The methods signature.
141    */
142   public static String getMethodSignature(Method m) {
143      StringBuilder sb = new StringBuilder(m.getName());
144      Class<?>[] pt = m.getParameterTypes();
145      if (pt.length > 0) {
146         sb.append('(');
147         for (int i = 0; i < pt.length; i++) {
148            if (i > 0)
149               sb.append(',');
150            sb.append(getReadableClassName(pt[i]));
151         }
152         sb.append(')');
153      }
154      return sb.toString();
155   }
