012import java.lang.reflect.*;
013import java.util.*;
014
015import com.ibm.juno.core.*;

021/**
022 * Serializes POJOs to HTTP responses as HTML documents.
023 *
024 *
025 * <h6 class='topic'>Media types</h6>
026 * <p>
027 *    Handles <code>Accept</code> types: <code>text/html</code>
028 * <p>
029 *    Produces <code>Content-Type</code> types: <code>text/html</code>
030 *
031 *
032 * <h6 class='topic'>Description</h6>
033 * <p>
034 *    Same as {@link HtmlSerializer}, except wraps the response in <code><xt>&lt;html&gt;</code>, <code><xt>&lt;head&gt;</code>,
035 *    and <code><xt>&lt;body&gt;</code> tags so that it can be rendered in a browser.
036 *
037 *
038 * <h6 class='topic'>Configurable properties</h6>
039 * <p>
040 *    This class has the following properties associated with it:
041 * <ul>
042 *    <li>{@link HtmlDocSerializerProperties}
043 *    <li>{@link HtmlSerializerProperties}
044 *    <li>{@link SerializerProperties}
045 *    <li>{@link BeanContextProperties}
046 * </ul>
047 *
048 *
049 * @author James Bognar (jbognar@us.ibm.com)
050 */
051@Produces("text/html")
052public class HtmlDocSerializer extends HtmlStrippedDocSerializer {
053
054   // Properties defined in RestServletProperties
055   private static final String
056      REST_method = "RestServlet.method",
057      REST_relativeServletURI = "RestServlet.relativeServletURI";
058
059
060   //--------------------------------------------------------------------------------
061   // Overridden methods
062   //--------------------------------------------------------------------------------
063
064   @Override /* Serializer */
065   protected void doSerialize(Object o, Writer out, SerializerContext ctx) throws IOException, SerializeException {
066
067      HtmlSerializerContext hctx = (HtmlSerializerContext)ctx;
068      HtmlSerializerWriter w = hctx.getWriter(out);
069
070      ObjectMap properties = hctx.getProperties();
071
072      boolean isOptionsPage = properties.containsKey(REST_method) && properties.getString(REST_method).equalsIgnoreCase("OPTIONS");
073
074      // Render the header.
075      w.sTag("html").nl();
076      w.sTag("head").nl();
077
078      String cssUrl = hctx.getCssUrl();
079      if (cssUrl == null)
080         cssUrl = properties.getString(REST_relativeServletURI) + "/style.css";
081
082      w.oTag(1, "style")
083         .attr("type", "text/css")
084         .appendln(">")
085         .append(2, "@import ").q().append(cssUrl).q().appendln(";");
086      if (hctx.isNoWrap())
087         w.appendln("\n* {white-space:nowrap;}");
088      if (hctx.getCssImports() != null)
089         for (String cssImport : hctx.getCssImports())
090            w.append(2, "@import ").q().append(cssImport).q().appendln(";");
091      w.eTag(1, "style").nl();
092      w.eTag("head").nl();
093      w.sTag("body").nl();
094      // Write the title of the page.
095      String title = hctx.getTitle();
096      if (title == null && isOptionsPage)
097         title = "Options";
098      String description = hctx.getDescription();
099      if (title != null)
100         w.oTag(1, "h3").attr("class", "title").append('>').encodeText(title).eTag("h3").nl();
101      if (description != null)
102         w.oTag(1, "h5").attr("class", "description").append('>').encodeText(description).eTag("h5").nl();
103
104      // Write the action links that render above the results.
105      List<Link> actions = new LinkedList<Link>();
106
107      // If this is an OPTIONS request, provide a 'back' link to return to the GET request page.
108      if (! isOptionsPage) {
109         ObjectMap htmlLinks = hctx.getLinks();
110         if (htmlLinks != null) {
111            for (Map.Entry<String,Object> e : htmlLinks.entrySet()) {
112               String uri = e.getValue().toString();
113               if (uri.indexOf("://") == -1 && ! StringUtils.startsWith(uri, '/')) {
114                  StringBuilder sb = new StringBuilder(properties.getString(REST_relativeServletURI));
115                  if (! (uri.isEmpty() || uri.charAt(0) == '?' || uri.charAt(0) == '/'))
116                     sb.append('/');
117                  sb.append(uri);
118                  uri = sb.toString();
119               }
120
121               actions.add(new Link(e.getKey(), uri));
122            }
123         }
124      }
125
126      if (actions.size() > 0) {
127         w.oTag(1, "p").attr("class", "links").append('>').nl();
128         for (Iterator<Link> i = actions.iterator(); i.hasNext();) {
129            Link h = i.next();
130            w.oTag(2, "a").attr("class", "link").attr("href", h.getHref(), true).append('>').append(h.getName()).eTag("a").nl();
131            if (i.hasNext())
132               w.append(3, " - ").nl();
133         }
134         w.eTag(1, "p").nl();
135      }
136
137      hctx.indent = 3;
138
139      // To allow for page formatting using CSS, we encapsulate the data inside two div tags:
140      // <div class='outerdata'><div class='data' id='data'>...</div></div>
141      w.oTag(1, "div").attr("class","outerdata").append('>').nl();
142      w.oTag(2, "div").attr("class","data").attr("id", "data").append('>').nl();
143      if (isEmptyList(o))
144         w.oTag(3, "p").append('>').append("no results").eTag("p");
145      else
146         super.doSerialize(o, w, hctx);
147      w.eTag(2, "div").nl();
148      w.eTag(1, "div").nl();
149
150      w.eTag("body").nl().eTag("html").nl();
151   }
152
153   private boolean isEmptyList(Object o) {
154      if (o == null)
155         return false;
156      if (o instanceof Collection && ((Collection<?>)o).size() == 0)
157         return true;
158      if (o.getClass().isArray() && Array.getLength(o) == 0)
159         return true;
160      return false;
161   }
162}
