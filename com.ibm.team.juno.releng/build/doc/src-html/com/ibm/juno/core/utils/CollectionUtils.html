009package com.ibm.juno.core.utils;
010
011import java.util.*;
012
013/**
014 * Utility methods for collections.
015 *
016 * @author James Bognar (jbognar@us.ibm.com)
017 */
018public class CollectionUtils {
019
020   /**
021    * Reverses the order of a {@link LinkedHashMap}.
022    *
023    * @param in The map to reverse the order on.
024    * @return A new {@link LinkedHashMap} with keys in reverse order.
025    */
026   public static <K,V> LinkedHashMap<K,V> reverse(LinkedHashMap<K,V> in) {
027      if (in == null)
028         return null;
029      LinkedHashMap<K,V> m = new LinkedHashMap<K,V>();
030
031      // Note:  Entry objects are reusable in an entry set, so we simply can't
032      // create a reversed iteration of that set.
033      List<K> keys = new ArrayList<K>(in.keySet());
034      List<V> values = new ArrayList<V>(in.values());
035      for (int i = in.size()-1; i >= 0; i--)
036         m.put(keys.get(i), values.get(i));
037
038      return m;
039   }
040
041   /**
042    * Add a value to a list if the value is not null.
043    *
044    * @param l The list to add to.
045    * @param o The element to add.
046    * @return The same list.
047    */
048   public static <T> List<T> addIfNotNull(List<T> l, T o) {
049      if (o != null)
050         l.add(o);
051      return l;
052   }
053}
