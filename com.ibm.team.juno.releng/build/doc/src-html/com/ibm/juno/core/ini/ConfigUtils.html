009package com.ibm.juno.core.ini;
010
011/**
012 * Internal utility methods.
013 *
014 * @author James Bognar (jbognar@us.ibm.com)
015 */
016public class ConfigUtils {
017
018   static final String getSectionName(String key) {
019      int i = key.indexOf('/');
020      if (i == -1)
021         return "default";
022      return key.substring(0, i);
023   }
024
025   static final String getSectionKey(String key) {
026      int i = key.indexOf('/');
027      if (i == -1)
028         return key;
029      return key.substring(i+1);
030   }
031
032   static final String getFullKey(String section, String key) {
033      if (section.equals("default"))
034         return key;
035      return section + '/' + key;
036   }
037
038   static final boolean isComment(String line) {
039      for (int i = 0; i < line.length(); i++) {
040         char c = line.charAt(i);
041         if (! Character.isWhitespace(c))
042            return c == '#';
043      }
044      return false;
045   }
046
047   static final boolean isAssignment(String line) {
048      int S1 = 1; // Looking for char;
049      int S2 = 2; // Found char, looking for whitespace or =
050      int S3 = 3; // Found whitespace, looking for =
051      int state = S1;
052      for (int i = 0; i < line.length(); i++) {
053         char c = line.charAt(i);
054         if (state == S1) {
055            if (! Character.isWhitespace(c))
056               state = S2;
057         } else if (state == S2) {
058            if (c == '=')
059               return true;
060            if (Character.isWhitespace(c))
061               state = S3;
062         } else if (state == S3) {
063            if (c == '=')
064               return true;
065         }
066      }
067      return false;
068   }
069
070   static final boolean isSection(String line) {
071      int S1 = 1; // Looking for [;
072      int S2 = 2; // Found [, looking for ]
073      int state = S1;
074      for (int i = 0; i < line.length(); i++) {
075         char c = line.charAt(i);
076         if (state == S1) {
077            if (! Character.isWhitespace(c)) {
078               if (c == '[')
079                  state = S2;
080               else
081                  return false;
082            }
083         } else if (state == S2) {
084            if (c == ']')
085               return true;
086         }
087      }
088      return false;
089   }
090}
