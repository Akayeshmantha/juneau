009package com.ibm.juno.core.utils;
010
011import static com.ibm.juno.core.utils.ThrowableUtils.*;
012
013/**
014 * Stores a set of language keywords for quick lookup.
015 * <p>
016 * Keywords must be:
017 * <ul>
018 *    <li>2 or more characters in length.
019 *    <li>Lowercase ASCII.
020 * </ul>
021 *
022 * @author James Bognar (jbognar@us.ibm.com)
023 */
024public final class KeywordSet {
025   final char[][][][] store;
026
027   /**
028    * Constructor.
029    *
030    * @param keywords The list of keywords.
031    */
032   public KeywordSet(String... keywords) {
033      this.store = new char[26][][][];
034
035      for (String keyword : keywords) {
036         if (keyword.length() < 2)
037            illegalArg("Invalid keyword '{0}' passed to KeywordStore.", keyword);
038         int c0 = keyword.charAt(0) - 'a';
039         int c1 = keyword.charAt(1) - 'a';
040         if (c0 < 0 || c0 > 25 || c1 < 0 || c1 > 25)
041            illegalArg("Invalid keyword '{0}' passed to KeywordStore.", keyword);
042         if (this.store[c0] == null)
043            this.store[c0] = new char[26][][];
044         char[][][] x1 = this.store[c0];
045         char[][] x2;
046         if (x1[c1] == null)
047            x2 = new char[1][];
048         else {
049            x2 = new char[x1[c1].length+1][];
050            System.arraycopy(x1[c1], 0, x2, 0, x1[c1].length);
051         }
052         x2[x2.length-1] = keyword.toCharArray();
053         x1[c1] = x2;
054      }
055   }
056
057   /**
058    * Returns <jk>true<jk> if the specified string exists in this store.
059    *
060    * @param s The string to check.
061    * @return <jk>true<jk> if the specified string exists in this store.
062    */
063   public boolean contains(String s) {
064      if (s == null || s.length() < 2)
065         return false;
066      int c0 = s.charAt(0) - 'a', c1 = s.charAt(1) - 'a';
067      if (c0 < 0 || c0 > 25 || c1 < 0 || c1 > 25)
068         return false;
069      char[][][] x1 = store[c0];
070      if (x1 == null)
071         return false;
072      char[][] x2 = x1[c1];
073      if (x2 == null)
074         return false;
075      for (int i = 0; i < x2.length; i++) {
076         char[] keyword = x2[i];
077         if (keyword.length == s.length()) {
078            for (int j = 0; j < keyword.length; j++)
079               if (keyword[j] != s.charAt(j))
080                  return false;
081            return true;
082         }
083      }
084      return false;
085   }
086}
