009package com.ibm.juno.core.utils;
010
011/**
012 * Various utility methods.
013 *
014 * @author James Bognar (jbognar@us.ibm.com)
015 */
016public class Utils {
017
018   /**
019    * Compare two integers numerically.
020    *
021    * @param i1 Integer #1
022    * @param i2 Integer #2
023     * @return the value <code>0</code> if Integer #1 is
024     *      equal to Integer #2; a value less than
025     *      <code>0</code> if Integer #1 numerically less
026     *      than Integer #2; and a value greater
027     *      than <code>0</code> if Integer #1 is numerically
028     *       greater than Integer #2 (signed
029     *       comparison).
030    */
031   public static final int compare(int i1, int i2) {
032      return (i1<i2 ? -1 : (i1==i2 ? 0 : 1));
033   }
034}
