009package com.ibm.juno.core.xml.annotation;
010
011/**
012 * XML format to use when serializing a POJO.
013 *
014 * @author James Bognar (jbognar@us.ibm.com)
015 */
016public enum XmlFormat {
017
018   /**
019    * Normal formatting (default)
020    */
021   NORMAL,
022
023   /**
024    * Render property as an attribute instead of an element.
025    * <p>
026    *    Can only be applied to properties (methods/fields) of simple types (e.g. <code>String</code>, <code>Number</code>).
027    */
028   ATTR,
029
030   /**
031    * Render property as an element instead of an attribute.
032    * <p>
033    *    Can be applied to URL and ID bean properties that would normally be rendered as attributes.
034    */
035   ELEMENT,
036
037   /**
038    * Prevents collections and arrays from being enclosed in <xt>&lt;array&gt;</xt> elements.
039    * <p>
040    *    Can only be applied to properties (methods/fields) of type collection or array, or collection classes.
041    */
042   COLLAPSED,
043
044   /**
045    * Render property value directly as content of element.
046    * <p>
047    *    By default, content is converted to plain text.
048    * <p>
049    *    Can be used in combination with {@link Xml#contentHandler()} to produce something other
050    *    than plain text, such as embedded XML.
051    */
052   CONTENT
053}
