009package com.ibm.juno.core;
010
011/**
012 * Common super class for all core-API serializers, parsers, and serializer/parser groups.
013 *
014 * <h6 class='topic'>Description</h6>
015 * <p>
016 *    Maintains an inner {@link BeanContextFactory} instance that can be used by serializer and parser subclasses
017 *       to work with beans in a consistent way.
018 * <p>
019 *    Provides several duplicate convenience methods from the {@link BeanContextFactory} class to set properties on that class from this class.
020 * <p>
021 *    Also implements the {@link Lockable} interface to allow for easy locking and cloning.
022 *
023 *
024 * @author James Bognar (jbognar@us.ibm.com)
025 */
026public abstract class CoreApi extends Lockable {
027
028   /** The bean context used by this object. */
029   protected transient BeanContextFactory beanContextFactory = new BeanContextFactory();
030   private BeanContext beanContext;
031
032
033   /**
034    * Returns the current value of the {@code beanContext} setting.
035    *
036    * @return The current setting value.
037    */
038   public final BeanContext getBeanContext() {
039      if (beanContext == null)
040         beanContext = beanContextFactory.getBeanContext();
041      return beanContext;
042   }
043
044   /**
045    * Sets a property on this class.
046    *
047    * @param property The property name.
048    * @param value The property value.
049    * @return This class (for method chaining).
050    * @throws LockedException If {@link #lock()} has been called on this object.
051    */
052   public CoreApi setProperty(String property, Object value) throws LockedException {
053      checkLock();
054      beanContextFactory.setProperty(property, value);
055      return this;
056   }
057
058   /**
059    * Sets multiple properties on this class.
060    *
061    * @param properties The properties to set on this class.
062    * @return This class (for method chaining).
063    * @throws LockedException If {@link #lock()} has been called on this object.
064    */
065   public CoreApi setProperties(ObjectMap properties) throws LockedException {
066      checkLock();
067      beanContextFactory.setProperties(properties);
068      return this;
069   }
070
071   /**
072    * Shortcut for calling <code>getBeanContext().addNotBeanClasses(Class...)</code>.
073    *
074    * @see BeanContextFactory#addNotBeanClasses(Class...)
075    * @param classes The new setting value for the bean context.
076    * @throws LockedException If {@link BeanContextFactory#lock()} was called on this class or the bean context.
077    * @return This object (for method chaining).
078    */
079   public CoreApi addNotBeanClasses(Class<?>...classes) throws LockedException {
080      checkLock();
081      beanContextFactory.addNotBeanClasses(classes);
082      return this;
083   }
084
085   /**
086    * Shortcut for calling <code>getBeanContext().addFilters(Class...)</code>.
087    *
088    * @see BeanContextFactory#addFilters(Class...)
089    * @param classes The new setting value for the bean context.
090    * @throws LockedException If {@link BeanContextFactory#lock()} was called on this class or the bean context.
091    * @return This object (for method chaining).
092    */
093   public CoreApi addFilters(Class<?>...classes) throws LockedException {
094      checkLock();
095      beanContextFactory.addFilters(classes);
096      return this;
097   }
098
099   /**
100    * Shortcut for calling <code>getBeanContext().addImplClass(Class, Class)</code>.
101    *
102    * @see BeanContextFactory#addImplClass(Class, Class)
103    * @param interfaceClass The interface class.
104    * @param implClass The implementation class.
105    * @throws LockedException If {@link BeanContextFactory#lock()} was called on this class or the bean context.
106    * @param <T> The class type of the interface.
107    * @return This object (for method chaining).
108    */
109   public <T> CoreApi addImplClass(Class<T> interfaceClass, Class<? extends T> implClass) throws LockedException {
110      checkLock();
111      beanContextFactory.addImplClass(interfaceClass, implClass);
112      return this;
113   }
114
115   /**
116    * Shortcut for calling <code>getBeanContext().setClassLoader(ClassLoader)</code>.
117    *
118    * @see BeanContextFactory#setClassLoader(ClassLoader)
119    * @param classLoader The new classloader.
120    * @throws LockedException If {@link BeanContextFactory#lock()} was called on this class or the bean context.
121    * @return This object (for method chaining).
122    */
123   public CoreApi setClassLoader(ClassLoader classLoader) throws LockedException {
124      checkLock();
125      beanContextFactory.setClassLoader(classLoader);
126      return this;
127   }
128
129   /**
130    * Shortcut for calling {@link BeanContext#object()}.
131    *
132    * @return The reusable {@link ClassMeta} for representing the {@link Object} class.
133    */
134   public ClassMeta<Object> object() {
135      return getBeanContext().object();
136   }
137
138   /**
139    * Shortcut for calling  {@link BeanContext#string()}.
140    *
141    * @return The reusable {@link ClassMeta} for representing the {@link String} class.
142    */
143   public ClassMeta<String> string() {
144      return getBeanContext().string();
145   }
146
147   //--------------------------------------------------------------------------------
148   // Overridden methods
149   //--------------------------------------------------------------------------------
150
151   @Override /* Lockable */
152   public void checkLock() {
153      super.checkLock();
154      beanContext = null;
155   }
156
157   @Override /* Lockable */
158   public CoreApi lock() {
159      super.lock();
160      beanContextFactory.lock();
161      beanContext = beanContextFactory.getBeanContext();
162      return this;
163   }
164
165   @Override /* Lockable */
166   public CoreApi clone() throws CloneNotSupportedException{
167      CoreApi c = (CoreApi)super.clone();
168      c.beanContextFactory = beanContextFactory.clone();
169      c.beanContext = null;
170      return c;
171   }
172}
