028/**
029 * Encapsulates all access to the properties of a bean class (like a souped-up {@link java.beans.BeanInfo}).
030 *
031 *
032 * <h6 class='topic'>Description</h6>
033 * <p>
034 *    Uses introspection to find all the properties associated with this class.  If the {@link Bean @Bean} annotation
035 *    is present on the class, or the class has a {@link BeanFilter} registered with it in the bean context,
036 *    then that information is used to determine the properties on the class.
037 *    Otherwise, the {@code BeanInfo} functionality in Java is used to determine the properties on the class.
038 *
039 *
040 * <h6 class='topic'>Bean property ordering</h6>
041 * <p>
042 *    The order of the properties are as follows:
043 *    <ul>
044 *       <li>If {@link Bean @Bean} annotation is specified on class, then the order is the same as the list of properties in the annotation.
045 *       <li>If {@link Bean @Bean} annotation is not specified on the class, then the order is based on the following.
046 *          <ul>
047 *             <li>Public fields (same order as {@code Class.getFields()}).
048 *             <li>Properties returned by {@code BeanInfo.getPropertyDescriptors()}.
049 *             <li>Non-standard getters/setters with {@link BeanProperty @BeanProperty} annotation defined on them.
050 *          </ul>
051 *    </ul>
052 *    <br>
053 *    The order can also be overridden through the use of an {@link BeanFilter}.
054 *
055 *
056 * @param <T> The class type that this metadata applies to.
057 * @author Barry M. Caceres
058 * @author James Bognar (jbognar@us.ibm.com)
059 */
060public class BeanMeta<T> {
061
062   /** The target class type that this meta object describes. */
063   protected ClassMeta<T> classMeta;
064
065   /** The target class that this meta object describes. */
066   protected Class<T> c;
067
068   /** The properties on the target class. */
069   protected Map<String,BeanPropertyMeta<T>> properties;
070
071   /** The getter properties on the target class. */
072   protected Map<Method,String> getterProps = new HashMap<Method,String>();
073
074   /** The setter properties on the target class. */
075   protected Map<Method,String> setterProps = new HashMap<Method,String>();
076
077   /** The bean context that created this metadata object. */
078   protected BeanContext ctx;
079
080   /** Optional bean filter associated with the target class. */
081   protected BeanFilter<? extends T> filter;
082
083   /** Type variables implemented by this bean. */
084   protected Map<Class<?>,Class<?>[]> typeVarImpls;
085
086   /** The constructor for this bean. */
087   protected Constructor<T> constructor;
088
089   /** For beans with constructors with BeanConstructor annotation, this is the list of constructor arg properties. */
090   protected String[] constructorArgs = new String[0];
091
092   /** XML-related metadata */
093   protected XmlBeanMeta<T> xmlMeta;
094
095   // Other fields
096   BeanPropertyMeta<T> uriProperty;                                 // The property identified as the URI for this bean (annotated with @BeanProperty.beanUri).
097   BeanPropertyMeta<T> subTypeIdProperty;                           // The property indentified as the sub type differentiator property (identified by @Bean.subTypeProperty annotation).
098   PropertyNamer propertyNamer;                                     // Class used for calculating bean property names.
099
100   BeanMeta() {}

126   /**
127    * Initializes this bean meta, and returns an error message if the specified class is not
128    * a bean for any reason.
129    *
130    * @return Reason why this class isn't a bean, or <jk>null</jk> if no problems detected.
131    * @throws BeanRuntimeException If unexpected error occurs such as invalid annotations on the bean class.
132    */
133   @SuppressWarnings("unchecked")
134   protected String init() throws BeanRuntimeException {
135
136      try {
137         Visibility
138            conVis = ctx.beanConstructorVisibility,
139            cVis = ctx.beanClassVisibility,
140            mVis = ctx.beanMethodVisibility,
141            fVis = ctx.beanFieldVisibility;
142
143         // If @Bean.interfaceClass is specified on the parent class, then we want
144         // to use the properties defined on that class, not the subclass.
145         Class<?> c2 = (filter != null && filter.getInterfaceClass() != null ? filter.getInterfaceClass() : c);
146
147         Class<?> stopClass = (filter != null ? filter.getStopClass() : Object.class);
148         if (stopClass == null)
149            stopClass = Object.class;
150
151         Map<String,BeanPropertyMeta<T>> normalProps = new LinkedHashMap<String,BeanPropertyMeta<T>>();
152
153         /// See if this class matches one the patterns in the exclude-class list.
154         if (ctx.isNotABean(c))
155            return "Class matches exclude-class list";
156
157         if (! cVis.isVisible(c.getModifiers()))
158            return "Class is not public";
159
160         if (c.isAnnotationPresent(BeanIgnore.class))
161            return "Class is annotated with @BeanIgnore";
162
163         // Make sure it's serializable.
164         if (filter == null && ctx.beansRequireSerializable && ! isParentClass(Serializable.class, c))
165            return "Class is not serializable";
166
167         // Look for @BeanConstructor constructor.
168         for (Constructor<?> x : c.getConstructors()) {
169            if (x.isAnnotationPresent(BeanConstructor.class)) {
170               if (constructor != null)
171                  throw new BeanRuntimeException(c, "Multiple instances of '@BeanConstructor' found.");
172               constructor = (Constructor<T>)x;
173               constructorArgs = x.getAnnotation(BeanConstructor.class).properties();
174               if (constructorArgs.length != x.getParameterTypes().length)
175                  throw new BeanRuntimeException(c, "Number of properties defined in '@BeanConstructor' annotation does not match number of parameters in constructor.");
176               if (! setAccessible(constructor))
177                  throw new BeanRuntimeException(c, "Could not set accessibility to true on method with @BeanConstructor annotation.  Method=''{0}''", constructor.getName());
178            }
179         }
180
181         // If this is an interface, look for impl classes defined in the context.
182         if (constructor == null)
183            constructor = (Constructor<T>)ctx.getImplClassConstructor(c, conVis);
184
185         if (constructor == null)
186            constructor = (Constructor<T>)ClassMeta.findNoArgConstructor(c, conVis);
187
188         if (constructor == null && filter == null && ctx.beansRequireDefaultConstructor)
189            return "Class does not have the required no-arg constructor";
190
191         if (! setAccessible(constructor))
192            throw new BeanRuntimeException(c, "Could not set accessibility to true on no-arg constructor");
193
194         // Explicitly defined property names in @Bean annotation.
195         Set<String> fixedBeanProps = new LinkedHashSet<String>();
196
197         if (filter != null) {
198
199            // Get the 'properties' attribute if specified.
200            if (filter.getProperties() != null)
201               for (String p : filter.getProperties())
202                  fixedBeanProps.add(p);
203
204            if (filter.getPropertyNamer() != null)
205               propertyNamer = filter.getPropertyNamer().newInstance();
206         }
207
208         if (propertyNamer == null)
209            propertyNamer = new PropertyNamerDefault();
210
211         // First populate the properties with those specified in the bean annotation to
212         // ensure that ordering first.
213         for (String name : fixedBeanProps)
214            normalProps.put(name, new BeanPropertyMeta<T>(this, name));
215
216         if (ctx.useJavaBeanIntrospector) {
217            BeanInfo bi = null;
218            if (! c2.isInterface())
219               bi = Introspector.getBeanInfo(c2, stopClass);
220            else
221               bi = Introspector.getBeanInfo(c2, null);
222            if (bi != null) {
223               for (PropertyDescriptor pd : bi.getPropertyDescriptors()) {
224                  String name = pd.getName();
225                  if (! normalProps.containsKey(name))
226                     normalProps.put(name, new BeanPropertyMeta<T>(this, name));
227                  normalProps.get(name).setGetter(pd.getReadMethod()).setSetter(pd.getWriteMethod());
228               }
229            }
230
231         } else /* Use 'better' introspection */ {
232
233            for (Field f : findBeanFields(c2, stopClass, fVis)) {
234               String name = findPropertyName(f, fixedBeanProps);
235               if (name != null) {
236                  if (! normalProps.containsKey(name))
237                     normalProps.put(name, new BeanPropertyMeta<T>(this, name));
238                  normalProps.get(name).setField(f);
239               }
240            }
241
242            List<BeanMethod> bms = findBeanMethods(c2, stopClass, mVis, fixedBeanProps, propertyNamer);
243
244            // Iterate through all the getters.
245            for (BeanMethod bm : bms) {
246               String pn = bm.propertyName;
247               Method m = bm.method;
248               if (! normalProps.containsKey(pn))
249                  normalProps.put(pn, new BeanPropertyMeta<T>(this, pn));
250               BeanPropertyMeta<?> bpm = normalProps.get(pn);
251               if (! bm.isSetter)
252                  bpm.setGetter(m);
253            }
254
255            // Now iterate through all the setters.
256            for (BeanMethod bm : bms) {
257               if (bm.isSetter) {
258                  BeanPropertyMeta<?> bpm = normalProps.get(bm.propertyName);
259                  if (bm.matchesPropertyType(bpm))
260                     bpm.setSetter(bm.method);
261               }
262            }
263         }
264
265         typeVarImpls = new HashMap<Class<?>,Class<?>[]>();
266         findTypeVarImpls(c, typeVarImpls);
267         if (typeVarImpls.isEmpty())
268            typeVarImpls = null;
269
270         // Eliminate invalid properties, and set the contents of getterProps and setterProps.
271         for (Iterator<BeanPropertyMeta<T>> i = normalProps.values().iterator(); i.hasNext();) {
272            BeanPropertyMeta<T> p = i.next();
273            try {
274               if (p.validate()) {
275
276                  if (p.getGetter() != null)
277                     getterProps.put(p.getGetter(), p.getName());
278
279                  if (p.getSetter() != null)
280                     setterProps.put(p.getSetter(), p.getName());
281
282                  if (p.isBeanUri())
283                     uriProperty = p;
284
285               } else {
286                  i.remove();
287               }
288            } catch (ClassNotFoundException e) {
289               throw new BeanRuntimeException(c, e.getLocalizedMessage());
290            }
291         }
292
293         // Check for missing properties.
294         for (String fp : fixedBeanProps)
295            if (! normalProps.containsKey(fp))
296               throw new BeanRuntimeException(c, "The property ''{0}'' was defined on the @Bean(properties=X) annotation but was not found on the class definition.", fp);
297
298         // Mark constructor arg properties.
299         for (String fp : constructorArgs) {
300            BeanPropertyMeta<T> m = normalProps.get(fp);
301            if (m == null)
302               throw new BeanRuntimeException(c, "The property ''{0}'' was defined on the @BeanConstructor(properties=X) annotation but was not found on the class definition.", fp);
303            m.setAsConstructorArg();
304         }
305
306         // Make sure at least one property was found.
307         if (filter == null && ctx.beansRequireSomeProperties && normalProps.size() == 0)
308            return "No properties detected on bean class";
309
310         properties = new LinkedHashMap<String,BeanPropertyMeta<T>>();
311
312         if (filter != null && filter.getSubTypeProperty() != null) {
313            String subTypeProperty = filter.getSubTypeProperty();
314            this.subTypeIdProperty = new SubTypePropertyMeta(subTypeProperty, filter.getSubTypes(), normalProps.remove(subTypeProperty));
315            properties.put(subTypeProperty, this.subTypeIdProperty);
316         }
317
318         properties.putAll(normalProps);
319
320         // If a filter is defined, look for inclusion and exclusion lists.
321         if (filter != null) {
322
323            // Eliminated excluded properties if BeanFilter.excludeKeys is specified.
324            String[] includeKeys = filter.getProperties();
325            String[] excludeKeys = filter.getExcludeProperties();
326            if (excludeKeys != null) {
327               for (String k : excludeKeys)
328                  properties.remove(k);
329
330            // Only include specified properties if BeanFilter.includeKeys is specified.
331            // Note that the order must match includeKeys.
332            } else if (includeKeys != null) {
333               Map<String,BeanPropertyMeta<T>> properties2 = new LinkedHashMap<String,BeanPropertyMeta<T>>();
334               for (String k : includeKeys) {
335                  if (properties.containsKey(k))
336                     properties2.put(k, properties.get(k));
337               }
338               properties = properties2;
339            }
340         }
341
342         xmlMeta = new XmlBeanMeta<T>(this, null);
343
344         // We return this through the Bean.keySet() interface, so make sure it's not modifiable.
345         properties = Collections.unmodifiableMap(properties);
346
347      } catch (BeanRuntimeException e) {
348         throw e;
349      } catch (Exception e) {
350         return "Exception:  " + StringUtils.getStackTrace(e);
351      }
352
353      return null;
354   }

395   /*
396    * Temporary getter/setter method struct.
397    */
398   private static class BeanMethod {
399      String propertyName;
400      boolean isSetter;
401      Method method;
402      Class<?> type;
403
404      BeanMethod(String propertyName, boolean isSetter, Method method) {
405         this.propertyName = propertyName;
406         this.isSetter = isSetter;
407         this.method = method;
408         if (isSetter)
409            this.type = method.getParameterTypes()[0];
410         else
411            this.type = method.getReturnType();
412      }
413
414      /*
415       * Returns true if this method matches the class type of the specified property.
416       * Only meant to be used for setters.
417       */
418      boolean matchesPropertyType(BeanPropertyMeta<?> b) {
419         if (b == null)
420            return false;
421
422         // Get the bean property type from the getter/field.
423         Class<?> pt = null;
424         if (b.getGetter() != null)
425            pt = b.getGetter().getReturnType();
426         else if (b.getField() != null)
427            pt = b.getField().getType();
428
429         // Doesn't match if no getter/field defined.
430         if (pt == null)
431            return false;
432
433         // Doesn't match if not same type or super type as getter/field.
434         if (! isParentClass(type, pt))
435            return false;
436
437         // If a setter was previously set, only use this setter if it's a closer
438         // match (e.g. prev type is a superclass of this type).
439         if (b.getSetter() == null)
440            return true;
441
442         Class<?> prevType = b.getSetter().getParameterTypes()[0];
443         return isParentClass(prevType, type, true);
444      }
445
446      @Override /* Object */
447      public String toString() {
448         return method.toString();
449      }
450   }

452   /*
453    * Find all the bean methods on this class.
454    *
455    * @param c The filtered class.
456    * @param stopClass Don't look above this class in the hierarchy.
457    * @param v The minimum method visibility.
458    * @param fixedBeanProps Only include methods whose properties are in this list.
459    * @param pn Use this property namer to determine property names from the method names.
460    */
461   private static List<BeanMethod> findBeanMethods(Class<?> c, Class<?> stopClass, Visibility v, Set<String> fixedBeanProps, PropertyNamer pn) {
462      List<BeanMethod> l = new LinkedList<BeanMethod>();
463
464      for (Class<?> c2 : findClasses(c, stopClass)) {
465         for (Method m : c2.getDeclaredMethods()) {
466            int mod = m.getModifiers();
467            if (Modifier.isStatic(mod) || Modifier.isTransient(mod))
468               continue;
469            if (m.isAnnotationPresent(BeanIgnore.class))
470               continue;
471            if (m.isBridge())   // This eliminates methods with covariant return types from parent classes on child classes.
472               continue;
473            if (! (v.isVisible(m) || m.isAnnotationPresent(BeanProperty.class)))
474               continue;
475            String n = m.getName();
476            Class<?>[] pt = m.getParameterTypes();
477            Class<?> rt = m.getReturnType();
478            boolean isGetter = false, isSetter = false;
479            if (pt.length == 1 && n.startsWith("set") && (isParentClass(rt, c) || rt.equals(Void.TYPE))) {
480               isSetter = true;
481               n = n.substring(3);
482            } else if (pt.length == 0 && n.startsWith("get") && (! rt.equals(Void.TYPE))) {
483               isGetter = true;
484               n = n.substring(3);
485            } else if (pt.length == 0 && n.startsWith("is") && (rt.equals(Boolean.TYPE) || rt.equals(Boolean.class))) {
486               isGetter = true;
487               n = n.substring(2);
488            }
489            n = pn.getPropertyName(n);
490            if (isGetter || isSetter) {
491               BeanProperty bp = m.getAnnotation(BeanProperty.class);
492               if (bp != null && ! bp.name().equals("")) {
493                  n = bp.name();
494                  if (! fixedBeanProps.isEmpty())
495                     if (! fixedBeanProps.contains(n))
496                        throw new BeanRuntimeException(c, "Method property ''{0}'' identified in @BeanProperty, but missing from @Bean", n);
497               }
498               l.add(new BeanMethod(n, isSetter, m));
499            }
500         }
501      }
502      return l;
503   }
504
505   private static Collection<Field> findBeanFields(Class<?> c, Class<?> stopClass, Visibility v) {
506      List<Field> l = new LinkedList<Field>();
507      for (Class<?> c2 : findClasses(c, stopClass)) {
508         for (Field f : c2.getDeclaredFields()) {
509            int m = f.getModifiers();
510            if (Modifier.isStatic(m) || Modifier.isTransient(m))
511               continue;
512            if (f.isAnnotationPresent(BeanIgnore.class))
513               continue;
514            if (! (v.isVisible(f) || f.isAnnotationPresent(BeanProperty.class)))
515               continue;
516            l.add(f);
517         }
518      }
519      return l;
520   }
521
522   private static List<Class<?>> findClasses(Class<?> c, Class<?> stopClass) {
523      LinkedList<Class<?>> l = new LinkedList<Class<?>>();
524      findClasses(c, l, stopClass);
525      return l;
526   }
527
528   private static void findClasses(Class<?> c, LinkedList<Class<?>> l, Class<?> stopClass) {
529      while (c != null && stopClass != c) {
530         l.addFirst(c);
531         for (Class<?> ci : c.getInterfaces())
532            findClasses(ci, l, stopClass);
533         c = c.getSuperclass();
534      }
535   }

581   /**
582    * Creates a new instance of this bean.
583    *
584    * @param outer The outer object if bean class is a non-static inner member class.
585    * @return A new instance of this bean if possible, or <jk>null</jk> if not.
586    * @throws IllegalArgumentException Thrown by constructor.
587    * @throws InstantiationException Thrown by constructor.
588    * @throws IllegalAccessException Thrown by constructor.
589    * @throws InvocationTargetException Thrown by constructor.
590    */
591   @SuppressWarnings("unchecked")
592   protected T newBean(Object outer) throws IllegalArgumentException, InstantiationException, IllegalAccessException, InvocationTargetException {
593      if (classMeta.isMemberClass) {
594         if (constructor != null)
595            return constructor.newInstance(outer);
596      } else {
597         if (constructor != null)
598            return constructor.newInstance((Object[])null);
599         InvocationHandler h = classMeta.getProxyInvocationHandler();
600         if (h != null) {
601            ClassLoader cl = classMeta.beanContext.classLoader;
602            if (cl == null)
603               cl = this.getClass().getClassLoader();
604            return (T)Proxy.newProxyInstance(cl, new Class[] { classMeta.innerClass, java.io.Serializable.class }, h);
605         }
606      }
607      return null;
608   }

610   /*
611    * Returns the property name of the specified field if it's a valid property.
612    * Returns null if the field isn't a valid property.
613    */
614   private String findPropertyName(Field f, Set<String> fixedBeanProps) {
615      BeanProperty bp = f.getAnnotation(BeanProperty.class);
616      if (bp != null && ! bp.name().equals("")) {
617         String name = bp.name();
618         if (fixedBeanProps.isEmpty() || fixedBeanProps.contains(name))
619            return name;
620         throw new BeanRuntimeException(c, "Method property ''{0}'' identified in @BeanProperty, but missing from @Bean", name);
621      }
622      String name = propertyNamer.getPropertyName(f.getName());
623      if (fixedBeanProps.isEmpty() || fixedBeanProps.contains(name))
624         return name;
625      return null;
626   }

628   /**
629    * Recursively determines the classes represented by parameterized types in the class hierarchy of
630    * the specified type, and puts the results in the specified map.<br>
631    * <p>
632    *    For example, given the following classes...
633    * <p class='bcode'>
634    *    public static class BeanA&lt;T> {
635    *       public T x;
636    *    }
637    *    public static class BeanB extends BeanA&lt;Integer>} {...}
638    * <p>
639    *    ...calling this method on {@code BeanB.class} will load the following data into {@code m} indicating
640    *    that the {@code T} parameter on the BeanA class is implemented with an {@code Integer}:
641    * <p class='bcode'>
642    *    {BeanA.class:[Integer.class]}
643    * <p>
644    *    TODO:  This code doesn't currently properly handle the following situation:
645    * <p class='bcode'>
646    *    public static class BeanB&ltT extends Number> extends BeanA&ltT>;
647    *    public static class BeanC extends BeanB&ltInteger>;
648    * <p>
649    *    When called on {@code BeanC}, the variable will be detected as a {@code Number}, not an {@code Integer}.<br>
650    *    If anyone can figure out a better way of doing this, please do so!
651    *
652    * @param t The type we're recursing.
653    * @param m Where the results are loaded.
654    */
655   private static void findTypeVarImpls(Type t, Map<Class<?>,Class<?>[]> m) {
656      if (t instanceof Class) {
657         Class<?> c = (Class<?>)t;
658         findTypeVarImpls(c.getGenericSuperclass(), m);
659         for (Type ci : c.getGenericInterfaces())
660            findTypeVarImpls(ci, m);
661      } else if (t instanceof ParameterizedType) {
662         ParameterizedType pt = (ParameterizedType)t;
663         Type rt = pt.getRawType();
664         if (rt instanceof Class) {
665            Type[] gImpls = pt.getActualTypeArguments();
666            Class<?>[] gTypes = new Class[gImpls.length];
667            for (int i = 0; i < gImpls.length; i++) {
668               Type gt = gImpls[i];
669               if (gt instanceof Class)
670                  gTypes[i] = (Class<?>)gt;
671               else if (gt instanceof TypeVariable) {
672                  TypeVariable<?> tv = (TypeVariable<?>)gt;
673                  for (Type upperBound : tv.getBounds())
674                     if (upperBound instanceof Class)
675                        gTypes[i] = (Class<?>)upperBound;
676               }
677            }
678            m.put((Class<?>)rt, gTypes);
679            findTypeVarImpls(pt.getRawType(), m);
680         }
681      }
682   }

685    * Bean property for getting and setting bean subtype.
686    */
687   @SuppressWarnings({"rawtypes","unchecked"})
688   private class SubTypePropertyMeta extends BeanPropertyMeta<T> {
689
690      private Map<Class<?>,String> subTypes;
691      private BeanPropertyMeta<T> realProperty;  // Bean property if bean actually has a real subtype field.
692
693      SubTypePropertyMeta(String subTypeAttr, Map<Class<?>,String> subTypes, BeanPropertyMeta<T> realProperty) {
694         super(BeanMeta.this, subTypeAttr, ctx.string());
695         this.subTypes = subTypes;
696         this.realProperty = realProperty;
697         this.htmlMeta = new HtmlBeanPropertyMeta<T>(this);
698         this.xmlMeta = new XmlBeanPropertyMeta<T>(this);
699         this.rdfMeta = new RdfBeanPropertyMeta<T>(this);
700      }
701
702      /*
703       * Setting this bean property causes the inner bean to be set to the subtype implementation.
704       */
705      @Override /* BeanPropertyMeta */
706      public Object set(BeanMap<T> m, Object value) throws BeanRuntimeException {
707         if (value == null)
708            throw new BeanRuntimeException("Attempting to set bean subtype property to null.");
709         String subTypeId = value.toString();
710         for (Entry<Class<?>,String> e : subTypes.entrySet()) {
711            if (e.getValue().equals(subTypeId)) {
712               Class subTypeClass = e.getKey();
713               m.meta = ctx.getBeanMeta(subTypeClass);
714               try {
715                  m.bean = (T)subTypeClass.newInstance();
716                  if (realProperty != null)
717                     realProperty.set(m, value);
718                  // If subtype attribute wasn't specified first, set them again from the temporary cache.
719                  if (m.propertyCache != null)
720                     for (Map.Entry<String,Object> me : m.propertyCache.entrySet())
721                        m.put(me.getKey(), me.getValue());
722               } catch (Exception e1) {
723                  throw new BeanRuntimeException(e1);
724               }
725               return null;
726            }
727         }
728         throw new BeanRuntimeException(c, "Unknown subtype ID ''{0}''", subTypeId);
729      }
730
731      @Override /* BeanPropertyMeta */
732      public Object get(BeanMap<T> m) throws BeanRuntimeException {
733         String subTypeId = filter.getSubTypes().get(c);
734         if (subTypeId == null)
735            throw new BeanRuntimeException(c, "Unmapped sub type class");
736         return subTypeId;
737      }
738   }

743      sb.append(" {\n");
744      for (BeanPropertyMeta<?> pm : this.properties.values())
