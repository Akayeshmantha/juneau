011import javax.xml.stream.*;
012import javax.xml.stream.util.*;

014import com.ibm.juno.core.*;

017/**
018 * Configurable properties on the {@link XmlParser} class.
019 * <p>
020 *    Use the {@link XmlParser#setProperty(String, Object)} method to set property values.
021 * <p>
022 *    In addition to these properties, the following properties are also applicable for {@link XmlParser}.
023 * <ul>
024 *    <li>{@link ParserProperties}
025 *    <li>{@link BeanContextProperties}
026 * </ul>
027 *
028 * @author James Bognar (jbognar@us.ibm.com)
029 */
030public final class XmlParserProperties implements Cloneable {
031
032   /**
033    * XMLSchema-instance namespace URI ({@link String}, default=<js>"http://www.w3.org/2001/XMLSchema-instance"</js>).
034    * <p>
035    * The XMLSchema namespace.
036    */
037   public static final String XML_xsiNs = "XmlParser.xsiNs";
038
039   /**
040    * Trim whitespace from text elements ({@link Boolean}, default=<jk>true</jk>).
041    * <p>
042    * If <jk>true</jk>, whitespace in text elements will be automatically trimmed.
043    */
044   public static final String XML_trimWhitespace = "XmlParser.trimWhitespace";
045
046   /**
047    * Set validating mode ({@link Boolean}, default=<jk>false</jk>).
048    * <p>
049    * If <jk>true</jk>, XML document will be validated.
050    * See {@link XMLInputFactory#IS_VALIDATING} for more info.
051    */
052   public static final String XML_validating = "XmlParser.validating";
053
054   /**
055    * Set coalescing mode ({@link Boolean}, default=<jk>false</jk>).
056    * <p>
057    * If <jk>true</jk>, XML text elements will be coalesced.
058    * See {@link XMLInputFactory#IS_COALESCING} for more info.
059    */
060   public static final String XML_coalescing = "XmlParser.coalescing";
061
062   /**
063    * Replace entity references ({@link Boolean}, default=<jk>true</jk>).
064    * <p>
065    * If <jk>true</jk>, entity references will be replace during parsing.
066    * See {@link XMLInputFactory#IS_REPLACING_ENTITY_REFERENCES} for more info.
067    */
068   public static final String XML_replaceEntityReferences = "XmlParser.replaceEntityReferences";
069
070   /**
071    * XML reporter ({@link XMLReporter}, default=<jk>null</jk>).
072    * <p>
073    * Associates an {@link XMLReporter} with this parser.
074    * <p>
075    * Note:  Reporters are not copied to new parsers during a clone.
076    */
077   public static final String XML_reporter = "XmlParser.reporter";
078
079   /**
080    * XML resolver ({@link XMLResolver}, default=<jk>null</jk>).
081    * <p>
082    * Associates an {@link XMLResolver} with this parser.
083    */
084   public static final String XML_resolver = "XmlParser.resolver";
085
086   /**
087    * XML event allocator. ({@link XMLEventAllocator}, default=<jk>false</jk>).
088    * <p>
089    * Associates an {@link XMLEventAllocator} with this parser.
090    */
091   public static final String XML_eventAllocator = "XmlParser.eventAllocator";
092
093   /**
094    * Preserve root element during generalized parsing ({@link Boolean}, default=<jk>false</jk>).
095    * <p>
096    * If <jk>true</jk>, when parsing into a generic {@link ObjectMap}, the map will
097    *    contain a single entry whose key is the root element name.
098    *
099    * Example:
100    * <table class='styled'>
101    *    <tr>
102    *       <td>XML</td>
103    *       <td>ObjectMap.toString(), preserveRootElement==false</td>
104    *       <td>ObjectMap.toString(), preserveRootElement==true</td>
105    *    </tr>
106    *    <tr>
107    *       <td><code><xt>&lt;root&gt;&lt;a&gt;</xt>foobar<xt>&lt;/a&gt;&lt;/root&gt;</xt><code></td>
108    *       <td><code>{ a:<js>'foobar'</js> }</code></td>
109    *       <td><code>{ root: { a:<js>'foobar'</js> }}</code></td>
110    *    </tr>
111    * </table>
112    *
113    */
114   public static final String XML_preserveRootElement = "XmlParser.preserveRootElement";
115
116   private String xsiNs = "http://www.w3.org/2001/XMLSchema-instance";
117   private boolean
118      trimWhitespace = true,
119      validating = false,
120      coalescing = false,
121      replaceEntityReferences = true,
122      preserveRootElement = false;
123   private XMLReporter reporter;
124   private XMLResolver resolver;
125   private XMLEventAllocator eventAllocator;
126
127   /**
128    * Sets the specified property value.
129    * @param property The property name.
130    * @param value The property value.
131    * @return <jk>true</jk> if property name was valid and property was set.
132    * @throws LockedException If bean context is locked.
133    */
134   protected boolean setProperty(String property, Object value) throws LockedException {
135      BeanContext bc = BeanContext.DEFAULT;
136      if (property.equals(XML_trimWhitespace))
137         trimWhitespace = bc.convertToType(value, Boolean.class);
138      else if (property.equals(XML_validating))
139         validating = bc.convertToType(value, Boolean.class);
140      else if (property.equals(XML_coalescing))
141         coalescing = bc.convertToType(value, Boolean.class);
142      else if (property.equals(XML_replaceEntityReferences))
143         replaceEntityReferences = bc.convertToType(value, Boolean.class);
144      else if (property.equals(XML_xsiNs))
145         xsiNs = value.toString();
146      else if (property.equals(XML_reporter) && value instanceof XMLReporter)
147         reporter = (XMLReporter)value;
148      else if (property.equals(XML_resolver) && value instanceof XMLResolver)
149         resolver = (XMLResolver)value;
150      else if (property.equals(XML_eventAllocator) && value instanceof XMLEventAllocator)
151         eventAllocator = (XMLEventAllocator)value;
152      else if (property.equals(XML_preserveRootElement))
153         preserveRootElement = bc.convertToType(value, Boolean.class);
154      else
155         return false;
156      return true;
157   }
158

160    * Returns the current {@link #XML_xsiNs} value.

162    */

164      return xsiNs;
165   }

173   }

175   /**
176    * Returns the current {@link #XML_preserveRootElement} value.
177    * @return The current {@link #XML_preserveRootElement} value.
178    */
179   public boolean isPreserveRootElement() {
180      return preserveRootElement;
181   }
182
183   /**
184    * Returns the current {@link #XML_validating} value.
185    * @return The current {@link #XML_validating} value.
186    */
187   public boolean isValidating() {
188      return validating;
189   }
190
191   /**
192    * Returns the current {@link #XML_coalescing} value.
193    * @return The current {@link #XML_coalescing} value.
194    */
195   public boolean isCoalescing() {
196      return coalescing;
197   }
198
199   /**
200    * Returns the current {@link #XML_replaceEntityReferences} value.
201    * @return The current {@link #XML_replaceEntityReferences} value.
202    */
203   public boolean isReplaceEntityReferences() {
204      return replaceEntityReferences;
205   }
206
207   /**
208    * Returns the current {@link #XML_reporter} value.
209    * @return The current {@link #XML_reporter} value.
210    */
211   public XMLReporter getReporter() {
212      return reporter;
213   }

216    * Returns the current {@link #XML_reporter} value.
217    * @return The current {@link #XML_reporter} value.

219   public XMLResolver getResolver() {
220      return resolver;
221   }

224    * Returns the current {@link #XML_eventAllocator} value.
225    * @return The current {@link #XML_eventAllocator} value.
226    */
227   public XMLEventAllocator getEventAllocator() {
228      return eventAllocator;
229   }
