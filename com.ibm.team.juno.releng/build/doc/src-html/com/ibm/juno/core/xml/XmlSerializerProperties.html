009package com.ibm.juno.core.xml;
010
011import com.ibm.juno.core.*;
012import com.ibm.juno.core.serializer.*;
013
014/**
015 * Configurable properties on the {@link XmlSerializer} class.
016 * <p>
017 *    Use the {@link XmlSerializer#setProperty(String, Object)} method to set property values.
018 * <p>
019 *    In addition to these properties, the following properties are also applicable for {@link XmlSerializer}.
020 * <ul>
021 *    <li>{@link SerializerProperties}
022 *    <li>{@link BeanContextProperties}
023 * </ul>
024 *
025 * @author James Bognar (jbognar@us.ibm.com)
026 */
027public final class XmlSerializerProperties implements Cloneable {
028
029   /**
030    * Add JSON type attributes to output ({@link Boolean}, default=<jk>false</jk>).
031    * <p>
032    * If <js>true</jk>, {@code type} attributes will be added to elements in the XML for number/boolean/null nodes.
033    */
034   public static final String XML_addJsonTypeAttrs = "XmlSerializer.addJsonTypeAttrs";
035
036   /**
037    * Add JSON type attributes for strings to output ({@link Boolean}, default=<jk>false</jk>).
038    * <p>
039    * If <jk>true</jk>, {@code type} attributes will be added to elements in the XML for string nodes.
040    * <p>
041    * By default, these attributes are not added, and the parser will assume that the content type
042    * of the node is string by default.
043    * <p>
044    * This feature is disabled if {@link #XML_addJsonTypeAttrs} is disabled.
045    */
046   public static final String XML_addJsonStringTypeAttrs = "XmlSerializer.addJsonStringTypeAttrs";
047
048   /**
049    * Enable support for XML namespaces ({@link Boolean}, default=<jk>true</jk>).
050    * <p>
051    * If not enabled, XML output will not contain any namespaces regardless of any other settings.
052    */
053   public static final String XML_enableNamespaces = "XmlSerializer.enableNamespaces";
054
055   /**
056    * Auto-detect namespace usage ({@link Boolean}, default=<jk>true</jk>).
057    * <p>
058    * Detect namespace usage before serialization.
059    * <p>
060    * Used in conjunction with {@link #XML_addNamespaceUrisToRoot} to reduce
061    * the list of namespace URLs appended to the root element to only those
062    * that will be used in the resulting document.
063    * <p>
064    * If enabled, then the data structure will first be crawled looking for
065    * namespaces that will be encountered before the root element is
066    * serialized.
067    * <p>
068    * This setting is ignored if {@link #XML_enableNamespaces} is not enabled.
069    * <p>
070    * <b>IMPORTANT NOTE:</b>
071    * Auto-detection of namespaces can be costly performance-wise.
072    * In high-performance environments, it's recommended that namespace detection be
073    *    disabled, and that namespaces be manually defined through the {@link #XML_namespaces} property.
074    */
075   public static final String XML_autoDetectNamespaces = "XmlSerializer.autoDetectNamespaces";
076
077   /**
078    * Add namespace URLs to the root element ({@link Boolean}, default=<jk>true</jk>).
079    * <p>
080    * Use this setting to add {@code xmlns:x} attributes to the root
081    * element for the default and all mapped namespaces.
082    * <p>
083    * This setting is ignored if {@link #XML_enableNamespaces} is not enabled.
084    */
085   public static final String XML_addNamespaceUrisToRoot = "XmlSerializer.addNamespaceUrisToRoot";
086
087   /**
088    * Default namespace URI ({@link String}, default=<jk>null</jk>).
089    * <p>
090    * Specifies the default namespace URI for this document.
091    */
092   public static final String XML_defaultNamespaceUri = "XmlSerializer.defaultNamespaceUri";
093
094   /**
095    * XMLSchema namespace ({@link Namespace}, default=<code>{name:<js>'xs'</js>,uri:<js>'http://www.w3.org/2001/XMLSchema'</js>}</code>).
096    * <p>
097    * Specifies the namespace for the <code>XMLSchema</code> namespace, used by the schema generated
098    * by the {@link XmlSchemaSerializer} class.
099    */
100   public static final String XML_xsNamespace = "XmlSerializer.xsNamespace";
101
102   /**
103    * XMLSchema-Instance namespace ({@link Namespace}, default=<code>{name:<js>'xsi'</js>,uri:<js>'http://www.w3.org/2001/XMLSchema-instance'</js>}</code>).
104    * <p>
105    * Specifies the namespace of the <code>XMLSchema-instance</code> namespace used for<code>nil=<jk>true</jk></code> attributes.
106    */
107   public static final String XML_xsiNamespace = "XmlSerializer.xsiNamespace";
108
109   /**
110    * Default namespaces (<code>Set&lt;Namespace&gt;</code>, default=empty set).
111    * <p>
112    * The default list of namespaces associated with this serializer.
113    */
114   public static final String XML_namespaces = "XmlSerializer.namespaces";
115
116   boolean
117      addJsonTypeAttrs = false,
118      addJsonStringTypeAttrs = false,
119      autoDetectNamespaces = true,
120      enableNamespaces = true,
121      addNamespaceUrlsToRoot = true;
122
123   String defaultNamespace = "{juno:'http://www.ibm.com/2013/Juno'}";
124
125   Namespace
126      xsiNamespace = NamespaceFactory.get("xsi", "http://www.w3.org/2001/XMLSchema-instance"),
127      xsNamespace = NamespaceFactory.get("xs", "http://www.w3.org/2001/XMLSchema");
128
129   Namespace[] namespaces = new Namespace[0];
130
131   /**
132    * Sets the specified property value.
133    * @param property The property name.
134    * @param value The property value.
135    * @return <jk>true</jk> if property name was valid and property was set.
136    * @throws LockedException If the bean context is locked.
137    */
138   public boolean setProperty(String property, Object value) throws LockedException {
139      BeanContext bc = BeanContext.DEFAULT;
140      if (property.equals(XML_addJsonTypeAttrs))
141         addJsonTypeAttrs = bc.convertToType(value, Boolean.class);
142      else if (property.equals(XML_addJsonStringTypeAttrs))
143         addJsonStringTypeAttrs = bc.convertToType(value, Boolean.class);
144      else if (property.equals(XML_enableNamespaces))
145         enableNamespaces = bc.convertToType(value, Boolean.class);
146      else if (property.equals(XML_autoDetectNamespaces))
147         autoDetectNamespaces = bc.convertToType(value, Boolean.class);
148      else if (property.equals(XML_addNamespaceUrisToRoot))
149         addNamespaceUrlsToRoot = bc.convertToType(value, Boolean.class);
150      else if (property.equals(XML_defaultNamespaceUri))
151         defaultNamespace = (String)value;
152      else if (property.equals(XML_xsiNamespace))
153         xsiNamespace = NamespaceFactory.parseNamespace(value);
154      else if (property.equals(XML_xsNamespace))
155         xsNamespace = NamespaceFactory.parseNamespace(value);
156      else if (property.equals(XML_namespaces))
157         namespaces = NamespaceFactory.parseNamespaces(value);
158      else
159         return false;
160      return true;
161   }
162
163   @Override /* Cloneable */
164   public XmlSerializerProperties clone() {
165      try {
166         return (XmlSerializerProperties)super.clone();
167      } catch (CloneNotSupportedException e) {
168         throw new RuntimeException(e); // Shouldn't happen
169      }
170   }
