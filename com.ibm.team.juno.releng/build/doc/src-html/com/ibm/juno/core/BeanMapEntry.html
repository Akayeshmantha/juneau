009package com.ibm.juno.core;
010
011import java.util.*;
012
013import com.ibm.juno.core.annotation.*;
014import com.ibm.juno.core.filter.*;
015
016/**
017 * Represents a single entry in a bean map.
018 * <p>
019 *    This class can be used to get and set property values on a bean, or to get metadata on a property.
020 *
021 * <h6 class='topic'>Examples</h6>
022 * <p class='bcode'>
023 *    <jc>// Construct a new bean</jc>
024 *    Person p = <jk>new</jk> Person();
025 *
026 *    <jc>// Wrap it in a bean map</jc>
027 *    BeanMap&lt;Person&gt; b = BeanContext.<jsf>DEFAULT</jsf>.forBean(p);
028 *
029 *    <jc>// Get a reference to the birthDate property</jc>
030 *    BeanMapEntry birthDate = b.getProperty(<js>"birthDate"</js>);
031 *
032 *    <jc>// Set the property value</jc>
033 *    birthDate.setValue(<jk>new</jk> Date(1, 2, 3, 4, 5, 6));
034 *
035 *    <jc>// Or if the DateFilter.DEFAULT_ISO8601DT is registered with the bean context, set a filtered value</jc>
036 *    birthDate.setFilteredValue(<js>"'1901-03-03T04:05:06-5000'"</js>);
037 * </p>
038 *
039 * @author James Bognar (jbognar@us.ibm.com)
040 *
041 * @param <T> The bean type.
042 */
043public class BeanMapEntry<T> implements Map.Entry<String,Object> {
044   private final BeanMap<T> beanMap;
045   private final BeanPropertyMeta<T> meta;
046
047   /**
048    * Constructor.
049    *
050    * @param beanMap The bean map that this entry belongs to.
051    * @param property The bean property.
052    */
053   protected BeanMapEntry(BeanMap<T> beanMap, BeanPropertyMeta<T> property) {
054      this.beanMap = beanMap;
055      this.meta = property;
056   }
057
058   @Override /* Map.Entry */
059   public String getKey() {
060      return meta.getName();
061   }
062
063   /**
064    * Returns the value of this property.
065    * <p>
066    * If there is a {@link PojoFilter} associated with this bean property or bean property type class, then
067    *    this method will return the filtered value.
068    * For example, if the bean property type class is a {@link Date} and the bean property has the
069    *    {@link com.ibm.juno.core.filters.DateFilter.ISO8601DT} filter associated with it through the
070    *    {@link BeanProperty#filter() @BeanProperty.filter()} annotation, this method will return a String
071    *    containing an ISO8601 date-time string value.
072    */
073   @Override /* Map.Entry */
074   public Object getValue() {
075      return meta.get(this.beanMap);
076   }
077
078   /**
079    * Sets the value of this property.
080    * <p>
081    * If the property is an array of type {@code X}, then the value can be a {@code Collection<X>} or {@code X[]} or {@code Object[]}.
082    * <p>
083    * If the property is a bean type {@code X}, then the value can either be an {@code X} or a {@code Map}.
084    * <p>
085    * If there is a {@link PojoFilter} associated with this bean property or bean property type class, then
086    *    you must pass in a filtered value.
087    * For example, if the bean property type class is a {@link Date} and the bean property has the
088    *    {@link com.ibm.juno.core.filters.DateFilter.ISO8601DT} filter associated with it through the
089    *    {@link BeanProperty#filter() @BeanProperty.filter()} annotation, the value being passed in must be
090    *    a String containing an ISO8601 date-time string value.
091    *
092    * @return  The set value after it's been converted.
093    */
094   @Override /* Map.Entry */
095   public Object setValue(Object value) {
096      return meta.set(this.beanMap, value);
097   }
098
099   /**
100    * Returns the bean map that contains this property.
101    *
102    * @return The bean map that contains this property.
103    */
104   public BeanMap<T> getBeanMap() {
105      return this.beanMap;
106   }
107
108   /**
109    * Returns the metadata about this bean property.
110    *
111    * @return Metadata about this bean property.
112    */
113   public BeanPropertyMeta<T> getMeta() {
114      return this.meta;
115   }
116
117   @Override /* Object */
118   public String toString() {
119      return this.getKey() + "=" + this.getValue();
120   }
121}
