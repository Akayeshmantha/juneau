022/**
023 * Context object that lives for the duration of a single serialization of {@link HtmlSerializer} and its subclasses.
024 * <p>
025 *    See {@link SerializerContext} for details.
026 * </p>
027 * <p>
028 *    This class is NOT thread safe.  It is meant to be discarded after one-time use.
029 * </p>
030 *
031 * @author James Bognar (jbognar@us.ibm.com)
032 */
033public final class HtmlSerializerContext extends XmlSerializerContext {
034
035   private final String uriAnchorText, title, description, cssUrl;
036   private final String[] cssImports;
037   private final ObjectMap links;
038   private final boolean nowrap;
039
040   /**
041    * Constructor.
042    *
043    * @param beanContext The bean context being used by the serializer.
044    * @param sp Default general serializer properties.
045    * @param xsp Default XML serializer properties.
046    * @param hsp Default HTML serializer properties.
047    * @param op Override properties.
048    * @param javaMethod Java method that invoked this serializer.
049    *    When using the REST API, this is the Java method invoked by the REST call.
050    *    Can be used to access annotations defined on the method or class.
051    */
052   protected HtmlSerializerContext(BeanContext beanContext, SerializerProperties sp, XmlSerializerProperties xsp, HtmlSerializerProperties hsp, ObjectMap op, Method javaMethod) {
053      super(beanContext, sp, xsp, op, javaMethod);
054      if (op == null || op.isEmpty()) {
055         uriAnchorText = hsp.uriAnchorText;
056         title = hsp.title;
057         description = hsp.description;
058         links = hsp.links;
059         cssUrl = hsp.cssUrl;
060         cssImports = hsp.cssImports;
061         nowrap = hsp.nowrap;
062      } else {
063         uriAnchorText = op.getString(HTML_uriAnchorText, hsp.uriAnchorText);
064         title = op.getString(HTMLDOC_title, hsp.title);
065         description = op.getString(HTMLDOC_description, hsp.description);
066         ObjectMap m = op.getObjectMap(HTMLDOC_links, hsp.links);
067         if (op.containsKey(HTMLDOC_addLinks))
068            if (m == null)
069               m = op.getObjectMap(HTMLDOC_addLinks, null);
070            else
071               m.putAll(op.getObjectMap(HTMLDOC_addLinks, null));
072         links = m;
073         cssUrl = op.getString(HTMLDOC_cssUrl, hsp.cssUrl);
074         cssImports = StringUtils.split(op.getString(HTMLDOC_cssImports, null), ',');
075         nowrap = op.getBoolean(HTMLDOC_cssUrl, hsp.nowrap);
076      }
077   }
078
079   final String getUriAnchorText() {
080      return uriAnchorText;
081   }
082
083   final String getTitle() {
084      return title;
085   }
086
087   final String getDescription() {
088      return description;
089   }
090
091   final ObjectMap getLinks() {
092      return links;
093   }
094
095   final String getCssUrl() {
096      return cssUrl;
097   }
098
099   final String[] getCssImports() {
100      return cssImports;
101   }
102
103   final boolean isNoWrap() {
104      return nowrap;
105   }
106
107   /**
108    * Wraps the specified writer in a {@link HtmlSerializerWriter}.
109    */
110   @Override /* XmlSerializerContext */
111   public HtmlSerializerWriter getWriter(Writer w) {
112      if (w instanceof HtmlSerializerWriter)
113         return (HtmlSerializerWriter)w;
114      return new HtmlSerializerWriter(w, isUseIndentation(), isTrimStrings(), getQuoteChar(), getRelativeUriBase(), getAbsolutePathUriBase());
115   }
116}
