009package com.ibm.juno.core.utils;
010
011import java.util.*;
012
013import com.ibm.juno.core.*;
014
015/**
016 * Represents a wrapped {@link BeanMap} where property values can be overridden, removed, or reordered
017 *    without affecting the underlying bean.
018 * <p>
019 *    Provides the {@link #filterKeys(List)} method for specifying the keys to keep in the bean map
020 *       and in what order they should appear.
021 *
022 * @author James Bognar (jbognar@us.ibm.com)
023 * @param <T> The class type of the wrapped bean.
024 */
025@SuppressWarnings("hiding")
026public class DelegateBeanMap<T> extends BeanMap<T> {
027
028   private Set<String> keys = Collections.newSetFromMap(new LinkedHashMap<String,Boolean>());
029   private ObjectMap overrideValues = new ObjectMap();
030
031   @SuppressWarnings("unchecked")
032   DelegateBeanMap(T bean, BeanContext bc) {
033      super(bean, bc.getBeanMeta((Class<T>)bean.getClass()));
034   }
035
036   void addKey(String key) {
037      this.keys.add(key);
038   }
039
040   @Override /* Map */
041   public Object put(String key, Object val) {
042      this.overrideValues.put(key, val);
043      this.keys.add(key);
044      return null;
045   }
046
047   @Override /* Map */
048   public Object get(Object key) {
049      if (overrideValues.containsKey(key))
050         return overrideValues.get(key);
051      return super.get(key);
052   }
053
054   @Override /* Map */
055   public Set<String> keySet() {
056      return keys;
057   }
058
059   /**
060    * Remove all but the specified properties from this bean map.
061    * <p>
062    * This does not affect the underlying bean.
063    *
064    * @param keys The remaining keys in the bean map (in the specified order).
065    */
066   public void filterKeys(List<String> keys) {
067      this.keys.clear();
068      this.keys.addAll(keys);
069   }
070
071   @Override /* Map */
072   public Object remove(Object key) {
073      keys.remove(key);
074      return null;
075   }
076
077   @Override /* BeanMap */
078   public BeanMeta<T> getMeta() {
079      return new BeanMetaFiltered<T>(super.getMeta(), keys);
080   }
081
082   @Override /* Map */
083   public Set<Entry<String,Object>> entrySet() {
084      Set<Entry<String,Object>> s = Collections.newSetFromMap(new LinkedHashMap<Map.Entry<String,Object>,Boolean>());
085      for (final String key : keys) {
086         BeanMapEntry<T> bme;
087         if (overrideValues.containsKey(key))
088            bme = new BeanMapEntryOverride<T>(this, this.getPropertyMeta(key), overrideValues.get(key));
089         else
090            bme = this.getProperty(key);
091         if (bme == null)
092            throw new BeanRuntimeException(super.getClassMeta().getInnerClass(), "Property ''{0}'' not found on class.", key);
093         s.add(bme);
094      }
095      return s;
096   }
097
098   private class BeanMapEntryOverride<T2> extends BeanMapEntry<T2> {
099      Object value;
100
101      private BeanMapEntryOverride(BeanMap<T2> bm, BeanPropertyMeta<T2> bpm, Object value) {
102         super(bm, bpm);
103         this.value = value;
104      }
105
106      @Override /* Map.Entry */
107      public Object getValue() {
108         return value;
109      }
110   }
111}
