024/**
025 * Parent class for all Juno parsers.
026 *
027 *
028 * <h6 class='topic'>@Consumes annotation</h6>
029 * <p>
030 *    The media types that this parser can handle is specified through the {@link Consumes @Consumes} annotation.
031 * <p>
032 *    However, the media types can also be specified programmatically by overriding the {@link #getMediaTypes()} method.
033 *
034 *
035 * <a id='ValidDataConversions'></a><h6 class='topic'>Valid data conversions</h6>
036 *    Parsers can parse any parsable POJO types, as specified in the <a class='doclink' href='../package-summary.html#PojoCategories'>POJO Categories</a>.
037 * <p>
038 *    Some examples of conversions are shown below...
039 * </p>
040 *    <table class='styled'>
041 *       <tr>
042 *          <th>Data type</th>
043 *          <th>Class type</th>
044 *          <th>JSON example</th>
045 *          <th>XML example</th>
046 *          <th>Class examples</th>
047 *       </tr>
048 *       <tr>
049 *          <td>object</td>
050 *          <td>Maps, Java beans</td>
051 *          <td class='code'>{name:<js>'John Smith'</js>,age:21}</td>
052 *          <td class='code'><xt>&lt;object&gt;
053 *    &lt;name</xt> <xa>type</xa>=<xs>'string'</xs><xt>&gt;</xt>John Smith<xt>&lt;/name&gt;
054 *    &lt;age</xt> <xa>type</xa>=<xs>'number'</xs><xt>&gt;</xt>21<xt>&lt;/age&gt;
055 * &lt;/object&gt;</xt></td>
056 *          <td class='code'>HashMap, TreeMap&lt;String,Integer&gt;</td>
057 *       </tr>
058 *       <tr>
059 *          <td>array</td>
060 *          <td>Collections, Java arrays</td>
061 *          <td class='code'>[1,2,3]</td>
062 *          <td class='code'><xt>&lt;array&gt;
063 *    &lt;number&gt;</xt>1<xt>&lt;/number&gt;
064 *    &lt;number&gt;</xt>2<xt>&lt;/number&gt;
065 *    &lt;number&gt;</xt>3<xt>&lt;/number&gt;
066 * &lt;/array&gt;</xt></td>
067 *          <td class='code'>List&lt;Integer&gt;, <jk>int</jk>[], Float[], Set&lt;Person&gt;</td>
068 *       </tr>
069 *       <tr>
070 *          <td>number</td>
071 *          <td>Numbers</td>
072 *          <td class='code'>123</td>
073 *          <td class='code'><xt>&lt;number&gt;</xt>123<xt>&lt;/number&gt;</xt></td>
074 *          <td class='code'>Integer, Long, Float, <jk>int</jk></td>
075 *       </tr>
076 *       <tr>
077 *          <td>boolean</td>
078 *          <td>Booleans</td>
079 *          <td class='code'><jk>true</jk></td>
080 *          <td class='code'><xt>&lt;boolean&gt;</xt>true<xt>&lt;/boolean&gt;</xt></td>
081 *          <td class='code'>Boolean</td>
082 *       </tr>
083 *       <tr>
084 *          <td>string</td>
085 *          <td>CharSequences</td>
086 *          <td class='code'><js>'foobar'</js></td>
087 *          <td class='code'><xt>&lt;string&gt;</xt>foobar<xt>&lt;/string&gt;</xt></td>
088 *          <td class='code'>String, StringBuilder</td>
089 *       </tr>
090 *    </table>
091 * <p>
092 *    In addition, any class types with {@link PojoFilter PojoFilters} associated with them on the registered
093 *       {@link #getBeanContext() beanContext} can also be passed in.
094 * <p>
095 *    For example, if the {@link CalendarFilter} filter is used to generalize {@code Calendar} objects to {@code String} objects.  When registered
096 *    with this parser, you can construct {@code Calendar} objects from {@code Strings} using the following syntax...
097 * <p class='bcode'>
098 *    Calendar c = parser.parse(<js>"'Sun Mar 03 04:05:06 EST 2001'"</js>, GregorianCalendar.<jk>class</jk>);
099 * <p>
100 *    If <code>Object.<jk>class</jk></code> is specified as the target type, then the parser
101 *    automatically determines the data types and generates the following object types...
102 * </p>
103 * <table class='styled'>
104 *    <tr><th>JSON type</th><th>Class type</th></tr>
105 *    <tr><td>object</td><td>{@link ObjectMap}</td></tr>
106 *    <tr><td>array</td><td>{@link ObjectList}</td></tr>
107 *    <tr><td>number</td><td>{@link Number} <br>(depending on length and format, could be {@link Integer}, {@link Double}, {@link Float}, etc...)</td></tr>
108 *    <tr><td>boolean</td><td>{@link Boolean}</td></tr>
109 *    <tr><td>string</td><td>{@link String}</td></tr>
110 * </table>
111 *
112 *
113 * <a id='SupportedTypes'></a><h6 class='topic'>Supported types</h6>
114 * <p>
115 *    Several of the methods below take {@link Type} parameters to identify the type of
116 *       object to create.  Any of the following types can be passed in to these methods...
117 * </p>
118 * <ul>
119 *    <li>{@link ClassMeta}
120 *    <li>{@link Class}
121 *    <li>{@link ParameterizedType}
122 *    <li>{@link GenericArrayType}
123 * </ul>
124 * <p>
125 *    However, {@code ParameterizedTypes} and {@code GenericArrayTypes} should not contain
126 *       {@link WildcardType WildcardTypes} or {@link TypeVariable TypeVariables}.
127 * <p>
128 *    Passing in <jk>null</jk> or <code>Object.<jk>class</jk></code> typically signifies that it's up to the parser
129 *    to determine what object type is being parsed parsed based on the rules above.
130
131 *
132 * @author James Bognar (jbognar@us.ibm.com)
133 * @param <R> The input type (e.g. Reader or InputStream)
134 */
135public abstract class Parser<R> extends CoreApi {

150    * Workhorse method.  Subclasses are expected to implement this method.

152    * @param in The input stream or reader containing the input.
153    * @param estimatedSize The estimated size of the input, or <code>-1</code> if unknown.
154    * @param type The class type of the object to create.
155    *    If <jk>null</jk> or <code>Object.<jk>class</jk></code>, object type is based on what's being parsed.
156    *    For example, when parsing JSON text, it may return a <code>String</code>, <code>Number</code>, <code>ObjectMap</code>, etc...
157    * @param ctx The runtime context object returned by {@link #createContext(ObjectMap, Method, Object)}.

161    * @throws ParseException If the input contains a syntax error or is malformed, or is not valid for the specified type.
162    * @throws IOException If a problem occurred trying to read from the reader.
163    */

177   /**
178    * Parses the content of the reader and creates an object of the specified type.
179    *
180    * @param in The input stream or reader containing the input.
181    * @param estimatedSize The estimated size of the input, or <code>-1</code> if unknown.
182    * @param type The class type of the object to create.
183    *    If <jk>null</jk> or <code>Object.<jk>class</jk></code>, object type is based on what's being parsed.
184    *    For example, when parsing JSON text, it may return a <code>String</code>, <code>Number</code>, <code>ObjectMap</code>, etc...
185    * @param ctx The runtime context object returned by {@link #createContext(ObjectMap, Method, Object)}.
186    * @param <T> The class type of the object to create.
187    * @return The parsed object.
188    * @throws ParseException If the input contains a syntax error or is malformed, or is not valid for the specified type.
189    * @throws IOException If a problem occurred trying to read from the reader.
190    */
191   public final <T> T parse(R in, int estimatedSize, ClassMeta<T> type, ParserContext ctx) throws ParseException, IOException {
192      try {
193         if (in == null)
194            throw new IOException("Null input stream or reader passed to parser.");
195         return doParse(in, estimatedSize, type, ctx);
196      } catch (RuntimeException e) {
197         throw new ParseException(e);
198      } finally {
199         ctx.close();
200      }
201   }

204    * Parses the content of the reader and creates an object of the specified type.
205    * <p>
206    * Equivalent to calling <code>parser.parse(in, type, <jk>null</jk>);</code>
207    *
208    * @param in The input stream or reader containing the input.
209    * @param estimatedSize The estimated size of the input, or <code>-1</code> if unknown.
210    * @param type The class type of the object to create.
211    *    If <jk>null</jk> or <code>Object.<jk>class</jk></code>, object type is based on what's being parsed.
212    *    For example, when parsing JSON text, it may return a <code>String</code>, <code>Number</code>, <code>ObjectMap</code>, etc...
213    * @param <T> The class type of the object to create.
214    * @return The parsed object.
215    * @throws ParseException If the input contains a syntax error or is malformed, or is not valid for the specified type.
216    * @throws IOException If a problem occurred trying to read from the reader.
217    */
218   public final <T> T parse(R in, int estimatedSize, ClassMeta<T> type) throws ParseException, IOException {
219      ParserContext ctx = createContext();
220      return parse(in, estimatedSize, type, ctx);
221   }

223   /**
224    * Parses input into the specified object type.
225    *
226    * <dl>
227    *    <dt>Example:</dt>
228    *    <dd>
229    *       <p class='bcode'>
230    *    ReaderParser p = JsonParser.<jsf>DEFAULT</jsf>;
231    *    MyBean b = p.parse(json, MyBean.<jk>class</jk>);
232    *       </p>
233    *       <p>
234    *       This method equivalent to the following code:
235    *       <p class='bcode'>
236    *    ReaderParser p = JsonParser.<jsf>DEFAULT</jsf>;
237    *    ClassMeta&lt;MyBean&gt; cm = p.getBeanContext().getClassMeta(MyBean.<jk>class</jk>);
238    *    MyBean b = p.parse(json, cm, <jk>null</jk>);
239    *       </p>
240    *    </dd>
241    * </dl>
242    *
243    * @param <T> The class type of the object to create.
244    * @param in The input stream or reader containing the input.
245    * @param estimatedSize The estimated size of the input, or <code>-1</code> if unknown.
246    * @param type The class type of the object to create.
247    * @return The parsed object.
248    * @throws ParseException If the input contains a syntax error or is malformed, or is not valid for the specified type.
249    * @throws IOException If a problem occurred trying to read from the reader.
250    */
251   public final <T> T parse(R in, int estimatedSize, Class<T> type) throws ParseException, IOException {
252      ClassMeta<T> cm = getBeanContext().getClassMeta(type);
253      return parse(in, estimatedSize, cm);
254   }

256   /**
257    * Parses input into a map with specified key and value types.
258    *
259    * <dl>
260    *    <dt>Example:</dt>
261    *    <dd>
262    *       <p class='bcode'>
263    *    ReaderParser p = JsonParser.<jsf>DEFAULT</jsf>;
264    *    Map&lt;String,MyBean&gt; m = p.parseMap(json, LinkedHashMap.<jk>class</jk>, String.<jk>class</jk>, MyBean.<jk>class</jk>);
265    *       </p>
266    *       <p>
267    *          A simpler approach is often to just extend the map class you want and just use the normal {@link #parse(Object, int, Class)} method:
268    *       </p>
269    *       <p class='bcode'>
270    *    <jk>public static class</jk> MyMap <jk>extends</jk> LinkedHashMap&lt;String,MyBean&gt; {}
271    *
272    *    ReaderParser p = JsonParser.<jsf>DEFAULT</jsf>;
273    *    Map&lt;String,MyBean&gt; m = p.parse(json, MyMap.<jk>class</jk>);
274    *       </p>
275    *       <p>
276    *          This method equivalent to the following code:
277    *       </p>
278    *       <p class='bcode'>
279    *    ReaderParser p = JsonParser.<jsf>DEFAULT</jsf>;
280    *    ClassMeta&lt;Map&lt;String,MyBean&gt;&gt; cm = p.getBeanContext().getMapClassMeta(LinkedList.<jk>class</jk>, String.<jk>class</jk>, MyBean.<jk>class</jk>);
281    *    Map&ltString,MyBean&gt; m = p.parse(json, cm, <jk>null</jk>);
282    *       </p>
283    *    </dd>
284    * </dl>
285    *
286    * @param <T> The class type of the object to create.
287    * @param in The input stream or reader containing the input.
288    * @param estimatedSize The estimated size of the input, or <code>-1</code> if unknown.
289    * @param mapClass The map class type.
290    * @param keyClass The key class type.
291    * @param valueClass The value class type.
292    * @return The parsed object.
293    * @throws ParseException If the input contains a syntax error or is malformed, or is not valid for the specified type.
294    * @throws IOException If a problem occurred trying to read from the reader.
295    */
296   public final <K,V,T extends Map<K,V>> T parseMap(R in, int estimatedSize, Class<T> mapClass, Class<K> keyClass, Class<V> valueClass) throws ParseException, IOException {
297      ClassMeta<T> cm = getBeanContext().getMapClassMeta(mapClass, keyClass, valueClass);
298      return parse(in, estimatedSize, cm);
299   }

399   /**
400    * Converts the specified string to the specified type.
401    *
402    * @param outer The outer object if we're converting to an inner object that needs to be created within the context of an outer object.
403    * @param s The string to convert.
404    * @param type The class type to convert the string to.
405    * @return The string converted as an object of the specified type.
406    * @throws ParseException If the input contains a syntax error or is malformed, or is not valid for the specified type.
407    * @param <T> The class type to convert the string to.
408    */
409   @SuppressWarnings({ "unchecked", "rawtypes" })
410   protected <T> T convertAttrToType(Object outer, String s, ClassMeta<T> type) throws ParseException {
411      if (s == null)
412         return null;
413
414      if (type == null)
415         type = (ClassMeta<T>)object();
416      PojoFilter filter = type.getPojoFilter();
417      ClassMeta<?> gType = type.getFilteredClassMeta();
418
419      Object o = s;
420      if (gType.isChar())
421         o = s.charAt(0);
422      else if (gType.isNumber())
423         o = parseNumber(s, (Class<? extends Number>)gType.getInnerClass());
424      else if (gType.isBoolean())
425         o = Boolean.parseBoolean(s);
426      else if (! (gType.isCharSequence() || gType.isObject())) {
427         if (gType.canCreateNewInstanceFromString(outer)) {
428            try {
429               o = gType.newInstanceFromString(outer, s);
430            } catch (Exception e) {
431               throw new ParseException("Unable to construct new object of type ''{0}'' from input string ''{1}''", type, s).initCause(e);
432            }
433         } else {
434            throw new ParseException("Invalid conversion from string to class ''{0}''", type);
435         }
436      }
437
438      if (filter != null)
439         o = filter.unfilter(o, type);
440
441      return (T)o;
442   }

485   /**
486    * Method that gets called when an unknown bean property name is encountered.
487    *
488    * @param ctx The parser context.
489    * @param propertyName The unknown bean property name.
490    * @param beanMap The bean that doesn't have the expected property.
491    * @param line The line number where the property was found.  <code>-1</code> if line numbers are not available.
492    * @param col The column number where the property was found.  <code>-1</code> if column numbers are not available.
493    * @throws ParseException Automatically thrown if {@link BeanContextProperties#BEAN_ignoreUnknownBeanProperties} setting
494    *    on this parser is <jk>false</jk>
495    * @param <T> The class type of the bean map that doesn't have the expected property.
496    */
497   protected <T> void onUnknownProperty(ParserContext ctx, String propertyName, BeanMap<T> beanMap, int line, int col) throws ParseException {
498      if (propertyName.equals("uri") || propertyName.equals("type") || propertyName.equals("_class"))
499         return;
500      if (! ctx.getBeanContext().isIgnoreUnknownBeanProperties())
501         throw new ParseException(line, col, "Unknown property ''{0}'' encountered while trying to parse into class ''{1}''", propertyName, beanMap.getClassMeta());
502      if (listeners.size() > 0)
503         for (ParserListener listener : listeners)
504            listener.onUnknownProperty(propertyName, beanMap.getClassMeta().getInnerClass(), beanMap.getBean(), line, col);
505   }

508   /**
509    * Returns the media types handled based on the value of the {@link Consumes} annotation on the parser class.
510    * <p>
511    * This method can be overridden by subclasses to determine the media types programatically.
512    *
513    * @return The list of media types.  Never <jk>null</jk>.
514    */
515   public String[] getMediaTypes() {
516      if (mediaTypes == null) {
517         Consumes c = ReflectionUtils.getAnnotation(Consumes.class, getClass());
518         if (c == null)
519            throw new RuntimeException(MessageFormat.format("Class ''{0}'' is missing the @Consumes annotation", getClass().getName()));
520         mediaTypes = c.value();
521      }
522      return mediaTypes;
523   }
