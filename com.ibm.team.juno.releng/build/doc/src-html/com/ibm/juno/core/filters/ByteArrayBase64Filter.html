009package com.ibm.juno.core.filters;
010
011import com.ibm.juno.core.*;
012import com.ibm.juno.core.filter.*;
013import com.ibm.juno.core.parser.*;
014import com.ibm.juno.core.serializer.*;
015import com.ibm.juno.core.utils.*;
016
017/**
018 * Transforms <code><jk>byte</jk>[]</code> arrays to BASE-64 encoded {@link String Strings}.
019 *
020 * @author James Bognar (jbognar@us.ibm.com)
021 */
022public class ByteArrayBase64Filter extends PojoFilter<byte[],String> {
023
024   /**
025    * Converts the specified <code><jk>byte</jk>[]</code> to a {@link String}.
026    */
027   @Override /* PojoFilter */
028   public String filter(byte[] b) throws SerializeException {
029      try {
030         return StringUtils.base64Encode(b);
031      } catch (Exception e) {
032         throw new SerializeException(e);
033      }
034   }
035
036   /**
037    * Converts the specified {@link String} to a <code><jk>byte</jk>[]</code>.
038    */
039   @Override /* PojoFilter */
040   public byte[] unfilter(String s, ClassMeta<?> hint) throws ParseException {
041      try {
042         return StringUtils.base64Decode(s);
043      } catch (Exception e) {
044         throw new ParseException(e);
045      }
046   }
047}
