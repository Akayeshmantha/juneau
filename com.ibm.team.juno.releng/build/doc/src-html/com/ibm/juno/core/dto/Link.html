009package com.ibm.juno.core.dto;
010
011import java.text.*;
012
013import com.ibm.juno.core.html.*;
014import com.ibm.juno.core.urlencoding.*;
015import com.ibm.juno.core.utils.*;
016
017/**
018 * Simple bean that implements a hyperlink for the HTML serializer.
019 * <p>
020 *    The name and url properties correspond to the following parts of a hyperlink in an HTML document...
021 * <p class='bcode'>
022 *    <xt>&lt;a</xt> <xa>href</xa>=<xs>'href'</xs><xt>&gt;</xt>name<xt>&lt;/a&gt;</xt>
023 * <p>
024 *    When encountered by the {@link HtmlSerializer} class, this object gets converted to a hyperlink.<br>
025 *    All other serializers simply convert it to a simple bean.
026 *
027 * @author James Bognar (jbognar@us.ibm.com)
028 */
029@HtmlLink(nameProperty = "name", hrefProperty = "href")
030public class Link implements Comparable<Link> {
031   private String name, href;
032
033   /** No-arg constructor. */
034   public Link() {}
035
036   /**
037    * Constructor.
038    *
039    * @param name Corresponds to the text inside of the <xt>&lt;A&gt;</xt> element.
040    * @param href Corresponds to the value of the <xa>href</xa> attribute of the <xt>&lt;A&gt;</xt> element.
041    * @param hrefArgs Optional arguments for {@link MessageFormat} style arguments in the href.
042    */
043   public Link(String name, String href, Object...hrefArgs) {
044      setName(name);
045      setHref(href, hrefArgs);
046   }
047
048   //--------------------------------------------------------------------------------
049   // Bean properties
050   //--------------------------------------------------------------------------------
051
052   /**
053    * Bean property getter:  <property>name</property>.
054    * Corresponds to the text inside of the <xt>&lt;A&gt;</xt> element.
055    *
056    * @return The value of the <property>name</property> property on this bean, or <jk>null</jk> if it is not set.
057    */
058   public String getName() {
059      return name;
060   }
061
062   /**
063    * Bean property setter:  <property>name</property>.
064    *
065    * @param name The new value for the <property>name</property> property on this bean.
066    * @return This object (for method chaining).
067    */
068   public Link setName(String name) {
069      this.name = name;
070      return this;
071   }
072
073   /**
074    * Bean property getter:  <property>href</property>.
075    * Corresponds to the value of the <xa>href</xa> attribute of the <xt>&lt;A&gt;</xt> element.
076    *
077    * @return The value of the <property>href</property> property on this bean, or <jk>null</jk> if it is not set.
078    */
079   public String getHref() {
080      return href;
081   }
082
083   /**
084    * Bean property setter:  <property>href</property>.
085    *
086    * @param href The new value for the <property>href</property> property on this bean.
087    * @return This object (for method chaining).
088    */
089   public Link setHref(String href) {
090      setHref(href, new Object[0]);
091      return this;
092   }
093
094   /**
095    * Bean property setter:  <property>href</property>.
096    * Same as {@link #setHref(String)} except allows for {@link MessageFormat} style arguments.
097    *
098    * @param href The new href.
099    * @param args Optional message format arguments.
100    * @return This object (for method chaining).
101    */
102   public Link setHref(String href, Object...args) {
103      for (int i = 0; i < args.length; i++)
104         args[i] = UrlEncodingSerializer.DEFAULT.serializeUrlPart(args[i]);
105      this.href = (args.length > 0 ? MessageFormat.format(href, args) : href);
106      return this;
107   }
108
109   /**
110    * Returns the name so that the {@link PojoQuery} class can search against it.
111    */
112   @Override /* Object */
113   public String toString() {
114      return name;
115   }
116
117   @Override /* Comparable */
118   public int compareTo(Link o) {
119      return name.compareTo(o.name);
120   }
121
122   @Override /* Object */
123   public boolean equals(Object o) {
124      if (! (o instanceof Link))
125         return false;
126      return (compareTo((Link)o) == 0);
127   }
128
129   @Override /* Object */
130   public int hashCode() {
131      return super.hashCode();
132   }
133}
