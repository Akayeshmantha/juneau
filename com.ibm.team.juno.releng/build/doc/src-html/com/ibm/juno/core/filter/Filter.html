009package com.ibm.juno.core.filter;
010
011import com.ibm.juno.core.*;
012
013/**
014 * Parent class for all bean and POJO filters.
015 *
016 *
017 * <h6 class='topic'>Description</h6>
018 * <p>
019 *    Filters are used to alter how POJOs are handled by bean contexts (and subsequently serializers and parsers).
020 *    The are a very powerful feature of the Juno framework that allows virtually any POJO to be serialized and parsed.
021 *    For example, they can be used to...
022 * <ul>
023 *    <li>Convert a non-serializable POJO into a serializable POJO during serialization (and optionally vis-versa during parsing).
024 *    <li>Control various aspects of beans, such as what properties are visible, bean subclasses, etc...
025 * </ul>
026 * <p>
027 *    There are 2 subclasses of filters:
028 * <ul>
029 *    <li>{@link PojoFilter} - Non-bean filters for converting POJOs into serializable equivalents.
030 *    <li>{@link BeanFilter} - Bean filters for configuring how beans are handled.
031 * </ul>
032 * <p>
033 *    Filters are associated with bean contexts (and serializers/parsers) through the {@link BeanContextFactory#addFilters(Class[])}
034 *       and {@link CoreApi#addFilters(Class[])} methods.
035 *
036 *
037 * <h6 class='topic'>Additional information</h6>
038 *    See {@link com.ibm.juno.core.filter} for more information.
039 *
040 *
041 * @author James Bognar (jbognar@us.ibm.com)
042 */
043public class Filter {
044
045   /** Represents no filter. */
046   public static class NULL extends Filter {}
047
048   /** The filter subtype */
049   public static enum FilterType {
050      /** PojoFilter */
051      POJO,
052      /** BeanFilter */
053      BEAN
054   }
055
056   /** The class that this filter applies to. */
057   protected Class<?> forClass;
058
059   /** The bean context that this filter instance belongs to. */
060   protected BeanContext beanContext;
061
062   /** Whether this is a BeanFilter or PojoFilter. */
063   protected FilterType type = FilterType.POJO;
064
065   Filter() {}
066
067   Filter(Class<?> forClass) {
068      this.forClass = forClass;
069   }
070
071
072   /**
073    * Returns the class that this filter applies to.
074    *
075    * @return The class that this filter applies to.
076    */
077   public Class<?> forClass() {
078      return forClass;
079   }
080
081   /**
082    * Returns the implementation class.
083    * Useful for debugging when calling {@link BeanContext#toString()}.
084    *
085    * @return The implementation class of this filter.
086    */
087   public Class<?> getImplClass() {
088      return this.getClass();
089   }
090
091   /**
092    * Returns whether this is an instance of {@link PojoFilter} or {@link BeanFilter}.
093    *
094    * @return The filter type.
095    */
096   public FilterType getType() {
097      return type;
098   }
099
100   /**
101    * Returns the {@link BeanContext} that created this filter.
102    *
103    * @return The bean context that created this filter.
104    */
105   protected BeanContext getBeanContext() {
106      return beanContext;
107   }
108
109   /**
110    * Sets the bean context that this filter instance was created by.
111    *
112    * @param beanContext The bean context that created this filter.
113    * @return This object (for method chaining).
114    */
115   public Filter setBeanContext(BeanContext beanContext) {
116      this.beanContext = beanContext;
117      return this;
118   }
119
120   @Override /* Object */
121   public int hashCode() {
122      return getClass().getName().hashCode() + forClass().getName().hashCode();
123   }
124
125   /**
126    * Checks if the specified filter class is the same as this one.
127    *
128    * @param f The filter to check.
129    * @return <jk>true</jk> if the specified filter is equivalent to this one.
130    */
131   public boolean isSameAs(Filter f) {
132      return f.getClass().equals(getClass()) && f.forClass().equals(forClass());
133   }
134}
