009package com.ibm.juno.core.json;
010
011import static com.ibm.juno.core.json.JsonSerializerProperties.*;
012
013import java.io.*;
014import java.lang.reflect.*;
015
016import com.ibm.juno.core.*;
017import com.ibm.juno.core.serializer.*;
018
019/**
020 * Context object that lives for the duration of a single serialization of {@link JsonSerializer} and its subclasses.
021 * <p>
022 *    See {@link SerializerContext} for details.
023 * </p>
024 * <p>
025 *    This class is NOT thread safe.  It is meant to be discarded after one-time use.
026 * </p>
027 *
028 * @author James Bognar (jbognar@us.ibm.com)
029 */
030public final class JsonSerializerContext extends SerializerContext {
031
032   private final boolean simpleMode, useWhitespace, escapeSolidus;
033
034   /**
035    * Constructor.
036    * @param beanContext The bean context being used by the serializer.
037    * @param sp Default general serializer properties.
038    * @param jsp Default JSON serializer properties.
039    * @param op Override properties.
040    * @param javaMethod Java method that invoked this serializer.
041    *    When using the REST API, this is the Java method invoked by the REST call.
042    *    Can be used to access annotations defined on the method or class.
043    */
044   protected JsonSerializerContext(BeanContext beanContext, SerializerProperties sp, JsonSerializerProperties jsp, ObjectMap op, Method javaMethod) {
045      super(beanContext, sp, op, javaMethod);
046      if (op == null || op.isEmpty()) {
047         simpleMode = jsp.simpleMode;
048         useWhitespace = jsp.useWhitespace;
049         escapeSolidus = jsp.escapeSolidus;
050      } else {
051         simpleMode = op.getBoolean(JSON_simpleMode, jsp.simpleMode);
052         useWhitespace = op.getBoolean(JSON_useWhitespace, jsp.useWhitespace);
053         escapeSolidus = op.getBoolean(JSON_escapeSolidus, jsp.escapeSolidus);
054      }
055   }
056
057   final boolean isSimpleMode() {
058      return simpleMode;
059   }
060
061   final boolean isUseWhitespace() {
062      return useWhitespace;
063   }
064
065   final boolean isEscapeSolidus() {
066      return escapeSolidus;
067   }
068
069   /**
070    * Wraps the specified writer inside a {@link JsonSerializerWriter}.
071    *
072    * @param out The writer being wrapped.
073    * @return The wrapped writer.
074    */
075   public JsonSerializerWriter getWriter(Writer out) {
076      if (out instanceof JsonSerializerWriter)
077         return (JsonSerializerWriter)out;
078      return new JsonSerializerWriter(out, isUseIndentation(), isUseWhitespace(), isEscapeSolidus(), getQuoteChar(), isSimpleMode(), isTrimStrings(), getRelativeUriBase(), getAbsolutePathUriBase());
079   }
080}
