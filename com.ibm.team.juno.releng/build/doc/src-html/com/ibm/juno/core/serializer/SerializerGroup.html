018/**
019 * Represents a group of {@link Serializer Serializers} that can be looked up by media type.
020 *
021 *
022 * <h6 class='topic'>Description</h6>
023 * <p>
024 *    Provides the following features:
025 * <ul>
026 *    <li>Finds serializers based on HTTP <code>Accept</code> header values.
027 *    <li>Sets common properties on all serializers in a single method call.
028 *    <li>Locks all serializers in a single method call.
029 *    <li>Clones existing groups and all serializers within the group in a single method call.
030 * </ul>
031 *
032 *
033 * <h6 class='topic'>Match ordering</h6>
034 * <p>
035 *    Serializers are matched against <code>Accept</code> strings in the order they exist in this group.
036 * <p>
037 *    Adding new entries will cause the entries to be prepended to the group.
038 *    This allows for previous serializers to be overridden through subsequent calls.
039 * <p>
040 *    For example, calling <code>g.append(S1.<jk>class</jk>,S2.<jk>class</jk>).append(S3.<jk>class</jk>,S4.<jk>class</jk>)</code>
041 *    will result in the order <code>S3, S4, S1, S2</code>.
042 *
043 *
044 * <h6 class='topic'>Examples</h6>
045 * <p class='bcode'>
046 *    <jc>// Construct a new serializer group</jc>
047 *    SerializerGroup g = <jk>new</jk> SerializerGroup();
048 *
049 *    <jc>// Add some serializers to it</jc>
050 *    g.append(JsonSerializer.<jk>class</jk>, XmlSerializer.<jk>class</jk>);
051 *
052 *    <jc>// Change settings for all serializers in the group and lock it.</jc>
053 *    g.setProperty(SerializerProperties.<jsf>SERIALIZER_useIndentation</jsf>, <jk>true</jk>)
054 *       .addFilters(CalendarFilter.ISO8601DT.<jk>class</jk>)
055 *       .lock();
056 *
057 *    <jc>// Find the appropriate serializer by Accept type</jc>
058 *    String mediaTypeMatch = g.findMatch(<js>"text/foo, text/json;q=0.8, text/*;q:0.6, *\/*;q=0.0"</js>);
059 *    WriterSerializer s = (WriterSerializer)g.getSerializer(mediaTypeMatch);
060 *
061 *    <jc>// Serialize a bean to JSON text </jc>
062 *    AddressBook addressBook = <jk>new</jk> AddressBook();  <jc>// Bean to serialize.</jc>
063 *    String json = s.serialize(addressBook);
064 * </p>
065 *
066 * @author James Bognar (jbognar@us.ibm.com)
067 */
068public final class SerializerGroup extends Lockable {
069
070   // Maps media-types to serializers.
071   private transient Map<String,SerializerEntry> entryMap = new HashMap<String,SerializerEntry>();
072   private transient LinkedList<SerializerEntry> tempEntries = new LinkedList<SerializerEntry>();
073   private transient SerializerEntry[] entries;
074
075
076   /**
077    * Registers the specified REST serializers with this serializer group.
078    *
079    * @param s The serializers to append to this group.
080    * @return This object (for method chaining).
081    */
082   public SerializerGroup append(Serializer<?>...s) {
083      checkLock();
084      entries = null;
085      for (Serializer<?> ss : reverse(s))  {
086         SerializerEntry e = new SerializerEntry(ss);
087         tempEntries.addFirst(e);
088         for (String mediaType : e.mediaTypes)
089            entryMap.put(mediaType, e);
090      }
091      return this;
092   }
093
094   /**
095    * Same as {@link #append(Serializer[])}, except specify classes instead of class instances
096    *     of {@link Serializer}.
097    * <p>
098    * Note that this can only be used on {@link Serializer Serializers} with public no-arg constructors.
099    *
100    * @param s The serializers to append to this group.
101    * @return This object (for method chaining).
102    * @throws Exception Thrown if {@link Serializer} could not be constructed.
103    */
104   public SerializerGroup append(Class<? extends Serializer<?>>...s) throws Exception {
105      checkLock();
106      for (Class<? extends Serializer<?>> ss : reverse(s))
107         try {
108            append(ss.newInstance());
109         } catch (NoClassDefFoundError e) {
110            // Ignore if dependent library not found (e.g. Jena).
111            System.err.println(e);
112         }
113      return this;
114   }
115
116   /**
117    * Same as {@link #append(Class[])}, except specify a single class to avoid unchecked compile warnings.
118    *
119    * @param c The serializer to append to this group.
120    * @return This object (for method chaining).
121    * @throws Exception Thrown if {@link Serializer} could not be constructed.
122    */
123   public SerializerGroup append(Class<? extends Serializer<?>> c) throws Exception {
124      checkLock();
125      try {
126         append(c.newInstance());
127      } catch (NoClassDefFoundError e) {
128         // Ignore if dependent library not found (e.g. Jena).
129         System.err.println(e);
130      }
131      return this;
132   }

134   /**
135    * Returns the serializer registered to handle the specified media type.
136    * <p>
137    * The media-type string must not contain any parameters or q-values.
138    *
139    * @param mediaType The media-type string (e.g. <js>"text/json"</js>
140    * @return The serializer that handles the specified accept content type, or <jk>null</jk> if
141    *       no serializer is registered to handle it.
142    */
143   public Serializer<?> getSerializer(String mediaType) {
144      SerializerEntry e = entryMap.get(mediaType);
145      return (e == null ? null : e.serializer);
146   }
147
148   /**
149    * Searches the group for a serializer that can handle the specified <code>Accept</code> value.
150    * <p>
151    *    The <code>accept</code> value complies with the syntax described in RFC2616, Section 14.1, as described below:
152    * <p class='bcode'>
153    *    Accept         = "Accept" ":"
154    *                      #( media-range [ accept-params ] )
155    *
156    *    media-range    = ( "*\/*"
157    *                      | ( type "/" "*" )
158    *                      | ( type "/" subtype )
159    *                      ) *( ";" parameter )
160    *    accept-params  = ";" "q" "=" qvalue *( accept-extension )
161    *    accept-extension = ";" token [ "=" ( token | quoted-string ) ]
162    * </p>
163    * <p>
164    *    The general idea behind having the serializer resolution be a two-step process is so that
165    *    the matched media type can be passed in to the {@link WriterSerializer#doSerialize(Object, Writer, SerializerContext)} method.
166    *    For example...
167    * <p class='bcode'>
168    *    String acceptHeaderValue = request.getHeader(<js>"Accept"</js>);
169    *    String matchingMediaType = group.findMatch(acceptHeaderValue);
170    *    if (matchingMediaType == <jk>null</jk>)
171    *       <jk>throw new</jk> RestException(<jsf>SC_NOT_ACCEPTABLE</jsf>);
172    *    WriterSerializer s = (WriterSerializer)group.getSerializer(matchingMediaType);
173    *  s.serialize(getPojo(), response.getWriter(), response.getProperties(), matchingMediaType);
174    * </p>
175    *
176    * @param accept The accept string.
177    * @return The media type registered by one of the parsers that matches the <code>accept</code> string,
178    *    or <jk>null</jk> if no media types matched.
179    */
180   public String findMatch(String accept) {
181      MediaRange[] mr = MediaRange.parse(accept);
182      if (mr.length == 0)
183         mr = MediaRange.parse("*/*");
184
185      for (MediaRange a : mr)
186         for (SerializerEntry e : getEntries())
187            for (MediaRange a2 : e.mediaRanges)
188               if (a.matches(a2))
189                  return a2.getMediaType();
190
191      return null;
192   }
193
194   /**
195    * Returns the media types that all serializers in this group can handle
196    * <p>
197    * Entries are ordered in the same order as the serializers in the group.
198    *
199    * @return The list of media types.
200    */
201   public List<String> getSupportedMediaTypes() {
202      List<String> l = new ArrayList<String>();
203      for (SerializerEntry e : getEntries())
204         for (String mt : e.mediaTypes)
205            if (! l.contains(mt))
206               l.add(mt);
207      return l;
208   }
209
210   private SerializerEntry[] getEntries() {
211      if (entries == null)
212         entries = tempEntries.toArray(new SerializerEntry[tempEntries.size()]);
213      return entries;
214   }
215
216   static class SerializerEntry {
217      Serializer<?> serializer;
218      MediaRange[] mediaRanges;
219      String[] mediaTypes;
220
221      SerializerEntry(Serializer<?> s) {
222         serializer = s;
223
224         mediaTypes = new String[s.getMediaTypes().length];
225         int i = 0;
226         for (String mt : s.getMediaTypes())
227            mediaTypes[i++] = mt.toLowerCase(Locale.ENGLISH);
228
229         List<MediaRange> l = new LinkedList<MediaRange>();
230         for (i = 0; i < mediaTypes.length; i++)
231            l.addAll(Arrays.asList(MediaRange.parse(mediaTypes[i])));
232         mediaRanges = l.toArray(new MediaRange[l.size()]);
233      }
234   }

241   /**
242    * Shortcut for calling {@link Serializer#setProperty(String, Object)} on all serializers in this group.
243    *
244    * @param property The property name.
245    * @param value The property value.
246    * @throws LockedException If {@link #lock()} was called on this object.
247    * @return This object (for method chaining).
248    */
249   public SerializerGroup setProperty(String property, Object value) throws LockedException {
250      checkLock();
251      for (SerializerEntry e : getEntries())
252         e.serializer.setProperty(property, value);
253      return this;
254   }
255
256   /**
257    * Shortcut for calling {@link Serializer#setProperties(ObjectMap)} on all serializers in this group.
258    *
259    * @param properties The properties to set.  Ignored if <jk>null</jk>.
260    * @throws LockedException If {@link #lock()} was called on this object.
261    * @return This object (for method chaining).
262    */
263   public SerializerGroup setProperties(ObjectMap properties) {
264      checkLock();
265      if (properties != null)
266         for (Map.Entry<String,Object> e : properties.entrySet())
267            setProperty(e.getKey(), e.getValue());
268      return this;
269   }

320   /**
321    * Locks this group and all serializers in this group.
322    */
323   @Override /* Lockable */
324   public SerializerGroup lock() {
325      super.lock();
326      for (SerializerEntry e : getEntries())
327         e.serializer.lock();
328      return this;
329   }
330
331   /**
332    * Clones this group and all serializers in this group.
333    */
334   @Override /* Lockable */
335   public SerializerGroup clone() throws CloneNotSupportedException {
336      SerializerGroup c = (SerializerGroup)super.clone();
337      c.entryMap = new HashMap<String,SerializerEntry>();
338      c.tempEntries = new LinkedList<SerializerEntry>();
339      c.entries = null;
340      SerializerEntry[] e = getEntries();
341      for (int i = e.length-1; i >= 0; i--)
342         c.append(e[i].serializer.clone());
343      return c;
344   }
