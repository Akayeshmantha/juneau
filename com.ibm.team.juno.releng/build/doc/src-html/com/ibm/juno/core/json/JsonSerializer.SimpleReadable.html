023/**
024 * Serializes POJO models to JSON.
025 *
026 *
027 * <h6 class='topic'>Media types</h6>
028 * <p>
029 *    Handles <code>Accept</code> types: <code>application/json, text/json</code>
030 * <p>
031 *    Produces <code>Content-Type</code> types: <code>application/json</code>
032 *
033 *
034 * <h6 class='topic'>Description</h6>
035 * <p>
036 *    The conversion is as follows...
037 *    <ul>
038 *       <li>Maps (e.g. {@link HashMap HashMaps}, {@link TreeMap TreeMaps}) are converted to JSON objects.
039 *       <li>Collections (e.g. {@link HashSet HashSets}, {@link LinkedList LinkedLists}) and Java arrays are converted to JSON arrays.
040 *       <li>{@link String Strings} are converted to JSON strings.
041 *       <li>{@link Number Numbers} (e.g. {@link Integer}, {@link Long}, {@link Double}) are converted to JSON numbers.
042 *       <li>{@link Boolean Booleans} are converted to JSON booleans.
043 *       <li>{@code nulls} are converted to JSON nulls.
044 *       <li>{@code arrays} are converted to JSON arrays.
045 *       <li>{@code beans} are converted to JSON objects.
046 *    </ul>
047 * <p>
048 *    The types above are considered "JSON-primitive" object types.  Any non-JSON-primitive object types are transformed
049 *       into JSON-primitive object types through {@link com.ibm.juno.core.filter.Filter Filters} associated through the {@link BeanContextFactory#addFilters(Class...)}
050 *       method.  Several default filters are provided for transforming Dates, Enums, Iterators, etc...
051 * <p>
052 *    This serializer provides several serialization options.  Typically, one of the predefined DEFAULT serializers will be sufficient.
053 *    However, custom serializers can be constructed to fine-tune behavior.
054 *
055 *
056 * <h6 class='topic'>Configurable properties</h6>
057 * <p>
058 *    This class has the following properties associated with it:
059 * <ul>
060 *    <li>{@link JsonSerializerProperties}
061 *    <li>{@link SerializerProperties}
062 *    <li>{@link BeanContextProperties}
063 * </ul>
064 *
065 *
066 * <h6 class='topic'>Behavior-specific subclasses</h6>
067 * <p>
068 *    The following direct subclasses are provided for convenience:
069 * <ul>
070 *    <li>{@link Simple} - Default serializer, single quotes, simple mode.
071 *    <li>{@link SimpleReadable} - Default serializer, single quotes, simple mode, with whitespace.
072 * </ul>
073 *
074 *
075 * <h6 class='topic'>Examples</h6>
076 * <p class='bcode'>
077 *    <jc>// Use one of the default serializers to serialize a POJO</jc>
078 *    String json = JsonSerializer.<jsf>DEFAULT</jsf>.serialize(someObject);
079 *
080 *    <jc>// Create a custom serializer for lax syntax using single quote characters</jc>
081 *    JsonSerializer serializer = <jk>new</jk> JsonSerializer()
082 *       .setProperty(JsonSerializerProperties.<jsf>JSON_simpleMode</jsf>, <jk>true</jk>)
083 *       .setProperty(SerializerProperties.<jsf>SERIALIZER_quoteChar</jsf>, <js>'\''</js>);
084 *
085 *    <jc>// Clone an existing serializer and modify it to use single-quotes</jc>
086 *    JsonSerializer serializer = JsonSerializer.<jsf>DEFAULT</jsf>.clone()
087 *       .setProperty(SerializerProperties.<jsf>SERIALIZER_quoteChar</jsf>, <js>'\''</js>);
088 *
089 *    <jc>// Serialize a POJO to JSON</jc>
090 *    String json = serializer.serialize(someObject);
091 * </p>
092 *
093 *
094 * @author James Bognar (jbognar@us.ibm.com)
095 */

160   /**
161    * Workhorse method. Determines the type of object, and then calls the
162    * appropriate type-specific serialization method.
163    */
164   @SuppressWarnings({ "rawtypes", "unchecked" })
165   SerializerWriter serializeAnything(JsonSerializerWriter out, Object o, ClassMeta<?> eType, JsonSerializerContext ctx, String attrName, BeanPropertyMeta pMeta) throws SerializeException {
166      try {
167         BeanContext bc = ctx.getBeanContext();
168
169         if (o == null) {
170            out.append("null");
171            return out;
172         }
173
174         if (eType == null)
175            eType = object();
176
177         boolean addClassAttr;      // Add "_class" attribute to element?
178         ClassMeta<?> aType;        // The actual type
179         ClassMeta<?> gType;        // The generic type
180
181         aType = ctx.push(attrName, o, eType);
182         boolean isRecursion = aType == null;
183
184         // Handle recursion
185         if (aType == null) {
186            o = null;
187            aType = object();
188         }
189
190         gType = aType.getFilteredClassMeta();
191         addClassAttr = (ctx.isAddClassAttrs() && ! eType.equals(aType));
192
193         // Filter if necessary
194         PojoFilter filter = aType.getPojoFilter();            // The filter
195         if (filter != null) {
196            o = filter.filter(o);
197
198            // If the filter's getFilteredClass() method returns Object, we need to figure out
199            // the actual type now.
200            if (gType.isObject())
201               gType = bc.getClassMetaForObject(o);
202         }
203
204         String wrapperAttr = gType.getJsonMeta().getWrapperAttr();
205         if (wrapperAttr != null) {
206            out.append('{').cr(ctx.indent).attr(wrapperAttr).append(':').s();
207            ctx.indent++;
208         }
209
210         // '\0' characters are considered null.
211         if (o == null || (gType.isChar() && ((Character)o).charValue() == 0))
212            out.append("null");
213         else if (gType.isNumber() || gType.isBoolean())
214            out.append(o);
215         else if (gType.hasToObjectMapMethod())
216            serializeMap(out, gType.toObjectMap(o), gType, ctx);
217         else if (gType.isBean())
218            serializeBeanMap(out, bc.forBean(o), addClassAttr, ctx);
219         else if (gType.isUri() || (pMeta != null && (pMeta.isUri() || pMeta.isBeanUri())))
220            out.q().appendUri(o).q();
221         else if (gType.isMap()) {
222            if (o instanceof BeanMap)
223               serializeBeanMap(out, (BeanMap)o, addClassAttr, ctx);
224            else
225               serializeMap(out, (Map)o, eType, ctx);
226         }
227         else if (gType.isCollection()) {
228            if (addClassAttr)
229               serializeCollectionMap(out, (Collection)o, gType, ctx);
230            else
231               serializeCollection(out, (Collection) o, eType, ctx);
232         }
233         else if (gType.isArray()) {
234            if (addClassAttr)
235               serializeCollectionMap(out, toList(gType.getInnerClass(), o), gType, ctx);
236            else
237               serializeCollection(out, toList(gType.getInnerClass(), o), eType, ctx);
238         }
239         else
240            out.stringValue(o);
241
242         if (wrapperAttr != null) {
243            ctx.indent--;
244            out.cr(ctx.indent-1).append('}');
245         }
246
247         if (! isRecursion)
248            ctx.pop();
249         return out;
250      } catch (SerializeException e) {
251         throw e;
252      } catch (StackOverflowError e) {
253         throw e;
254      } catch (Throwable e) {
255         throw new SerializeException("Exception occured trying to process object of type ''{0}''", (o == null ? null : o.getClass().getName())).initCause(e);
256      }
257   }

259   @SuppressWarnings({ "rawtypes", "unchecked" })
260   private SerializerWriter serializeMap(JsonSerializerWriter out, Map m, ClassMeta<?> type, JsonSerializerContext ctx) throws IOException, SerializeException {
261
262      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
263
264      m = ctx.sort(m);
265
266      int depth = ctx.getIndent();
267      out.append('{');
268
269      Iterator mapEntries = m.entrySet().iterator();
270
271      while (mapEntries.hasNext()) {
272         Map.Entry e = (Map.Entry) mapEntries.next();
273         Object value = e.getValue();
274
275         Object key = ctx.generalize(e.getKey(), keyType);
276
277         out.cr(depth).attr(key).append(':').s();
278
279         serializeAnything(out, value, valueType, ctx, (key == null ? null : key.toString()), null);
280
281         if (mapEntries.hasNext())
282            out.append(',').s();
283      }
284
285      out.cr(depth-1).append('}');
286
287      return out;
288   }

290   @SuppressWarnings({ "rawtypes" })
291   private SerializerWriter serializeCollectionMap(JsonSerializerWriter out, Collection o, ClassMeta<?> type, JsonSerializerContext ctx) throws IOException, SerializeException {
292      int i = ctx.getIndent();
293      out.append('{');
294      out.cr(i).attr("_class").append(':').s().q().append(type.getInnerClass().getName()).q().append(',').s();
295      out.cr(i).attr("items").append(':').s();
296      ctx.indent++;
297      serializeCollection(out, o, type, ctx);
298      ctx.indent--;
299      out.cr(i-1).append('}');
300      return out;
301   }

303   @SuppressWarnings({ "rawtypes" })
304   private SerializerWriter serializeBeanMap(JsonSerializerWriter out, BeanMap m, boolean addClassAttr, JsonSerializerContext ctx) throws IOException, SerializeException {
305      int depth = ctx.getIndent();
306      out.append('{');
307
308      Iterator mapEntries = m.entrySet().iterator();
309
310      // Print out "_class" attribute on this bean if required.
311      if (addClassAttr) {
312         String attr = "_class";
313         out.cr(depth).attr(attr).append(':').s().q().append(m.getClassMeta().getInnerClass().getName()).q();
314         if (mapEntries.hasNext())
315            out.append(',').s();
316      }
317
318      boolean addComma = false;
319
320      while (mapEntries.hasNext()) {
321         BeanMapEntry p = (BeanMapEntry)mapEntries.next();
322         BeanPropertyMeta pMeta = p.getMeta();
323
324         String key = p.getKey();
325         Object value = null;
326         try {
327            value = p.getValue();
328         } catch (StackOverflowError e) {
329            throw e;
330         } catch (Throwable t) {
331            ctx.addBeanGetterWarning(pMeta, t);
332         }
333
334         if (ctx.canIgnoreValue(pMeta.getClassMeta(), key, value))
335            continue;
336
337         if (addComma)
338            out.append(',').s();
339
340         out.cr(depth).attr(key).append(':').s();
341
342         serializeAnything(out, value, pMeta.getClassMeta(), ctx, key, pMeta);
343
344         addComma = true;
345      }
346      out.cr(depth-1).append('}');
347      return out;
348   }
349
350   @SuppressWarnings({"rawtypes", "unchecked"})
351   private SerializerWriter serializeCollection(JsonSerializerWriter out, Collection c, ClassMeta<?> type, JsonSerializerContext ctx) throws IOException, SerializeException {
352
353      ClassMeta<?> elementType = type.getElementType();
354
355      c = ctx.sort(c);
356
357      out.append('[');
358      int depth = ctx.getIndent();
359
360      for (Iterator i = c.iterator(); i.hasNext();) {
361
362         Object value = i.next();
363
364         out.cr(depth);
365
366         serializeAnything(out, value, elementType, ctx, "<iterator>", null);
367
368         if (i.hasNext())
369            out.append(',').s();
370      }
371      out.cr(depth-1).append(']');
372      return out;
373   }

375   /**
376    * Returns the schema serializer based on the settings of this serializer.
377    * @return The schema serializer.
378    */
379   public JsonSchemaSerializer getSchemaSerializer() {
380      JsonSchemaSerializer s = new JsonSchemaSerializer();
381      s.beanContextFactory = this.beanContextFactory;
382      s.sp = this.sp;
383      s.jsp = this.jsp;
384      return s;

391   @Override /* Serializer */
392   public JsonSerializerContext createContext(ObjectMap properties, Method javaMethod) {
393      return new JsonSerializerContext(getBeanContext(), sp, jsp, properties, javaMethod);
394   }

396   @Override /* Serializer */
397   protected void doSerialize(Object o, Writer out, SerializerContext ctx) throws IOException, SerializeException {
398      JsonSerializerContext jctx = (JsonSerializerContext)ctx;
399      serializeAnything(jctx.getWriter(out), o, null, jctx, "root", null);
400   }

402   @Override /* CoreApi */
403   public JsonSerializer setProperty(String property, Object value) throws LockedException {
404      checkLock();
405      if (! jsp.setProperty(property, value))
406         super.setProperty(property, value);
407      return this;
408   }
