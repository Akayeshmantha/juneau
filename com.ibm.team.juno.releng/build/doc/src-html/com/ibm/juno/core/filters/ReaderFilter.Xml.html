009package com.ibm.juno.core.filters;
010
011import java.io.*;
012
013import com.ibm.juno.core.filter.*;
014import com.ibm.juno.core.html.*;
015import com.ibm.juno.core.json.*;
016import com.ibm.juno.core.parser.*;
017import com.ibm.juno.core.serializer.*;
018import com.ibm.juno.core.utils.*;
019import com.ibm.juno.core.xml.*;
020
021/**
022 * Transforms the contents of a {@link Reader} into an {@code Object}.
023 *
024 *
025 * <h6 class='topic'>Description</h6>
026 * <p>
027 *    The {@code Reader} must contain JSON, Juno-generated XML (output from {@link XmlSerializer}),
028 *       or Juno-generated HTML (output from {@link JsonSerializer}) in order to be parsed correctly.
029 * <p>
030 *    Useful for serializing models that contain {@code Readers} created by {@code RestCall} instances.
031 * <p>
032 *    This is a one-way filter, since {@code Readers} cannot be reconstituted.
033 *
034 *
035 * <h6 class='topic'>Behavior-specific subclasses</h6>
036 * <p>
037 *    The following direct subclasses are provided for convenience:
038 * <ul>
039 *    <li>{@link Json} - Parses JSON text.
040 *    <li>{@link Xml} - Parses XML text.
041 *    <li>{@link Html} - Parses HTML text.
042 *    <li>{@link PlainText} - Parses plain text.
043 * </ul>
044 *
045 *
046 * @author James Bognar (jbognar@us.ibm.com)
047 */
048public class ReaderFilter extends PojoFilter<Reader,Object> {
049
050   /** Reader filter for reading JSON text. */
051   public static class Json extends ReaderFilter {
052      /** Constructor */
053      public Json() {
054         super(JsonParser.DEFAULT);
055      }
056   }
057
058   /** Reader filter for reading XML text. */
059   public static class Xml extends ReaderFilter {
060      /** Constructor */
061      public Xml() {
062         super(XmlParser.DEFAULT);
063      }
064   }
065
066   /** Reader filter for reading HTML text. */
067   public static class Html extends ReaderFilter {
068      /** Constructor */
069      public Html() {
070         super(HtmlParser.DEFAULT);
071      }
072   }
073
074   /** Reader filter for reading plain text. */
075   public static class PlainText extends ReaderFilter {
076      /** Constructor */
077      public PlainText() {
078         super(null);
079      }
080   }
081
082   /** The parser to use to parse the contents of the Reader. */
083   private ReaderParser parser;
084
085   /**
086    * @param parser The parser to use to convert the contents of the reader to Java objects.
087    */
088   public ReaderFilter(ReaderParser parser) {
089      this.parser = parser;
090   }
091
092   /**
093    * Converts the specified {@link Reader} to an {@link Object} whose type is determined
094    * by the contents of the reader.
095    */
096   @Override /* PojoFilter */
097   public Object filter(Reader o) throws SerializeException {
098      try {
099         if (parser == null)
100            return IOUtils.read(o);
101         return parser.parse(o, -1, beanContext.object());
102      } catch (IOException e) {
103         return e.getLocalizedMessage();
104      } catch (Exception e) {
105         throw new SerializeException("ReaderFilter could not filter object of type ''{0}''", o == null ? null : o.getClass().getName()).initCause(e);
106      }
107   }
108}
