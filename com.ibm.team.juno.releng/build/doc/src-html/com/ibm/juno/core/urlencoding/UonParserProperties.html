009package com.ibm.juno.core.urlencoding;
010
011import com.ibm.juno.core.*;
012import com.ibm.juno.core.parser.*;
013
014/**
015 * Configurable properties on the {@link UonParser} and {@link UrlEncodingParser} classes.
016 * <p>
017 *    Use the {@link UonParser#setProperty(String, Object)} method to set property values.
018 * <p>
019 *    In addition to these properties, the following properties are also applicable for {@link UonParser}.
020 * <ul>
021 *    <li>{@link ParserProperties}
022 *    <li>{@link BeanContextProperties}
023 * </ul>
024 *
025 * @author James Bognar (jbognar@us.ibm.com)
026 */
027public class UonParserProperties implements Cloneable {
028
029   /**
030    * Decode <js>"%xx"</js> sequences. ({@link Boolean}, default=<jk>false</jk> for {@link UonParser}, <jk>true</jk> for {@link UrlEncodingParser}).
031    * <p>
032    * Specify <jk>true</jk> if URI encoded characters should be decoded, <jk>false</jk>
033    *    if they've already been decoded before being passed to this parser.
034    */
035   public static final String UON_decodeChars = "UonParser.decodeChars";
036
037   /**
038    * Expect input to contain readable whitespace characters from using the {@link UonSerializerProperties#UON_useWhitespace} setting ({@link Boolean}, default=<jk>false</jk>).
039    */
040   public static final String UON_whitespaceAware = "UonParser.whitespaceAware";
041
042   boolean
043      decodeChars = false,
044      whitespaceAware = false;
045
046   /**
047    * Sets the specified property value.
048    * @param property The property name.
049    * @param value The property value.
050    * @return <jk>true</jk> if property name was valid and property was set.
051    */
052   public boolean setProperty(String property, Object value) {
053      BeanContext bc = BeanContext.DEFAULT;
054      if (property.equals(UON_decodeChars))
055         decodeChars = bc.convertToType(value, Boolean.class);
056      else if (property.equals(UON_whitespaceAware))
057         whitespaceAware = bc.convertToType(value, Boolean.class);
058      else
059         return false;
060      return true;
061   }
062
063
064   //--------------------------------------------------------------------------------
065   // Overridden methods
066   //--------------------------------------------------------------------------------
067
068   @Override /* Cloneable */
069   public UonParserProperties clone() {
070      try {
071         return (UonParserProperties)super.clone();
072      } catch (CloneNotSupportedException e) {
073         throw new RuntimeException(e); // Shouldn't happen
074      }
075   }
076}
