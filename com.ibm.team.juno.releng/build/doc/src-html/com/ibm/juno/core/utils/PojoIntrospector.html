009package com.ibm.juno.core.utils;
010
011import java.io.*;
012import java.lang.reflect.*;
013
014import com.ibm.juno.core.*;
015import com.ibm.juno.core.json.*;
016import com.ibm.juno.core.parser.*;
017
018/**
019 * Used to invoke methods on {@code Objects} using arguments in serialized form.
020 * <p>
021 * Example:
022 * <p class='bcode'>
023 *    String s = <js>"foobar"</js>;
024 *    String s2 = (String)<jk>new</jk> PojoIntrospector(s).invoke(<js>"substring(int,int)"</js>, <js>"[3,6]"</js>);  <jc>// "bar"</jc>
025 * </p>
026 *
027 * @author James Bognar (jbognar@us.ibm.com)
028 */
029public final class PojoIntrospector {
030
031   private final Object o;
032   private final ReaderParser p;
033
034   /**
035    * Constructor.
036    *
037    * @param o The object on which Java methods will be invoked.
038    * @param p The parser to use to parse the method arguments.  If <jk>null</jk>, {@link JsonParser#DEFAULT} is used.
039    */
040   public PojoIntrospector(Object o, ReaderParser p) {
041      if (p == null)
042         p = JsonParser.DEFAULT;
043      this.o = o;
044      this.p = p;
045   }
046
047   /**
048    * Shortcut for calling <code><jk>new</jk> PojoIntrospector(o, <jk>null</jk>);</code>
049    *
050    * @param o The object on which Java methods will be invoked.
051    */
052   public PojoIntrospector(Object o) {
053      this(o, null);
054   }
055
056   /**
057    * Primary method.  Invokes the specified method on this bean.
058    *
059    * @param method The method being invoked.
060    * @param args The arguments to pass as parameters to the method.<br>
061    *    These will automatically be converted to the appropriate object type if possible.<br>
062    *    Can be <jk>null</jk> if method has no arguments.
063    * @return The object returned by the call to the method, or <jk>null</jk> if target object is <jk>null</jk>.
064    * @throws IllegalAccessException If the <code>Constructor</code> object enforces Java language access control and the underlying constructor is inaccessible.
065    * @throws IllegalArgumentException If one of the following occurs:
066    *    <ul>
067    *       <li>The number of actual and formal parameters differ.
068    *       <li>An unwrapping conversion for primitive arguments fails.
069    *       <li>A parameter value cannot be converted to the corresponding formal parameter type by a method invocation conversion.
070    *       <li>The constructor pertains to an enum type.
071    *    </ul>
072    * @throws InvocationTargetException If the underlying constructor throws an exception.
073    * @throws ParseException If the input contains a syntax error or is malformed.
074    * @throws IOException
075    */
076   public Object invokeMethod(Method method, Reader args) throws InvocationTargetException, IllegalArgumentException, IllegalAccessException, ParseException, IOException {
077      if (o == null)
078         return null;
079      ClassMeta<?>[] argTypes = p.getBeanContext().getClassMetas(method.getParameterTypes());
080      Object[] params = args == null ? null : p.parseArgs(args, -1, argTypes);
081      return method.invoke(o, params);
082   }
083
084   /**
085    * Convenience method for invoking argument from method signature (@see {@link ClassUtils#getMethodSignature(Method)}.
086    *
087    * @param method The method being invoked.
088    * @param args The arguments to pass as parameters to the method.<br>
089    *    These will automatically be converted to the appropriate object type if possible.<br>
090    *    Can be <jk>null</jk> if method has no arguments.
091    * @return The object returned by the call to the method, or <jk>null</jk> if target object is <jk>null</jk>.
092    * @throws NoSuchMethodException If method does not exist.
093    * @throws IllegalAccessException If the <code>Constructor</code> object enforces Java language access control and the underlying constructor is inaccessible.
094    * @throws IllegalArgumentException If one of the following occurs:
095    *    <ul>
096    *       <li>The number of actual and formal parameters differ.
097    *       <li>An unwrapping conversion for primitive arguments fails.
098    *       <li>A parameter value cannot be converted to the corresponding formal parameter type by a method invocation conversion.
099    *       <li>The constructor pertains to an enum type.
100    *    </ul>
101    * @throws InvocationTargetException If the underlying constructor throws an exception.
102    * @throws ParseException If the input contains a syntax error or is malformed.
103    * @throws IOException
104    */
105   public Object invokeMethod(String method, String args) throws NoSuchMethodException, IllegalArgumentException, InvocationTargetException, IllegalAccessException, ParseException, IOException {
106      if (o == null)
107         return null;
108      Method m = p.getBeanContext().getClassMeta(o.getClass()).getPublicMethods().get(method);
109      if (m == null)
110         throw new NoSuchMethodException(method);
111      return invokeMethod(m, args == null ? null : new StringReader(args));
112   }
113}
