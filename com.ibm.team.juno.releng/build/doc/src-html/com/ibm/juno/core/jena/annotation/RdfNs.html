009package com.ibm.juno.core.jena.annotation;
010
011import static java.lang.annotation.RetentionPolicy.*;
012
013import java.lang.annotation.*;
014
015/**
016 * Namespace name/URL mapping pair.
017 * <p>
018 *    Used to identify a namespace/URI pair on a {@link RdfSchema#rdfNs()} annotation.
019 *
020 * @author James Bognar (jbognar@us.ibm.com)
021 */
022@Documented
023@Target({})
024@Retention(RUNTIME)
025@Inherited
026public @interface RdfNs {
027
028   /**
029    * RDF namespace prefix.
030    */
031   String prefix();
032
033   /**
034    * RDF namespace URL.
035    */
036   String namespaceURI();
037}
