009package com.ibm.juno.core.utils;
010
011import java.util.*;
012
013/**
014 * A collection of {@link SafeResourceBundle} objects.
015 * Allows servlets to define resource bundles for different classes in the class hierarchy.
016 *
017 * @author James Bognar (jbognar@us.ibm.com)
018 */
019public final class SafeResourceMultiBundle extends SafeResourceBundle {
020
021   private List<SafeResourceBundle> bundles = new ArrayList<SafeResourceBundle>();
022
023   /**
024    * Constructor.
025    *
026    * @param bundles The resource bundles to wrap.  Can be <jk>null</jk> to indicate a missing bundle.
027    */
028   public SafeResourceMultiBundle(Collection<SafeResourceBundle> bundles) {
029      this.bundles.addAll(bundles);
030   }
031
032   @Override /* SafeResourceBundle */
033   public String findFirstString(String...keys) {
034      for (String key : keys)
035         for (SafeResourceBundle srb : bundles)
036            if (srb.containsKey(key))
037               return srb.getString(key);
038      return null;
039   }
040
041   @Override /* ResourceBundle */
042   public boolean containsKey(String key) {
043      for (SafeResourceBundle srb : bundles)
044         if (srb.containsKey(key))
045            return true;
046      return false;
047   }
048
049   @Override /* SafeResourceBundle */
050   public String getString(String key, Object...args) {
051      for (SafeResourceBundle srb : bundles)
052         if (srb.containsKey(key))
053            return srb.getString(key, args);
054      return "{!" + key + "}";
055   }
056
057   @Override /* ResourceBundle */
058   @SuppressWarnings("unchecked")
059   public Set<String> keySet() {
060      MultiSet<String> s = new MultiSet<String>();
061      for (SafeResourceBundle rb : bundles)
062         s.append(rb.keySet());
063      return s;
064   }
065
066   @Override /* ResourceBundle */
067   @SuppressWarnings("unchecked")
068   public Enumeration<String> getKeys() {
069      MultiSet<String> s = new MultiSet<String>();
070      for (SafeResourceBundle rb : bundles)
071         s.append(rb.keySet());
072      return s.enumerator();
073   }
074
075   @Override /* ResourceBundle */
076   protected Object handleGetObject(String key) {
077      for (SafeResourceBundle srb : bundles)
078         if (srb.containsKey(key))
079            return srb.handleGetObject(key);
080      return "{!"+key+"}";
081   }
082}
