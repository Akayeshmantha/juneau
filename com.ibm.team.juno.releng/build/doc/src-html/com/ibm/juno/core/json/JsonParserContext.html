009package com.ibm.juno.core.json;
010
011import static com.ibm.juno.core.json.JsonParserProperties.*;
012
013import java.io.*;
014import java.lang.reflect.*;
015
016import com.ibm.juno.core.*;
017import com.ibm.juno.core.parser.*;
018
019/**
020 * Context object that lives for the duration of a single parsing of {@link JsonParser}.
021 * <p>
022 *    This class is NOT thread safe.  It is meant to be discarded after one-time use.
023 * </p>
024 *
025 * @author James Bognar (jbognar@us.ibm.com)
026 */
027public final class JsonParserContext extends ParserContext {
028
029   private final boolean strictMode;
030
031   /**
032    * Create a new parser context with the specified options.
033    *
034    * @param beanContext The bean context being used.
035    * @param jpp The JSON parser properties.
036    * @param pp The default parser properties.
037    * @param op The override properties.
038    * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
039    * @param outer The outer object for instantiating top-level non-static inner classes.
040    */
041   public JsonParserContext(BeanContext beanContext, JsonParserProperties jpp, ParserProperties pp, ObjectMap op, Method javaMethod, Object outer) {
042      super(beanContext, pp, op, javaMethod, outer);
043      if (op == null || op.isEmpty()) {
044         strictMode = jpp.isStrictMode();
045      } else {
046         strictMode = op.getBoolean(JSON_strictMode, jpp.isStrictMode());
047      }
048   }
049
050   final boolean isStrictMode() {
051      return strictMode;
052   }
053
054   /**
055    * Returns the reader associated with this context wrapped in a {@link ParserReader}.
056    *
057    * @param in The reader being wrapped.
058    * @param estimatedSize The estimated size of the input.
059    * @return The reader wrapped in a specialized parser reader.
060    */
061   public ParserReader getReader(Reader in, int estimatedSize) {
062      if (in instanceof ParserReader)
063         return (ParserReader)in;
064      return new ParserReader(in, Math.min(8096, estimatedSize));
065   }
066}
