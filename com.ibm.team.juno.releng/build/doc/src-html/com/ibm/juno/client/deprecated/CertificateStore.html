009package com.ibm.juno.client.deprecated;
010
011import java.io.*;
012import java.security.*;
013import java.security.cert.*;
014import java.security.cert.Certificate;
015import java.util.*;
016
017/**
018 * Specialized certificate storage based on {@link KeyStore} for managing trusted certificates.
019 */
020@Deprecated // Use SimpleX509TrustManager
021public class CertificateStore {
022
023   private final KeyStore keyStore;
024
025   /**
026    * Get the underlying KeyStore.
027    */
028   KeyStore getKeyStore() {
029      return keyStore;
030   }
031
032   /**
033    * Helper method that creates a {@link KeyStore} by reading it from a file.
034    */
035   static KeyStore load(File file, String password) throws NoSuchAlgorithmException, CertificateException, IOException, KeyStoreException {
036      KeyStore ks = KeyStore.getInstance(KeyStore.getDefaultType());
037      try {
038         InputStream input = new FileInputStream(file);
039         try {
040            ks.load(input, password == null ? null : password.toCharArray());
041         } finally {
042            input.close();
043         }
044      } catch (IOException e) {
045         // Return an empty initialized KeyStore
046         ks.load(null, null);
047      }
048      return ks;
049   }
050
051   /**
052    * Helper method that writes a {@link KeyStore} to a file.
053    */
054   static void store(KeyStore ks, File file, String password) throws KeyStoreException, NoSuchAlgorithmException, CertificateException, IOException {
055      OutputStream output = new FileOutputStream(file);
056      try {
057         ks.store(output, password == null ? null : password.toCharArray());
058      } finally {
059         output.close();
060      }
061   }
062
063   /**
064    * Helper to compute a unique alias within the trust store for a specified certificate.
065    * @param cert The certificate to compute an alias for.
066    */
067   static String computeAlias(Certificate cert) {
068      // There appears to be no standard way to construct certificate aliases,
069      // but this class never depends on looking up a certificate by its
070      // computed alias, so just create an alias that's unique and be done.
071      return UUID.randomUUID().toString();
072   }
073
074   /**
075    * Construct a new TrustStore initially containing no certificates.
076    */
077   public CertificateStore() throws NoSuchAlgorithmException, CertificateException, IOException {
078      try {
079         keyStore = KeyStore.getInstance(KeyStore.getDefaultType());
080      } catch (KeyStoreException e) {
081         // If the code above caused a KeyStoreException, then the JVM classpath is probably messed up.
082         throw new RuntimeException("KeyStoreException: ["+e.getLocalizedMessage()+"]. "
083            + "Likely cause is that the Java Cryptography Extension libraries are missing from the JRE classpath.  "
084            + "Make sure %JAVA_HOME%/lib/ext is specified in your JVM's java.ext.dirs system property.");
085      }
086      keyStore.load(null, null);
087   }
088
089   /**
090    * Does the trust store contain the specified certificate?
091    */
092   public boolean containsCertificate(Certificate cert) throws KeyStoreException {
093      return (keyStore.getCertificateAlias(cert) != null);
094   }
095
096   /**
097    * Enter the specified certificate into the trust store.
098    */
099   public void enterCertificate(Certificate cert) throws KeyStoreException {
100      if (! containsCertificate(cert))
101         keyStore.setCertificateEntry(computeAlias(cert), cert);
102   }
103
104   /*
105    * Helper to copy all the certificate entries, and none of the other
106    * entries, from a {@link KeyStore} into the trust store.
107    */
108   private void enterCertificates(KeyStore ks) throws KeyStoreException {
109      for (Enumeration<String> e = ks.aliases(); e.hasMoreElements();) {
110         String alias = e.nextElement();
111         if (ks.isCertificateEntry(alias)) {
112            Certificate cert = ks.getCertificate(alias);
113            enterCertificate(cert);
114         }
115      }
116   }
117
118   /**
119    * Load the specified {@link KeyStore} file and copy all of the certificates
120    * it contains into the trust store. Only certificates, and not any other
121    * entries, are loaded.
122    */
123   public void loadCertificates(File file, String password) throws NoSuchAlgorithmException, CertificateException, KeyStoreException, IOException {
124      KeyStore ks = load(file, password);
125      enterCertificates(ks);
126   }
127}
