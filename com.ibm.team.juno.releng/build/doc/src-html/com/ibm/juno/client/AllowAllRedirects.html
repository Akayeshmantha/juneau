009package com.ibm.juno.client;
010
011import org.apache.http.impl.client.*;
012
013/**
014 * Redirect strategy that allows for redirects on any request type, not just <code>GET</code> or <code>HEAD</code>.
015 * <p>
016 * Note:  This class is similar to <code>org.apache.http.impl.client.LaxRedirectStrategy</code>
017 *    in Apache HttpClient 4.2, but also allows for redirects on <code>PUTs</code> and <code>DELETEs</code>.
018 *
019 * @author James Bognar (jbognar@us.ibm.com)
020 */
021public class AllowAllRedirects extends DefaultRedirectStrategy {
022
023   @Override /* DefaultRedirectStrategy */
024   protected boolean isRedirectable(final String method) {
025      return true;
026   }
027}
