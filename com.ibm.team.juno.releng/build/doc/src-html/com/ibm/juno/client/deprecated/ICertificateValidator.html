009package com.ibm.juno.client.deprecated;
010
011import java.security.cert.*;
012
013/**
014 * Validator of certificates presented by a server when establishing an SSL
015 * connection.
016 */
017@Deprecated // Use SimpleX509TrustManager
018public interface ICertificateValidator {
019
020   /** Action to take for a server-supplied certificate. */
021   public enum Trust {
022
023      /** Do not accept the certificate. */
024      REJECT,
025
026      /** Accept the certificate temporarily for the current connection. */
027      ACCEPT_CONNECTION,
028
029      /** Accept the certificate temporarily for the current session. */
030      ACCEPT_SESSION,
031
032      /** Accept the certificate permanently, by saving it in the user's trust store.*/
033      ACCEPT_PERMANENT
034   }
035
036   /**
037    * There is a problem accepting the server-supplied certificate. What should
038    * be done?
039    *
040    * @param cert The problematic certificate presented by the server
041    * @param problem The {@link CertificateException} that may indicate the specific
042    *    problem with the certificate, e.g. {@link CertificateExpiredException}.
043    * @return The disposition on the certificate.
044    */
045   Trust validate(X509Certificate cert, CertificateException problem);
046}
