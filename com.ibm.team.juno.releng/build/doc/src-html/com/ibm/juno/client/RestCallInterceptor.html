009package com.ibm.juno.client;
010
011import org.apache.http.*;
012
013/**
014 * Used to intercept http connection responses to allow modification of that response before processing
015 * and for listening for call lifecycle events.
016 * <p>
017 * Useful if you want to prevent {@link RestCallException RestCallExceptions} from being thrown on error conditions.
018 */
019public abstract class RestCallInterceptor {
020
021   /**
022    * Called when {@link RestCall} object is created.
023    *
024    * @param restCall The restCall object invoking this method.
025    */
026   public void onInit(RestCall restCall) {}
027
028   /**
029    * Called immediately after an HTTP response has been received.
030    *
031    * @param statusCode The HTTP status code received.
032    * @param restCall The restCall object invoking this method.
033    * @param req The HTTP request object.
034    * @param res The HTTP response object.
035    */
036   public void onConnect(RestCall restCall, int statusCode, HttpRequest req, HttpResponse res) {}
037
038   /**
039    * Called if retry is going to be attempted.
040    *
041    * @param statusCode The HTTP status code received.
042    * @param restCall The restCall object invoking this method.
043    * @param req The HTTP request object.
044    * @param res The HTTP response object.
045    * @param ex The exception thrown from the client.
046    */
047   public void onRetry(RestCall restCall, int statusCode, HttpRequest req, HttpResponse res, Exception ex) {}
048
049   /**
050    * Called when {@link RestCall#close()} is called.
051    *
052    * @param restCall The restCall object invoking this method.
053    * @throws RestCallException
054    */
055   public void onClose(RestCall restCall) throws RestCallException {}
056}
