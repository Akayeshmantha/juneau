009package com.ibm.juno.client;
010
011/**
012 * Used to determine whether a request should be retried based on the HTTP response code.
013 *
014 * @author James Bognar (jbognar@us.ibm.com)
015 */
016public interface RetryOn {
017
018   /**
019    * Default RetryOn that returns <jk>true</jk> of any HTTP response &gt;= 400 is received.
020    */
021   public static final RetryOn DEFAULT = new RetryOn() {
022      @Override /* RetryOn */
023      public boolean onCode(int httpResponseCode) {
024         return httpResponseCode <= 0 || httpResponseCode >= 400;
025      }
026   };
027
028   /**
029    * Subclasses should override this method to determine whether the HTTP response is retryable.
030    *
031    * @param httpResponseCode The HTTP response code.
032    * @return <jk>true</jk> if the specified response code is retryable.
033    */
034   boolean onCode(int httpResponseCode);
035}
