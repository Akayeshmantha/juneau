009package com.ibm.juno.client;
010
011import java.util.*;
012
013import org.apache.http.*;
014import org.apache.http.client.entity.*;
015
016/**
017 * Convenience class for constructing instances of <code>List&lt;NameValuePair&gt;</code>
018 *    for the {@link UrlEncodedFormEntity} class.
019 *
020 * <h6 class='topic'>Example</h6>
021 * <p class='bcode'>
022 *    NameValuePairs params = <jk>new</jk> NameValuePairs()
023 *       .append(<jk>new</jk> BasicNameValuePair(<js>"j_username"</js>, user))
024 *       .append(<jk>new</jk> BasicNameValuePair(<js>"j_password"</js>, pw));
025 *    request.setEntity(<jk>new</jk> UrlEncodedFormEntity(params));
026 * </p>
027 *
028 * @author James Bognar (jbognar@us.ibm.com)
029 */
030public final class NameValuePairs extends LinkedList<NameValuePair> {
031
032   private static final long serialVersionUID = 1L;
033
034   /**
035    * Appends the specified pair to the end of this list.
036    *
037    * @param pair The pair to append to this list.
038    * @return This object (for method chaining).
039    */
040   public NameValuePairs append(NameValuePair pair) {
041      super.add(pair);
042      return this;
043   }
044}
