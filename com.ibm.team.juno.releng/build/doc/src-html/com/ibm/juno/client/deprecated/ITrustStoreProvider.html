009package com.ibm.juno.client.deprecated;
010
011import java.io.*;
012import java.security.*;
013import java.security.cert.*;
014import java.security.cert.Certificate;
015
016/**
017 * Utility class for handling certificate stores.
018 */
019@Deprecated // Use SimpleX509TrustManager
020public interface ITrustStoreProvider {
021
022   /**
023    * Returns the store of all certificates trusted for the lifetime
024    * of this trust provider
025    */
026   CertificateStore getSessionTrustStore();
027
028   /**
029    * Returns the store of all permanently trusted certificates.
030    */
031   CertificateStore getRuntimeTrustStore();
032
033    /**
034     * Install a certificate in the user's application-specific on-disk key
035     * store, if possible.
036     */
037    public void installCertificate(Certificate cert) throws NoSuchAlgorithmException, CertificateException, KeyStoreException, IOException;
038
039}
