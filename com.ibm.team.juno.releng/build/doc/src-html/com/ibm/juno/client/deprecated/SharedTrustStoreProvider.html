009package com.ibm.juno.client.deprecated;
010
011import java.io.*;
012import java.security.*;
013import java.security.cert.*;
014import java.security.cert.Certificate;
015
016/**
017 * Trust store provider with shared static certificate stores.
018 */
019@Deprecated // Use SimpleX509TrustManager
020public final class SharedTrustStoreProvider implements ITrustStoreProvider {
021
022   // In-memory trust store of all certificates explicitly accepted by the
023   // certificate validator during this session. The validator will not be
024   // called again during this session for any of these certificates. These may
025   // include expired, not yet valid, or otherwise untrusted certificates.
026   // These are kept distinctly, rather than merged into the runtime trust
027   // store, because the base trust manager will never accept expired, etc.
028   // certificates, even if from a trusted source.
029   private static CertificateStore sessionCerts;
030
031   // In-memory trust store of all permanently trusted certificates, assembled
032   // from a number of key store files. These are provided to the base trust
033   // manager as the basis for its decision making.
034   private static CertificateStore runtimeCerts;
035
036   // Location and password of the user's private trust store for this application.
037   private static String userTrustStoreLocation;
038   private static String userTrustStorePassword;
039
040   static {
041      init();
042   }
043
044   private static final void init() {
045      try {
046         String userHome = System.getProperty("user.home");
047         String javaHome = System.getProperty("java.home");
048
049         userTrustStoreLocation = userHome + "/.jazzcerts";
050         userTrustStorePassword = "ibmrationaljazz";
051
052         sessionCerts = new CertificateStore();
053
054         runtimeCerts = new CertificateStore();
055
056         // JRE keystore override
057         String file = System.getProperty("javax.net.ssl.trustStore");
058         String password = System.getProperty("javax.net.ssl.trustStorePassword");
059         addCertificatesFromStore(runtimeCerts, file, password);
060
061         // JRE Signer CA keystore
062         file = javaHome + "/lib/security/cacerts";
063         addCertificatesFromStore(runtimeCerts, file, null);
064
065         // JRE Secure Site CA keystore
066         file =  (javaHome + "/lib/security/jssecacerts");
067         addCertificatesFromStore(runtimeCerts, file, null);
068
069         // Application-specific keystore for the current user
070         addCertificatesFromStore(runtimeCerts, userTrustStoreLocation, userTrustStorePassword);
071
072      } catch (Exception e) {
073         e.printStackTrace();
074      }
075   }
076
077   private static void addCertificatesFromStore(CertificateStore store, String file, String password) {
078      try {
079         File f = new File(file);
080         if (f.canRead())
081            store.loadCertificates(f, password);
082      } catch (Exception e) {
083         // Discard errors
084      }
085   }
086
087   @Override /* ITrustStoreProvider */
088   public CertificateStore getRuntimeTrustStore() {
089      return runtimeCerts;
090   }
091
092   @Override /* ITrustStoreProvider */
093   public CertificateStore getSessionTrustStore() {
094      return sessionCerts;
095   }
096
097   @Override /* ITrustStoreProvider */
098   public void installCertificate(Certificate cert) throws NoSuchAlgorithmException, CertificateException, KeyStoreException, IOException {
099      File  f = new File(userTrustStoreLocation);
100      KeyStore ks = CertificateStore.load(f, userTrustStorePassword);
101      ks.setCertificateEntry(CertificateStore.computeAlias(cert), cert);
102      CertificateStore.store(ks, f, userTrustStorePassword);
103   }
104}
