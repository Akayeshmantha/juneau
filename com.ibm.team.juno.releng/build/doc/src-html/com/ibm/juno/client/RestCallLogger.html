009package com.ibm.juno.client;
010
011import java.io.*;
012import java.text.*;
013import java.util.logging.*;
014
015import org.apache.http.*;
016import org.apache.http.client.methods.*;
017import org.apache.http.util.*;
018
019/**
020 * Specialized interceptor for logging calls to a log file.
021 * <p>
022 * Causes a log entry to be created that shows all the request and response headers and content
023 *    at the end of the request.
024 * <p>
025 * Use the {@link RestClient#logTo(Level, Logger)} and {@link RestCall#logTo(Level, Logger)}
026 * <p>
027 * methods to create instances of this class.
028 */
029public class RestCallLogger extends RestCallInterceptor {
030
031   private Level level;
032   private Logger log;
033
034   /**
035    * Constructor.
036    *
037    * @param level The log level to log messages at.
038    * @param log The logger to log to.
039    */
040   protected RestCallLogger(Level level, Logger log) {
041      this.level = level;
042      this.log = log;
043   }
044
045   @Override /* RestCallInterceptor */
046   public void onInit(RestCall restCall) {
047      if (log.isLoggable(level))
048         restCall.captureResponse();
049   }
050
051   @Override /* RestCallInterceptor */
052   public void onConnect(RestCall restCall, int statusCode, HttpRequest req, HttpResponse res) {
053      // Do nothing.
054   }
055
056   @Override /* RestCallInterceptor */
057   public void onRetry(RestCall restCall, int statusCode, HttpRequest req, HttpResponse res, Exception ex) {
058      if (log.isLoggable(level)) {
059         if (ex == null)
060         log.log(level, MessageFormat.format("Call to {0} returned {1}.  Will retry.", req.getRequestLine().getUri(), statusCode)); //$NON-NLS-1$
061         else
062            log.log(level, MessageFormat.format("Call to {0} caused exception {1}.  Will retry.", req.getRequestLine().getUri(), ex.getLocalizedMessage()), ex); //$NON-NLS-1$
063      }
064   }
065
066   @Override /* RestCallInterceptor */
067   public void onClose(RestCall restCall) throws RestCallException {
068      try {
069         if (log.isLoggable(level)) {
070            String output = restCall.getCapturedResponse();
071            StringBuilder sb = new StringBuilder();
072            HttpUriRequest req = restCall.getRequest();
073            HttpResponse res = restCall.getResponse();
074            if (req != null) {
075               sb.append("\n=== HTTP Call ==================================================================");
076
077               sb.append("\n=== REQUEST ===\n").append(req);
078               sb.append("\n---request headers---");
079               for (Header h : req.getAllHeaders())
080                  sb.append("\n").append(h);
081               if (req instanceof HttpEntityEnclosingRequestBase) {
082                  sb.append("\n---request entity---");
083                  HttpEntityEnclosingRequestBase req2 = (HttpEntityEnclosingRequestBase)req;
084                  HttpEntity e = req2.getEntity();
085                  if (e == null)
086                     sb.append("\nEntity is null");
087                  else {
088                     if (e.getContentType() != null)
089                        sb.append("\n").append(e.getContentType());
090                     if (e.getContentEncoding() != null)
091                        sb.append("\n").append(e.getContentEncoding());
092                     if (e.isRepeatable()) {
093                        try {
094                           sb.append("\n---request content---\n").append(EntityUtils.toString(e));
095                        } catch (Exception ex) {
096                           throw new RuntimeException(ex);
097                        }
098                     }
099                  }
100               }
101            }
102            if (res != null) {
103               sb.append("\n=== RESPONSE ===\n").append(res.getStatusLine());
104               sb.append("\n---response headers---");
105               for (Header h : res.getAllHeaders())
106                  sb.append("\n").append(h);
107               sb.append("\n---response content---\n").append(output);
108               sb.append("\n=== END ========================================================================");
109            }
110            log.log(level, sb.toString());
111         }
112      } catch (IOException e) {
113         log.log(Level.SEVERE, e.getLocalizedMessage(), e);
114      }
115   }
116}
