009package com.ibm.juno.client.deprecated;
010
011import java.io.*;
012import java.security.*;
013import java.security.cert.*;
014
015import javax.net.ssl.*;
016
017/**
018 * A trust manager that will call a registered {@link ICertificateValidator} in
019 * the event that a problematic (e.g. expired, not yet valid) or untrusted
020 * certificate is presented by a server, and react appropriately. This trust
021 * manager will rely on multiple key stores, and manage one of its own. The
022 * managed key store and the session-accepted key store are shared by all trust
023 * manager instances.
024 */
025@Deprecated // Use SimpleX509TrustManager
026public final class ValidatingX509TrustManager implements X509TrustManager {
027
028   // The JRE-provided trust manager used to validate certificates presented by a server.
029   private X509TrustManager baseTrustManager;
030
031   // The registered certificate validator, may be null, called when the base
032   // trust manager rejects a certificate presented by a server.
033   private ICertificateValidator validator;
034
035   private ITrustStoreProvider trustStoreProvider;
036
037   /**
038    * Construct a new ValidatingX509TrustManager.
039    *
040    * @param validator Certificate validator to consult regarding problematic
041    *    certificates, or <code>null</code> to always reject them.
042    */
043   public ValidatingX509TrustManager(ICertificateValidator validator) throws KeyStoreException, NoSuchAlgorithmException {
044      this.validator = validator;
045      this.trustStoreProvider = new SharedTrustStoreProvider();
046
047      // Initialize the base X509 trust manager that will be used to evaluate
048      // certificates presented by the server against the runtime trust store.
049      TrustManagerFactory factory = TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
050      factory.init(trustStoreProvider.getRuntimeTrustStore().getKeyStore());
051      TrustManager[] managers = factory.getTrustManagers();
052      for (TrustManager manager : managers) {
053         if (manager instanceof X509TrustManager) {
054            baseTrustManager = (X509TrustManager) manager; // Take the first X509TrustManager we find
055            return;
056         }
057      }
058      throw new IllegalStateException("Couldn't find JRE's X509TrustManager"); //$NON-NLS-1$
059   }
060
061   @Override /* X509TrustManager */
062   public X509Certificate[] getAcceptedIssuers() {
063      return baseTrustManager.getAcceptedIssuers();
064   }
065
066   @Override /* X509TrustManager */
067   public void checkClientTrusted(X509Certificate[] chain, String authType) throws CertificateException {
068      baseTrustManager.checkClientTrusted(chain, authType);
069   }
070
071   @Override /* X509TrustManager */
072   public void checkServerTrusted(X509Certificate[] chain, String authType) throws CertificateException {
073      X509Certificate cert = chain[0];
074
075      // Has the certificate been OK'd for the session?
076      try {
077         if (trustStoreProvider.getSessionTrustStore().containsCertificate(cert))
078            return;
079      } catch (KeyStoreException e) {
080         // Ignore; proceed to try base trust manager
081      }
082
083      try {
084         // Rely on the base trust manager to check the certificate against the assembled runtime key store
085         baseTrustManager.checkServerTrusted(chain, authType);
086      } catch (CertificateException certEx) {
087
088         // Done if there isn't a validator to consult
089         if (validator == null)
090            throw certEx; // Rejected!
091
092         // Ask the registered certificate validator to rule on the certificate
093         ICertificateValidator.Trust disposition = validator.validate(cert, certEx);
094         switch (disposition) {
095            case REJECT:            throw certEx;
096            case ACCEPT_CONNECTION: break;
097            case ACCEPT_SESSION:    enterCertificate(cert, false); break;
098            case ACCEPT_PERMANENT:  enterCertificate(cert, true); break;
099         }
100      }
101   }
102
103   private void enterCertificate(X509Certificate cert, boolean permanent) throws CertificateException {
104      try {
105         trustStoreProvider.getSessionTrustStore().enterCertificate(cert);
106         if (permanent)
107            trustStoreProvider.installCertificate(cert);
108      } catch (KeyStoreException e) {
109      } catch (NoSuchAlgorithmException e) {
110      } catch (IOException e) {
111      }
112   }
113}
